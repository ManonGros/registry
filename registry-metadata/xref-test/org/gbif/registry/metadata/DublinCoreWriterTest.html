42    private static final String OAI_DC_SCHEMA = "http://www.openarchives.org/OAI/2.0/oai_dc.xsd";
43  
44    @Test
45    public void testWrite() throws Exception {
46      Dataset d = new Dataset();
47      d.setKey(UUID.fromString("bdd601cc-00a7-431c-9724-d5b03170fcb2"));
48      //d.setDoi(new DOI("10.1234/5679"));
49      d.setTitle("This is a keyboard dataset");
50      d.setDataLanguage(Language.FRENCH);
51  
52      Calendar calendar = Calendar.getInstance();
53      calendar.set(2015, 8, 14);
54      d.setPubDate(calendar.getTime());
55      d.setDescription("Description d'un dataset sur les claviers qwerty");
56  
57      d.setHomepage(URI.create("http:///www.gbif.org"));
58      d.setPurpose("Le but de ce dataset est de ...");
59      List<KeywordCollection> listKeyWordColl = Lists.newArrayList();
60      KeywordCollection keyword = new KeywordCollection();
61      keyword.addKeyword("keyboard");
62      keyword.addKeyword("qwerty");
63      listKeyWordColl.add(keyword);
64      d.setKeywordCollections(listKeyWordColl);
65  
66      //try additional properties
67      Map<String, Object> additionalProperties = Maps.newHashMap();
68      additionalProperties.put(DublinCoreWriter.ADDITIONAL_PROPERTY_OCC_COUNT, 3l);
69      additionalProperties.put(DublinCoreWriter.ADDITIONAL_PROPERTY_DC_FORMAT, "application/dwca+zip");
70  
71      List<Citation> citationList = Lists.newArrayList();
72      Citation citation = new Citation();
73      citation.setText("Qwerty U (2015). The World Register of Keyboards. 2015-08-09");
74      citationList.add(citation);
75      d.setBibliographicCitations(citationList);
76  
77      Contact originatorContact = new Contact();
78      originatorContact.setFirstName("Carey");
79      originatorContact.setLastName("Price");
80      originatorContact.setType(ContactType.ORIGINATOR);
81      originatorContact.setPrimary(true);
82  
83      //add the same name twice sith a different ContactType to make sure it will appear only once at the end
84      Contact metadataAuthorContact = new Contact();
85      metadataAuthorContact.setFirstName("Carey");
86      metadataAuthorContact.setLastName("Price");
87      metadataAuthorContact.setType(ContactType.METADATA_AUTHOR);
88      metadataAuthorContact.setPrimary(true);
89  
90      d.setContacts(Lists.newArrayList(originatorContact, metadataAuthorContact));
91  
92      d.setGeographicCoverageDescription("Description de la portée");
93      Date endDate = calendar.getTime();
94      //subtract a day
95      calendar.roll(Calendar.DAY_OF_MONTH, false);
96      Date startDate = calendar.getTime();
97      TemporalCoverage tp = new DateRange(startDate, endDate);
98      d.setTemporalCoverages(Lists.newArrayList(tp));
99  
100     d.setGeographicCoverageDescription("Plusieurs pays");
101     GeospatialCoverage geospatialCoverage = new GeospatialCoverage();
102     geospatialCoverage.setBoundingBox(new BoundingBox(-10, 10, 20, -20));
103     geospatialCoverage.setDescription("Defined by bounding box");
104     d.setGeographicCoverages(Lists.newArrayList(geospatialCoverage));
105 
106     Organization organization = new Organization();
107     organization.setTitle("Qwerty U");
108 
109     StringWriter writer = new StringWriter();
110     DublinCoreWriter.newInstance().writeTo(organization, d, additionalProperties, writer);
111 
112     //Load test file
113     String expectedContent = FileUtils.readFileToString(org.gbif.utils.file.FileUtils.getClasspathFile("dc/qwerty_dc.xml"));
114 
115     //compare without the whitespace characters
116     String expectedFileContent = CharMatcher.WHITESPACE.removeFrom(expectedContent);
117     String actualFileContent = CharMatcher.WHITESPACE.removeFrom(writer.toString());
118 
119     assertEquals(expectedFileContent, actualFileContent);
120 
121     // ensure we have a valid XML file according to the schema
122     XMLValidator.assertXMLAgainstXSD(writer.toString(), OAI_DC_SCHEMA);
123 
124   }
