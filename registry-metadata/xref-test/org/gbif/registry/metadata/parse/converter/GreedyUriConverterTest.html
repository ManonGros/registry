5   import org.junit.Test;

7   import static org.junit.Assert.assertEquals;
8   import static org.junit.Assert.assertNotNull;
9   import static org.junit.Assert.assertNull;

11  public class GreedyUriConverterTest {
12  
13    private static final GreedyUriConverter CONVERTER = new GreedyUriConverter();
14  
15    @Test
16    public void testConvertToType() throws Throwable {
17      // Malformed URLs, including unrecognized protocols
18      URI uri = (URI) CONVERTER.convertToType(URI.class, "- - - ");
19      assertNull(uri);
20      // test a very complete uri
21      uri = (URI) CONVERTER.convertToType(URI.class, "http://gbif.org:8080/my/path/search?q=abies&rank=species#images");
22      assertEquals("images", uri.getFragment());
23      assertEquals("http", uri.getScheme());
24      // test missing scheme
25      uri = (URI) CONVERTER.convertToType(URI.class, "www.gbif.org:8080/my/path/search?q=abies&rank=species#images");
26      assertEquals("images", uri.getFragment());
27      assertEquals("http", uri.getScheme());
28  
29      uri = (URI) CONVERTER.convertToType(URI.class, "//**##");
30      assertNull(uri);
31      uri = (URI) CONVERTER.convertToType(URI.class, "      ");
32      assertNull(uri);
33      uri = (URI) CONVERTER.convertToType(URI.class, "torrent://www.gbif.org");
34      assertEquals("torrent", uri.getScheme());
35  
36      // Invalid URL with recognized protocols
37      uri = (URI) CONVERTER.convertToType(URI.class, "ftp://ftp.gbif.org //h");
38      assertNull(uri);
39  
40      // Conversion here, to greedily capture such URLs starting with www - otherwise they are discarded
41      uri = (URI) CONVERTER.convertToType(URI.class, "www.gbif.org");
42      assertNotNull(uri);
43      assertEquals(URI.create("http://www.gbif.org"), uri);
44      uri = (URI) CONVERTER.convertToType(URI.class, "ftp://ftp.gbif.org");
45      assertNotNull(uri);
46      assertEquals("ftp", uri.getScheme());
47      uri = (URI) CONVERTER.convertToType(URI.class, "http://www.gbif.org");
48      assertNotNull(uri);
49      assertEquals("http", uri.getScheme());
50  
51      assertEquals("http://www.wii.gov.in", CONVERTER.convert("http://www.wii.gov.in").toString());
52      assertEquals("http://www.wii.gov.in", CONVERTER.convert("www.wii.gov.in").toString());
53      assertEquals("http://www.wii.gov.in/hello?q=world", CONVERTER.convert(" www.wii.gov.in/hello?q=world").toString());
54      assertEquals("https://ftp.gbif.org/mine.zip", CONVERTER.convert("https://ftp.gbif.org/mine.zip").toString());
55    }
