001package org.gbif.registry.metadata.parse.converter;
002
003
004import org.gbif.api.util.VocabularyUtils;
005import org.gbif.api.vocabulary.ContactType;
006
007import org.apache.commons.beanutils.converters.AbstractConverter;
008
009/**
010 * {@link org.apache.commons.beanutils.Converter} implementation that handles conversion
011 * to and from <b>InterpretedEnum(value, ContactType)</b> objects.
012 */
013public final class ContactTypeConverter extends AbstractConverter {
014  private static final ContactType DEFAULT_CONTACT_TYPE = ContactType.ADMINISTRATIVE_POINT_OF_CONTACT;
015
016  /**
017   * Construct a <b>InterpretedEnum(value, ContactType)</b> <i>Converter</i> that throws
018   * a {@code ConversionException} if an error occurs.
019   */
020  public ContactTypeConverter() {
021  }
022
023  /**
024   * Construct a <b>InterpretedEnum<String, ContactType</b> <i>Converter</i> that returns
025   * a default value if an error occurs.
026   *
027   * @param defaultValue The default value to be returned
028   *                     if the value to be converted is missing or an error
029   *                     occurs converting the value.
030   */
031  public ContactTypeConverter(Object defaultValue) {
032    super(defaultValue);
033  }
034
035  /**
036   * Return the default type this {@code Converter} handles.
037   *
038   * @return The default type this {@code Converter} handles.
039   */
040  protected Class getDefaultType() {
041    return ContactType.class;
042  }
043
044  /**
045   * <p>Convert String into an InterpretedEnum<String, ContactType.</p> Checks map with alternative values for each
046   * ContactType to get the interpreted ContactType.
047   *
048   * @param type  Data type to which this value should be converted.
049   * @param value The input value to be converted.
050   *
051   * @return The converted InterpretedEnum with interpreted ContactType, or default ContactType if it could not be
052   *         interpreted.
053   *
054   * @throws Throwable if an error occurs converting to the specified type
055   */
056  protected Object convertToType(Class type, Object value) throws Throwable {
057
058    ContactType infer = (ContactType) VocabularyUtils.lookupEnum(value.toString(), ContactType.class);
059    return infer == null ? DEFAULT_CONTACT_TYPE : infer;
060  }
061}
