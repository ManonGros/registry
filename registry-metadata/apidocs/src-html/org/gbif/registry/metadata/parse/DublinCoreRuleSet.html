001package org.gbif.registry.metadata.parse;
002
003import org.gbif.api.vocabulary.Language;
004import org.gbif.registry.metadata.parse.converter.GreedyUriConverter;
005import org.gbif.registry.metadata.parse.converter.LanguageTypeConverter;
006
007import java.net.URI;
008
009import org.apache.commons.beanutils.ConvertUtils;
010import org.apache.commons.digester3.Digester;
011import org.apache.commons.digester3.RuleSetBase;
012
013/**
014 * Digester rules to parse Dublin Core metadata documents together with a DatasetDelegator digester model.
015 */
016public class DublinCoreRuleSet extends RuleSetBase {
017
018    public DublinCoreRuleSet( ) {
019        super("http://purl.org/dc/terms/");
020    }
021
022  private void setupTypeConverters() {
023
024    GreedyUriConverter uriConverter = new GreedyUriConverter();
025    ConvertUtils.register(uriConverter, URI.class);
026
027    LanguageTypeConverter langConverter = new LanguageTypeConverter();
028    ConvertUtils.register(langConverter, Language.class);
029  }
030
031    public void addRuleInstances(Digester digester) {
032      setupTypeConverters();
033
034      // add the rules
035      digester.addCallMethod("*/protocol", "throwIllegalArgumentException");
036      digester.addBeanPropertySetter("*/title", "title");
037      digester.addCallMethod("*/abstract", "addAbstract", 0);
038      digester.addBeanPropertySetter("*/description", "description");
039      digester.addCallMethod("*/subject", "addSubjects", 0);
040      digester.addBeanPropertySetter("*/language", "dataLanguage");
041      digester.addBeanPropertySetter("*/source", "homepage");
042      digester.addCallMethod("*/isFormatOf", "addDataUrl", 0, new Class[]{URI.class});
043      digester.addCallMethod("*/creator", "addCreator", 0);
044      digester.addCallMethod("*/created", "setPubDateAsString", 0);
045      digester.addBeanPropertySetter("*/rights", "rights");
046      digester.addCallMethod("*/license", "addLicense", 0);
047      digester.addCallMethod("*/bibliographicCitation", "addBibCitation", 0);
048      digester.addCallMethod("*/identifier", "addIdentifier", 0);
049    }
050}
