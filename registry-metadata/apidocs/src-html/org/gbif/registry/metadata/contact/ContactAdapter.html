001package org.gbif.registry.metadata.contact;
002
003import org.gbif.api.model.registry.Contact;
004import org.gbif.api.model.registry.Dataset;
005import org.gbif.api.vocabulary.ContactType;
006
007import java.util.List;
008
009import com.google.common.base.Joiner;
010import com.google.common.collect.Lists;
011
012/**
013 *
014 * Adapt the {@link Dataset} {@link Contact} list for what metadata documents generally want.
015 *
016 * @author cgendreau
017 */
018public class ContactAdapter {
019
020  private static final Joiner JOINER = Joiner.on(" ").skipNulls();
021
022  private List<Contact> contactList;
023
024  public ContactAdapter(List<Contact> contacts) {
025    this.contactList = contacts;
026  }
027
028  /**
029   * Get the list of AssociatedParties.
030   * This is defined as all non-primary {@link Contact}.
031   *
032   * @return first preferred ResourceCreator found or null if none were found
033   */
034  public List<Contact> getAssociatedParties() {
035    List<Contact> contacts = Lists.newArrayList();
036    for (Contact c : this.contactList) {
037      if (!c.isPrimary()) {
038        contacts.add(c);
039      }
040    }
041    return contacts;
042  }
043
044  /**
045   * Get the ResourceCreator {@link Contact}.
046   * This is defined as the first primary {@link Contact} of type ContactType.ORIGINATOR.
047   *
048   * @return first preferred ResourceCreator found or null if none were found
049   */
050  public Contact getResourceCreator() {
051    return getFirstPreferredType(ContactType.ORIGINATOR);
052  }
053
054
055  /**
056   * Format the name of the contact as "FirstName LastName".
057   *
058   * @param contact
059   * @return formatted name or "" if the contact is null or empty
060   */
061  public static String formatContactName(Contact contact){
062    if(contact == null){
063      return "";
064    }
065    return JOINER.join(contact.getFirstName(), contact.getLastName()).trim();
066  }
067
068  /**
069   * Get the AdministrativeContact {@link Contact}.
070   * This is defined as the first primary {@link Contact} of type ContactType.ADMINISTRATIVE_POINT_OF_CONTACT.
071   *
072   * @return first preferred AdministrativeContact found or null if none were found
073   */
074  public Contact getAdministrativeContact() {
075    return getFirstPreferredType(ContactType.ADMINISTRATIVE_POINT_OF_CONTACT);
076  }
077
078  /**
079   * Filter contacts based on the provided contact types.
080   * The order in which the ContactType are provided will be respected in the response except missing
081   * ContactType will be ommited.
082   * Filtering is done by the {@link #getFirstPreferredType} method.
083
084   *
085   * @param types
086   * @return filtered contacts or an empty list if none matched
087   */
088  public List<Contact> getFilteredContacts(ContactType ... types){
089    List<Contact> contacts = Lists.newArrayList();
090    Contact contact;
091    for(ContactType type : types){
092      contact = getFirstPreferredType(type);
093      if(contact != null){
094        contacts.add(contact);
095      }
096    }
097    return contacts;
098  }
099
100  /**
101   * Get the first primary {@link Contact} for the provided type.
102   *
103   * @return first preferred type contact found or null if nothing were found
104   */
105  public Contact getFirstPreferredType(ContactType type) {
106    Contact pref = null;
107    for (Contact c : contactList) {
108      if (type == c.getType()) {
109        if (pref == null || c.isPrimary()) {
110          pref = c;
111        }
112      }
113    }
114    return pref;
115  }
116
117}
