031/**
032 * Test class for DublinCoreWriter.
033 *
034 * @author cgendreau
035 */
036public class DublinCoreWriterTest {
037
038  private static final String OAI_DC_SCHEMA = "http://www.openarchives.org/OAI/2.0/oai_dc.xsd";
039
040  @Test
041  public void testWrite() throws Exception {
042    Dataset d = new Dataset();
043    d.setKey(UUID.fromString("bdd601cc-00a7-431c-9724-d5b03170fcb2"));
044    //d.setDoi(new DOI("10.1234/5679"));
045    d.setTitle("This is a keyboard dataset");
046    d.setDataLanguage(Language.FRENCH);
047
048    Calendar calendar = Calendar.getInstance();
049    calendar.set(2015, 8, 14);
050    d.setPubDate(calendar.getTime());
051    d.setDescription("Description d'un dataset sur les claviers qwerty");
052
053    d.setHomepage(URI.create("http:///www.gbif.org"));
054    d.setPurpose("Le but de ce dataset est de ...");
055    List<KeywordCollection> listKeyWordColl = Lists.newArrayList();
056    KeywordCollection keyword = new KeywordCollection();
057    keyword.addKeyword("keyboard");
058    keyword.addKeyword("qwerty");
059    listKeyWordColl.add(keyword);
060    d.setKeywordCollections(listKeyWordColl);
061
062    //try additional properties
063    Map<String, Object> additionalProperties = Maps.newHashMap();
064    additionalProperties.put(DublinCoreWriter.ADDITIONAL_PROPERTY_OCC_COUNT, 3l);
065    additionalProperties.put(DublinCoreWriter.ADDITIONAL_PROPERTY_DC_FORMAT, "application/dwca+zip");
066
067    List<Citation> citationList = Lists.newArrayList();
068    Citation citation = new Citation();
069    citation.setText("Qwerty U (2015). The World Register of Keyboards. 2015-08-09");
070    citationList.add(citation);
071    d.setBibliographicCitations(citationList);
072
073    Contact originatorContact = new Contact();
074    originatorContact.setFirstName("Carey");
075    originatorContact.setLastName("Price");
076    originatorContact.setType(ContactType.ORIGINATOR);
077    originatorContact.setPrimary(true);
078
079    //add the same name twice sith a different ContactType to make sure it will appear only once at the end
080    Contact metadataAuthorContact = new Contact();
081    metadataAuthorContact.setFirstName("Carey");
082    metadataAuthorContact.setLastName("Price");
083    metadataAuthorContact.setType(ContactType.METADATA_AUTHOR);
084    metadataAuthorContact.setPrimary(true);
085
086    // ADMINISTRATIVE_POINT_OF_CONTACT should be displayed first in the generated DublinCore document
087    Contact administrativeContact = new Contact();
088    administrativeContact.setFirstName("Patrick");
089    administrativeContact.setLastName("Roy");
090    administrativeContact.setType(ContactType.ADMINISTRATIVE_POINT_OF_CONTACT);
091
092    d.setContacts(Lists.newArrayList(originatorContact, metadataAuthorContact, administrativeContact));
093
094    d.setGeographicCoverageDescription("Description de la port√©e");
095    Date endDate = calendar.getTime();
096    //subtract a day
097    calendar.roll(Calendar.DAY_OF_MONTH, false);
098    Date startDate = calendar.getTime();
099    TemporalCoverage tp = new DateRange(startDate, endDate);
100    d.setTemporalCoverages(Lists.newArrayList(tp));
101
102    d.setGeographicCoverageDescription("Plusieurs pays");
103    GeospatialCoverage geospatialCoverage = new GeospatialCoverage();
104    geospatialCoverage.setBoundingBox(new BoundingBox(-10, 10, 20, -20));
105    geospatialCoverage.setDescription("Defined by bounding box");
106    d.setGeographicCoverages(Lists.newArrayList(geospatialCoverage));
107
108    Organization organization = new Organization();
109    organization.setTitle("Qwerty U");
110
111    StringWriter writer = new StringWriter();
112    DublinCoreWriter.newInstance().writeTo(organization, d, additionalProperties, writer);
113
114    //Load test file
115    String expectedContent = FileUtils.readFileToString(org.gbif.utils.file.FileUtils.getClasspathFile("dc/qwerty_dc.xml"));
116
117    //compare without the whitespace characters
118    String expectedFileContent = CharMatcher.WHITESPACE.removeFrom(expectedContent);
119    String actualFileContent = CharMatcher.WHITESPACE.removeFrom(writer.toString());
120
121    assertEquals(expectedFileContent, actualFileContent);
122
123    // ensure we have a valid XML file according to the schema
124    XMLValidator.assertXMLAgainstXSD(writer.toString(), OAI_DC_SCHEMA);
125
126  }
127
