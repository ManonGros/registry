1   package org.gbif.registry.metadata.parse;
2   
3   import org.gbif.api.vocabulary.Language;
4   import org.gbif.registry.metadata.parse.converter.GreedyUriConverter;
5   import org.gbif.registry.metadata.parse.converter.LanguageTypeConverter;
6   
7   import java.net.URI;
8   
9   import org.apache.commons.beanutils.ConvertUtils;
10  import org.apache.commons.digester3.Digester;
11  import org.apache.commons.digester3.RuleSetBase;
12  
13  /**
14   * Digester rules to parse Dublin Core metadata documents together with a DatasetDelegator digester model.
15   */
16  public class DublinCoreRuleSet extends RuleSetBase {
17  
18      public DublinCoreRuleSet( ) {
19          super("http://purl.org/dc/terms/");
20      }
21  
22    private void setupTypeConverters() {
23  
24      GreedyUriConverter uriConverter = new GreedyUriConverter();
25      ConvertUtils.register(uriConverter, URI.class);
26  
27      LanguageTypeConverter langConverter = new LanguageTypeConverter();
28      ConvertUtils.register(langConverter, Language.class);
29    }
30  
31      public void addRuleInstances(Digester digester) {
32        setupTypeConverters();
33  
34        // add the rules
35        digester.addCallMethod("*/protocol", "throwIllegalArgumentException");
36        digester.addBeanPropertySetter("*/title", "title");
37        digester.addCallMethod("*/abstract", "addAbstract", 0);
38        digester.addBeanPropertySetter("*/description", "description");
39        digester.addCallMethod("*/subject", "addSubjects", 0);
40        digester.addBeanPropertySetter("*/language", "dataLanguage");
41        digester.addBeanPropertySetter("*/source", "homepage");
42        digester.addCallMethod("*/isFormatOf", "addDataUrl", 0, new Class[]{URI.class});
43        digester.addCallMethod("*/creator", "addCreator", 0);
44        digester.addCallMethod("*/created", "setPubDateAsString", 0);
45        digester.addBeanPropertySetter("*/rights", "rights");
46        digester.addCallMethod("*/license", "addLicense", 0);
47        digester.addCallMethod("*/bibliographicCitation", "addBibCitation", 0);
48        digester.addCallMethod("*/identifier", "addIdentifier", 0);
49      }
50  }
