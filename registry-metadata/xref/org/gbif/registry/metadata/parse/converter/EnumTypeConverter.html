1   package org.gbif.registry.metadata.parse.converter;
2   
3   import com.google.common.collect.Maps;
4   import org.apache.commons.beanutils.converters.AbstractConverter;
5   import org.gbif.api.util.VocabularyUtils;
6   
7   import java.util.Map;
8   
9   /**
10   * {@link org.apache.commons.beanutils.Converter} implementation that handles conversion
11   * to and from java enumerations.
12   */
13  public class EnumTypeConverter<T extends Enum<?>> extends AbstractConverter {
14  
15      private final Class<T> clazz;
16      private final T defaultValue;
17  
18      private final Map<String, T> lookup = Maps.newHashMap();
19  
20      public EnumTypeConverter(Class<T> clazz, T defaultValue) {
21          this.clazz = clazz;
22          this.defaultValue = defaultValue;
23      }
24  
25  
26      public void addMappings(Map<String, T> mapping) {
27          for (Map.Entry<String, T> entry : mapping.entrySet()) {
28              lookup.put(entry.getKey().toLowerCase(), entry.getValue());
29          }
30      }
31  
32      /**
33       * Return the default type this <code>Converter</code> handles.
34       *
35       * @return The default type this <code>Converter</code> handles.
36       */
37      protected Class getDefaultType() {
38          return clazz;
39      }
40  
41      /**
42       * <p>Convert a PreservationMethodType or object into a String.</p> Checks map with alternative values for each
43       * PreservationMethodType before returning the default value.
44       *
45       * @param type  Data type to which this value should be converted
46       * @param value The input value to be converted
47       * @return The converted value.
48       * @throws Throwable if an error occurs converting to the specified type
49       */
50      protected Object convertToType(Class type, Object value) throws Throwable {
51          // never null, super class implements this as:
52          // return value.toString();
53          final String val = value.toString();
54          if (lookup.containsKey(val)) {
55              return lookup.get(val);
56          }
57          // try regular enum values
58          try {
59              T eVal = (T) VocabularyUtils.lookupEnum(val, clazz);
60              return eVal == null ? defaultValue : eVal;
61          } catch (IllegalArgumentException e) {
62              // cant parse, return default
63              return defaultValue;
64          }
65      }
66  }
