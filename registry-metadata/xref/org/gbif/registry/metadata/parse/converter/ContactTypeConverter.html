1   package org.gbif.registry.metadata.parse.converter;
2   
3   
4   import org.gbif.api.util.VocabularyUtils;
5   import org.gbif.api.vocabulary.ContactType;
6   
7   import org.apache.commons.beanutils.converters.AbstractConverter;
8   
9   /**
10   * {@link org.apache.commons.beanutils.Converter} implementation that handles conversion
11   * to and from <b>InterpretedEnum(value, ContactType)</b> objects.
12   */
13  public final class ContactTypeConverter extends AbstractConverter {
14    private static final ContactType DEFAULT_CONTACT_TYPE = ContactType.ADMINISTRATIVE_POINT_OF_CONTACT;
15  
16    /**
17     * Construct a <b>InterpretedEnum(value, ContactType)</b> <i>Converter</i> that throws
18     * a {@code ConversionException} if an error occurs.
19     */
20    public ContactTypeConverter() {
21    }
22  
23    /**
24     * Construct a <b>InterpretedEnum<String, ContactType</b> <i>Converter</i> that returns
25     * a default value if an error occurs.
26     *
27     * @param defaultValue The default value to be returned
28     *                     if the value to be converted is missing or an error
29     *                     occurs converting the value.
30     */
31    public ContactTypeConverter(Object defaultValue) {
32      super(defaultValue);
33    }
34  
35    /**
36     * Return the default type this {@code Converter} handles.
37     *
38     * @return The default type this {@code Converter} handles.
39     */
40    protected Class getDefaultType() {
41      return ContactType.class;
42    }
43  
44    /**
45     * <p>Convert String into an InterpretedEnum<String, ContactType.</p> Checks map with alternative values for each
46     * ContactType to get the interpreted ContactType.
47     *
48     * @param type  Data type to which this value should be converted.
49     * @param value The input value to be converted.
50     *
51     * @return The converted InterpretedEnum with interpreted ContactType, or default ContactType if it could not be
52     *         interpreted.
53     *
54     * @throws Throwable if an error occurs converting to the specified type
55     */
56    protected Object convertToType(Class type, Object value) throws Throwable {
57  
58      ContactType infer = (ContactType) VocabularyUtils.lookupEnum(value.toString(), ContactType.class);
59      return infer == null ? DEFAULT_CONTACT_TYPE : infer;
60    }
61  }
