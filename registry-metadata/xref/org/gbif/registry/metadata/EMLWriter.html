23   * A simple tool to serialize a dataset object into an EML GBIF profile compliant xml document.
24   */
25  public class EMLWriter {

27    private static final String TEMPLATE_PATH = "/gbif-eml-profile-template";
28    private static final String EML_TEMPLATE = String.format("eml-dataset-%s.ftl", EMLProfileVersion.GBIF_1_0_2.getVersion());
29    private final Configuration freemarkerConfig;
30    private final boolean useDoiAsIdentifier;
31  
32    private static final Configuration FTL = DatasetXMLWriterConfigurationProvider.provideFreemarker(TEMPLATE_PATH);

35     * Private constructor, use {@link #newInstance()}

37     */
38    private EMLWriter(Configuration cfg, boolean useDoiAsIdentifier) {
39      this.freemarkerConfig = cfg;

44     * Get a new instance of EMLWriter with default Freemarker configuration.

64     * Write a document from a Dataset object.

75    /**
76     * Wrapper for a dataset instance that exposes some very EML specific methods.
77     * Mostly used for generating EML, see EMLWriter.
78     * This class requires to be public to be used in the Freemarker template.
79     */
80    public static class EmlDatasetWrapper {
81  
82      private final Dataset dataset;
83      private final ContactAdapter contactAdapter;
84  
85      public EmlDatasetWrapper(Dataset dataset) {
86        this.dataset = dataset;
87        this.contactAdapter = new ContactAdapter(dataset.getContacts());
88      }
89  
90      public List<Contact> getAssociatedParties() {
91        return contactAdapter.getAssociatedParties();
92      }
93  
94      public Contact getResourceCreator() {
95        return contactAdapter.getResourceCreator();
96      }
97  
98      public Contact getAdministrativeContact() {
99        return contactAdapter.getAdministrativeContact();
100     }
101 
102     public Contact getMetadataProvider() {
103       return contactAdapter.getFirstPreferredType(ContactType.METADATA_AUTHOR);
104     }
105 
106     public List getFormationPeriods() {
107       return getTimePeriods(VerbatimTimePeriodType.FORMATION_PERIOD);
108     }
109 
110     public List getLivingTimePeriods() {
111       return getTimePeriods(VerbatimTimePeriodType.LIVING_TIME_PERIOD);
112     }
113 
114     private List<VerbatimTimePeriod> getTimePeriods(VerbatimTimePeriodType type) {
115       List<VerbatimTimePeriod> periods = Lists.newArrayList();
116       for (TemporalCoverage tc : dataset.getTemporalCoverages()) {
117         if (tc instanceof VerbatimTimePeriod) {
118           VerbatimTimePeriod tp = (VerbatimTimePeriod) tc;
119           if (type.equals(tp.getType())) {
120             periods.add(tp);
121           }
122         }
123       }
124       return periods;
125     }
126 
127   }

142   /**
143    * This method will be deprecated soon, please use an instance {@link #newInstance()}
144    *
145    * Write an EML document from a Dataset object.
146    *
147    * @param dataset non null dataset object
148    * @param writer where the output document will go. The writer is not closed by this method.
149    * @param _useDoiAsIdentifier should the packageId be the dataset.doi? If true, the dataset.doi won't be included in the list of alternate identifiers.
150    * @throws IOException if an error occurs while processing the template
151    */
152   public static void write(Dataset dataset, Writer writer, boolean _useDoiAsIdentifier) throws IOException {
153     Preconditions.checkNotNull(dataset, "Dataset can't be null");
154 
155     Map<String, Object> map = ImmutableMap.of
156             ("dataset", dataset, "eml", new EmlDatasetWrapper(dataset), "useDoiAsIdentifier", _useDoiAsIdentifier);
157 
158     try {
159       FTL.getTemplate(EML_TEMPLATE).process(map, writer);
160     } catch (TemplateException e) {
161       throw new IOException("Error while processing the EML Freemarker template for dataset " + dataset.getKey(), e);
162     }
163   }
