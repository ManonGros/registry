13  /**
14   * Configuration provider for XML based dataset writer.
15   * Currently providing Freemarker configuration.
16   *
17   */
18  class DatasetXMLWriterConfigurationProvider {
19  
20    public static final String FM_UTILS_NAME = "fmUtil";
21    private static final String TEMPLATE_PATH = "/";
22    private static final Logger LOG = LoggerFactory.getLogger(DatasetXMLWriterConfigurationProvider.class);
23  
24    /**
25     * Provides a Freemarker template loader. It is configured to access the UTF-8 templates folder on the classpath.
26     */
27    public static Configuration provideFreemarker() {
28      return provideFreemarker(TEMPLATE_PATH);
29    }
30  
31    /**
32     * Provides a Freemarker template loader. It is configured to access the UTF-8 templates folder on the classpath.
33     * @param templatePath Freemarker base package path
34     */
35    public static Configuration provideFreemarker(String templatePath) {
36      // load templates from classpath by prefixing /templates
37      TemplateLoader tl = new ClassTemplateLoader(DatasetXMLWriterConfigurationProvider.class, templatePath);
38  
39      // Using deprecated constructor
40      // From Freemarker documentation:
41      // Configuration cfg = new Configuration(VERSION_X_Y_Z));
42      // Where X, Y, Z enables the not-100%-backward-compatible fixes introduced in
43      // FreeMarker version X.Y.Z  and earlier (see Configuration(Version)).
44      // To be safe, we do not use any of the new features but it would be useful to have a minimum version
45      // of Freemarker to use.
46      Configuration fm = new Configuration();
47  
48      fm.setDefaultEncoding(StandardCharsets.UTF_8.toString());
49      fm.setTemplateLoader(tl);
50      try {
51        fm.setSharedVariable(FM_UTILS_NAME, new FreemarkerUtils());
52      } catch (TemplateModelException e) {
53        LOG.error("Can't set sharedVariable in Freemarker Configuration", e);
54      }
55      return fm;
56    }
57  
58    /**
59     * Nested class to expose some Java utilities to Freemarker template.
60     * This class requires to be public to be visible by the Freemarker template.
61     *
62     */
63    public static final class FreemarkerUtils{
64  
65      /**
66       * Allows to return a different string based on the number provided.
67       * Useful for pluralization or words that depends on a variable in templates
68       * @see ChoiceFormat for patterns
69       * @param pattern "-1#is negative| 0#is zero or fraction | 1#is one |1.0<is 1+ |2#is two |2<is more than 2."
70       * @param number
71       */
72      public String choiceFormat(String pattern, double number){
73        ChoiceFormat fmt = new ChoiceFormat(pattern);
74        return fmt.format(number);
75      }
76    }
77  }
