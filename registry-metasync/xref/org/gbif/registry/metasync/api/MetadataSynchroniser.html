1   package org.gbif.registry.metasync.api;
2   
3   import java.util.List;
4   import java.util.UUID;
5   
6   /**
7    * This interface specifies all operations necessary to do a synchronisation of an Installation from our registry
8    * against the actual Endpoint.
9    */
10  public interface MetadataSynchroniser {
11  
12    /**
13     * Runs a synchronisation against the installation provided.
14     * <p/>
15     * All datasets from this installation will be updated, old ones deleted and new ones registered if needed.
16     *
17     * @param key of the installation to synchronise
18     *
19     * @return the result of the synchronisation
20     *
21     * @throws IllegalArgumentException if no technical installation with the given key exists, the installation doesn't
22     *                                  have any endpoints or none that we can use to gather metadata (might also be
23     *                                  missing the proper protocol handler)
24     */
25    SyncResult synchroniseInstallation(UUID key);
26  
27    /**
28     * Synchronises all registered Installations ignoring any failures (because there will definitely be failures due to
29     * unsupported Installation types and missing Endpoints). This will run single threaded.
30     */
31    List<SyncResult> synchroniseAllInstallations();
32  
33    /**
34     * Synchronises all registered Installations ignoring any failures (because there will definitely be failures due to
35     * unsupported Installation types and missing Endpoints).
36     *
37     * @param parallel how many threads to run in parallel
38     */
39    List<SyncResult> synchroniseAllInstallations(int parallel);
40  
41  }
