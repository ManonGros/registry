3   import java.util.concurrent.TimeUnit;
4   
5   import org.apache.http.client.HttpClient;
6   import org.apache.http.client.params.ClientPNames;
7   import org.apache.http.conn.ClientConnectionManager;
8   import org.apache.http.conn.scheme.PlainSocketFactory;
9   import org.apache.http.conn.scheme.Scheme;
10  import org.apache.http.conn.scheme.SchemeRegistry;
11  import org.apache.http.impl.client.DecompressingHttpClient;
12  import org.apache.http.impl.client.DefaultHttpClient;
13  import org.apache.http.impl.conn.PoolingClientConnectionManager;
14  import org.apache.http.params.BasicHttpParams;
15  import org.apache.http.params.HttpConnectionParams;
16  import org.apache.http.params.HttpParams;
17  
18  /**
19   * A factory that builds new Apache Http Clients to be used with the Metadata synchroniser.
20   *
21   * This allows for easy swapping of implementations to make testing easier.
22   */
23  public class HttpClientFactory {
24  
25    private static final int HTTP_PORT = 80;
26    private static final int HTTPS_PORT = 443;
27  
28    private static final int MAX_TOTAL_CONNECTIONS = 200;
29    private static final int MAX_CONNECTIONS_PER_HOST = 20;
30  
31    private final ClientConnectionManager connectionManager;
32  
33    private final int timeout;
34  
35    /**
36     * Builds a new Factory with a default timeout covering various things (connection timeout, read timeout, ...).
37     */
38    public HttpClientFactory(long timeout, TimeUnit timeUnit) {
39      long millis = TimeUnit.MILLISECONDS.convert(timeout, timeUnit);
40  
41      System.setProperty("sun.net.client.defaultConnectTimeout", String.valueOf(millis));
42      System.setProperty("sun.net.client.defaultReadTimeout", String.valueOf(millis));
43  
44      connectionManager = setupConnectionManager();
45      this.timeout = (int) millis;
46    }
47  
48    public HttpClient provideHttpClient() {
49      HttpParams params = new BasicHttpParams();
50      HttpConnectionParams.setConnectionTimeout(params, timeout);
51      HttpConnectionParams.setSoTimeout(params, timeout);
52      params.setLongParameter(ClientPNames.CONN_MANAGER_TIMEOUT, timeout);
53      return new DecompressingHttpClient(new DefaultHttpClient(connectionManager, params));
54    }
55  
56    private ClientConnectionManager setupConnectionManager() {
57      SchemeRegistry schemeRegistry = new SchemeRegistry();
58      schemeRegistry.register(new Scheme("http", HTTP_PORT, PlainSocketFactory.getSocketFactory()));
59      schemeRegistry.register(new Scheme("https", HTTPS_PORT, PlainSocketFactory.getSocketFactory()));
60  
61      PoolingClientConnectionManager newConnectionManager = new PoolingClientConnectionManager(schemeRegistry);
62      newConnectionManager.setMaxTotal(MAX_TOTAL_CONNECTIONS);
63      newConnectionManager.setDefaultMaxPerRoute(MAX_CONNECTIONS_PER_HOST);
64      return newConnectionManager;
65    }
66  
67  }
