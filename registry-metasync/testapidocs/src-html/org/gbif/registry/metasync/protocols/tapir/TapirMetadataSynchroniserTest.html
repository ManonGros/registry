043  @Before
044  public void setup() {
045    synchroniser = new TapirMetadataSynchroniser(client);
046
047    installation = new Installation();
048    installation.setType(InstallationType.TAPIR_INSTALLATION);
049    Endpoint endpoint = new Endpoint();
050    endpoint.setUrl(URI.create("http://localhost/nmr"));
051    installation.addEndpoint(endpoint);
052  }
053
054  @Test
055  public void testCanHandle() {
056    installation.setType(InstallationType.BIOCASE_INSTALLATION);
057    assertThat(synchroniser.canHandle(installation)).isFalse();
058
059    installation.setType(InstallationType.TAPIR_INSTALLATION);
060    assertThat(synchroniser.canHandle(installation)).isTrue();
061  }
062
063  /**
064   * A simple test to see if multiple datasets are parsed successfully.
065   */
066  @Test
067  public void testAddedDatasets() throws Exception {
068    when(client.execute(argThat(HttpGetMatcher.matchUrl("http://localhost/nmr?op=capabilities")))).thenReturn(
069      prepareResponse(200, "tapir/capabilities1.xml"));
070    when(client.execute(argThat(HttpGetMatcher.matchUrl("http://localhost/nmr")))).thenReturn(
071      prepareResponse(200, "tapir/metadata1.xml"));
072    when(
073      client.execute(argThat(HttpGetMatcher
074        .matchUrl("http://localhost/nmr?op=s&t=http%3A%2F%2Frs.gbif.org%2Ftemplates%2Ftapir%2Fdwc%2F1.4%2Fsci_name_range.xml&count=true&start=0&limit=1&lower=AAA&upper=zzz"))))
075      .thenReturn(
076        prepareResponse(200, "tapir/search1.xml"));
077    SyncResult syncResult = synchroniser.syncInstallation(installation, new ArrayList<Dataset>());
078    assertThat(syncResult.deletedDatasets).isEmpty();
079    assertThat(syncResult.existingDatasets).isEmpty();
080    assertThat(syncResult.addedDatasets).hasSize(1);
081    assertThat(syncResult.addedDatasets.get(0).getContacts()).hasSize(2);
082    assertThat(syncResult.addedDatasets.get(0).getMachineTags()).hasSize(2);
083    assertThat(syncResult.addedDatasets.get(0).getDoi()).isEqualTo(new DOI("10.1234/doi"));
084
085    // Assert the declared record count machine tag was found, and that its value was 167348
086    MachineTag count = null;
087    for (MachineTag tag : syncResult.addedDatasets.get(0).getMachineTags()) {
088      if (tag.getName().equalsIgnoreCase(Constants.DECLARED_COUNT)) {
089        count = tag;
090      }
091    }
092    assertThat(count).isNotNull();
093    assertThat(Integer.valueOf(count.getValue())).isEqualTo(167348);
094  }

146  public HttpResponse prepareResponse(int responseStatus, String fileName) throws IOException {
147    HttpResponse response =
148      new BasicHttpResponse(new BasicStatusLine(new ProtocolVersion("HTTP", 1, 1), responseStatus, ""));
149    response.setStatusCode(responseStatus);
150    byte[] bytes = Resources.toByteArray(Resources.getResource(fileName));
151    response.setEntity(new ByteArrayEntity(bytes));
152    return response;
153  }
