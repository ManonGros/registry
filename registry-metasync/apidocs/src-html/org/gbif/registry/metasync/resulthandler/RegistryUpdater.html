004import org.gbif.api.model.registry.Dataset;
005import org.gbif.api.model.registry.Endpoint;
006import org.gbif.api.model.registry.Identifier;
007import org.gbif.api.model.registry.MachineTag;
008import org.gbif.api.service.registry.DatasetService;
009import org.gbif.api.service.registry.MetasyncHistoryService;
010import org.gbif.api.vocabulary.DatasetType;
011import org.gbif.api.vocabulary.EndpointType;
012import org.gbif.registry.metasync.api.SyncResult;
013import org.gbif.registry.metasync.util.Constants;
014
015import java.util.Map;
016import java.util.UUID;
017
018import com.google.common.annotations.VisibleForTesting;
019import org.slf4j.Logger;
020import org.slf4j.LoggerFactory;

022/**
023 * Takes synchronisation results and saves those back to the registry.
024 */
025public class RegistryUpdater {
026
027  private static final Logger LOG = LoggerFactory.getLogger(RegistryUpdater.class);
028  private final DatasetService datasetService;
029  private final MetasyncHistoryService historyService;
030
031  public RegistryUpdater(DatasetService datasetService, MetasyncHistoryService historyService) {
032    this.datasetService = datasetService;
033    this.historyService = historyService;
034  }
035
036  @VisibleForTesting
037  public DatasetService getDatasetService() {
038    return datasetService;
039  }
040
041  public void saveSyncResultsToRegistry(Iterable<SyncResult> syncResults) {
042    for (SyncResult syncResult : syncResults) {
043      saveSyncResults(syncResult);
044    }
045  }
046
047  /**
048   * Iterate through the updated datasets. For each dataset, update the dataset itself, and then update its contacts,
049   * identifiers, machine tags, and endpoints. Tags remain unchanged by the metadata synchronizer, and therefore aren't
050   * updated. The following assumptions apply:
051   * </br>
052   * Contacts: are derived 100% from the metadata response.
053   * </br>
054   * Machine tags: with metasync.gbif.org namespace are derived 100% from metadata synchronizer
055   * </br>
056   * Endpoints: only 1 endpoint exists per dataset.
057   * </br>
058   * Identifiers:
059   *
060   * @param result SyncResult
061   */
062  void saveUpdatedDatasets(SyncResult result) {
063    for (Map.Entry<Dataset, Dataset> entry : result.existingDatasets.entrySet()) {
064      Dataset existingDataset = entry.getKey();
065      UUID datasetKey = existingDataset.getKey();
066      if (skipDatasetUpdate(existingDataset)) {
067        LOG.info("Dataset [{}] updated at source untouched in Registry because it's locked or contains a DwC-A "
068          + "endpoint, which takes precedence", datasetKey);
069      } else {
070        LOG.info("Updating dataset [{}]", datasetKey);
071
072        // the updated dataset created from the metadata synchronization
073        Dataset updated = entry.getValue();
074
075        // safest, preserve existing dataset, copying over only those properties that could have changed from the sync
076        existingDataset.setHomepage(updated.getHomepage());
077        existingDataset.setLanguage(updated.getLanguage());
078        existingDataset.setCitation(updated.getCitation());
079        existingDataset.setTitle(updated.getTitle());
080        existingDataset.setDescription(updated.getDescription());
081        existingDataset.setLogoUrl(updated.getLogoUrl());
082        existingDataset.setRights(updated.getRights());
083        // perform update
084        datasetService.update(existingDataset);
085
086        // Contacts are derived 100% from the metadata
087        // delete existing contacts, and replace with new/updated contacts
088        for (Contact contact : existingDataset.getContacts()) {
089          datasetService.deleteContact(datasetKey, contact.getKey());
090        }
091        for (Contact contact : updated.getContacts()) {
092          datasetService.addContact(datasetKey, contact);
093        }
094
095        // Machine tags with namepace "metasync.gbif.org" are derived 100% from the metadata sync
096        // delete existing machine tags in this namespace, and replace with new/updated machine tags
097        for (MachineTag machineTag : existingDataset.getMachineTags()) {
098          // TODO: replace below with datasetService.deleteMachineTags(uuid, namespace); once implemented
099          if (machineTag.getNamespace().equalsIgnoreCase(Constants.METADATA_NAMESPACE)) {
100            datasetService.deleteMachineTag(datasetKey, machineTag.getKey());
101          }
102        }
103        for (MachineTag machineTag : updated.getMachineTags()) {
104          datasetService.addMachineTag(datasetKey, machineTag);
105        }
106
107        // Only 1 endpoint exists per Dataset
108        // delete existing endpoint, and replace with new/updated endpoint
109        for (Endpoint endpoint : existingDataset.getEndpoints()) {
110          datasetService.deleteEndpoint(datasetKey, endpoint.getKey());
111        }
112        for (Endpoint endpoint : updated.getEndpoints()) {
113          datasetService.addEndpoint(datasetKey, endpoint);
114        }
115
116        // Only add identifiers that don't yet exist
117        // Identifiers like GBIF Portal ID aren't created during metadata sync, so never delete these
118        if (!updated.getIdentifiers().isEmpty()) {
119          for (Identifier identifier : updated.getIdentifiers()) {
120            // does this identifier exist already?
121            boolean found = false;
122            for (Identifier existingIdentifier : existingDataset.getIdentifiers()) {
123              if (identifier.lenientEquals(existingIdentifier)) {
124                found = true;
125                break;
126              }
127            }
128            if (!found) {
129              datasetService.addIdentifier(datasetKey, identifier);
130            }
131          }
132        }
133      }
134    }
135  }
136
137  /**
138   * Processes the result of a synchronisation by:
139   * <ul>
140   * <li>Creating new Datasets and Endpoints</li>
141   * <li>Deleting existing Datasets</li>
142   * <li>Updating existing Installations, Datasets and Endpoints</li>
143   * </ul>
144   */
145  private void saveSyncResults(SyncResult result) {
146    if (result.exception == null) {
147      saveAddedDatasets(result);
148      saveDeletedDatasets(result);
149      saveUpdatedDatasets(result);
150    } else {
151      LOG.warn("Installation [{}] failed sync because of [{}]",
152        result.installation.getKey(),
153        result.exception.getMessage());
154    }
155    historyService.createMetasync(result.buildHistory());
156
157  }
158
159  /**
160   * Check if the dataset update should be skipped. Currently, there are 2 reasons for skipping:
161   * 1) The dataset has been locked for automatic updates
162   * 2) The dataset has been migrated to DwC-A, in other words, it has an endpoint of type Darwin Core.
163   *
164   * @param existingDataset existing dataset (dataset as it currently exists in registry)
165   * @return true if the dataset update should be skipped, false otherwise
166   */
167  private boolean skipDatasetUpdate(Dataset existingDataset) {
168    // is the dataset locked for auto update?
169    if (existingDataset.isLockedForAutoUpdate()) {
170      return true;
171    }
172    // is the dataset migrated to DwC-A?
173    for (Endpoint endpoint : existingDataset.getEndpoints()) {
174      if (endpoint.getType() == EndpointType.DWC_ARCHIVE) {
175        return true;
176      }
177    }
178    // otherwise, proceed with update
179    return false;
180  }
181
182  private void saveDeletedDatasets(SyncResult result) {
183    for (Dataset dataset : result.deletedDatasets) {
184      if (dataset.isLockedForAutoUpdate()) {
185        LOG.info("Dataset [{}] deleted at source but left in Registry because it's locked", dataset.getKey());
186      } else {
187        LOG.info("Deleting dataset [{}]", dataset.getKey());
188        datasetService.delete(dataset.getKey());
189      }
190    }
191  }
192
193  private void saveAddedDatasets(SyncResult result) {
194    for (Dataset dataset : result.addedDatasets) {
195      dataset.setPublishingOrganizationKey(result.installation.getOrganizationKey());
196      dataset.setInstallationKey(result.installation.getKey());
197      dataset.setType(DatasetType.OCCURRENCE);
198
199      UUID uuid = datasetService.create(dataset);
200      LOG.info("Created new Dataset with id [{}]", uuid);
201      for (Contact contact : dataset.getContacts()) {
202        datasetService.addContact(uuid, contact);
203      }
204      for (MachineTag machineTag : dataset.getMachineTags()) {
205        datasetService.addMachineTag(uuid, machineTag);
206      }
207      for (Endpoint endpoint : dataset.getEndpoints()) {
208        datasetService.addEndpoint(uuid, endpoint);
209      }
210      for (Identifier identifier : dataset.getIdentifiers()) {
211        datasetService.addIdentifier(uuid, identifier);
212      }
213    }
214  }
