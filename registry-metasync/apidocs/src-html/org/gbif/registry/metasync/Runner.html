013package org.gbif.registry.metasync;
014
015import org.gbif.api.service.registry.DatasetService;
016import org.gbif.api.service.registry.MetasyncHistoryService;
017import org.gbif.registry.metasync.api.SyncResult;
018import org.gbif.registry.metasync.protocols.biocase.BiocaseMetadataSynchroniser;
019import org.gbif.registry.metasync.protocols.digir.DigirMetadataSynchroniser;
020import org.gbif.registry.metasync.protocols.tapir.TapirMetadataSynchroniser;
021import org.gbif.registry.metasync.resulthandler.DebugHandler;
022import org.gbif.registry.metasync.resulthandler.RegistryUpdater;
023import org.gbif.registry.metasync.util.HttpClientFactory;
024import org.gbif.registry.ws.client.guice.RegistryWsClientModule;
025import org.gbif.ws.client.guice.SingleUserAuthModule;
026
027import java.util.List;
028import java.util.Properties;
029import java.util.concurrent.TimeUnit;
030
031import com.google.inject.AbstractModule;
032import com.google.inject.Guice;
033import com.google.inject.Injector;
034import com.google.inject.Scopes;
035import org.slf4j.Logger;
036import org.slf4j.LoggerFactory;
037import org.slf4j.bridge.SLF4JBridgeHandler;
038
039public final class Runner {
040
041  private static final Logger LOG = LoggerFactory.getLogger(Runner.class);
042
043  public static void main(String[] args) {
044    SLF4JBridgeHandler.removeHandlersForRootLogger();
045    SLF4JBridgeHandler.install();
046
047    HttpClientFactory clientFactory = new HttpClientFactory(10, TimeUnit.SECONDS);
048
049    Properties props = new Properties();
050    props.setProperty("registry.ws.url", "http://localhost:8080");
051
052    Injector injector =
053      Guice.createInjector(new RegistryWsClientModule(props), new SingleUserAuthModule("username", "password"),
054        new AbstractModule() {
055
056          @Override
057          protected void configure() {
058            bind(MetadataSynchroniserImpl.class).in(Scopes.SINGLETON);
059          }
060        });
061
062    MetadataSynchroniserImpl synchroniser = injector.getInstance(MetadataSynchroniserImpl.class);
063
064
065    synchroniser.registerProtocolHandler(new DigirMetadataSynchroniser(clientFactory.provideHttpClient()));
066    synchroniser.registerProtocolHandler(new TapirMetadataSynchroniser(clientFactory.provideHttpClient()));
067    synchroniser.registerProtocolHandler(new BiocaseMetadataSynchroniser(clientFactory.provideHttpClient()));
068
069    List<SyncResult> syncResults = synchroniser.synchroniseAllInstallations(100);
070    LOG.info("Done syncing. Processing results");
071    DebugHandler.processResults(syncResults);
072
073    DatasetService datasetService = injector.getInstance(DatasetService.class);
074    MetasyncHistoryService historyService = injector.getInstance(MetasyncHistoryService.class);
075    RegistryUpdater updater = new RegistryUpdater(datasetService, historyService);
076    updater.saveSyncResultsToRegistry(syncResults);
077  }
078
079  private Runner() {
080    throw new UnsupportedOperationException("Can't initialize class");
081  }
082
083}
