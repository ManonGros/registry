001package org.gbif.registry.metasync.protocols.tapir.model.search;
002
003import com.google.common.base.Objects;
004import org.apache.commons.digester3.annotations.rules.ObjectCreate;
005import org.apache.commons.digester3.annotations.rules.SetProperty;
006
007/**
008 * Class used to represent the response from a TAPIR search request, sent in order to discover the number of records.
009 */
010@ObjectCreate(pattern = "response/search/summary")
011public class TapirSearch {
012
013  @SetProperty(pattern = "response/search/summary", attributeName = "totalMatched")
014  private int numberOfRecords;
015
016  /**
017   * Get the total number of records for the Dataset behind the endpoint.
018   *
019   * @return total number of records for the Dataset
020   */
021  public int getNumberOfRecords() {
022    return numberOfRecords;
023  }
024
025  public void setNumberOfRecords(int numberOfRecords) {
026    this.numberOfRecords = numberOfRecords;
027  }
028
029  @Override
030  public String toString() {
031    return Objects.toStringHelper(this)
032      .toString();
033  }
034}
