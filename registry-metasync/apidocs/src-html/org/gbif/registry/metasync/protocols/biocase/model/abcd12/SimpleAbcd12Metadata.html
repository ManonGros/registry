001package org.gbif.registry.metasync.protocols.biocase.model.abcd12;
002
003import org.gbif.api.model.registry.Contact;
004import org.gbif.api.vocabulary.ContactType;
005
006import java.net.URI;
007import java.util.List;
008
009import com.google.common.collect.Lists;
010import org.apache.commons.digester3.annotations.rules.BeanPropertySetter;
011import org.apache.commons.digester3.annotations.rules.CallMethod;
012import org.apache.commons.digester3.annotations.rules.CallParam;
013import org.apache.commons.digester3.annotations.rules.ObjectCreate;
014
015/**
016 * This object extracts the same information from ABCD 1.2 as the "old" registry did.
017 */
018@ObjectCreate(pattern = "response/content/DataSets/DataSet")
019public class SimpleAbcd12Metadata {
020
021  private static final String BASE_PATH = "response/content/DataSets/DataSet/";
022  private final List<String> termsOfUses = Lists.newArrayList();
023  private final List<String> iprDeclarations = Lists.newArrayList();
024  private final List<String> rightsUrls = Lists.newArrayList();
025  private final List<String> organisationNames = Lists.newArrayList();
026  private final List<String> supplierUrls = Lists.newArrayList();
027  private final List<Contact> contacts = Lists.newArrayList();
028  @BeanPropertySetter(pattern = BASE_PATH + "OriginalSource/SourceName")
029  private String code;
030  @BeanPropertySetter(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Description")
031  private String description;
032  @BeanPropertySetter(pattern = BASE_PATH + "OriginalSource/SourceWebAddress")
033  private URI homepage;
034  @BeanPropertySetter(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Statements/LogoURL")
035  private URI logoUrl;
036  @BeanPropertySetter(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Supplier/Addresses/Address")
037  private String address;
038  @BeanPropertySetter(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Supplier/EmailAddresses/EmailAddress")
039  private String email;
040  @BeanPropertySetter(pattern = BASE_PATH
041                                + "DatasetDerivations/DatasetDerivation/Supplier/TelephoneNumbers/TelephoneNumber/Number")
042  private String phone;
043  @BeanPropertySetter(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Rights/TermsOfUse")
044  private String rights;
045  @BeanPropertySetter(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Statements/Acknowledgement")
046  private String citationText;
047  @BeanPropertySetter(pattern = BASE_PATH + "Units/Unit[0]/RecordBasis")
048  private String basisOfRecord;
049
050  public String getCode() {
051    return code;
052  }
053
054  public void setCode(String code) {
055    this.code = code;
056  }
057
058  public String getDescription() {
059    return description;
060  }
061
062  public void setDescription(String description) {
063    this.description = description;
064  }
065
066  public URI getHomepage() {
067    return homepage;
068  }
069
070  public void setHomepage(URI homepage) {
071    this.homepage = homepage;
072  }
073
074  public URI getLogoUrl() {
075    return logoUrl;
076  }
077
078  public void setLogoUrl(URI logoUrl) {
079    this.logoUrl = logoUrl;
080  }
081
082  public String getAddress() {
083    return address;
084  }
085
086  public void setAddress(String address) {
087    this.address = address;
088  }
089
090  public String getEmail() {
091    return email;
092  }
093
094  public void setEmail(String email) {
095    this.email = email;
096  }
097
098  public String getPhone() {
099    return phone;
100  }
101
102  public void setPhone(String phone) {
103    this.phone = phone;
104  }
105
106  public String getRights() {
107    return rights;
108  }
109
110  public void setRights(String rights) {
111    this.rights = rights;
112  }
113
114  public String getCitationText() {
115    return citationText;
116  }
117
118  public void setCitationText(String citationText) {
119    this.citationText = citationText;
120  }
121
122  public String getBasisOfRecord() {
123    return basisOfRecord;
124  }
125
126  public void setBasisOfRecord(String basisOfRecord) {
127    this.basisOfRecord = basisOfRecord;
128  }
129
130  public List<String> getTermsOfUses() {
131    return termsOfUses;
132  }
133
134  public List<String> getIprDeclarations() {
135    return iprDeclarations;
136  }
137
138  public List<String> getRightsUrls() {
139    return rightsUrls;
140  }
141
142  public List<String> getOrganisationNames() {
143    return organisationNames;
144  }
145
146  public List<String> getSupplierUrls() {
147    return supplierUrls;
148  }
149
150  public List<Contact> getContacts() {
151    return contacts;
152  }
153
154  @CallMethod(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Rights/TermsOfUse")
155  public void addTermsOfUse(
156    @CallParam(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Rights/TermsOfUse") String termsOfUse
157  ) {
158    termsOfUses.add(termsOfUse);
159  }
160
161  @CallMethod(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Rights/IPRDeclaration")
162  public void addIprDeclaration(
163    @CallParam(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Rights/IPRDeclaration") String iprDeclaration
164  ) {
165    iprDeclarations.add(iprDeclaration);
166  }
167
168  @CallMethod(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Rights/RightsURL")
169  public void addRightsUrl(
170    @CallParam(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Rights/RightsURL") String iprDeclaration
171  ) {
172    rightsUrls.add(iprDeclaration);
173  }
174
175  @CallMethod(pattern = BASE_PATH + "DatasetDerivation/Supplier/OrganisationName")
176  public void addOrganisationName(
177    @CallParam(pattern = BASE_PATH + "DatasetDerivation/Supplier/OrganisationName") String iprDeclaration
178  ) {
179    organisationNames.add(iprDeclaration);
180  }
181
182  @CallMethod(pattern = BASE_PATH + "DatasetDerivation/Supplier/URLs/URL")
183  public void addSupplierUrl(
184    @CallParam(pattern = BASE_PATH + "DatasetDerivation/Supplier/URLs/URL") String iprDeclaration
185  ) {
186    supplierUrls.add(iprDeclaration);
187  }
188
189  @CallMethod(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Supplier")
190  public void addTechnicalContact(
191    @CallParam(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Supplier/Person/PersonName") String name,
192    @CallParam(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Supplier/EmailAddresses/EmailAddress")
193    String email,
194    @CallParam(pattern = BASE_PATH
195                         + "DatasetDerivations/DatasetDerivation/Supplier/TelephoneNumbers/TelephoneNumber/Number")
196    String phone,
197    @CallParam(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Supplier/Addresses/Address") String address
198  ) {
199    Contact contact = new Contact();
200    contact.setFirstName(name);
201    contact.setEmail(Lists.newArrayList(email));
202    contact.setPhone(Lists.newArrayList(phone));
203    contact.setAddress(Lists.newArrayList(address));
204    contact.setType(ContactType.TECHNICAL_POINT_OF_CONTACT);
205    contacts.add(contact);
206  }
207
208  @CallMethod(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Rights/LegalOwner")
209  public void addAdministrativeContact(
210    @CallParam(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Rights/LegalOwner/Person/PersonName")
211    String name,
212    @CallParam(pattern = BASE_PATH
213                         + "DatasetDerivations/DatasetDerivation/Rights/LegalOwner/EmailAddresses/EmailAddress")
214    String email,
215    @CallParam(pattern = BASE_PATH
216                         + "DatasetDerivations/DatasetDerivation/Rights/LegalOwner/TelephoneNumbers/TelephoneNumber/Number")
217    String phone,
218    @CallParam(pattern = BASE_PATH + "DatasetDerivations/DatasetDerivation/Rights/LegalOwner/Addresses/Address")
219    String address
220  ) {
221    Contact contact = new Contact();
222    contact.setFirstName(name);
223    contact.setEmail(Lists.newArrayList(email));
224    contact.setPhone(Lists.newArrayList(phone));
225    contact.setAddress(Lists.newArrayList(address));
226    contact.setType(ContactType.ADMINISTRATIVE_POINT_OF_CONTACT);
227    contacts.add(contact);
228  }
