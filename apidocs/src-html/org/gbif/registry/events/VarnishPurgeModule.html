001package org.gbif.registry.events;
002
003import org.gbif.utils.HttpUtil;
004
005import java.net.URI;
006import java.util.Properties;
007
008import com.google.inject.PrivateModule;
009import org.apache.commons.lang3.StringUtils;
010import org.apache.http.impl.client.CloseableHttpClient;
011import org.slf4j.Logger;
012import org.slf4j.LoggerFactory;
013
014/**
015 * A Guice module that sets up implementation of the varnish purging via the event bus.
016 * If no or a bad api.cache.purge.url property is configured the listener will not be wired up.
017 */
018public class VarnishPurgeModule extends PrivateModule {
019
020  public static final String API_CACHE_PURGE_URL_PROPERTY = "api.cache.purge.url";
021
022  private static final Logger LOG = LoggerFactory.getLogger(VarnishPurgeModule.class);
023  private static final int DEFAULT_HTTP_TIMEOUT_MSECS = 2000;
024  private final int httpThreads;
025  private final URI apiRoot;
026
027  public VarnishPurgeModule(Properties properties) {
028    String apiCachePurgeUrl = properties.getProperty(API_CACHE_PURGE_URL_PROPERTY);
029
030    try {
031      if(StringUtils.isNotBlank(apiCachePurgeUrl)) {
032        apiRoot = URI.create(apiCachePurgeUrl);
033      }
034      else{
035        apiRoot = null;
036      }
037      httpThreads = Integer.valueOf(properties.getProperty("purging.threads", "100").trim());
038    } catch (RuntimeException e) {
039      LOG.error("Failed to initialize varnish purger because of invalid properties", e.getMessage());
040      throw e;
041    }
042  }
043
044  @Override
045  protected void configure() {
046    if(apiRoot == null){
047      LOG.warn("No varnish purging configured: {} not found or empty", API_CACHE_PURGE_URL_PROPERTY);
048      return;
049    }
050    if (httpThreads > 0) {
051      bind(URI.class).toInstance(apiRoot);
052      bind(CloseableHttpClient.class).toInstance(HttpUtil.newMultithreadedClient(
053        DEFAULT_HTTP_TIMEOUT_MSECS, httpThreads, httpThreads));
054      bind(VarnishPurgeListener.class).asEagerSingleton();
055      LOG.info("Varnish purging enabled with {} threads and API {}", httpThreads, apiRoot);
056    } else {
057      LOG.warn("No varnish purging configured. Please set purging.threads greater than zero if you want it");
058    }
059  }
060
061}
