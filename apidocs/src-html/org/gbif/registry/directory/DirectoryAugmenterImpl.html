001package org.gbif.registry.directory;
002
003import org.gbif.api.model.directory.Participant;
004import org.gbif.api.model.directory.ParticipantPerson;
005import org.gbif.api.model.directory.Person;
006import org.gbif.api.model.registry.Contact;
007import org.gbif.api.model.registry.Identifier;
008import org.gbif.api.model.registry.Node;
009import org.gbif.api.service.directory.NodeService;
010import org.gbif.api.service.directory.ParticipantService;
011import org.gbif.api.service.directory.PersonService;
012import org.gbif.api.vocabulary.ContactType;
013import org.gbif.api.vocabulary.IdentifierType;
014import org.gbif.api.vocabulary.directory.ParticipantPersonRole;
015
016import java.net.URI;
017import java.util.List;
018
019import com.google.common.collect.ImmutableMap;
020import com.google.common.collect.Lists;
021import com.google.common.primitives.Ints;
022import com.google.inject.Inject;
023import com.google.inject.Singleton;
024import org.slf4j.Logger;
025import org.slf4j.LoggerFactory;
026
027@Singleton
028public class DirectoryAugmenterImpl implements Augmenter {
029
030  /**
031   * Maps contact types to participant roles.
032   */
033  private static final ImmutableMap<ParticipantPersonRole,ContactType> PARTICIPANT_ROLE_TO_CONTACT_TYPE = ImmutableMap.of(ParticipantPersonRole.ADDITIONAL_DELEGATE, ContactType.ADDITIONAL_DELEGATE, ParticipantPersonRole.HEAD_OF_DELEGATION, ContactType.HEAD_OF_DELEGATION, ParticipantPersonRole.TEMPORARY_DELEGATE, ContactType.TEMPORARY_DELEGATE, ParticipantPersonRole.TEMPORARY_HEAD_OF_DELEGATION, ContactType.TEMPORARY_HEAD_OF_DELEGATION);
034  private static Logger LOG = LoggerFactory.getLogger(DirectoryAugmenterImpl.class);
035
036
037  private ParticipantService participantService;
038  private NodeService nodeService;
039  private PersonService personService;
040
041  @Inject
042  public DirectoryAugmenterImpl(ParticipantService participantService, NodeService nodeService, PersonService personService) {
043    this.participantService = participantService;
044    this.nodeService = nodeService;
045    this.personService = personService;
046  }
047
048  /**
049   * Gets the participantID from the Node.
050   */
051  private static Integer findParticipantID(Node node) {
052    for (Identifier id : node.getIdentifiers()) {
053      if (IdentifierType.GBIF_PARTICIPANT == id.getType()) {
054        try {
055          return Integer.parseInt(id.getIdentifier());
056        } catch (NumberFormatException e) {
057          LOG.error("IMS Participant ID is no integer: %s", id.getIdentifier());
058        }
059      }
060    }
061    return null;
062  }
063
064  @Override
065  public Node augment(Node node) {
066    if (node != null) {
067      try {
068        Integer participantID = findParticipantID(node);
069        if (participantID != null) {
070          Participant participant = participantService.get(participantID);
071          if (participant != null) {
072            // update node with Directory info if it exists
073            List<org.gbif.api.model.directory.Node> participantNodes = getParticipantNodes(participant);
074            node.setParticipantTitle(participant.getName());
075            node.setContacts(getContactsFromPersons(participant));
076            node.setAbbreviation(participant.getAbbreviatedName());
077            node.setDescription(participant.getComments());
078            if(participant.getMembershipStart() != null) {
079              node.setParticipantSince(getParticipantSinceYear(participant.getMembershipStart()));
080            }
081            node.setAddress(getNodesAddresses(participantNodes));
082            node.setHomepage(getWebUrls(participant,participantNodes));
083            if(!participantNodes.isEmpty()){
084              node.setEmail(getEmails(participantNodes));
085              node.setPhone(getPhones(participantNodes));
086            }
087          }
088        }
089      } catch (Exception e) {
090        LOG.error("Failed to augment node %s with Directory information", node.getKey(), e);
091      }
092    }
093    return node;
094  }
095
096  /**
097   * Gets the year part of the membershipStart field.
098   */
099  private static Integer getParticipantSinceYear(String membershipStart) {
100    String[] membershipStartComponents = membershipStart.split(" ");
101    if(membershipStartComponents.length > 1) {
102      return Ints.tryParse(membershipStartComponents[1].trim());
103    }
104    return null;
105  }
106
107  /**
108   * Gets all the nodes associated to a participant.
109   */
110  private List<org.gbif.api.model.directory.Node> getParticipantNodes(Participant participant){
111    List<org.gbif.api.model.directory.Node> nodes = Lists.newArrayList();
112    if(participant.getNodes() != null){
113      for(org.gbif.api.model.directory.Node node : participant.getNodes()) {
114        nodes.add(nodeService.get(node.getId()));
115      }
116    }
117    return nodes;
118  }
119
120  /**
121   * Gets all the addresses associated to a participant.
122   */
123  private static List<String> getNodesAddresses(List<org.gbif.api.model.directory.Node> participantNodes){
124    List<String> addresses = Lists.newArrayList();
125    for(org.gbif.api.model.directory.Node node : participantNodes) {
126      addresses.add(node.getAddress());
127    }
128    return  addresses;
129  }
130
131  /**
132   * Gets all the web pages/urls of participant and its nodes.
133   */
134  private static List<URI> getWebUrls(Participant participant, List<org.gbif.api.model.directory.Node> participantNodes){
135    List<URI> addresses = Lists.newArrayList();
136    if(participant.getParticipantUrl() != null) {
137      addresses.add(URI.create(participant.getParticipantUrl()));
138    }
139    for(org.gbif.api.model.directory.Node node : participantNodes) {
140      if(node.getNodeUrl() != null) {
141        addresses.add(URI.create(node.getNodeUrl()));
142      }
143    }
144    return  addresses;
145  }
146
147  /**
148   * Gets all the emails from the nodes list.
149   */
150  private static List<String> getEmails(List<org.gbif.api.model.directory.Node> participantNodes){
151    List<String> emails = Lists.newArrayList();
152    for(org.gbif.api.model.directory.Node node : participantNodes) {
153      if(node.getEmail() != null) {
154        emails.add(node.getEmail());
155      }
156    }
157    return  emails;
158  }
159
160  /**
161   * Gets all the phones numbers from the nodes list.
162   */
163  private static List<String> getPhones(List<org.gbif.api.model.directory.Node> participantNodes){
164    List<String> phones = Lists.newArrayList();
165    for(org.gbif.api.model.directory.Node node : participantNodes) {
166      if(node.getPhone() != null) {
167        phones.add(node.getPhone());
168      }
169    }
170    return  phones;
171  }
172
173  /**
174   * Transforms the persons associated to a participant into a list of contacts.
175   */
176  private List<Contact> getContactsFromPersons(Participant participant){
177    List<Contact> contacts = Lists.newArrayList();
178    if(participant.getPeople() != null){
179      for(ParticipantPerson participantPerson : participant.getPeople()) {
180        Person person = personService.get(participantPerson.getPersonId());
181        Contact contact = new Contact();
182        contact.addAddress(person.getAddress());
183        contact.setCountry(person.getCountryCode());
184        if(person.getEmail() != null) {
185          contact.addEmail(person.getEmail());
186        }
187        contact.setFirstName(person.getFirstName());
188        contact.setLastName(person.getSurname());
189        if(person.getPhone() != null) {
190          contact.addPhone(person.getPhone());
191        }
192        contact.setModified(person.getModified());
193        contact.setOrganization(participant.getName());
194        if( participantPerson.getRole() != null) {
195          contact.setType(PARTICIPANT_ROLE_TO_CONTACT_TYPE.get(participantPerson.getRole()));
196        }
197        contacts.add(contact);
198      }
199    }
200    return contacts;
201  }
202}
