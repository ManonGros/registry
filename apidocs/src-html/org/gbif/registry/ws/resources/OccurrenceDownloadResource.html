059/**
060 * Occurrence download resource/web service.
061 */
062@Singleton
063@Path("occurrence/download")
064@Produces({MediaType.APPLICATION_JSON, ExtraMediaTypes.APPLICATION_JAVASCRIPT})
065@Consumes(MediaType.APPLICATION_JSON)
066public class OccurrenceDownloadResource implements OccurrenceDownloadService {
067
068  private final OccurrenceDownloadMapper occurrenceDownloadMapper;
069  private final DatasetOccurrenceDownloadMapper datasetOccurrenceDownloadMapper;
070  private final TitleLookup titleLookup;
071  private final UserService userService;
072  private final DoiGenerator doiGenerator;
073
074  //Download final/failed states
075  private final EnumSet<Download.Status> FAILED_STATES = EnumSet.of(Download.Status.KILLED, Download.Status.CANCELLED,
076                                                                    Download.Status.FAILED);
077  //Page size to iterate over dataset usages
078  private static final int USAGES_PAGE_SIZE = 400;
079
080  //DOI logging marker
081  private static Marker DOI_SMTP = MarkerFactory.getMarker("DOI_SMTP");
082  private static Logger LOG = LoggerFactory.getLogger(DoiGenerator.class);
083
084  // we use the optional injection only for tests !!!
085  @Inject(optional = true)
086  @Context
087  private SecurityContext securityContext;
088
089  @Inject
090  public OccurrenceDownloadResource(OccurrenceDownloadMapper occurrenceDownloadMapper, DatasetOccurrenceDownloadMapper datasetOccurrenceDownloadMapper,
091    DoiGenerator doiGenerator, UserService userService, TitleLookup titleLookup) {
092    this.occurrenceDownloadMapper = occurrenceDownloadMapper;
093    this.datasetOccurrenceDownloadMapper = datasetOccurrenceDownloadMapper;
094    this.doiGenerator = doiGenerator;
095    this.userService = userService;
096    this.titleLookup = titleLookup;
097  }
098
099
100  @POST
101  @Trim
102  @Transactional
103  @Validate(groups = {PrePersist.class, Default.class})
104  @RolesAllowed(ADMIN_ROLE)
105  @Override
106  public void create(@Valid @NotNull @Trim Download occurrenceDownload) {
107    occurrenceDownload.setDoi(doiGenerator.newDownloadDOI());
108    occurrenceDownloadMapper.create(occurrenceDownload);
109  }
110
111  @GET
112  @Path("{key}")
113  @Nullable
114  @NullToNotFound
115  @Override
116  public Download get(@PathParam("key") String key) {
117    Download download = occurrenceDownloadMapper.get(key);
118    if (download != null) { // the user can request a non-existing download
119      clearSensitiveData(securityContext, download);
120    }
121    return download;
122  }
123
124  /**
125   * Lists all the downloads. This operation can be executed by role ADMIN only.
126   */
127  @GET
128  @RolesAllowed(ADMIN_ROLE)
129  @Override
130  public PagingResponse<Download> list(@Context Pageable page, @Nullable @QueryParam("status") Set<Download.Status> status) {
131    if(status == null ||status.isEmpty()) {
132      return new PagingResponse<Download>(page, (long) occurrenceDownloadMapper.count(), occurrenceDownloadMapper.list(page));
133    } else {
134      return new PagingResponse<Download>(page, (long) occurrenceDownloadMapper.countByStatus(status), occurrenceDownloadMapper.listByStatus(page,status));
135    }
136  }
137
138
139  @GET
140  @Path("user/{user}")
141  @NullToNotFound
142  public PagingResponse<Download> listByUser(@PathParam("user") String user, @Context Pageable page, @Nullable @QueryParam("status")
143  Set<Download.Status> status) {
144    checkUserIsInSecurityContext(user, securityContext);
145    return new PagingResponse<Download>(page, (long) occurrenceDownloadMapper.countByUser(user,status),
146                                        occurrenceDownloadMapper.listByUser(user,page,status));
147  }
148
149
150  @PUT
151  @Path("{key}")
152  @Transactional
153  @Override
154  public void update(Download download) {
155    // The current download is retrieved because its user could be modified during the update
156    Download currentDownload = get(download.getKey());
157    Preconditions.checkNotNull(currentDownload);
158    checkUserIsInSecurityContext(currentDownload.getRequest().getCreator(), securityContext);
159    User user = userService.get(securityContext.getUserPrincipal().getName());
160    updateDownloadDOI(download, currentDownload, user);
161    occurrenceDownloadMapper.update(download);
162  }
163
164  /**
165   * Updates the download DOI according to the download status.
166   * If the download succeeded its DOI is registered; if the download status is one the FAILED_STATES
167   * the DOI is removed, otherwise doesn't nothing.
168   */
169  private void updateDownloadDOI(Download download, Download previousDownload, User user){
170    if(download.isAvailable() && previousDownload.getStatus() != Download.Status.SUCCEEDED){
171      try {
172        doiGenerator.registerDownload(download.getDoi(), buildMetadata(download, user), download.getKey());
173      } catch(Exception error) {
174        LOG.error(DOI_SMTP, "Invalid metadata for download {} with doi {} ", download.getKey(), download.getDoi(),  error);
175      }
176
177    } else if(FAILED_STATES.contains(download.getStatus())){
178      doiGenerator.delete(download.getDoi());
179    }
180  }
181
182  /**
183   * Creates the DataCite metadata for a download object.
184   */
185  private DataCiteMetadata buildMetadata(Download download, User user) {
186    List<DatasetOccurrenceDownloadUsage> response = null;
187    List<DatasetOccurrenceDownloadUsage> usages = Lists.newArrayList();
188    PagingRequest pagingRequest = new PagingRequest(0, USAGES_PAGE_SIZE);
189
190    while (response == null || !response.isEmpty()) {
191      response = datasetOccurrenceDownloadMapper.listByDownload(download.getKey(), pagingRequest);
192      usages.addAll(response);
193      pagingRequest.nextPage();
194    }
195
196    return DataCiteConverter.convert(download, user, usages, titleLookup);
197  }
198
199  @GET
200  @Path("{key}/datasets")
201  @Override
202  @NullToNotFound
203  public PagingResponse<DatasetOccurrenceDownloadUsage> listDatasetUsages(@PathParam("key") String downloadKey,
204                                                                   @Context Pageable page){
205    Download download = get(downloadKey);
206    if(download != null) {
207      return new PagingResponse<DatasetOccurrenceDownloadUsage>(page,
208                                   download.getNumberDatasets(),
209                                   datasetOccurrenceDownloadMapper.listByDownload(downloadKey, page));
210    }
211    throw new NotFoundException();
212  }
