040/**
041 * Handle all legacy web service Dataset requests (excluding IPT requests), previously handled by the GBRDS.
042 */
043@Singleton
044@Path("registry/resource")
045public class LegacyDatasetResource {
046
047  private static final Logger LOG = LoggerFactory.getLogger(LegacyDatasetResource.class);
048
049  private final OrganizationService organizationService;
050  private final DatasetService datasetService;
051  private final InstallationService installationService;
052  private final IptResource iptResource;
053
054  @Inject
055  public LegacyDatasetResource(OrganizationService organizationService, DatasetService datasetService,
056    IptResource iptResource, InstallationService installationService) {
057    this.organizationService = organizationService;
058    this.datasetService = datasetService;
059    this.iptResource = iptResource;
060    this.installationService = installationService;
061  }

064  /**
065   * Register GBRDS dataset, handling incoming request with path /resource. The primary contact, publishing organization
066   * key, and resource name are mandatory. Only after both the dataset and primary contact have been persisted is a
067   * Response with Status.CREATED (201) returned.
068   *
069   * @param dataset IptDataset with HTTP form parameters having been injected from Jersey
070   * @param security SecurityContext (security related information)
071   * @return Response
072   * @see IptResource#registerDataset(org.gbif.registry.ws.model.LegacyDataset, javax.ws.rs.core.SecurityContext)
073   */
074  @POST
075  @Produces(MediaType.APPLICATION_XML)
076  @Consumes(MediaType.APPLICATION_FORM_URLENCODED)
077  public Response registerDataset(@InjectParam LegacyDataset dataset, @Context SecurityContext security) {
078    // reuse existing subresource
079    return iptResource.registerDataset(dataset, security);
080  }

082  /**
083   * Update GBRDS Dataset, handling incoming request with path /resource/{key}. The publishing organization key is
084   * mandatory (supplied in the credentials not the parameters). The primary contact is not required, but if any
085   * of the primary contact parameters were included in the request, it is required. This is the difference between this
086   * method and registerDataset. Only after both the dataset and optional primary contact have been updated is a
087   * Response with Status.OK (201) returned.
088   *
089   * @param datasetKey dataset key (UUID) coming in as path param
090   * @param dataset IptDataset with HTTP form parameters having been injected from Jersey
091   * @param security SecurityContext (security related information)
092   * @return Response with Status.CREATED (201) if successful
093   */
094  @POST
095  @Path("{key}")
096  @Produces(MediaType.APPLICATION_XML)
097  @Consumes(MediaType.APPLICATION_FORM_URLENCODED)
098  public Response updateDataset(@PathParam("key") UUID datasetKey, @InjectParam LegacyDataset dataset,
099    @Context SecurityContext security) {
100    if (dataset != null) {
101      // set required fields
102      String user = security.getUserPrincipal().getName();
103      dataset.setCreatedBy(user);
104      dataset.setModifiedBy(user);
105      dataset.setKey(datasetKey);
106      // retrieve existing dataset
107      Dataset existing = datasetService.get(datasetKey);
108      // populate dataset with existing primary contact so it gets updated, not duplicated
109      dataset.setContacts(existing.getContacts());
110      // if primary contact wasn't supplied, set existing one here so that it doesn't respond BAD_REQUEST
111      if (dataset.getPrimaryContactAddress() == null && dataset.getPrimaryContactEmail() == null
112        && dataset.getPrimaryContactType() == null && dataset.getPrimaryContactPhone() == null
113        && dataset.getPrimaryContactName() == null && dataset.getPrimaryContactDescription() == null) {
114        dataset.setPrimaryContact(LegacyResourceUtils.getPrimaryContact(existing));
115      }
116      // otherwise, update primary contact and type
117      else {
118        dataset.prepare();
119      }
120      // If installation key wasn't provided, reuse existing dataset's installation key
121      // Reason: non-IPT consumers weren't aware they could supply the parameter iptKey on dataset updates before
122      if (dataset.getInstallationKey() == null) {
123        dataset.setInstallationKey(existing.getInstallationKey());
124      }
125      // Dataset can only have 1 installation key, log if the hosting installation is being changed
126      else if (dataset.getInstallationKey() != existing.getInstallationKey()) {
127        LOG.debug("The dataset's technical installation is being changed from {} to {}", dataset.getInstallationKey(), existing.getInstallationKey());
128      }
129      // type can't be derived from endpoints, since there are no endpoints supplied on this update, so re-set existing
130      dataset.setType(existing.getType());
131      // populate publishing organization from credentials
132      dataset.setPublishingOrganizationKey( LegacyAuthorizationFilter.extractOrgKeyFromSecurity(security) );
133      // ensure the publishing organization exists, the installation exists, primary contact exists, etc
134      Contact contact = dataset.getPrimaryContact();
135      if (contact != null && LegacyResourceUtils.isValidOnUpdate(dataset,
136                                datasetService, organizationService, installationService)) {
137        // update only fields that could have changed
138        existing.setModifiedBy(user);
139        existing.setTitle(dataset.getTitle());
140        existing.setDescription(dataset.getDescription());
141        existing.setHomepage(dataset.getHomepage());
142        existing.setLogoUrl(dataset.getLogoUrl());
143        existing.setLanguage(dataset.getLanguage());
144        existing.setInstallationKey(dataset.getInstallationKey());
145
146        existing.setPublishingOrganizationKey(dataset.getPublishingOrganizationKey());
147
148        // persist changes
149        datasetService.update(existing);
150
151        // set primary contact's required field(s)
152        contact.setModifiedBy(user);
153        // add/update primary contact: Contacts are mutable, so try to update if the Contact already exists
154        if (contact.getKey() == null) {
155          contact.setCreatedBy(user);
156          datasetService.addContact(datasetKey, contact);
157        } else {
158          datasetService.updateContact(datasetKey, contact);
159        }
160
161        // endpoint changes are done through Service API
162
163        LOG.info("Dataset updated successfully, key=%s", datasetKey.toString());
164        return Response.status(Response.Status.CREATED).entity(dataset)
165          .cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED).build();
166
167      } else {
168        LOG.error("Request invalid. Dataset missing required fields or using stale keys!");
169      }
170    }
171    LOG.error("Dataset update failed");
172    return Response.status(Response.Status.BAD_REQUEST).cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED)
173      .build();
174  }

176  /**
177   * Retrieve all Datasets owned by an organization, handling incoming request with path /resource.
178   * The publishing organization query parameter is mandatory. Only after both
179   * the organizationKey is verified to correspond to an existing organization, is a Response including the list of
180   * Datasets returned.
181   *
182   * @param organizationKey organization key (UUID) coming in as query param
183   * @return Response with list of Datasets or empty list with error message if none found
184   */
185  @GET
186  @Produces({MediaType.APPLICATION_XML, MediaType.APPLICATION_JSON})
187  @Consumes(MediaType.TEXT_PLAIN)
188  public Response datasetsForOrganization(@QueryParam("organisationKey") UUID organizationKey) {
189
190    if (organizationKey != null) {
191      try {
192        LOG.debug("Get all Datasets owned by Organization, key={}", organizationKey);
193        // verify organization with key exists, otherwise NotFoundException gets thrown
194        organizationService.get(organizationKey);
195
196        List<LegacyDatasetResponse> datasets = Lists.newArrayList();
197        PagingRequest page = new PagingRequest(0, LegacyResourceConstants.WS_PAGE_SIZE);
198        PagingResponse<Dataset> response;
199        do {
200          LOG.debug("Requesting {} datasets starting at offset {}", page.getLimit(), page.getOffset());
201          response = organizationService.publishedDatasets(organizationKey, page);
202          for (Dataset d : response.getResults()) {
203            Contact contact = LegacyResourceUtils.getPrimaryContact(d);
204            datasets.add(new LegacyDatasetResponse(d, contact));
205          }
206          page.nextPage();
207        } while (!response.isEndOfRecords());
208        LOG.debug("Get all Datasets owned by Organization finished");
209        // return array, required for serialization otherwise get com.sun.jersey.api.MessageException: A message body
210        // writer for Java class java.util.ArrayList
211        LegacyDatasetResponse[] array = datasets.toArray(new LegacyDatasetResponse[datasets.size()]);
212        return Response.status(Response.Status.OK).entity(array).build();
213      } catch (NotFoundException e) {
214        LOG.error("The organization with key {} specified by query parameter does not exist", organizationKey);
215      }
216    }
217    return Response.status(Response.Status.OK).entity(new ErrorResponse("No organisation matches the key provided"))
218      .build();
219  }

221  /**
222   * Read GBRDS Dataset, handling incoming request with path /resource/{key}.
223   *
224   * @param datasetKey dataset key (UUID) coming in as path param
225   * @return Response with Status.OK (200) if dataset exists
226   */
227  @GET
228  @Path("{key}")
229  @Produces({MediaType.APPLICATION_XML, MediaType.APPLICATION_JSON})
230  @Consumes(MediaType.TEXT_PLAIN)
231  public Response readDataset(@PathParam("key") UUID datasetKey) {
232    if (datasetKey != null) {
233      try {
234        LOG.debug("Get Dataset, key={}", datasetKey);
235        // verify Dataset with key exists, otherwise NotFoundException gets thrown
236        Dataset dataset = datasetService.get(datasetKey);
237        Contact contact = LegacyResourceUtils.getPrimaryContact(dataset);
238        return Response.status(Response.Status.OK).entity(new LegacyDatasetResponse(dataset, contact))
239          .cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED).build();
240      } catch (NotFoundException e) {
241        LOG.error("The dataset with key {} specified by path parameter does not exist", datasetKey);
242      }
243    }
244    return Response.status(Response.Status.OK).entity(new ErrorResponse("No resource matches the key provided"))
245      .cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED).build();
246  }
