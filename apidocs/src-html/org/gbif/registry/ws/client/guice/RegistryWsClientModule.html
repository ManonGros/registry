038import java.util.Properties;

040import com.google.inject.Provides;
041import com.google.inject.Scopes;
042import com.google.inject.Singleton;
043import com.google.inject.name.Named;
044import com.sun.jersey.api.client.Client;
045import com.sun.jersey.api.client.WebResource;
046
047/**
048 * A Module for Guice doing all the necessary wiring with the exception of the authentication filters.
049 * In order to use this module an authentication module needs to be installed first, such as {@link AnonymousAuthModule}
050 * for anonymous access or {@link GbifApplicationAuthModule} for trusted applications.
051 * With an authentication module installed, the only thing left for clients to do is to provide a property
052 * to be bound to {@code registry.ws.url}.
053 * If you want to use this module remember to also depend on Guice, jersey-apache-client4 and jersey-json in your
054 * pom.xml file as they are declared as optional dependencies in this project.
055 */
056public class RegistryWsClientModule extends GbifWsClientModule {
057
058  public RegistryWsClientModule(Properties properties) {
059    super(properties, NodeWsClient.class.getPackage());
060  }
061
062  @Override
063  protected void configureClient() {
064    install(new InternalRegistryWsClientModule(getProperties()));
065    expose(NodeService.class);
066    expose(OrganizationService.class);
067    expose(InstallationService.class);
068    expose(DatasetService.class);
069    expose(DatasetSearchService.class);
070    expose(NetworkService.class);
071    expose(OccurrenceDownloadService.class);
072    expose(DatasetOccurrenceDownloadUsageService.class);
073    expose(DatasetProcessStatusService.class);
074    expose(MetasyncHistoryService.class);
075
076  }
077
078
079  // To allow the prefixing of the properties
080  private class InternalRegistryWsClientModule extends PrivateServiceModule {
081
082    private static final String PREFIX = "registry.";
083
084    private InternalRegistryWsClientModule(Properties properties) {
085      super(PREFIX, properties);
086    }
087
088    @Override
089    protected void configureService() {
090
091      bind(NodeService.class).to(NodeWsClient.class).in(Scopes.SINGLETON);
092      bind(OrganizationService.class).to(OrganizationWsClient.class).in(Scopes.SINGLETON);
093      bind(InstallationService.class).to(InstallationWsClient.class).in(Scopes.SINGLETON);
094      bind(MetasyncHistoryService.class).to(InstallationWsClient.class).in(Scopes.SINGLETON);
095      bind(DatasetService.class).to(DatasetWsClient.class).in(Scopes.SINGLETON);
096      bind(DatasetSearchService.class).to(DatasetSearchWsClient.class).in(Scopes.SINGLETON);
097      bind(NetworkService.class).to(NetworkWsClient.class).in(Scopes.SINGLETON);
098      bind(OccurrenceDownloadService.class).to(OccurrenceDownloadWsClient.class).in(Scopes.SINGLETON);
099      bind(DatasetOccurrenceDownloadUsageService.class).to(DatasetOccurrenceDownloadUsageWsClient.class).in(
100        Scopes.SINGLETON);
101      bind(DatasetProcessStatusService.class).to(DatasetWsClient.class).in(Scopes.SINGLETON);
102
103      expose(NodeService.class);
104      expose(OrganizationService.class);
105      expose(InstallationService.class);
106      expose(DatasetService.class);
107      expose(DatasetSearchService.class);
108      expose(NetworkService.class);
109      expose(OccurrenceDownloadService.class);
110      expose(DatasetOccurrenceDownloadUsageService.class);
111      expose(DatasetProcessStatusService.class);
112      expose(MetasyncHistoryService.class);
113    }
114
115    @Provides
116    @Singleton
117    @RegistryWs
118    private WebResource provideBaseWsWebResource(Client client, @Named("ws.url") String url) {
119      return client.resource(url);
120    }
121  }
122}
