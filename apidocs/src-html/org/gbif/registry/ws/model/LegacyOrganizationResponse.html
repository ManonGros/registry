016/**
017 * Class used to generate response for legacy (GBRDS/IPT) API. Previously known as an Organisation with an s in the
018 * GBRDS.
019 * </br>
020 * JAXB annotations allow the class to be converted into an XML document or JSON response. @XmlElement is used to
021 * specify element names that consumers of legacy services expect to find.
022 */
023@XmlRootElement(name = "organisation")
024public class LegacyOrganizationResponse {
025
026  private String key;
027  private String name;
028  private String nameLanguage;
029  private String homepageURL;
030  private String description;
031  private String descriptionLanguage;
032  private String nodeKey;
033  private String nodeName;
034  private String nodeContactEmail;
035  private String primaryContactType;
036  private String primaryContactName;
037  private String primaryContactEmail;
038  private String primaryContactAddress;
039  private String primaryContactPhone;
040  private String primaryContactDescription;
041
042  private static final Joiner CONTACT_NAME = Joiner.on(" ").skipNulls();
043
044  public LegacyOrganizationResponse(Organization organization, Contact contact, Node node) {
045    key = organization.getKey() == null ? "" : organization.getKey().toString();
046    name = Strings.nullToEmpty(organization.getTitle());
047    nameLanguage = organization.getLanguage() == null ? "" : organization.getLanguage().getIso2LetterCode();
048    homepageURL = organization.getHomepage() == null ? "" : organization.getHomepage().toString();
049    description = Strings.nullToEmpty(organization.getDescription());
050    descriptionLanguage = organization.getLanguage() == null ? "" : organization.getLanguage().getIso2LetterCode();
051    primaryContactAddress = contact == null || contact.getAddress().isEmpty() ? "" :
052      Strings.nullToEmpty(contact.getAddress().get(0));
053    primaryContactDescription = contact == null ? "" : Strings.nullToEmpty(contact.getDescription());
054    primaryContactEmail = contact == null || contact.getEmail().isEmpty() ? "" :
055      Strings.nullToEmpty(contact.getEmail().get(0));
056    primaryContactPhone = contact == null || contact.getPhone().isEmpty() ? "" :
057      Strings.nullToEmpty(contact.getPhone().get(0));
058    primaryContactName =
059      contact == null ? "" : CONTACT_NAME.join(new String[] {contact.getFirstName(), contact.getLastName()});
060    nodeKey = node.getKey() == null ? "" : node.getKey().toString();
061    nodeName = node.getTitle() == null ? "" : node.getTitle();
062    nodeContactEmail = node.getEmail().isEmpty() ? "" : node.getEmail().get(0);
063    // conversion of contact type, defaulting to empty
064    primaryContactType = "";
065    if (contact != null) {
066      ContactType type = contact.getType();
067      if (type != null) {
068        if (type.compareTo(ContactType.ADMINISTRATIVE_POINT_OF_CONTACT) == 0) {
069          primaryContactType = LegacyResourceConstants.ADMINISTRATIVE_CONTACT_TYPE;
070        } else if (type.compareTo(ContactType.TECHNICAL_POINT_OF_CONTACT) == 0) {
071          primaryContactType = LegacyResourceConstants.TECHNICAL_CONTACT_TYPE;
072        }
073      }
074    }
075  }
