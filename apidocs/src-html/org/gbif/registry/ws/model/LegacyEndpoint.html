001package org.gbif.registry.ws.model;
002
003import org.gbif.api.model.registry.Endpoint;
004import org.gbif.api.vocabulary.EndpointType;
005import org.gbif.registry.ws.util.LegacyResourceConstants;
006import org.gbif.registry.ws.util.LegacyResourceUtils;
007
008import java.net.URI;
009import java.net.URISyntaxException;
010import java.util.UUID;
011import javax.annotation.Nullable;
012import javax.validation.constraints.NotNull;
013import javax.ws.rs.FormParam;
014import javax.xml.bind.annotation.XmlElement;
015import javax.xml.bind.annotation.XmlRootElement;
016
017import com.google.common.base.Strings;
018import org.slf4j.Logger;
019import org.slf4j.LoggerFactory;
020
021/**
022 * Class used to create or update an Endpoint for legacy (GBRDS/IPT) API. Previously known as a Service in the GBRDS.
023 * A set of HTTP Form parameters coming from a POST request are injected.
024 * </br>
025 * Its fields are injected using the @FormParam. It is assumed the following parameters exist in the HTTP request:
026 * 'resourceKey', 'description', 'descriptionLanguage', 'type', 'accessPointURL'.
027 * </br>
028 * JAXB annotations allow the class to be converted into an XML document, that gets included in the Response following
029 * a successful registration or update. @XmlElement is used to specify element names that consumers of legacy services
030 * expect to find.
031 */
032@XmlRootElement(name = "service")
033public class LegacyEndpoint extends Endpoint {
034
035  private static final Logger LOG = LoggerFactory.getLogger(LegacyEndpoint.class);
036
037  // injected from HTTP form parameters
038  private UUID datasetKey;
039
040  private UUID organizationKey;
041
042  /**
043   * Set the endpoint's dataset key. Mandatory field, injected on both create and update requests.
044   *
045   * @param resourceKey dataset key as UUID
046   */
047  @FormParam(LegacyResourceConstants.RESOURCE_KEY_PARAM)
048  public void setDatasetKey(String resourceKey) {
049    try {
050      datasetKey = UUID.fromString(Strings.nullToEmpty(resourceKey));
051    } catch (IllegalArgumentException e) {
052      LOG.error("Dataset key is not a valid UUID: {}", Strings.nullToEmpty(resourceKey));
053    }
054  }
055
056  /**
057   * Get the endpoint's dataset key.
058   *
059   * @return dataset key
060   */
061  @XmlElement(name = LegacyResourceConstants.RESOURCE_KEY_PARAM)
062  @NotNull
063  public UUID getDatasetKey() {
064    return datasetKey;
065  }
066
067  /**
068   * Set the endpoint description.
069   *
070   * @param description of the endpoint
071   */
072  @FormParam(LegacyResourceConstants.DESCRIPTION_PARAM)
073  public void setEndpointDescription(String description) {
074    setDescription(LegacyResourceUtils.validateField(description, 10));
075  }
076
077  /**
078   * Get the endpoint description. This method is not used but it is needed otherwise this Object
079   * can't be converted into an XML document via JAXB.
080   *
081   * @return description of the endpoint
082   */
083  @XmlElement(name = LegacyResourceConstants.DESCRIPTION_PARAM)
084  @Nullable
085  public String getEndpointDescription() {
086    return getDescription();
087  }
088
089  /**
090   * Set endpoint type. First, check if it is not null or empty. The incoming type always comes from the GBRDS Service
091   * Type vocabulary. Note: this field is required, and the old web services would throw 400 response if not found.
092   *
093   * @param type endpoint type
094   */
095  @FormParam(LegacyResourceConstants.TYPE_PARAM)
096  public void setEndpointType(String type) {
097    String injected = Strings.nullToEmpty(type);
098    EndpointType lookup = EndpointType.fromString(injected);
099    if (lookup == null) {
100      // try to match some endpoints with their variant name
101      if (injected.equalsIgnoreCase(LegacyResourceConstants.CHECKLIST_SERVICE_TYPE_1)
102          || injected.equalsIgnoreCase(LegacyResourceConstants.CHECKLIST_SERVICE_TYPE_2)
103          || injected.equalsIgnoreCase(LegacyResourceConstants.OCCURRENCE_SERVICE_TYPE_1)
104          || injected.equalsIgnoreCase(LegacyResourceConstants.OCCURRENCE_SERVICE_TYPE_2)
105          || injected.equalsIgnoreCase(LegacyResourceConstants.SAMPLING_EVENT_SERVICE_TYPE)) {
106        setType(EndpointType.DWC_ARCHIVE);
107      } else {
108        LOG.error("Endpoint type could not be interpreted: {}", injected);
109      }
110    } else {
111      setType(lookup);
112    }
113  }
114
115  /**
116   * Get endpoint type. This method is not used but it is needed otherwise this Object can't be converted into
117   * an XML document via JAXB.
118   *
119   * @return primary contact type
120   */
121  @XmlElement(name = LegacyResourceConstants.TYPE_PARAM)
122  @NotNull
123  public EndpointType getEndpointType() {
124    return getType();
125  }
126
127  /**
128   * Set the endpoint URL.
129   *
130   * @param url of the endpoint
131   */
132  @FormParam(LegacyResourceConstants.ACCESS_POINT_URL_PARAM)
133  public void setEndpointUrl(String url) {
134    if (!Strings.isNullOrEmpty(url)) {
135      try {
136        URI uri = new URI(url);
137        setUrl(uri);
138      } catch (URISyntaxException e) {
139        LOG.warn("Endpoint URL was invalid: {}", Strings.nullToEmpty(url));
140      }
141    }
142  }
143
144  /**
145   * Get the endpoint URL. This method is not used but it is needed otherwise this Object
146   * can't be converted into an XML document via JAXB.
147   *
148   * @return url of the endpoint
149   */
150  @XmlElement(name = LegacyResourceConstants.ACCESS_POINT_URL_PARAM)
151  @NotNull
152  public String getEndpointUrl() {
153    return getUrl().toASCIIString();
154  }
155
156  /**
157   * Set the publishing organization key of the dataset to which the endpoint is associated.
158   *
159   * @param organizationKey organization key as UUID
160   */
161  public void setOrganizationKey(UUID organizationKey) {
162    this.organizationKey = organizationKey;
163  }
164}
