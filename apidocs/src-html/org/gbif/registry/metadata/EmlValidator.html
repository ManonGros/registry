001package org.gbif.registry.metadata;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.io.StringReader;
006import javax.xml.transform.Source;
007import javax.xml.transform.stream.StreamSource;
008import javax.xml.validation.Schema;
009import javax.xml.validation.SchemaFactory;
010import javax.xml.validation.Validator;
011
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014import org.xml.sax.SAXException;
015
016/**
017 * EML gbif profile schema validator utility
018 */
019public class EmlValidator {
020  private static final Logger LOG = LoggerFactory.getLogger(EmlValidator.class);
021  private static final String SCHEMA_LANG = "http://www.w3.org/2001/XMLSchema";
022  private static final String EML_SCHEMA = "http://rs.gbif.org/schema/eml-gbif-profile/1.1/eml.xsd";
023  private static Validator EML_VALIDATOR;
024
025  public static void validate(String xml) throws InvalidEmlException {
026    validate(new StreamSource(new StringReader(xml)));
027  }
028
029  public static void validate(InputStream xml) throws InvalidEmlException {
030    validate(new StreamSource(xml));
031  }
032
033  public static void validate(Source source) throws InvalidEmlException {
034    try {
035      getValidator().validate(source);
036      LOG.debug("EML XML passed validation");
037
038    } catch (Exception e) {
039      throw new InvalidEmlException(e);
040    }
041  }
042
043  private static Validator getValidator() throws IOException, SAXException {
044    if (EML_VALIDATOR == null) {
045      // define the type of schema - we use W3C:
046      // resolve validation driver:
047      SchemaFactory factory = SchemaFactory.newInstance(SCHEMA_LANG);
048      // create schema by reading it from gbif online resources:
049      Schema schema = factory.newSchema(new StreamSource(EML_SCHEMA));
050      EML_VALIDATOR = schema.newValidator();
051    }
052    return EML_VALIDATOR;
053  }
054}
