001package org.gbif.registry.metadata;
002
003import java.beans.PropertyDescriptor;
004import java.lang.reflect.Field;
005import java.lang.reflect.ParameterizedType;
006import java.util.Collection;
007import java.util.Iterator;
008
009import org.apache.commons.beanutils.PropertyUtils;
010import org.apache.commons.lang3.StringUtils;
011
012public class CleanUtils {
013
014  /**
015   * Takes a bean and goes through all String properties and Collection properties with a generic String type
016   * and replaces all blank values, i.e. empty strings or pure whitespace values, with nulls.
017   * In case of collections those values will be removed
018   * @param obj bean to clean
019   */
020  public static <T> void removeEmptyStrings(T obj) {
021    for (PropertyDescriptor pd : PropertyUtils.getPropertyDescriptors(obj)) {
022      try {
023        if (String.class.equals(pd.getPropertyType()) && StringUtils
024          .isBlank((String) PropertyUtils.getProperty(obj, pd.getName()))) {
025          PropertyUtils.setProperty(obj, pd.getName(), null);
026        } else if (Collection.class.isAssignableFrom(pd.getPropertyType())) {
027          // inspect also list string values
028          Field collField = obj.getClass().getDeclaredField(pd.getName());
029          ParameterizedType collType = (ParameterizedType) collField.getGenericType();
030          Class<?> collClass = (Class<?>) collType.getActualTypeArguments()[0];
031          if (collClass.equals(String.class)) {
032            Collection<String> list = (Collection<String>) PropertyUtils.getProperty(obj, pd.getName());
033            if (list != null) {
034              Iterator<String> iter = list.iterator();
035              while (iter.hasNext()) {
036                if (StringUtils.isBlank(iter.next())) {
037                  iter.remove();
038                }
039              }
040            }
041          }
042        }
043      } catch (Exception e) {
044        // ignore
045      }
046    }
047  }
048
049}
