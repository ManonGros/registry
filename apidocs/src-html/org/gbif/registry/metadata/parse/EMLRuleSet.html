048/**
049 * Digester rules to parse EML dataset metadata documents together with a DatasetDelegator digester model.
050 * The rules here ignore any namespace to be able to work with any eml versions after 2.0.
051 */
052public class EMLRuleSet extends RuleSetBase {
053
054  private void setupTypeConverters() {
055
056    GreedyUriConverter uriConverter = new GreedyUriConverter();
057    ConvertUtils.register(uriConverter, URI.class);
058
059    LanguageTypeConverter langConverter = new LanguageTypeConverter();
060    ConvertUtils.register(langConverter, Language.class);
061
062
063    // handles the Identifier.type string -> IdentifierType conversion
064    IdentifierTypeConverter identifierTypeConverter = new IdentifierTypeConverter(IdentifierType.UNKNOWN);
065    ConvertUtils.register(identifierTypeConverter, IdentifierType.class);
066
067    DateConverter dateConverter = new DateConverter();
068    ConvertUtils.register(dateConverter, Date.class);
069
070    CountryTypeConverter countryTypeConverter = new CountryTypeConverter();
071    ConvertUtils.register(countryTypeConverter, Country.class);
072
073    ContactTypeConverter typeConverter = new ContactTypeConverter(ContactType.ADMINISTRATIVE_POINT_OF_CONTACT);
074    ConvertUtils.register(typeConverter, ContactType.class);
075
076    // Collection with CuratorialUnits
077    Converter intConverter = new IntegerConverter();
078    ConvertUtils.register(intConverter, Integer.class);
079
080    // handles the Collection.specimenPreservationType string -> PreservationMethodType enum conversion
081    PreservationMethodTypeConverter preservationMethodTypeConverter = new PreservationMethodTypeConverter(PreservationMethodType.OTHER);
082    ConvertUtils.register(preservationMethodTypeConverter, PreservationMethodType.class);
083  }
084  /**
085   * Creates a new digester with all rules to parse an EML document.
086   */
087  public void addRuleInstances(Digester digester) {
088
089    setupTypeConverters();
090
091    // language as xml:lang attribute
092    digester.addCallMethod("eml", "setLanguage", 1, new Class[]{Language.class} );
093    digester.addCallParam("eml", 0, "xml:lang");
094
095    // DOI as packageId attribute
096    digester.addCallMethod("eml", "setPackageId", 1);
097    digester.addCallParam("eml", 0, "packageId");
098
099    // identifiers
100    digester.addObjectCreate("eml/dataset/alternateIdentifier", Identifier.class);
101    digester.addBeanPropertySetter("eml/dataset/alternateIdentifier", "identifier");
102    digester.addBeanPropertySetter("eml/dataset/alternateIdentifier", "type");
103    digester.addSetNext("eml/dataset/alternateIdentifier", "addIdentifier");
104
105    // title (no language support in Dataset)
106    digester.addBeanPropertySetter("eml/dataset/title", "title");
107
108    // SamplingDescription
109    addSamplingDescriptionRules(digester, "eml/dataset/methods", "setSamplingDescription");
110
111    // WritableDataset properties
112    digester.addBeanPropertySetter("eml/dataset/language", "dataLanguage");
113    digester.addBeanPropertySetter("eml/dataset/distribution/online/url", "homepage");
114    digester.addBeanPropertySetter("eml/additionalMetadata/metadata/gbif/resourceLogoUrl", "logoURL");
115
116    // multi paragraph description
117    digester.addObjectCreate("eml/dataset/abstract", ParagraphContainer.class);
118    digester.addCallMethod("eml/dataset/abstract/para", "appendParagraph", 0);
119    digester.addRule("eml/dataset/abstract", new SetRootRule("addDescription", ParagraphContainer.class));
120
121    // Citation
122    addCitationRules(digester, "eml/additionalMetadata/metadata/gbif/citation", "setCitation");
123
124    // digester.addBeanPropertySetter("eml/additionalMetadata/metadata/gbif/hierarchyLevel", "hierarchyLevel");
125
126    // Publication date
127    digester.addCallMethod("eml/dataset/pubDate", "setPubDateAsString", 1);
128    digester.addCallParam("eml/dataset/pubDate", 0);
129
130    // digester.addCallMethod("eml/additionalMetadata/metadata/gbif/dateStamp", "setDateStamp", 1);
131    // digester.addCallParam("eml/additionalMetadata/metadata/gbif/dateStamp", 0);
132
133    addContactRules(digester, "eml/dataset/creator", "addPreferredOriginatorContact");
134    addContactRules(digester, "eml/dataset/metadataProvider", "addPreferredMetadataContact");
135    addContactRules(digester, "eml/dataset/associatedParty", "addContact");
136
137    addContactRules(digester, "eml/dataset/contact", "addPreferredAdministrativeContact");
138
139    digester.addBeanPropertySetter("eml/dataset/purpose/para", "purpose");
140    digester.addBeanPropertySetter("eml/dataset/additionalInfo/para", "additionalInfo");
141
142    // POR-2792: We prefer to use the title if it is given, otherwise we'll take the blurb
143    // Digester provides no easy way of concatenating the content across the elements, and this is seen
144    // as an acceptable solution, recognising that one could craft EML that would make this behaviour
145    // unacceptable.  However, the risk of that is considered tiny.
146    digester.addBeanPropertySetter("eml/dataset/intellectualRights/para/ulink/citetitle", "rights");
147    // Note: because this fires on leaving the element, any ulink/citetitle will already have populated the value
148    digester.addCallMethod("eml/dataset/intellectualRights/para", "setRightsIfEmpty", 1);
149    digester.addCallParam("eml/dataset/intellectualRights/para", 0);
150
151    // KeywordCollections
152    addKeywordCollectionRules(digester, "eml/dataset/keywordSet", "addKeywordCollection");
153
154    // BibliographicCitations
155    addCitationRules(digester, "eml/additionalMetadata/metadata/gbif/bibliography/citation",
156      "addBibliographicCitation");
157
158    // Data Description (Physical Data aka External Links)
159    addDataDescriptionRules(digester, "eml/additionalMetadata/metadata/gbif/physical", "addDataDescription");
160
161    // Geographic coverage
162    addGeographicCoverageRules(digester, "eml/dataset/coverage/geographicCoverage", "addGeographicCoverage");
163
164    // Temporal coverage
165    addTemporalCoverageRules(digester);
166
167    // Taxonomic coverage
168    addTaxonomicCoverageRules(digester);
169
170    // Project
171    addProjectRules(digester, "eml/dataset/project", "setProject");
172
173    addCollectionRules(digester, "eml/additionalMetadata/metadata/gbif/collection", "addCollection");
174
175    digester.addBeanPropertySetter("eml/additionalMetadata/metadata/gbif/specimenPreservationMethod",
176      "specimenPreservationMethod");
177
178  }

180  private void addTaxonomicCoverageRules(Digester digester) {
181    final String prefix = "eml/dataset/coverage/taxonomicCoverage";
182
183    digester.addObjectCreate(prefix, TaxonomicCoverages.class);
184    digester.addBeanPropertySetter(prefix + "/generalTaxonomicCoverage", "description");
185
186    digester.addObjectCreate(prefix + "/taxonomicClassification", TaxonomicCoverage.class);
187    digester.addBeanPropertySetter(prefix + "/taxonomicClassification/taxonRankValue", "scientificName");
188    digester.addBeanPropertySetter(prefix + "/taxonomicClassification/commonName", "commonName");
189
190    digester.addFactoryCreate(prefix + "/taxonomicClassification/taxonRankName", new InterpretedRankEnumFactory());
191    digester.addBeanPropertySetter(prefix + "/taxonomicClassification/taxonRankName", "verbatim");
192
193    digester.addSetNext(prefix + "/taxonomicClassification/taxonRankName", "setRank");
194
195    digester.addSetNext(prefix + "/taxonomicClassification", "addCoverages");
196
197    digester.addSetNext(prefix, "addTaxonomicCoverages");
198  }

200  public class InterpretedRankEnumFactory extends AbstractObjectCreationFactory {
201    @Override
202    public Object createObject(Attributes attributes) throws Exception {
203      return new InterpretedEnum<String, Rank>();
204    }

206  /**
207   * This is a reusable set of rules to build Citation and add the Citation as per the method called on parent object
208   * which is the previous stack object.
209   *
210   * @param digester     to add the rules to
211   * @param path         The XPath for extracting the Citation information
212   * @param parentMethod Of the previous stack object to call and add the Citation to
213   */
214  private void addCitationRules(Digester digester, String path, String parentMethod) {
215    digester.addObjectCreate(path, Citation.class);
216    digester.addBeanPropertySetter(path, "text");
217    digester.addCallMethod(path, "setIdentifier", 1);
218    digester.addCallParam(path, 0, "identifier");
219    digester.addSetNext(path, parentMethod);
220  }

222  /**
223   * This is a reusable set of rules to build a Collection and add the Collection as per the method
224   * called on parent object which is the previous stack object.
225   *
226   * @param digester     to add the rules to
227   * @param prefix       The XPath prefix to prepend for extracting the Collection information

230  private void addCollectionRules(Digester digester, String prefix, String parentMethod) {
231    digester.addObjectCreate(prefix, Collection.class);
232    digester.addBeanPropertySetter(prefix + "/parentCollectionIdentifier", "parentIdentifier");
233    digester.addBeanPropertySetter(prefix + "/collectionIdentifier", "identifier");
234    digester.addBeanPropertySetter(prefix + "/collectionName", "name");
235    addCuratorialUnit(digester, "eml/additionalMetadata/metadata/gbif/jgtiCuratorialUnit", "addCuratorial");
236    digester.addSetNext(prefix, parentMethod);
237  }

239  /**
240   * This is a reusable set of rules to build Contacts and add the Contact as per the method called on parent object
241   * which is the previous stack object.
242   *
243   * @param digester     to add the rules to
244   * @param prefix       The XPath prefix to prepend for extracting the Contact information
245   * @param parentMethod Of the previous stack object to call and add the Contact to
246   */
247  private void addContactRules(Digester digester, String prefix, String parentMethod) {
248    digester.addObjectCreate(prefix, Contact.class);
249
250    digester.addCallMethod(prefix + "/userId", "addUserId", 2);
251    digester.addCallParam(prefix + "/userId", 0, "directory");
252    digester.addCallParam(prefix + "/userId", 1);
253
254    digester.addBeanPropertySetter(prefix + "/individualName/givenName", "firstName");
255    digester.addBeanPropertySetter(prefix + "/individualName/surName", "lastName");
256    digester.addBeanPropertySetter(prefix + "/organizationName", "organization");
257    digester.addCallMethod(prefix + "/positionName", "addPosition", 0);
258    digester.addCallMethod(prefix + "/phone", "addPhone", 0);
259    digester.addCallMethod(prefix + "/electronicMailAddress", "addEmail", 0);
260    digester.addCallMethod(prefix + "/onlineUrl", "addHomepage", 0, new Class[]{URI.class});
261    digester.addBeanPropertySetter(prefix + "/role", "type");
262    digester.addBeanPropertySetter(prefix + "/address/city", "city");
263    digester.addBeanPropertySetter(prefix + "/address/administrativeArea", "province");
264    digester.addBeanPropertySetter(prefix + "/address/postalCode", "postalCode");
265    digester.addBeanPropertySetter(prefix + "/address/country", "country");
266    digester.addCallMethod(prefix + "/address/deliveryPoint", "addAddress", 0);
267    digester.addSetNext(prefix, parentMethod);
268  }

270  /**
271   * This is a reusable set of rules to build a CuratorialUnit and add the CuratorialUnit as per the method
272   * called on parent object which is the previous stack object.
273   *
274   * @param digester     to add the rules to
275   * @param prefix       The XPath prefix to prepend for extracting the CuratorialUnit information
276   * @param parentMethod Of the previous stack object to call and add the CuratorialUnit to
277   */
278  private void addCuratorialUnit(Digester digester, String prefix, String parentMethod) {
279    digester.addObjectCreate(prefix, CuratorialUnitComposite.class);
280    digester.addCallMethod(prefix + "/jgtiUnits", "addDeviation", 1);
281    digester.addCallParam(prefix + "/jgtiUnits", 0, "uncertaintyMeasure");
282    digester.addBeanPropertySetter(prefix + "/jgtiUnitType", "typeVerbatim");
283    digester.addBeanPropertySetter(prefix + "/jgtiUnits", "count");
284    digester.addBeanPropertySetter(prefix + "/jgtiUnitRange/beginRange", "lower");
285    digester.addBeanPropertySetter(prefix + "/jgtiUnitRange/endRange", "upper");
286    digester.addSetNext(prefix + "/jgtiUnitType", parentMethod);
287  }

307  /**
308   * Adds rules to get the geographic coverage.
309   *
310   * @param digester     to add the rules to
311   * @param prefix       The XPath prefix to prepend for extracting the geographic information
312   * @param parentMethod Of the previous stack object to call and add the geographic information to
313   */
314  private void addGeographicCoverageRules(Digester digester, String prefix, String parentMethod) {
315    digester.addObjectCreate(prefix, GeospatialCoverage.class);
316    digester.addBeanPropertySetter(prefix + "/geographicDescription", "description");
317    digester.addObjectCreate(prefix + "/boundingCoordinates", BoundingBox.class);
318    digester.addBeanPropertySetter(prefix + "/boundingCoordinates/westBoundingCoordinate", "minLongitude");
319    digester.addBeanPropertySetter(prefix + "/boundingCoordinates/eastBoundingCoordinate", "maxLongitude");
320    digester.addBeanPropertySetter(prefix + "/boundingCoordinates/northBoundingCoordinate", "maxLatitude");
321    digester.addBeanPropertySetter(prefix + "/boundingCoordinates/southBoundingCoordinate", "minLatitude");
322    digester.addSetNext(prefix + "/boundingCoordinates", "setBoundingBox"); // add the BBox to the GeospatialCoverage
323    digester.addSetNext(prefix, parentMethod); // add the GeospatialCoverage to the list in EML
324  }

386  /**
387   * Adds rules to extract the temporal coverages, which can be single dates, date ranges, living time periods
388   * or formation periods.
389   *
390   * @param digester to add the rules to
391   */
392  private void addTemporalCoverageRules(Digester digester) {
393
394    digester.addObjectCreate("eml/dataset/coverage/temporalCoverage/singleDateTime", SingleDate.class);
395    digester.addBeanPropertySetter("eml/dataset/coverage/temporalCoverage/singleDateTime/calendarDate", "date");
396    digester.addSetNext("eml/dataset/coverage/temporalCoverage/singleDateTime", "addTemporalCoverage");
397
398    digester.addObjectCreate("eml/dataset/coverage/temporalCoverage/rangeOfDates", DateRange.class);
399    digester
400      .addBeanPropertySetter("eml/dataset/coverage/temporalCoverage/rangeOfDates/beginDate/calendarDate", "start");
401    digester.addBeanPropertySetter("eml/dataset/coverage/temporalCoverage/rangeOfDates/endDate/calendarDate", "end");
402    digester.addSetNext("eml/dataset/coverage/temporalCoverage/rangeOfDates", "addTemporalCoverage");
403
404    digester.addObjectCreate("eml/additionalMetadata/metadata/gbif/livingTimePeriod", VerbatimTimePeriod.class);
405    digester.addBeanPropertySetter("eml/additionalMetadata/metadata/gbif/livingTimePeriod", "period");
406    digester.addCallMethod("eml/additionalMetadata/metadata/gbif/livingTimePeriod", "setType", 1,
407      new Class[] {VerbatimTimePeriodType.class});
408    digester.addObjectParam("eml/additionalMetadata/metadata/gbif/livingTimePeriod", 0,
409      VerbatimTimePeriodType.LIVING_TIME_PERIOD);
410    digester.addSetNext("eml/additionalMetadata/metadata/gbif/livingTimePeriod", "addTemporalCoverage");
411
412    digester.addObjectCreate("eml/additionalMetadata/metadata/gbif/formationPeriod", VerbatimTimePeriod.class);
413    digester.addBeanPropertySetter("eml/additionalMetadata/metadata/gbif/formationPeriod", "period");
414    digester.addCallMethod("eml/additionalMetadata/metadata/gbif/formationPeriod", "setType", 1,
415      new Class[] {VerbatimTimePeriodType.class});
416    digester.addObjectParam("eml/additionalMetadata/metadata/gbif/formationPeriod", 0,
417      VerbatimTimePeriodType.FORMATION_PERIOD);
418    digester.addSetNext("eml/additionalMetadata/metadata/gbif/formationPeriod", "addTemporalCoverage");
419  }
