001package org.gbif.registry.metadata.parse.converter;
002
003import org.gbif.api.util.VocabularyUtils;
004
005import java.util.Map;
006
007import com.google.common.collect.Maps;
008import org.apache.commons.beanutils.converters.AbstractConverter;
009
010/**
011 * {@link org.apache.commons.beanutils.Converter} implementation that handles conversion
012 * to and from java enumerations.
013 */
014public class EnumTypeConverter<T extends Enum<?>> extends AbstractConverter {
015
016    private final Class<T> clazz;
017    private final T defaultValue;
018
019    private final Map<String, T> lookup = Maps.newHashMap();
020
021    public EnumTypeConverter(Class<T> clazz, T defaultValue) {
022        this.clazz = clazz;
023        this.defaultValue = defaultValue;
024    }
025
026
027    public void addMappings(Map<String, T> mapping) {
028        for (Map.Entry<String, T> entry : mapping.entrySet()) {
029            lookup.put(entry.getKey().toLowerCase(), entry.getValue());
030        }
031    }
032
033    /**
034     * Return the default type this <code>Converter</code> handles.
035     *
036     * @return The default type this <code>Converter</code> handles.
037     */
038    protected Class getDefaultType() {
039        return clazz;
040    }
041
042    /**
043     * <p>Convert a PreservationMethodType or object into a String.</p> Checks map with alternative values for each
044     * PreservationMethodType before returning the default value.
045     *
046     * @param type  Data type to which this value should be converted
047     * @param value The input value to be converted
048     * @return The converted value.
049     * @throws Throwable if an error occurs converting to the specified type
050     */
051    protected Object convertToType(Class type, Object value) throws Throwable {
052        // never null, super class implements this as:
053        // return value.toString();
054        final String val = value.toString();
055        if (lookup.containsKey(val)) {
056            return lookup.get(val);
057        }
058        // try regular enum values
059        try {
060            T eVal = (T) VocabularyUtils.lookupEnum(val, clazz);
061            return eVal == null ? defaultValue : eVal;
062        } catch (IllegalArgumentException e) {
063            // cant parse, return default
064            return defaultValue;
065        }
066    }
067}
