053/**
054 * Synchronises metadata from a BioCASe Installation.
055 * <p/>
056 * Every BioCASe Installation can have multiple Endpoints, each of those Endpoints can have multiple Datasets and each
057 * of those Datasets can have multiple Endpoints (Web service, ABCD Archive, DwC-A).
058 * <p/>
059 * BioCASe synchronisation happens in the following steps:
060 * <ul>
061 * <li>For all endpoints of an Installation make a {@code capabilities} request, followed by a dataset inventory
062 * request, then a metadata and count (to get the number of records) request for each dataset</li>
063 * <li>The type of inventory request depends on the capabilities. Newer version of BioCASe (3.4 and greater) support a
064 * separate inventory request.</li>
065 * </ul>
066 * <p/>
067 * Unfortunately BioCASe does not have good (stable) identifiers for Datasets so we need to rely on the Dataset title.
068 */
069public class BiocaseMetadataSynchroniser extends BaseProtocolHandler {
070  private static final Logger LOG = LoggerFactory.getLogger(BiocaseMetadataSynchroniser.class);
071
072  public BiocaseMetadataSynchroniser(HttpClient httpClient) {
073    super(httpClient);
074  }
075
076  @Override
077  public boolean canHandle(Installation installation) {
078    return installation.getType() == InstallationType.BIOCASE_INSTALLATION;
079  }
080
081  @Override
082  public SyncResult syncInstallation(
083    Installation installation, List<Dataset> datasets
084  ) throws MetadataException {
085    checkArgument(installation.getType() == InstallationType.BIOCASE_INSTALLATION,
086                  "Only supports BioCASe Installations");
087
088    List<Dataset> added = Lists.newArrayList();
089    List<Dataset> deleted = Lists.newArrayList();
090    Map<Dataset, Dataset> updated = Maps.newHashMap();
091
092    for (Endpoint endpoint : installation.getEndpoints()) {
093      LOG.info("Starting synchronization of endpoint: {}", endpoint.getUrl());
094
095      Capabilities capabilities = getCapabilities(endpoint);
096      if (capabilities.getPreferredSchema() == null) {
097        throw new MetadataException("No preferred schema", ErrorCode.PROTOCOL_ERROR);
098      }
099      List<String> datasetInventory = getDatasetInventory(capabilities, endpoint);
100      for (String datasetTitle : datasetInventory) {
101        Dataset newDataset;
102
103        if (capabilities.getPreferredSchema().equals(Constants.ABCD_12_SCHEMA)) {
104          SimpleAbcd12Metadata metadata = get12Metadata(endpoint, datasetTitle, capabilities);
105          newDataset = convertToDataset(metadata, endpoint, capabilities);
106        } else {
107          SimpleAbcd206Metadata metadata = get206Metadata(endpoint, datasetTitle, capabilities);
108          newDataset = convertToDataset(metadata, endpoint, capabilities);
109        }
110
111        Dataset existingDataset = findDataset(datasetTitle, datasets);
112        if (existingDataset == null) {
113          added.add(newDataset);
114        } else {
115          updated.put(existingDataset, newDataset);
116        }
117      }
118    }
119
120    // All Datasets that weren't updated must have been deleted
121    for (Dataset dataset : datasets) {
122      if (!updated.containsKey(dataset)) {
123        deleted.add(dataset);
124      }
125    }
126
127    return new SyncResult(updated, added, deleted, installation);
128  }

130  public URI buildUri(URI url, String parameter, String value) throws MetadataException {
131    try {
132      return new URIBuilder(url).addParameter(parameter, value).build();
133    } catch (URISyntaxException e) {
134      throw new MetadataException(e, ErrorCode.OTHER_ERROR);
135    }
136  }

138  /**
139   * Does a Capabilities request against the Endpoint.
140   */
141  private Capabilities getCapabilities(Endpoint endpoint) throws MetadataException {
142    return doHttpRequest(endpoint.getUrl(), newDigester(Capabilities.class));
143  }

145  /**
146   * Tries to get an inventory (list) of Datasets for this BioCASe Endpoint. Depending on the version of the
147   * Installation there are two ways to do this.
148   */
149  private List<String> getDatasetInventory(Capabilities capabilities, Endpoint endpoint) throws MetadataException {
150    String version = capabilities.getVersions().get("pywrapper");
151    if (checkIfSupportsNewInventory(version)) {
152      return doNewStyleInventory(endpoint);
153    } else {
154      return doOldStyleInventory(endpoint, capabilities);
155    }
156  }
157
158  /**
159   * Given a version string in the normal "major.minor.patch" format evaluates whether this version of BioCASe supports
160   * the new style inventory request or not. All versions of BioCASe 3.4 and above do support this.
161   */
162  private boolean checkIfSupportsNewInventory(String version) {
163    if (version == null) {
164      return false;
165    }
166
167    String[] versionParts = version.split("\\.");
168    try {
169      // Trying to parse the "major" part first, if we succeed and it is greater than 3 we do support the nev inventory
170      // style, if it is less than three we don't support it. If it is equal to 3 we need to check the "minor" component
171      int majorVersion = Integer.valueOf(versionParts[0]);
172      if (majorVersion < 3) {
173        return false;
174      }
175      if (majorVersion > 3) {
176        return true;
177      }
178
179      // "major" version is 3 but there is no "minor" part
180      if (versionParts.length < 2) {
181        return false;
182      }
183
184      // Check whether the "minor" version is greater than or equal to 4
185      int minorVersion = Integer.valueOf(versionParts[1]);
186      return minorVersion >= 4;
187    } catch (NumberFormatException ignored) {
188      return false;
189    }
190  }

192  /**
193   * Does a request against the dedicated {@code inventory} endpoint which lists all Datasets that are available as well
194   * as all Archives.
195   */
196  // TODO: Need to return information about archives
197  private List<String> doNewStyleInventory(Endpoint endpoint) throws MetadataException {
198    URI uri = buildUri(endpoint.getUrl(), "inventory", "1");
199    NewDatasetInventory inventory = doHttpRequest(uri, newDigester(NewDatasetInventory.class));
200    List<String> datasets = Lists.newArrayList();
201    if (inventory == null) return datasets;
202    for (InventoryDataset inventoryDataset : inventory.getDatasets()) {
203      datasets.add(inventoryDataset.getTitle());
204    }
205    return datasets;
206  }
207
208  /**
209   * Does a search request against this Endpoint specially crafted to only find all Dataset titles.
210   */
211  private List<String> doOldStyleInventory(Endpoint endpoint, Capabilities capabilities) throws MetadataException {
212    String requestParameter = TemplateUtils.getBiocaseInventoryRequest(capabilities.getPreferredSchema());
213    URI uri = buildUri(endpoint.getUrl(), "request", requestParameter);
214    OldDatasetInventory inventory = doHttpRequest(uri, newDigester(OldDatasetInventory.class));
215    return inventory.getDatasets();
216  }

238  private Dataset convertToDataset(
239    SimpleAbcd206Metadata metadata, Endpoint installationEndpoint, Capabilities capabilities
240  ) {
241    Dataset dataset = new Dataset();
242    dataset.setTitle(metadata.getName());
243    dataset.setDescription(metadata.getDetails());
244    dataset.setHomepage(metadata.getHomepage());
245    dataset.setLogoUrl(metadata.getLogoUrl());
246    dataset.setRights(metadata.getRights());
247    // Respect publisher issued DOIs if provided.
248    if (DOI.isParsable(metadata.getDatasetGUID())) {
249      dataset.setDoi(new DOI(metadata.getDatasetGUID()));
250    }
251
252    Citation citation = new Citation();
253    citation.setText(metadata.getCitationText());
254    dataset.setCitation(citation);
255
256    dataset.setContacts(metadata.getContacts());
257
258    Endpoint endpoint = new Endpoint();
259    endpoint.setType(EndpointType.BIOCASE);
260    endpoint.setUrl(installationEndpoint.getUrl());
261    endpoint.addMachineTag(MachineTag.newInstance(Constants.METADATA_NAMESPACE,
262                                                  Constants.CONCEPTUAL_SCHEMA,
263                                                  capabilities.getPreferredSchema()));
264
265    dataset.addEndpoint(endpoint);
266    return dataset;
267  }

296  }

298  /**
299   * Tries to find a matching Dataset in the list of provided Datasets by looking at the title.
300   *
301   * @return the matching Dataset or {@code null} if it could not be found
302   */
303  @Nullable
304  private Dataset findDataset(String datasetTitle, Iterable<Dataset> datasets) {
305    for (Dataset dataset : datasets) {
306      if (dataset.getTitle().equals(datasetTitle)) {
307        return dataset;
308      }
309    }
310
311    return null;
312  }
