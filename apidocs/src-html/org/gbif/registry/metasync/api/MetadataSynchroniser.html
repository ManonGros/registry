001package org.gbif.registry.metasync.api;
002
003import java.util.List;
004import java.util.UUID;
005
006/**
007 * This interface specifies all operations necessary to do a synchronisation of an Installation from our registry
008 * against the actual Endpoint.
009 */
010public interface MetadataSynchroniser {
011
012  /**
013   * Runs a synchronisation against the installation provided.
014   * <p/>
015   * All datasets from this installation will be updated, old ones deleted and new ones registered if needed.
016   *
017   * @param key of the installation to synchronise
018   *
019   * @return the result of the synchronisation
020   *
021   * @throws IllegalArgumentException if no technical installation with the given key exists, the installation doesn't
022   *                                  have any endpoints or none that we can use to gather metadata (might also be
023   *                                  missing the proper protocol handler)
024   */
025  SyncResult synchroniseInstallation(UUID key);
026
027  /**
028   * Synchronises all registered Installations ignoring any failures (because there will definitely be failures due to
029   * unsupported Installation types and missing Endpoints). This will run single threaded.
030   */
031  List<SyncResult> synchroniseAllInstallations();
032
033  /**
034   * Synchronises all registered Installations ignoring any failures (because there will definitely be failures due to
035   * unsupported Installation types and missing Endpoints).
036   *
037   * @param parallel how many threads to run in parallel
038   */
039  List<SyncResult> synchroniseAllInstallations(int parallel);
040
041}
