001package org.gbif.registry.metasync.util;
002
003import java.util.concurrent.TimeUnit;
004
005import org.apache.http.client.HttpClient;
006import org.apache.http.client.params.ClientPNames;
007import org.apache.http.conn.ClientConnectionManager;
008import org.apache.http.conn.scheme.PlainSocketFactory;
009import org.apache.http.conn.scheme.Scheme;
010import org.apache.http.conn.scheme.SchemeRegistry;
011import org.apache.http.impl.client.DecompressingHttpClient;
012import org.apache.http.impl.client.DefaultHttpClient;
013import org.apache.http.impl.conn.PoolingClientConnectionManager;
014import org.apache.http.params.BasicHttpParams;
015import org.apache.http.params.HttpConnectionParams;
016import org.apache.http.params.HttpParams;
017
018/**
019 * A factory that builds new Apache Http Clients to be used with the Metadata synchroniser.
020 *
021 * This allows for easy swapping of implementations to make testing easier.
022 */
023public class HttpClientFactory {
024
025  private static final int HTTP_PORT = 80;
026  private static final int HTTPS_PORT = 443;
027
028  private static final int MAX_TOTAL_CONNECTIONS = 200;
029  private static final int MAX_CONNECTIONS_PER_HOST = 20;
030
031  private final ClientConnectionManager connectionManager;
032
033  private final int timeout;
034
035  /**
036   * Builds a new Factory with a default timeout covering various things (connection timeout, read timeout, ...).
037   */
038  public HttpClientFactory(long timeout, TimeUnit timeUnit) {
039    long millis = TimeUnit.MILLISECONDS.convert(timeout, timeUnit);
040
041    System.setProperty("sun.net.client.defaultConnectTimeout", String.valueOf(millis));
042    System.setProperty("sun.net.client.defaultReadTimeout", String.valueOf(millis));
043
044    connectionManager = setupConnectionManager();
045    this.timeout = (int) millis;
046  }
047
048  public HttpClient provideHttpClient() {
049    HttpParams params = new BasicHttpParams();
050    HttpConnectionParams.setConnectionTimeout(params, timeout);
051    HttpConnectionParams.setSoTimeout(params, timeout);
052    params.setLongParameter(ClientPNames.CONN_MANAGER_TIMEOUT, timeout);
053    return new DecompressingHttpClient(new DefaultHttpClient(connectionManager, params));
054  }
055
056  private ClientConnectionManager setupConnectionManager() {
057    SchemeRegistry schemeRegistry = new SchemeRegistry();
058    schemeRegistry.register(new Scheme("http", HTTP_PORT, PlainSocketFactory.getSocketFactory()));
059    schemeRegistry.register(new Scheme("https", HTTPS_PORT, PlainSocketFactory.getSocketFactory()));
060
061    PoolingClientConnectionManager newConnectionManager = new PoolingClientConnectionManager(schemeRegistry);
062    newConnectionManager.setMaxTotal(MAX_TOTAL_CONNECTIONS);
063    newConnectionManager.setDefaultMaxPerRoute(MAX_CONNECTIONS_PER_HOST);
064    return newConnectionManager;
065  }
066
067}
