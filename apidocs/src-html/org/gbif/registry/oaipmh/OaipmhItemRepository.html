087  /**
088   * Build a CacheLoader<UUID, Organization> around the organizationMapper instance.
089   *
090   * @return

101  @Override
102  public Item getItem(String s) throws IdDoesNotExistException, OAIException {
103
104    Dataset dataset = null;
105
106    // the fully augmented dataset
107    try {
108      dataset = datasetResource.get(UUID.fromString(s));
109    }
110    catch (IllegalArgumentException ignoreEx){}
111
112    if (dataset != null) {
113      try {
114        return toOaipmhItem(dataset);
115      } catch (Exception e) {
116        throw new ServiceUnavailableException("Failed to serialize dataset " + s + " to DC/EML", e);
117      }
118    }
119    throw new IdDoesNotExistException();
120  }

157  /**
158   * Get items identifier as {@link ListItemIdentifiersResult} matching the provided filters.
159   *
160   * @param list
161   * @param offset
162   * @param length
163   * @param set a valid set or null
164   * @param from from date (inclusive) or null
165   * @param until until date (exclusive) or null
166   * @return
167   * @throws OAIException
168   */
169  @Override
170  public ListItemIdentifiersResult getItemIdentifiers(List<ScopedFilter> list, int offset, int length, String set, Date from, Date until) throws OAIException {
171    // ask for length+1 to determine if there are more results
172    List<Dataset> datasetList = getDatasetListFromFilters(offset, length+1, set, from, until);
173    List<ItemIdentifier> results = Lists.newArrayListWithCapacity(datasetList.size());
174
175    boolean hasMoreResults = (datasetList.size() == length+1);
176    // remove last element, it was only retrieve to determine hasMoreResults
177    if (hasMoreResults) {
178      datasetList.remove(datasetList.size()-1);
179    }
180
181    for (Dataset dataset : datasetList) {
182      results.add(toOaipmhItemIdentifier(dataset));
183    }
184
185    return new ListItemIdentifiersResult(hasMoreResults, results);
186  }

288  /**
289   * Get items as {@link ListItemsResults} matching the provided filters.
290   *
291   * @param list
292   * @param offset
293   * @param length
294   * @param set a valid set or null
295   * @param from from date (inclusive) or null
296   * @param until until date (exclusive) or null
297   * @return
298   * @throws OAIException
299   */
300  @Override
301  public ListItemsResults getItems(List<ScopedFilter> list, int offset, int length, String set, Date from, Date until) throws OAIException {
302
303    // ask for length+1 to determine if there are more results
304    List<Dataset> datasetList = getDatasetListFromFilters(offset, length+1, set, from, until);
305    List<Item> results = Lists.newArrayListWithCapacity(datasetList.size());
306
307    boolean hasMoreResults = (datasetList.size() == length+1);
308    // remove last element, it was only retrieve to determine hasMoreResults
309    if (hasMoreResults) {
310      datasetList.remove(datasetList.size()-1);
311    }
312
313    PagingResponse<Dataset> pagingResponse = new PagingResponse<>();
314    pagingResponse.setResults(datasetList);
315    pagingResponse = datasetResource.augmentWithMetadata(pagingResponse);
316
317    try {
318      for (Dataset dataset : pagingResponse.getResults() ) {
319        results.add(toOaipmhItem(dataset));
320      }
321    } catch (IOException e) {
322      // caused by https://github.com/DSpace/xoai/issues/31
323      LOG.error("Failed to serialize datasets to DC/EML", e);
324    }
325    return new ListItemsResults(hasMoreResults, results);
326  }

328  /**
329   * Build a {@OaipmhItem} instance from a {@link Dataset} and the {@link Set} it belongs to.
330   *
331   * @param dataset
332   * @return
333   * @throws IOException
334   */
335  private OaipmhItem toOaipmhItem(Dataset dataset) throws IOException {
336
337    Organization organization = null;
338    try {
339      organization = ORGANIZATION_CACHE.get(dataset.getPublishingOrganizationKey());
340    } catch (ExecutionException e) {
341      LOG.error("Error while loading Organization from cache fro dataset {}", dataset, e);
342    }
343    List<Set> sets = getSets(organization, dataset);
344    Map<String, Object> additionalProperties = Maps.newHashMap();
345    additionalProperties.put(DublinCoreWriter.ADDITIONAL_PROPERTY_DC_FORMAT, ExtraMediaTypes.APPLICATION_DWCA);
346
347    // get the occurrence counts for this dataset, only used in DublinCore
348    // This is designed to fail fast (short http timeout) and on failures which are expected to be exceptional
349    // events, it is simply omitted.  See the Guice RegistryWsSevletListener for the configuration of the timeout.
350    ReadBuilder readBuilder = new ReadBuilder();
351    readBuilder.at(OccurrenceCube.DATASET_KEY, dataset.getKey());
352    try {
353      long occurrenceCount = cubeService.get(readBuilder);
354      if (occurrenceCount > 0) {
355        additionalProperties.put(DublinCoreWriter.ADDITIONAL_PROPERTY_OCC_COUNT, occurrenceCount);
356      }
357    } catch (ClientHandlerException ex) {
358      LOG.warn("Unable to get occurrence count from cubeService for dataset {}. Omitting count.", dataset.getKey(), ex);
359    }
360
361    /**
362     * The XOAI library doesn't provide us with the metadata type (EML / OAI DC), so both must be produced.
363     * An XSLT transform pulls out the one that's required.
364     * This is ugly, so see https://github.com/DSpace/xoai/issues/31
365     */
366    StringWriter xml = new StringWriter();
367
368    xml.write("<root>");
369
370    xml.write("<oaidc>\n");
371    dublinCoreWriter.writeTo(organization, dataset, additionalProperties, xml);
372    xml.write("</oaidc>\n");
373
374    xml.write("<eml>\n");
375    emlWriter.write(dataset, xml);
376    xml.write("</eml>\n");
377
378    xml.write("</root>\n");
379    return new OaipmhItem(dataset, xml.toString(), sets);
380  }
381
382  private OaipmhItem toOaipmhItemIdentifier(Dataset dataset) {
383    Organization organization = null;
384    try {
385      organization = ORGANIZATION_CACHE.get(dataset.getPublishingOrganizationKey());
386    } catch (ExecutionException e) {
387      LOG.error("Error while loading Organization from cache fro dataset {}", dataset, e);
388    }
389    return new OaipmhItem(dataset, getSets(organization, dataset));
390  }

392  /**
393   * Get the list of {@link org.dspace.xoai.dataprovider.model.Set} for a {@link Dataset}.
394   *
395   * @param organization {@link Organization}, can be null
396   * @param dataset non-null {@link Dataset}
397   * @return list of all {@link org.dspace.xoai.dataprovider.model.Set} that the {@link Dataset} belongs to. Never null.
398   */
399  private List<Set> getSets(Organization organization, @NotNull Dataset dataset) {
400    Country publishingCountry = null;
401
402    if(organization != null) {
403      publishingCountry = organization.getCountry();
404    }
405
406    List<Set> sets = Lists.newArrayList();
407    sets.add(new Set(OaipmhSetRepository.SetType.INSTALLATION.getSubsetPrefix() + dataset.getInstallationKey().toString()));
408    sets.add(new Set(OaipmhSetRepository.SetType.DATASET_TYPE.getSubsetPrefix() + dataset.getType().toString()));
409    if(publishingCountry != null) {
410      sets.add(new Set(OaipmhSetRepository.SetType.COUNTRY.getSubsetPrefix() + publishingCountry.getIso2LetterCode()));
411    }
412    return sets;
413  }

415  /**
416   * Get a list of {@link Dataset} based on filter(s).
417   *
418   * @param offset
419   * @param length
420   * @param set set name in the form of set:subset {@see http://www.openarchives.org/OAI/openarchivesprotocol.html#Set}
421   *            XOAI library validates the set before calling the ItemRepository so we do not validate it again here.
422   * @param from
423   * @param until
424   * @return list of matching {@link Dataset}. Never null.
425   */
426  private List<Dataset> getDatasetListFromFilters(int offset, int length, String set, Date from, Date until) {
427
428    Optional<OaipmhSetRepository.SetIdentification> setIdentification = OaipmhSetRepository.parseSetName(set);
429
430    List<Dataset> datasetList;
431    if(setIdentification.isPresent()){
432      Country country = null;
433      UUID installationKey = null;
434      DatasetType datasetType = null;
435
436      String subSet = setIdentification.get().getSubSet();
437      switch (setIdentification.get().getSetType()) {
438        case COUNTRY:
439          country = Country.fromIsoCode(subSet);
440          break;
441        case INSTALLATION:
442          installationKey = UUID.fromString(subSet);
443          break;
444        case DATASET_TYPE:
445          datasetType = DatasetType.fromString(subSet);
446          break;
447      }
448
449      datasetList = datasetMapper.listWithFilter(country, datasetType, installationKey,
450              from, until, new PagingRequest(offset, length));
451    }
452    else {
453      datasetList = datasetMapper.listWithFilter(null, null, null, from, until, new PagingRequest(offset, length));
454    }
455
456    return datasetList;
457  }
