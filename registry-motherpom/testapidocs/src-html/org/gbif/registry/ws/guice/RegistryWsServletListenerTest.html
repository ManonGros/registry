001package org.gbif.registry.ws.guice;
002
003import org.gbif.api.model.common.paging.PagingRequest;
004import org.gbif.api.service.common.UserService;
005import org.gbif.api.service.registry.DatasetService;
006import org.gbif.api.service.registry.OrganizationService;
007import org.gbif.registry.database.LiquibaseInitializer;
008import org.gbif.registry.guice.RegistryTestModules;
009import org.gbif.utils.file.properties.PropertiesUtil;
010import org.gbif.ws.security.GbifAuthService;
011
012import java.io.IOException;
013import java.util.Properties;
014
015import com.google.inject.Injector;
016import org.junit.ClassRule;
017import org.junit.Test;
018
019import static org.junit.Assert.assertNotNull;
020
021public class RegistryWsServletListenerTest {
022
023  // Flushes the database on each run
024  @ClassRule
025  public static final LiquibaseInitializer liquibaseRule = new LiquibaseInitializer(RegistryTestModules.database());
026
027  private static Properties properties;
028
029  static {
030    try {
031      properties = PropertiesUtil.loadProperties("registry-test.properties");
032    } catch (IOException e) {
033      throw new RuntimeException(e);
034    }
035  }
036
037  /**
038   * Makes sure that two mybatis projects each with its own Datasource work fine together.
039   * Tests the complete listener module, calling real methods to force guice to finalize the bindings.
040   */
041  @Test
042  public void testListenerModule() {
043    RegistryWsServletListener mod = new RegistryWsServletListener(properties);
044    Injector injector = mod.getInjector();
045
046    GbifAuthService auth = injector.getInstance(GbifAuthService.class);
047    assertNotNull(auth);
048
049    DatasetService datasetService = injector.getInstance(DatasetService.class);
050    datasetService.list(new PagingRequest());
051
052    UserService userService = injector.getInstance(UserService.class);
053    userService.get("admin");
054
055    OrganizationService orgService = injector.getInstance(OrganizationService.class);
056    orgService.list(new PagingRequest());
057  }
058
059}
