046    @Test
047    public void testConvert() throws Exception {
048        Organization publisher = new Organization();
049        publisher.setTitle("X-Publisher");
050        publisher.setKey(UUID.randomUUID());
051
052        final DOI doi = new DOI("10.1234/5678");
053        Dataset d = new Dataset();
054        d.setKey(UUID.randomUUID());
055        d.setType(DatasetType.OCCURRENCE);
056        d.setTitle("my title");
057        d.setCreated(new Date());
058        d.setModified(new Date());
059        d.setCreatedBy("Markus");
060        d.setLanguage(Language.NORWEGIAN);
061        d.setDataLanguage(Language.NORWEGIAN);
062
063        DataCiteMetadata m = convertAndValidate(doi, d, publisher);
064        assertEquals("my title", m.getTitles().getTitle().get(0).getValue());
065        assertEquals("Markus", m.getCreators().getCreator().get(0).getCreatorName());
066        assertEquals(doi.getDoiName(), m.getIdentifier().getValue());
067
068        d.setDoi(doi);
069        d.setDescription("bla bla bla bla bla bla bla bla - I talk too much");
070        List<GeospatialCoverage> geos = Lists.newArrayList();
071        d.setGeographicCoverages(geos);
072        GeospatialCoverage g = new GeospatialCoverage();
073        geos.add(g);
074        g.setBoundingBox(new BoundingBox(1, 2, 3, 4));
075
076        m = convertAndValidate(doi, d, publisher);
077        assertEquals("my title", m.getTitles().getTitle().get(0).getValue());
078        assertEquals("Markus", m.getCreators().getCreator().get(0).getCreatorName());
079        assertEquals("10.1234/5678", m.getIdentifier().getValue());
080        assertEquals(Lists.<Double>newArrayList(1d, 3d, 2d, 4d), m.getGeoLocations().getGeoLocation().get(0).getGeoLocationBox());
081        assertEquals(d.getDescription(), m.getDescriptions().getDescription().get(0).getContent().get(0));
082    }
083
084
085    private DataCiteMetadata convertAndValidate(DOI doi, Dataset d, Organization publisher) throws InvalidMetadataException {
086        DataCiteMetadata m = DataCiteConverter.convert(d, publisher);
087        DataCiteValidator.toXml(doi, m);
088        return m;
089    }

091    @Test
092    public void testConvertDownload() throws Exception {
093        DatasetOccurrenceDownloadUsage du1 = new DatasetOccurrenceDownloadUsage();
094        du1.setDatasetKey(UUID.randomUUID());
095        du1.setDatasetTitle("my title");
096        du1.setDatasetDOI(new DOI("10.1234/5679"));
097        du1.setNumberRecords(101);
098
099        DatasetOccurrenceDownloadUsage du2 = new DatasetOccurrenceDownloadUsage();
100        du2.setDatasetKey(UUID.randomUUID());
101        du2.setDatasetTitle("my title #2");
102        du2.setDatasetDOI(new DOI("10.1234/klimbim"));
103        du2.setNumberRecords(2002);
104
105        Download download = new Download();
106        download.setCreated(new Date());
107        download.setDoi(new DOI("10.1234/5678"));
108        download.setKey("1");
109        download.setModified(new Date());
110        download.setNumberDatasets(1l);
111        download.setSize(100);
112        download.setStatus(Download.Status.SUCCEEDED);
113        download.setTotalRecords(10);
114        DownloadRequest downloadRequest = new DownloadRequest();
115        downloadRequest.setCreator("dev@gbif.org");
116        download.setRequest(downloadRequest);
117
118        User user = new User();
119        user.setUserName("peta");
120        user.setEmail("aha@music.com");
121        user.setFirstName("Pete");
122        user.setEmail("Doherty");
123
124        // mock title lookup API
125        TitleLookup tl = mock(TitleLookup.class);
126        when(tl.getDatasetTitle(anyString())).thenReturn("PonTaurus");
127        when(tl.getSpeciesName(anyString())).thenReturn("Abies alba Mill.");
128
129        DataCiteMetadata metadata = DataCiteConverter.convert(download, user, Lists.newArrayList(du1, du2), tl);
130        String xml = DataCiteValidator.toXml(download.getDoi(), metadata);
131        System.out.println(xml);
132        assertTrue(xml.contains(du1.getDatasetDOI().getDoiName()));
133        assertTrue(xml.contains(du2.getDatasetDOI().getDoiName()));
134        assertTrue(xml.contains(String.valueOf(du1.getNumberRecords())));
135        assertTrue(xml.contains(String.valueOf(du2.getNumberRecords())));
136    }

138    @Test
139    public void testTruncateDesription() throws Exception {
140        DOI doi = new DOI("10.15468/dl.v8zc57");
141        String xml = Resources.toString(Resources.getResource("metadata/datacite-large.xml"), Charsets.UTF_8);
142        String xml2 = DataCiteConverter.truncateDescription(doi, xml, URI.create("http://gbif.org"));
143        //System.out.println(xml2);
144        DataCiteValidator.validateMetadata(xml2);
145        assertTrue(xml2.contains("for full list of all constituents"));
146        assertFalse(xml2.contains("University of Ghent"));
147        assertTrue(xml2.contains("10.15468/siye1z"));
148        assertEquals(3648, xml2.length());
149    }
