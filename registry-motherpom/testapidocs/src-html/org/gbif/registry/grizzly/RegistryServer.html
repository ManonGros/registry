013package org.gbif.registry.grizzly;
014
015import org.gbif.registry.guice.TestRegistryWsServletListener;
016import org.gbif.registry.search.DatasetIndexUpdateListener;
017
018import java.io.IOException;
019
020import com.google.common.base.Preconditions;
021import com.google.common.base.Throwables;
022import com.google.inject.servlet.GuiceFilter;
023import com.sun.grizzly.http.embed.GrizzlyWebServer;
024import com.sun.grizzly.http.servlet.ServletAdapter;
025import org.apache.solr.client.solrj.SolrClient;
026import org.junit.rules.TestRule;
027import org.junit.runner.Description;
028import org.junit.runners.model.Statement;
029import org.slf4j.Logger;
030import org.slf4j.LoggerFactory;
031
032/**
033 * An embedded web server using Grizzly that run the registry ws application.
034 * To allow WS tests to flush SOLR between runs, this allows the registration of the SOLR server.
035 */
036public class RegistryServer implements TestRule {
037
038  /**
039   * The system property one can set to override the port.
040   */
041  public static final String PORT_PROPERTY = "grizzly.port"; // to override as a system property
042
043  /**
044   * The default port to use, should no system property be supplied.
045   */
046  public static final int DEFAULT_PORT = 7001;
047
048  private static final Logger LOG = LoggerFactory.getLogger(RegistryServer.class);
049  private final GrizzlyWebServer webServer;
050
051  public static final RegistryServer INSTANCE = new RegistryServer();
052
053  private SolrClient solrClient;
054  private DatasetIndexUpdateListener datasetUpdater;
055
056  /**
057   * Gets the port that grizzly will run on. This will either be {@link #DEFAULT_PORT} or the value supplied as a
058   * system
059   * property named {@link #PORT_PROPERTY}.
060   */
061  public static int getPort() {
062    String port = System.getProperty(PORT_PROPERTY);
063    try {
064      return (port == null) ? DEFAULT_PORT : Integer.parseInt(port);
065    } catch (NumberFormatException e) {
066      throw new IllegalArgumentException(PORT_PROPERTY + " does not hold a valid port: " + port);
067    }
068  }
069
070  // we are a singleton instance
071  private RegistryServer() {
072    webServer = new GrizzlyWebServer(getPort());
073    ServletAdapter sa = new ServletAdapter();
074    sa.addServletListener(TestRegistryWsServletListener.class.getName());
075    sa.addFilter(new GuiceFilter(), "Guice", null);
076    sa.setContextPath("/");
077    sa.setServletPath("/");
078    webServer.addGrizzlyAdapter(sa, null);
079  }
080
081  public void start() {
082    LOG.info("Starting registry WS for tests");
083    try {
084      webServer.start();
085    } catch (IOException e) {
086      Throwables.propagate(e);
087    }
088  }
089
090  public void stop() {
091    webServer.stop();
092    LOG.info("Stopping test registry WS");
093  }
094
095  public synchronized void setSolrClient(SolrClient solrClient) {
096    this.solrClient = solrClient;
097  }
098
099  public synchronized void setDatasetUpdater(DatasetIndexUpdateListener datasetUpdater) {
100    this.datasetUpdater = datasetUpdater;
101  }
102
103  /**
104   * Utility to allow this to be used as a rule that will start and stop a server around the statement base.
105   */
106  @Override
107  public Statement apply(final Statement base, Description description) {
108    return new Statement() {
109
110      @Override
111      public void evaluate() throws Throwable {
112        RegistryServer server = new RegistryServer();
113        server.start();
114        try {
115          base.evaluate();
116        } finally {
117          server.stop();
118        }
119      }
120    };
121  }
122
123  public SolrClient getSolrClient() {
124    Preconditions.checkNotNull(solrClient, "Misuse of class. SolrClient has not been set");
125    return solrClient;
126  }
127
128  public DatasetIndexUpdateListener getDatasetUpdater() {
129    Preconditions.checkNotNull(datasetUpdater, "Misuse of class. DatasetIndexUpdateListener has not been set");
130    return datasetUpdater;
131  }
132}
