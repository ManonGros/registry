092  @Parameters

094    final Injector webservice = webservice();

097    return ImmutableList.<Object[]>of(

103  /**
104   * Creates {@link Download} instance with test data.
105   * The key is generated randomly using the class java.util.UUID.
106   * The instance generated should be ready and valid to be persisted.
107   */
108  protected static Download getTestInstance() {
109    Download download = new Download();
110    final Collection<String> emails = Arrays.asList("downloadtest@gbif.org");
111    DownloadRequest request =
112      new DownloadRequest(new EqualsPredicate(OccurrenceSearchParameter.TAXON_KEY, "212"), TEST_ADMIN_USER, emails,
113        true, DownloadFormat.DWCA);
114    download.setKey(UUID.randomUUID().toString());
115    download.setStatus(Download.Status.PREPARING);
116    download.setRequest(request);
117    download.setDoi(new DOI("doi:10.1234/1ASCDU"));
118    download.setDownloadLink("testUrl");
119    return download;
120  }

183  /**
184   * Creates several occurrence download with the same user name and attempts to get them with a different user name.
185   */
186  @Test(expected = AccessControlException.class)
187  public void testListByUnauthorizedUser() {
188    // This test applies to web service calls only, requires a security context.
189    if (simplePrincipalProvider != null) {
190      for (int i = 1; i <= 5; i++) {
191        occurrenceDownloadService.create(getTestInstance());
192      }
193      final Injector clientBasicAuth = webserviceBasicAuthClient(TEST_USER, TEST_USER);
194      OccurrenceDownloadService downloadServiceAuth = clientBasicAuth.getInstance(OccurrenceDownloadService.class);
195      assertTrue("List by user operation should return 5 records",
196        downloadServiceAuth.listByUser(TEST_ADMIN_USER, new PagingRequest(3, 5),null).getResults().size() > 0);
197
198    } else {
199      // Just to make the test pass for the webservice version
200      throw new AccessControlException("Fake exception");
201    }
202  }

240  /**
241   * Tests the status update of {@link Download}.
242   */
243  @Test
244  public void testUpdateStatus() {
245    Download occurrenceDownload = getTestInstance();
246    occurrenceDownloadService.create(occurrenceDownload);
247    occurrenceDownload.setStatus(Download.Status.RUNNING);
248    occurrenceDownload.setSize(200L);
249    occurrenceDownload.setTotalRecords(600L);
250    occurrenceDownload.setDoi(new DOI("doi:10.1234/1ASCDU"));
251    occurrenceDownloadService.update(occurrenceDownload);
252    Download occurrenceDownload2 = occurrenceDownloadService.get(occurrenceDownload.getKey());
253    assertTrue(occurrenceDownload2.getStatus() == Download.Status.RUNNING);
254    assertTrue(occurrenceDownload2.getSize() == 200L);
255    assertTrue(occurrenceDownload2.getTotalRecords() == 600L);
256  }

258
259  /**
260   * Tests the status update of {@link Download}.
261   */
262  @Test
263  public void testUpdateStatusCompleted() {
264    Download occurrenceDownload = getTestInstance();
265    occurrenceDownloadService.create(occurrenceDownload);
266    // reload to get latest db modifications like created date
267    occurrenceDownload = occurrenceDownloadService.get(occurrenceDownload.getKey());
268
269    occurrenceDownload.setStatus(Download.Status.SUCCEEDED);
270    occurrenceDownload.setSize(200L);
271    occurrenceDownload.setTotalRecords(600L);
272    occurrenceDownload.setDoi(new DOI("doi:10.1234/1ASCDU"));
273    occurrenceDownloadService.update(occurrenceDownload);
274    Download occurrenceDownload2 = occurrenceDownloadService.get(occurrenceDownload.getKey());
275    assertTrue(occurrenceDownload2.getStatus() == Download.Status.SUCCEEDED);
276    assertTrue(occurrenceDownload2.getModified() != null);
277    assertTrue(occurrenceDownload2.getSize() == 200L);
278    assertTrue(occurrenceDownload2.getTotalRecords() == 600L);
279  }
