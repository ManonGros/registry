016package org.gbif.registry.utils;
017
018import org.gbif.api.model.registry.Organization;
019import org.gbif.api.service.registry.NodeService;
020import org.gbif.api.service.registry.OrganizationService;
021import org.gbif.registry.guice.RegistryTestModules;
022import org.gbif.registry.ws.resources.NodeResource;
023import org.gbif.registry.ws.resources.OrganizationResource;
024
025import java.util.UUID;
026
027import com.google.inject.Injector;
028import org.apache.http.auth.UsernamePasswordCredentials;
029import org.codehaus.jackson.type.TypeReference;
030
031public class Organizations extends JsonBackedData<Organization> {
032
033  private static final Organizations INSTANCE = new Organizations();
034  private static NodeService nodeService;
035  private static OrganizationService organizationService;
036
037  private Organizations() {
038    super("data/organization.json", new TypeReference<Organization>() {});
039    Injector i = RegistryTestModules.webservice();
040    nodeService = i.getInstance(NodeResource.class);
041    organizationService = i.getInstance(OrganizationResource.class);
042  }
043
044  public static Organization newInstance(UUID endorsingNodeKey) {
045    Organization o = INSTANCE.newTypedInstance();
046    o.setEndorsingNodeKey(endorsingNodeKey);
047    o.setPassword("password");
048    return o;
049  }
050
051  /**
052   * Persist a new Organization for use in Unit Tests.
053   *
054   * @return persisted Organization
055   */
056  public static Organization newPersistedInstance() {
057    UUID nodeKey = nodeService.create(Nodes.newInstance());
058    Organization organization = newInstance(nodeKey);
059    // password was not included in organization.json, so set it here
060    organization.setPassword("password");
061    UUID organizationKey = organizationService.create(organization);
062    return organizationService.get(organizationKey);
063  }
064
065  /**
066   * Populate credentials used in ws requests.
067   *
068   * @param organization Organization
069   *
070   * @return credentials
071   */
072  public static UsernamePasswordCredentials credentials(Organization organization) {
073    return new UsernamePasswordCredentials(organization.getKey().toString(), organization.getPassword());
074  }
075}
