035  /**
036   * Initializes and configures the HttpUtil instance, which is what the IPT/Legacy WS consumers use to issue HTTP
037   * requests. This method copies the same configuration as the IPT uses, to ensure the exact same structured requests
038   * are sent during unit testing.
039   */
040  public static HttpUtil getHttp() {
041    // new threadsafe, multithreaded http client with support for http and https.
042    DefaultHttpClient client = HttpUtil.newMultithreadedClient(1000000, 1, 1);
043    // the registry requires Preemptive authentication, so make this the very first interceptor in the protocol chain
044    client.addRequestInterceptor(new PreemptiveAuthenticationInterceptor(), 0);
045    return new HttpUtil(client);
046  }
047
048  /**
049   * Construct request URI, appending path to local Grizzly server address.
050   *
051   * @param path path
052   *
053   * @return request URI
054   */
055  public static String getRequestUri(String path) {
056    StringBuilder uri = new StringBuilder();
057    uri.append("http://localhost:");
058    uri.append(RegistryServer.getPort());
059    uri.append(path);
060    return uri.toString();
061  }
