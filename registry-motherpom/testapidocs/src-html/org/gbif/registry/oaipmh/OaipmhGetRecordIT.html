031/**
032 * Test the OAI-PMH endpoint using the XOAI OAI-PMH client library.
033 * Test the GetRecord verb of the OAI-PMH endpoint.
034 */
035@RunWith(Parameterized.class)
036public class OaipmhGetRecordIT extends AbstractOaipmhEndpointIT {
037  
038  public OaipmhGetRecordIT(NodeService nodeService, OrganizationService organizationService, InstallationService installationService, DatasetService datasetService) {
039    super(nodeService, organizationService, installationService, datasetService);
040  }
041

043  public void getRecordNotFound() throws Throwable {
044
045    serviceProvider.getRecord(
046            GetRecordParameters.request()
047                    .withIdentifier("non-existent-record-identifier")
048                    .withMetadataFormatPrefix(EML_FORMAT.getMetadataPrefix())

052  @Test(expected = CannotDisseminateFormatException.class)

057
058    String key = org1Installation1Dataset1.getKey().toString();
059
060    serviceProvider.getRecord(GetRecordParameters.request().withIdentifier(key).withMetadataFormatPrefix("made-up-metadata-format"));
061  }
062
063  @Test
064  public void getRecordFound() throws Exception {

067    Dataset org1Installation1Dataset1 = createDataset(org1.getKey(), org1Installation1.getKey(), DatasetType.CHECKLIST, new Date());

077
078  @Test
079  public void getRecordWithAugmentedMetadata() throws Exception {
080    Organization org1 = createOrganization(Country.ZAMBIA);
081    Installation org1Installation1 = createInstallation(org1.getKey());
082    Dataset org1Installation1Dataset1 = createDataset(org1.getKey(), org1Installation1.getKey(), DatasetType.CHECKLIST, new Date());
083
084    insertMetadata(org1Installation1Dataset1.getKey(), FileUtils.classpathStream("metadata/sample.xml"));
085
086    String key = org1Installation1Dataset1.getKey().toString();
087
088    Record record = serviceProvider.getRecord(
089            GetRecordParameters.request()
090                    .withIdentifier(key)
091                    .withMetadataFormatPrefix(EML_FORMAT.getMetadataPrefix()));
092    assertEquals(key, record.getHeader().getIdentifier());
093
094    // the xoai library doesn't return the <metadata> content as we expect so test the returned document directly.
095    String result = IOUtils.toString(new URI(baseUrl + "?verb=GetRecord&metadataPrefix=" + EML_FORMAT.getMetadataPrefix() +
096            "&identifier=" + record.getHeader().getIdentifier().toString()).toURL(), "UTF-8");
097
098    // ensure we get the augmented metadata data
099    assertTrue("GetRecord verb returns augmented metadata", result.contains("<citation identifier=\"doi:tims-ident.2136.ex43.33.d\">title 1</citation>"));
100  }
101
102  @Test
103  public void testGetDeletedRestoredRecord() throws Exception {
104    Organization org1 = createOrganization(Country.ZAMBIA);
105    Installation org1Installation1 = createInstallation(org1.getKey());
106    Dataset org1Installation1Dataset1 = createDataset(org1.getKey(), org1Installation1.getKey(), DatasetType.CHECKLIST, new Date());
107
108    String key = org1Installation1Dataset1.getKey().toString();
109
110    Record record = serviceProvider.getRecord(
111            GetRecordParameters.request()
112                    .withIdentifier(key)
113                    .withMetadataFormatPrefix(EML_FORMAT.getMetadataPrefix())
114    );
115    assertEquals(key, record.getHeader().getIdentifier());
116    assertFalse("An active record is returned and not identified as deleted", record.getHeader().isDeleted());
117
118    // deleted the record and make sure it is flagged as 'deleted'
119    deleteDataset(org1Installation1Dataset1.getKey());
120
121    record = serviceProvider.getRecord(
122            GetRecordParameters.request()
123                    .withIdentifier(key)
124                    .withMetadataFormatPrefix(EML_FORMAT.getMetadataPrefix()));
125    assertEquals(key, record.getHeader().getIdentifier());
126    assertTrue("A deleted record is returned and is identified as deleted", record.getHeader().isDeleted());
127
128    // restore the dataset
129    org1Installation1Dataset1 = getDataset(org1Installation1Dataset1.getKey());
130    assertNotNull(org1Installation1Dataset1.getDeleted());
131
132    org1Installation1Dataset1.setDeleted(null);
133    updateDataset(org1Installation1Dataset1);
134    record = serviceProvider.getRecord(
135            GetRecordParameters.request()
136                    .withIdentifier(key)
137                    .withMetadataFormatPrefix(EML_FORMAT.getMetadataPrefix()));
138    assertEquals(key, record.getHeader().getIdentifier());
139    assertFalse("A restored record is returned and not identified as deleted", record.getHeader().isDeleted());
140  }
