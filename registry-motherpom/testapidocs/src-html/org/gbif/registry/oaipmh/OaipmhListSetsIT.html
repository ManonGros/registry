025/**
026 * Test the ListSets verb of the OAI-PMH endpoint.
027 *
028 * @author cgendreau
029 */
030public class OaipmhListSetsIT  extends AbstractOaipmhEndpointIT {
031
032  // OaipmhSetRepository.SetType represents the root of the set hierarchy to the respective node,
033  // DatasetType.values() represents the static subsets of OaipmhSetRepository.SetType.DATASET_TYPE
034  private static final int NUMBER_OF_STATIC_SETS =  OaipmhSetRepository.SetType.values().length + (DatasetType.values().length);
035
036  private static final Function<Set, String> SET_TO_SETSPEC =
037    new Function<Set, String>() {
038      public String apply(Set set) {
039        return set.getSpec();
040      }
041    };
042
043  public OaipmhListSetsIT(NodeService nodeService, OrganizationService organizationService, InstallationService installationService, DatasetService datasetService) {
044    super(nodeService, organizationService, installationService, datasetService);
045  }
046
047  @Test
048  public void testSetsWithEmptyDatabase() throws NoSetHierarchyException {
049
050    Iterator<Set> sets = serviceProvider.listSets();
051    List<Set> setsList = Lists.newArrayList(sets);
052
053    assertEquals("ListSets verb returns only static Sets when database is empty", NUMBER_OF_STATIC_SETS, setsList.size());
054  }
055
056  @Test
057  public void testSetsWithDatabaseContent() throws Exception {
058    Organization orgIceland = createOrganization(Country.ICELAND);
059    Installation orgIcelandInstallation1 = createInstallation(orgIceland.getKey());
060
061    Iterator<Set> sets = serviceProvider.listSets();
062    List<Set> setsList = Lists.newArrayList(sets);
063    assertEquals("ListSets verb returns only static Sets when there is not datasets", NUMBER_OF_STATIC_SETS, setsList.size());
064
065    createDataset(orgIceland.getKey(), orgIcelandInstallation1.getKey(), DatasetType.OCCURRENCE, new Date());
066
067    // refresh data
068    sets = serviceProvider.listSets();
069    setsList = Lists.newArrayList(sets);
070
071    List<String> setsSpecList = Lists.transform(setsList, SET_TO_SETSPEC);
072    assertTrue("ListSets verb returns a Set corresponding to the country of the datatset publishing Organization",
073            setsSpecList.contains(OaipmhSetRepository.SetType.COUNTRY.getSubsetPrefix() + Country.ICELAND.getIso2LetterCode()));
074
075    assertTrue("ListSets verb returns a Set corresponding to the Installation from which the datatset was published",
076            setsSpecList.contains(OaipmhSetRepository.SetType.INSTALLATION.getSubsetPrefix() + orgIcelandInstallation1.getKey().toString()));
077  }
