059  @Test
060  public void testListRecordsBySets() throws Exception {
061    Calendar calendar = Calendar.getInstance();
062    calendar.set(2015,9,16);
063    Organization orgIceland = createOrganization(Country.ICELAND);
064    Installation orgIcelandInstallation1 = createInstallation(orgIceland.getKey());
065    Dataset orgIcelandInstallation1Dataset1 = createDataset(orgIceland.getKey(), orgIcelandInstallation1.getKey(), DatasetType.CHECKLIST, calendar.getTime());
066
067    Installation orgIcelandInstallation2 = createInstallation(orgIceland.getKey());
068    Dataset orgIcelandInstallation2Dataset1 = createDataset(orgIceland.getKey(), orgIcelandInstallation2.getKey(), DatasetType.OCCURRENCE, new Date());
069
070    Organization org2 = createOrganization(Country.NEW_ZEALAND);
071    Installation org2Installation1 = createInstallation(org2.getKey());
072    createDataset(org2.getKey(), org2Installation1.getKey(), DatasetType.CHECKLIST, new Date());
073
074    // SetType.COUNTRY
075    Iterator<Record> records = serviceProvider.listRecords(
076            ListRecordsParameters.request()
077                    .withMetadataPrefix(EML_FORMAT.getMetadataPrefix())
078                    .withSetSpec(OaipmhSetRepository.SetType.COUNTRY.getSubsetPrefix() + Country.ICELAND.getIso2LetterCode()));
079    assertTrue("ListRecords verb with set Country return at least 1 record", records.hasNext());
080
081    // SetType.INSTALLATION
082    records = serviceProvider.listRecords(
083            ListRecordsParameters.request()
084                    .withMetadataPrefix(EML_FORMAT.getMetadataPrefix())
085                    .withSetSpec(OaipmhSetRepository.SetType.INSTALLATION.getSubsetPrefix() + orgIcelandInstallation1.getKey().toString()));
086    assertTrue("ListRecords verb with set Installation return at least 1 record", records.hasNext());
087    assertEquals(orgIcelandInstallation1Dataset1.getKey().toString(), records.next().getHeader().getIdentifier());
088
089    // SetType.DATASET_TYPE
090    records = serviceProvider.listRecords(
091            ListRecordsParameters.request()
092                    .withMetadataPrefix(EML_FORMAT.getMetadataPrefix())
093                    .withSetSpec(OaipmhSetRepository.SetType.DATASET_TYPE.getSubsetPrefix() + DatasetType.OCCURRENCE.toString()));
094    assertTrue("ListRecords verb with set DatasetType return at least 1 record", records.hasNext());
095    assertEquals(orgIcelandInstallation2Dataset1.getKey().toString(), records.next().getHeader().getIdentifier());
096
097    // Non-existing Set
098    records = serviceProvider.listRecords(
099            ListRecordsParameters.request()
100                    .withMetadataPrefix(EML_FORMAT.getMetadataPrefix())
101                    .withSetSpec("non-existing-set"));
102    assertFalse("ListRecords verb with non-existing set should return no record", records.hasNext());
103  }

105  /**
106   *
107   * Test that ListRecords verb return all records when the number of records is higher than 'MaxListRecords'.
108   * When the number of records is higher than 'MaxListRecords', at least 2 requests will be sent and a 'resumptionToken'
109   * will be used.
110   *
111   * @throws Exception
112   */
113  @Test
114  public void testListRecordsPaging() throws Exception {
115    int numberOfDataset = 3;
116    Organization orgIceland = createOrganization(Country.ICELAND);
117    Installation orgIcelandInstallation1 = createInstallation(orgIceland.getKey());
118    createDataset(orgIceland.getKey(), orgIcelandInstallation1.getKey(), DatasetType.CHECKLIST, new Date());
119
120    Installation orgIcelandInstallation2 = createInstallation(orgIceland.getKey());
121    createDataset(orgIceland.getKey(), orgIcelandInstallation2.getKey(), DatasetType.OCCURRENCE, new Date());
122
123    Organization org2 = createOrganization(Country.NEW_ZEALAND);
124    Installation org2Installation1 = createInstallation(org2.getKey());
125    createDataset(org2.getKey(), org2Installation1.getKey(), DatasetType.CHECKLIST, new Date());
126
127    // ensure the test will run under the expected configuration
128    assertTrue("OaipmhMockModule 'MaxListRecords' should be set to a value less than " + numberOfDataset, numberOfDataset > OaipmhTestConfiguration.MAX_LIST_RECORDS);
129
130    Iterator<Record> records = serviceProvider.listRecords(
131            ListRecordsParameters.request()
132                    .withMetadataPrefix(EML_FORMAT.getMetadataPrefix()));
133    List<Record> recordList = Lists.newArrayList(records);
134    assertEquals("ListRecords verb return all records when the number of records is higher than 'MaxListRecords'", numberOfDataset, recordList.size());
135  }

137  @Test
138  public void getListRecordsWithAugmentedMetadata() throws Exception {
139    Organization org1 = createOrganization(Country.ZAMBIA);
140    Installation org1Installation1 = createInstallation(org1.getKey());
141    Dataset org1Installation1Dataset1 = createDataset(org1.getKey(), org1Installation1.getKey(), DatasetType.CHECKLIST, new Date());
142
143    insertMetadata(org1Installation1Dataset1.getKey(), FileUtils.classpathStream("metadata/sample.xml"));
144
145    String key = org1Installation1Dataset1.getKey().toString();
146
147    Record record = serviceProvider.getRecord(
148            GetRecordParameters.request()
149                    .withIdentifier(key)
150                    .withMetadataFormatPrefix(EML_FORMAT.getMetadataPrefix()));
151    assertEquals(key, record.getHeader().getIdentifier());
152
153    // the xoai library doesn't return the <metadata> content as we expect so test the returned document directly.
154    String result = IOUtils.toString(new URI(baseUrl + "?verb=ListRecords&metadataPrefix=" + EML_FORMAT.getMetadataPrefix()).toURL(), "UTF-8");
155
156    // ensure we get the augmented metadata data
157    assertTrue("ListRecords verb returns augmented metadata", result.contains("<citation identifier=\"doi:tims-ident.2136.ex43.33.d\">title 1</citation>"));
158  }
