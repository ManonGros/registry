004package org.gbif.registry.util;
005
006import org.gbif.api.model.common.paging.PagingRequest;
007import org.gbif.api.model.common.paging.PagingResponse;
008import org.gbif.api.model.registry.Dataset;
009import org.gbif.api.service.registry.DatasetService;
010import org.gbif.registry.ws.client.guice.RegistryWsClientModule;
011import org.gbif.ws.client.guice.SingleUserAuthModule;
012
013import java.util.Properties;
014
015import com.google.inject.Guice;
016import com.google.inject.Injector;
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;
019
020
021/**
022 * Tool to loop over all live datasets and issue an unchanged dataset update for all datasets without a DOI
023 * so a new GBIF one will get issued.
024 */
025public class DatasetDOIUpdater {
026  private static final String API = "http://api.gbif.org/v1";
027  private static final String USER = "crawler.gbif.org";
028  private static final String PASSWORD = "";
029
030  private static final Logger LOG = LoggerFactory.getLogger(DatasetDOIUpdater.class);
031
032  public static void main(String[] args) throws Exception {
033    Properties p = new Properties();
034    p.put("registry.ws.url", API);
035    Injector injector = Guice.createInjector(new RegistryWsClientModule(p), new SingleUserAuthModule(USER, PASSWORD));
036
037    LOG.info("Starting test");
038    DatasetService ds = injector.getInstance(DatasetService.class);
039
040    PagingRequest req = new PagingRequest(0,100);
041    PagingResponse<Dataset> resp = null;
042    int cntExist = 0;
043    int cntIssued= 0;
044    int cntFailed= 0;
045    while (resp == null || !resp.isEndOfRecords()) {
046      LOG.info("Loading new dataset page {}", req);
047      resp = ds.list(req);
048      for (Dataset d : resp.getResults()) {
049        if (d.getDoi() == null) {
050          try {
051            ds.update(d);
052            cntIssued++;
053          } catch (Exception e) {
054            LOG.error("Dataset update failed for {}", d.getKey(), e);
055            cntFailed++;
056          }
057        } else {
058          cntExist++;
059        }
060      }
061      LOG.info("Dataset DOI existed={}, issued={}, failed={}", cntExist, cntIssued, cntFailed);
062      req.nextPage();
063    }
064    LOG.info("Dataset DOI Update Finished. DOI existed={}, issued={}, failed={}", cntExist, cntIssued, cntFailed);
065  }
066}
