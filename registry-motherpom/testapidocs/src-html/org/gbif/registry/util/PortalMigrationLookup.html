027/**
028 * A quick hack to create a test file for inspecting dataset keys the migration.
029 * Can be deleted once registry 2 goes live. Expects to run on an input file created with:
030 * mysql> select id, gbif_registry_uuid, name, occurrence_count, data_provider_id from data_resource where
031 * occurrence_count>0 and deleted
032 * is null into outfile '/tmp/live_portal.txt';
033 * Query OK, 14744 rows affected (0.14 sec)
034 */
035public class PortalMigrationLookup {
036
037  private static final Logger LOG = LoggerFactory.getLogger(PortalMigrationLookup.class);
038
039  public static void main(String[] args) throws Exception {
040    Properties p = new Properties();
041    p.put("registry.ws.url", args[0]);
042    Injector injector = Guice.createInjector(new RegistryWsClientModule(p), new AnonymousAuthModule());
043
044    LOG.info("Starting test");
045    DatasetService service = injector.getInstance(DatasetService.class);
046
047    BufferedReader reader = new BufferedReader(new FileReader("/tmp/live_portal.txt"));
048    String line = reader.readLine();
049    Pattern tab = Pattern.compile("\t");
050    int goodCount = 0;
051    int missingCount = 0;
052    int ambigiousCount = 0;
053    int problemOccurrences = 0;
054
055    BufferedWriter writer = new BufferedWriter(new FileWriter("/tmp/problemDatasets.txt"));
056    writer.write("dataResourceId\tdataProviderId\toccurrenceCount\tname\n");
057
058    while (line != null) {
059      String atoms[] = tab.split(line);
060
061      PagingResponse<Dataset> results =
062        service.listByIdentifier(IdentifierType.GBIF_PORTAL, atoms[0], new PagingRequest());
063
064      // sorry to anyone looking at this - hacky as hell...
065      String id = atoms[0];
066      String name = atoms[2];
067      int occurrenceCount = Integer.parseInt(atoms[3]);
068      String provider_id = atoms[4];
069
070      if (results.getCount() == 1) {
071        LOG.info("Good {}: {}", id, results.getResults().get(0).getKey());
072        goodCount++;
073      } else if (results.getCount() > 1) {
074        LOG.info("BAD  {} has {} options", atoms[0], results.getCount());
075        ambigiousCount++;
076        problemOccurrences += occurrenceCount;
077      } else {
078        LOG.info("BAD  {} has no options", atoms[0]);
079        missingCount++;
080        problemOccurrences += occurrenceCount;
081
082        writer.write(id);
083        writer.write("\t");
084        writer.write(provider_id);
085        writer.write("\t");
086        writer.write("" + occurrenceCount);
087        writer.write("\t");
088        writer.write(name);
089        writer.write("\n");
090      }
091
092      line = reader.readLine();
093    }
094    writer.close();
095
096    LOG.info("Good[{}] missing[{}] ambigious[{}] problemOccurrences[{}]", goodCount, missingCount, ambigiousCount,
097      problemOccurrences);
098  }
099}
