1   package org.gbif.registry.metasync.protocols.tapir.model.metadata;
2   
3   import com.google.common.base.Objects;
4   import org.apache.commons.digester3.annotations.rules.ObjectCreate;
5   import org.apache.commons.digester3.annotations.rules.SetProperty;
6   import org.joda.time.DateTime;
7   import org.joda.time.IllegalFieldValueException;
8   import org.joda.time.Period;
9   import org.joda.time.format.ISODateTimeFormat;
10  import org.slf4j.Logger;
11  import org.slf4j.LoggerFactory;
12  
13  @ObjectCreate(pattern = "response/metadata/indexingPreferences")
14  public class IndexingPreferences {
15  
16    private static final Logger LOG = LoggerFactory.getLogger(IndexingPreferences.class);
17  
18    @SetProperty(pattern = "response/metadata/indexingPreferences", attributeName = "startTime")
19    private String startTime;
20    private DateTime parsedStartTime;
21  
22    @SetProperty(pattern = "response/metadata/indexingPreferences", attributeName = "maxDuration")
23    private Period duration;
24    @SetProperty(pattern = "response/metadata/indexingPreferences", attributeName = "frequency")
25    private Period frequency;
26  
27    public DateTime getParsedStartTime() {
28      return parsedStartTime;
29    }
30  
31    public String getStartTime() {
32      return startTime;
33    }
34  
35    public void setStartTime(String startTime) {
36      this.startTime = startTime;
37      try {
38        parsedStartTime = ISODateTimeFormat.timeParser().parseDateTime(startTime);
39      } catch (IllegalFieldValueException ignored) {
40        LOG.debug("Could not parse time: [{}]", startTime);
41      }
42    }
43  
44    public Period getDuration() {
45      return duration;
46    }
47  
48    public void setDuration(Period duration) {
49      this.duration = duration;
50    }
51  
52    public Period getFrequency() {
53      return frequency;
54    }
55  
56    public void setFrequency(Period frequency) {
57      this.frequency = frequency;
58    }
59  
60    @Override
61    public String toString() {
62      return Objects.toStringHelper(this)
63        .add("startTime", parsedStartTime)
64        .add("duration", duration)
65        .add("frequency", frequency)
66        .toString();
67    }
68  }
