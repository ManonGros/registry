247   /**
248    * Returns the parsed, preferred metadata document as a dataset.
249    */
250   private Dataset getPreferredMetadataDataset(UUID key) {
251     List<Metadata> docs = listMetadata(key, null);
252     if (!docs.isEmpty()) {
253       InputStream stream = null;
254       try {
255         // the list is sorted by priority already, just pick the first!
256         stream = getMetadataDocument(docs.get(0).getKey());
257         return DatasetParser.build(stream);
258       } catch (IOException e) {
259         LOG.error("Stored metadata document {} cannot be read", docs.get(0).getKey(), e);
260       } finally {
261         Closeables.closeQuietly(stream);
262       }
263     }
264 
265     return null;
266   }

282   /**
283    * Augments the target dataset with all persistable properties from the supplementary dataset.
284    * Typically the target would be a dataset built from rich XML metadata, and the supplementary would be the persisted
285    * view of the same dataset. NULL values in the supplementary dataset overwrite existing values in the target.
286    * Developers please note:
287    * <ul>
288    * <li>If the target is null, then the supplementary dataset object itself is returned - not a copy</li>
289    * <li>These objects are all mutable, and care should be taken that the returned object may be one or the other of the
290    * supplied, thus you need to {@code Dataset result = merge(Dataset emlView, Dataset dbView);}</li>
291    * </ul>
292    *
293    * @param target that will be modified with persitable values from the supplementary
294    * @param supplementary holding the preferred properties for the target
295    * @return the modified target dataset, or the supplementary dataset if the target is null
296    */
297   @Nullable
298   private Dataset merge(@Nullable Dataset target, @Nullable Dataset supplementary) {
299     // nothing to merge, return the target (which may be null)
300     if (supplementary == null) {
301       return target;
302     }
303 
304     // nothing to overlay into
305     if (target == null) {
306       return supplementary;
307     }
308 
309     // otherwise, copy all persisted values
310     target.setKey(supplementary.getKey());
311     target.setDoi(supplementary.getDoi());
312     target.setParentDatasetKey(supplementary.getParentDatasetKey());
313     target.setDuplicateOfDatasetKey(supplementary.getDuplicateOfDatasetKey());
314     target.setInstallationKey(supplementary.getInstallationKey());
315     target.setPublishingOrganizationKey(supplementary.getPublishingOrganizationKey());
316     target.setExternal(supplementary.isExternal());
317     target.setNumConstituents(supplementary.getNumConstituents());
318     target.setType(supplementary.getType());
319     target.setSubtype(supplementary.getSubtype());
320     target.setTitle(supplementary.getTitle());
321     target.setAlias(supplementary.getAlias());
322     target.setAbbreviation(supplementary.getAbbreviation());
323     target.setDescription(supplementary.getDescription());
324     target.setLanguage(supplementary.getLanguage());
325     target.setHomepage(supplementary.getHomepage());
326     target.setLogoUrl(supplementary.getLogoUrl());
327     target.setCitation(supplementary.getCitation());
328     target.setRights(supplementary.getRights());
329     target.setLockedForAutoUpdate(supplementary.isLockedForAutoUpdate());
330     target.setCreated(supplementary.getCreated());
331     target.setCreatedBy(supplementary.getCreatedBy());
332     target.setModified(supplementary.getModified());
333     target.setModifiedBy(supplementary.getModifiedBy());
334     target.setDeleted(supplementary.getDeleted());
335     // nested properties
336     target.setComments(supplementary.getComments());
337     target.setContacts(supplementary.getContacts());
338     target.setEndpoints(supplementary.getEndpoints());
339     target.setIdentifiers(supplementary.getIdentifiers());
340     target.setMachineTags(supplementary.getMachineTags());
341     target.setTags(supplementary.getTags());
342 
343     return target;
344   }

351     // the fully augmented dataset
352     Dataset dataset = get(datasetKey);
353     if (dataset != null) {
354       // generate new EML
355       try {
356         StringWriter eml = new StringWriter();
357         EMLWriter.write(dataset, eml);
358         return new ByteArrayInputStream(eml.toString().getBytes("UTF-8"));

361         throw new ServiceUnavailableException("Failed to serialize dataset " + datasetKey, e);
362       }
363     }
364     return null;

380   private Metadata insertMetadata(UUID datasetKey, InputStream document, String user) {
381     // check if the dataset actually exists
382     Dataset dataset = super.get(datasetKey);
383     if (dataset == null) {
384       throw new NotFoundException("Dataset " + datasetKey + " not existing");
385     } else if (dataset.getDeleted() != null) {
386       throw new NotFoundException("Dataset " + datasetKey + " has been deleted");
387     }
388 
389     // first keep document as byte array so we can analyze it as much as we want and store it later
390     byte[] data;
391     try {
392       data = ByteStreams.toByteArray(document);
393     } catch (IOException e) {
394       throw new IllegalArgumentException("Unreadable document", e);
395     }
396 
397     // now detect type and create a new metadata record
398     MetadataType type;
399     try (InputStream in = new ByteArrayInputStream(data)){
400       type = DatasetParser.detectParserType(in);
401       //TODO: should we not also validate the EML/DC document ???
402     } catch (IOException e) {
403       throw new IllegalArgumentException("Unreadable document", e);
404     }
405 
406 
407     // first, determine if this document is already stored, returning it with no action
408     // we do this, because updating metadata when nothing has changed, results in registry change events being
409     // propagated which can trigger crawlers which will run an update etc.
410     List<Metadata> existingDocs = listMetadata(datasetKey, type);
411     for (Metadata existing : existingDocs) {
412       try (InputStream in = getMetadataDocument(existing.getKey())){
413         String existingContent = CharStreams.toString(new InputStreamReader(in, Charsets.UTF_8));
414         if (existingContent != null) {
415           if (existingContent.equals(new String(data))) {
416             LOG.debug("This metadata document already exists - returning existing");
417             return existing;
418           }
419         }
420       } catch (Exception e) {
421         // swallow - we'll delete it anyway
422       }
423     }
424 
425     // persist metadata & data, which we know is not already stored
426     // first remove all existing metadata of the same type (so we end up storing only one document per type)
427     Metadata metadata = new Metadata();
428     metadata.setDatasetKey(datasetKey);
429     metadata.setType(type);
430     metadata.setCreatedBy(user);
431     metadata.setModifiedBy(user);
432     for (Metadata existing : existingDocs) {
433       deleteMetadata(existing.getKey());
434     }
435     int metaKey = metadataMapper.create(metadata, data);
436     metadata.setKey(metaKey);
437 
438     // check if we should update our registered base information
439     if (dataset.isLockedForAutoUpdate()) {
440       LOG.info(
441         "Dataset {} locked for automatic updates. Uploaded metadata document not does not modify registered dataset information",
442         datasetKey);
443 
444     } else {
445       // we retrieve the preferred document and only update if this new metadata is the preferred one
446       // e.g. we could put a DC document while an EML document exists that takes preference
447       Dataset updDataset = getPreferredMetadataDataset(datasetKey);
448       // keep some of the original properties
449       updDataset.setKey(dataset.getKey());
450       updDataset.setParentDatasetKey(dataset.getParentDatasetKey());
451       updDataset.setDuplicateOfDatasetKey(dataset.getDuplicateOfDatasetKey());
452       updDataset.setInstallationKey(dataset.getInstallationKey());
453       updDataset.setPublishingOrganizationKey(dataset.getPublishingOrganizationKey());
454       updDataset.setExternal(dataset.isExternal());
455       updDataset.setNumConstituents(dataset.getNumConstituents());
456       updDataset.setType(dataset.getType());
457       updDataset.setSubtype(dataset.getSubtype());
458       updDataset.setLockedForAutoUpdate(dataset.isLockedForAutoUpdate());
459       updDataset.setCreatedBy(dataset.getCreatedBy());
460       updDataset.setCreated(dataset.getCreated());
461       updDataset.setModifiedBy(user);
462       updDataset.setModified(new Date());
463       // persist contacts, overwriting any existing ones
464       replaceContacts(datasetKey, updDataset.getContacts(), user);
465       addIdentifiers(datasetKey, updDataset.getIdentifiers(), user);
466       addTags(datasetKey, updDataset.getTags(), user);
467 
468       // now update the core dataset only, remove associated data to avoid confusion and potential validation problems
469       updDataset.getContacts().clear();
470       updDataset.getIdentifiers().clear();
471       updDataset.getTags().clear();
472       updDataset.getMachineTags().clear();
473       update(updDataset);
474 
475       LOG.info("Dataset {} updated with base information from metadata document {}", datasetKey, metaKey);
476     }
477 
478     return metadata;
479   }

481   private <T extends LenientEquals> boolean containedIn(T id, Collection<T> ids) {
482     for (T id2 : ids) {
483       if (id.lenientEquals(id2)) {
484         return true;
485       }
486     }
487     return false;
488   }

490   /**
491    * Add all not yet existing identifiers to the db!
492    */
493   private void addIdentifiers(UUID datasetKey, List<Identifier> newIdentifiers, String user) {
494     List<Identifier> existing = datasetMapper.listIdentifiers(datasetKey);
495     for (Identifier id : newIdentifiers) {
496       if (IdentifierType.UNKNOWN != id.getType() && !containedIn(id, existing)) {
497         // insert into db
498         id.setCreatedBy(user);
499         id.setCreated(new Date());
500         WithMyBatis.addIdentifier(identifierMapper, datasetMapper, datasetKey, id);
501         // keep it in list for subsequent tests
502         existing.add(id);
503       }
504     }
505   }

507   /**
508    * Add all not yet existing identifiers to the db!
509    */
510   private void addTags(UUID datasetKey, List<Tag> newTags, String user) {
511     List<Tag> existing = datasetMapper.listTags(datasetKey);
512     for (Tag tag : newTags) {
513       if (!containedIn(tag, existing)) {
514         // insert into db
515         tag.setCreatedBy(user);
516         tag.setCreated(new Date());
517         WithMyBatis.addTag(tagMapper, datasetMapper, datasetKey, tag);
518         // keep it in list for subsequent tests
519         existing.add(tag);
520       }
521     }
522   }

524   private void replaceContacts(UUID datasetKey, List<Contact> contacts, String user) {
525     // persist contacts, overwriting any existing ones
526     datasetMapper.deleteContacts(datasetKey);
527     for (Contact c : contacts) {
528       c.setCreatedBy(user);
529       c.setCreated(new Date());
530       c.setModifiedBy(user);
531       c.setModified(new Date());
532       WithMyBatis.addContact(contactMapper, datasetMapper, datasetKey, c);
533     }
534   }

557   private DataCiteMetadata buildMetadata(Dataset d, @Nullable DOI related, @Nullable RelationType relationType) {
558     Organization publisher = organizationMapper.get(d.getPublishingOrganizationKey());
559     DataCiteMetadata m = DataCiteConverter.convert(d, publisher);
560     // add previous relationship
561     if (related != null) {
562       m.getRelatedIdentifiers().getRelatedIdentifier()
563         .add(DataCiteMetadata.RelatedIdentifiers.RelatedIdentifier.builder()
564           .withRelationType(relationType)
565           .withValue(related.getDoiName())
566           .withRelatedIdentifierType(RelatedIdentifierType.DOI)
567           .build()
568       );
569     }
570     return m;
571   }

583   /**
584    * This method does a regular dataset update as defined in the super.update(), but also deals with setting, changing
585    * or removing DOIs from the dataset.doi property and the list of attached identifiers.
586    *
587    * DOI update logic:
588    * <ul>
589    *   <li>If oldDoi exists and the new DOI is the same nothing happens</li>
590    *   <li>If oldDoi exists and the new DOI is different, the new one is used for the dataset and the old one is moved
591    *   to the identifiers table. If the new DOI existed in the identifiers table it will be removed.</li>
592    *   <li>If the dataset has no DOI and no oldDoi exists a new GBIF DOI is issued</li>
593    *   <li>If the dataset has no DOI and the oldDoi is a GBIF DOI, the oldDoi is kept</li>
594    *   <li>If the dataset has no DOI and the oldDoi is not a GBIF DOI, the oldDoi is moved to the identifiers table.
595    *   In case the identifiers table already contains a GBIF DOI this is removed and used for the dataset. If there was
596    *   no GBIF DOI yet a new one is issued</li>
597    * </ul>
598    *
599    * Also see http://dev.gbif.org/issues/browse/POR-2554 for a discussion.
600    *
601    * @param dataset the dataset to be used to update the dataset table in postgres
602    * @param existingIds the complete list of identifiers linked in postgres to the dataset before the update
603    * @param oldDoi the doi as found in postgres for the dataset before this update
604    * @param user the gbif user doing the update
605    */
606   private void update(Dataset dataset, List<Identifier> existingIds, @Nullable final DOI oldDoi, final String user) {
607     // no need to parse EML for the DOI, just get the current mybatis dataset props
608     if (dataset.getDoi() == null) {
609       // a dataset must have a DOI. If it came in with none a GBIF DOI needs to exist
610       if (oldDoi != null && doiGenerator.isGbif(oldDoi)) {
611         dataset.setDoi(oldDoi);
612       } else {
613         // we have a non GBIF DOI before that we need to deprecate
614         reactivatePreviousGbifDoiOrMintNew(existingIds, dataset);
615         // add old DOI to list of alt identifiers
616         if (oldDoi != null) {
617           addAltId(oldDoi, dataset.getKey(), user);
618         }
619       }
620 
621     } else if (oldDoi != null && !dataset.getDoi().equals(oldDoi)) {
622       // the doi has changed. Add old DOI to list of alt identifiers
623       addAltId(oldDoi, dataset.getKey(), user);
624       removeAltIdIfExists(dataset.getKey(), dataset.getDoi(), existingIds);
625     }
626     // update database for core dataset only
627     super.update(dataset);
628 
629     // if the old doi was a GBIF one and the new one is different, update its metadata with a version relationship
630     if (oldDoi != null && doiGenerator.isGbif(oldDoi) && !dataset.getDoi().equals(oldDoi)) {
631       scheduleRegistration(oldDoi, buildMetadata(dataset, dataset.getDoi(), RelationType.IS_PREVIOUS_VERSION_OF),
632         dataset.getKey());
633     }
634     // if the current doi was a GBIF DOI finally schedule a metadata update in datacite
635     if (doiGenerator.isGbif(dataset.getDoi())) {
636       // if DOIs changed establish relationship
637       DataCiteMetadata metadata;
638       // to get the latest timestamps we need to read a new copy of the dataset
639       if (oldDoi == null || dataset.getDoi().equals(oldDoi)) {
640         metadata = buildMetadata(get(dataset.getKey()));
641       } else {
642         metadata = buildMetadata(get(dataset.getKey()), oldDoi, RelationType.IS_NEW_VERSION_OF);
643       }
644       scheduleRegistration(dataset.getDoi(), metadata, dataset.getKey());
645     }
646   }
647 
648   private void scheduleRegistration(DOI doi, DataCiteMetadata metadata, UUID datasetKey) {
649     try {
650       doiGenerator.registerDataset(doi, metadata, datasetKey);
651     } catch (InvalidMetadataException e) {
652       LOG.error(DOI_SMTP, "Failed to schedule DOI update for {}, dataset {}", doi, datasetKey, e);
653       doiGenerator.failed(doi, e);
654     }
655   }

657   /**
658    * Add old DOI to list of alt identifiers in dataset.
659    */
660   private void addAltId(DOI altId, UUID datasetKey, String user) {
661     // update alt ids of dataset
662     Identifier id = new Identifier();
663     id.setType(IdentifierType.DOI);
664     id.setIdentifier(altId.toString());
665     id.setCreatedBy(user);
666     id.setCreated(new Date());
667     LOG.info("DOI changed. Adding previous DOI {} to alternative identifier list for dataset {}", altId, datasetKey);
668     WithMyBatis.addIdentifier(identifierMapper, datasetMapper, datasetKey, id);
669   }
670 
671   /**
672    * Removes a DOI from the alternative identifiers list of a dataset if it exists.
673    */
674   private void removeAltIdIfExists(UUID key, DOI doiToRemove, List<Identifier> existingIds) {
675     for (Identifier id : existingIds) {
676       if (DOI.isParsable(id.getIdentifier())) {
677         DOI doi = new DOI(id.getIdentifier());
678         if (doiToRemove.equals(doi)) {
679           // remove from id list
680           datasetMapper.deleteIdentifier(key, id.getKey());
681         }
682       }
683     }
684   }
685 
686   /**
687    * Scan list of alternate identifiers to find a previous, deleted GBIF DOI and update the dataset instance.
688    * If none can be found use a newly generated one.
689    */
690   private void reactivatePreviousGbifDoiOrMintNew(List<Identifier> existingIds, Dataset d) {
691     for (Identifier id : existingIds) {
692       if (DOI.isParsable(id.getIdentifier())) {
693         DOI doi = new DOI(id.getIdentifier());
694         if (doiGenerator.isGbif(doi)) {
695           // remove from id list and make primary DOI
696           LOG.info("Reactivating old GBIF DOI {} for dataset {}", doi, d.getKey());
697           datasetMapper.deleteIdentifier(d.getKey(), id.getKey());
698           d.setDoi(doi);
699           return;
700         }
701       }
702     }
703     // we never had a GBIF DOI for this dataset, give it a new one
704     DOI doi = doiGenerator.newDatasetDOI();
705     LOG.info("Create new GBIF DOI {} for dataset {}", doi, d.getKey());
706     d.setDoi(doi);
707   }

792   /**
793    * This is a REST only (e.g. not part of the Java API) method that allows the registry console to trigger the
794    * crawling of the dataset. This simply emits a message to rabbitmq requesting the crawl, and applies
795    * necessary security.
796    */
797   @POST
798   @Path("{key}/crawl")
799   @RolesAllowed({ADMIN_ROLE, EDITOR_ROLE})
800   public void crawl(@PathParam("key") UUID datasetKey) {
801     if (messagePublisher != null) {
802       LOG.info("Requesting crawl of dataset[{}]", datasetKey);
803       try {
804         // we'll bump this to the top of the queue since it is a user initiated
805         messagePublisher.send(new StartCrawlMessage(datasetKey, Priority.CRITICAL));
806       } catch (IOException e) {
807         LOG.error("Unable to send message requesting crawl", e);
808       }
809 
810     } else {
811       LOG.warn("Registry is configured to run without messaging capabilities.  Unable to crawl dataset[{}]",
812         datasetKey);
813     }
814   }
