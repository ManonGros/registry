3   import org.gbif.api.model.registry.Dataset;
4   
5   import java.util.ArrayList;
6   import java.util.Date;
7   import java.util.List;
8   
9   import org.dspace.xoai.dataprovider.model.Item;
10  import org.dspace.xoai.dataprovider.model.Set;
11  import org.dspace.xoai.model.oaipmh.About;
12  import org.dspace.xoai.model.oaipmh.Metadata;
13  
14  public class OaipmhItem implements Item {
15  
16    private Dataset dataset;
17    private Metadata metadata;
18    private List<Set> sets;
19  
20    /**
21     * Creates a new OaipmhItem instance with no metadata content.
22     * Mostly to be used as ItemIdentifier.
23     *
24     * @param dataset
25     * @param sets
26     */
27    public OaipmhItem(Dataset dataset, List<Set> sets) {
28      this(dataset, null, sets);
29    }
30  
31    public OaipmhItem(Dataset dataset, String metadata, List<Set> sets) {
32      this.dataset = dataset;
33      this.sets = sets;
34  
35      if(metadata != null){
36        this.metadata = new Metadata(metadata);
37      }
38    }
39  
40    @Override
41    public List<About> getAbout() {
42      return new ArrayList<>();
43    }
44  
45    @Override
46    public Metadata getMetadata() {
47      return metadata;
48    }
49  
50    @Override
51    public String getIdentifier() {
52      return dataset.getKey().toString();
53    }
54  
55    @Override
56    public Date getDatestamp() {
57      return dataset.getModified();
58    }
59  
60    @Override
61    public List<Set> getSets() {
62      return sets;
63    }
64  
65    @Override
66    public boolean isDeleted() {
67      return dataset.getDeleted() != null;
68    }
69  }
