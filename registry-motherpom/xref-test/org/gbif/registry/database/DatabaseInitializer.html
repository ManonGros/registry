13  package org.gbif.registry.database;
14  
15  import java.sql.Connection;
16  import java.sql.SQLException;
17  
18  import javax.sql.DataSource;
19  
20  import com.google.common.base.Throwables;
21  import org.junit.rules.ExternalResource;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  
25  /**
26   * A Rule that will truncate the tables ready for a new test. It is expected to do this before each test by using the
27   * following:
28   * 
29   * <pre>
30   * @Rule
31   * public DatabaseInitializer = new DatabaseInitializer(getDatasource()); // developer required to provide datasource
32   * </pre>
33   */
34  public class DatabaseInitializer extends ExternalResource {
35  
36    private static final Logger LOG = LoggerFactory.getLogger(DatabaseInitializer.class);
37    private final DataSource dataSource;
38  
39    public DatabaseInitializer(DataSource dataSource) {
40      this.dataSource = dataSource;
41    }
42  
43    @Override
44    protected void before() throws Throwable {
45      LOG.info("Truncating registry tables");
46      Connection connection = dataSource.getConnection();
47      try {
48        connection.setAutoCommit(false);
49        /*
50         * connection.createStatement().execute("TRUNCATE TABLE " +
51         * "node, node_identifier, node_machine_tag, node_tag, node_comment, " +
52         * "organization, organization_contact, organization_endpoint, organization_machine_tag, " +
53         * "organization_tag, organization_identifier, organization_comment, " +
54         * "installation, installation_contact, installation_endpoint, installation_machine_tag, " +
55         * "installation_tag, installation_comment, " +
56         * "dataset, dataset_contact, dataset_endpoint, dataset_machine_tag, dataset_tag, " +
57         * "dataset_identifier, dataset_comment, " +
58         * "network, network_contact, network_endpoint, network_machine_tag, network_tag, network_comment, " +
59         * "contact, endpoint, machine_tag, tag, identifier, comment, METADATA CASCADE");
60         */
61  
62        // Profiling on laptop shows this to be far quicker, but above left it for easier reverting
63        connection.prepareStatement("DELETE FROM contact").execute();
64        connection.prepareStatement("DELETE FROM endpoint").execute();
65        connection.prepareStatement("DELETE FROM tag").execute();
66        connection.prepareStatement("DELETE FROM identifier").execute();
67        connection.prepareStatement("DELETE FROM comment").execute();
68        connection.prepareStatement("DELETE FROM node_identifier").execute();
69        connection.prepareStatement("DELETE FROM node_machine_tag").execute();
70        connection.prepareStatement("DELETE FROM node_tag").execute();
71        connection.prepareStatement("DELETE FROM node_comment").execute();
72        connection.prepareStatement("DELETE FROM organization_contact").execute();
73        connection.prepareStatement("DELETE FROM organization_endpoint").execute();
74        connection.prepareStatement("DELETE FROM organization_machine_tag").execute();
75        connection.prepareStatement("DELETE FROM organization_tag").execute();
76        connection.prepareStatement("DELETE FROM organization_identifier").execute();
77        connection.prepareStatement("DELETE FROM organization_comment").execute();
78        connection.prepareStatement("DELETE FROM installation_contact").execute();
79        connection.prepareStatement("DELETE FROM installation_endpoint").execute();
80        connection.prepareStatement("DELETE FROM installation_machine_tag").execute();
81        connection.prepareStatement("DELETE FROM installation_tag").execute();
82        connection.prepareStatement("DELETE FROM installation_comment").execute();
83        connection.prepareStatement("DELETE FROM dataset_contact").execute();
84        connection.prepareStatement("DELETE FROM dataset_endpoint").execute();
85        connection.prepareStatement("DELETE FROM dataset_machine_tag").execute();
86        connection.prepareStatement("DELETE FROM dataset_tag").execute();
87        connection.prepareStatement("DELETE FROM dataset_identifier").execute();
88        connection.prepareStatement("DELETE FROM dataset_comment").execute();
89        connection.prepareStatement("DELETE FROM network_contact").execute();
90        connection.prepareStatement("DELETE FROM network_endpoint").execute();
91        connection.prepareStatement("DELETE FROM network_machine_tag").execute();
92        connection.prepareStatement("DELETE FROM network_tag").execute();
93        connection.prepareStatement("DELETE FROM network_comment").execute();
94        connection.prepareStatement("DELETE FROM machine_tag").execute();
95        connection.prepareStatement("DELETE FROM metadata").execute();
96        connection.prepareStatement("DELETE FROM network").execute();
97        connection.prepareStatement("DELETE FROM dataset").execute();
98        connection.prepareStatement("DELETE FROM installation").execute();
99        connection.prepareStatement("DELETE FROM organization").execute();
100       connection.prepareStatement("DELETE FROM node").execute();
101       connection.prepareStatement("TRUNCATE gbif_doi").execute();
102       connection.commit();
103 
104     } catch (SQLException e) {
105       Throwables.propagate(e);
106     } finally {
107       if (connection != null) {
108         connection.close();
109       }
110     }
111     LOG.info("Registry tables truncated");
112   }
113 }
