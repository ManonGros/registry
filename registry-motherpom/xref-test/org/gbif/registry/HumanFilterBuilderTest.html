11  import static org.junit.Assert.assertTrue;

13  public class HumanFilterBuilderTest {
14    // make sure this is the right bundle as used by the human query builder
15    private static final ResourceBundle resourceBundle = ResourceBundle.getBundle("org/gbif/occurrence/query/filter");
16  
17    /**
18     * Test all available search parameters and make sure we have a bundle entry for all possible enum values.
19     * Test copied from download-query-tools, but added here again as we might have a different GBIF API version in use.
20     */
21    @Test
22    public void testEnumResourceBundle() throws Exception {
23      for (OccurrenceSearchParameter p : OccurrenceSearchParameter.values()) {
24        if (p.type().isEnum()) {
25          if (p.type() != Country.class && p.type() != Continent.class) {
26            Class<Enum<?>> vocab = (Class<Enum<?>>) p.type();
27            // make sure we have en entry for all possible enum values
28            for (Enum<?> e : vocab.getEnumConstants()) {
29              assertTrue("Missing enum resource bundle entry for " + vocab.getSimpleName() + "." + e.name(),
30                resourceBundle.containsKey("enum." + vocab.getSimpleName().toLowerCase() + "." + e.name()));
31            }
32          }
33        }
34      }
35    }

37  }
