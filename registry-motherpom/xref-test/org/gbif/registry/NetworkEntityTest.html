189   /**
190    * Creates 5 entities, and then pages over them using differing paging strategies, confirming the correct number of
191    * records are returned for each strategy.
192    */
193   @Test
194   public void testPaging() {
195     for (int i = 1; i <= 5; i++) {
196       create(newEntity(), i);
197     }
198 
199     // the expected number of records returned when paging at different page sizes
200     int[][] expectedPages = new int[][] { {1, 1, 1, 1, 1, 0}, // page size of 1
201       {2, 2, 1, 0}, // page size of 2
202       {3, 2, 0}, // page size of 3
203       {4, 1, 0}, // page size of 4
204       {5, 0}, // page size of 5
205       {5, 0}, // page size of 6
206     };
207 
208     // test the various paging strategies (e.g. page size of 1,2,3 etc to verify they behave as outlined above)
209     for (int pageSize = 1; pageSize <= expectedPages.length; pageSize++) {
210       int offset = 0; // always start at beginning
211       for (int page = 0; page < expectedPages[pageSize - 1].length; page++, offset += pageSize) {
212         // request the page using the page size and offset
213         List<T> results = service.list(new PagingRequest(offset, expectedPages[pageSize - 1][page])).getResults();
214         // confirm it is the correct number of results as outlined above
215         assertEquals("Paging is not operating as expected when requesting pages of size " + pageSize,
216           expectedPages[pageSize - 1][page],
217           results.size());
218       }
219     }
220   }

222   /**
223    * Confirm that the list method and its paging return entities in creation time order.
224    */
225   @Test
226   public void testPagingOrder() {
227     // keeps a list of all uuids created in that creation order
228     List<UUID> uuids = Lists.newArrayList();
229     for (int i = 1; i <= 5; i++) {
230       T d = create(newEntity(), i);
231       uuids.add(d.getKey());
232     }
233     uuids = Lists.reverse(uuids);
234 
235     // test the various paging strategies (e.g. page size of 1,2,3 etc to verify they behave as outlined above)
236     for (int pageSize = 1; pageSize <= uuids.size(); pageSize++) {
237       for (int offset = 0; offset < uuids.size() + 1; offset++) {
238         // request a page using the page size and offset
239         PagingResponse<T> resp = service.list(new PagingRequest(offset, pageSize));
240         // confirm it is the correct number of results as outlined above
241         assertEquals("Paging is not operating as expected when requesting pages of size " + pageSize,
242           Math.min(pageSize, uuids.size() - offset),
243           resp.getResults().size());
244         assertEquals("Count wrong", Long.valueOf(uuids.size()), resp.getCount());
245         int lastIdx = -1;
246         for (T d : resp.getResults()) {
247           int idx = uuids.indexOf(d.getKey());
248           // make sure the datasets are in the same order as the reversed uuid list
249           assertTrue(idx > lastIdx);
250           lastIdx = idx;
251         }
252       }
253     }
254   }

274     assertEquals("Search should return no hits", Long.valueOf(0), service.search("New", null).getCount());
275     assertEquals("Search should return no hits", Long.valueOf(0), service.search("TITILE", null).getCount());

437   // Repeatable entity creation with verification tests
438   protected T create(T orig, int expectedCount) {
439     try {
440       @SuppressWarnings("unchecked")
441       T entity = (T) BeanUtils.cloneBean(orig);
442       Preconditions.checkNotNull(entity, "Cannot create a non existing entity");
443       UUID key = service.create(entity);
444       entity.setKey(key);
445       T written = service.get(key);
446       assertNotNull(written.getCreated());
447       assertNotNull(written.getModified());
448       assertNull(written.getDeleted());
449       assertLenientEquals("Persisted does not reflect original", entity, written);
450       assertEquals("List service does not reflect the number of created entities",
451         expectedCount,
452         service.list(new PagingRequest()).getResults().size());
453       return written;
454     } catch (Exception e) {
455       throw Throwables.propagate(e);
456     }
457   }

466     assertEquals("Unexpected result size for current test state", size, results.getResults().size());
