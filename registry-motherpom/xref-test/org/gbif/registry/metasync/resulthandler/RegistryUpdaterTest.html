97    /**
98     * Synchronize a TAPIR installation, updating a single existing Dataset. The HTTP responses, and the Registry web
99     * service client are mocked.
100    * </br>
101    * The test verifies that the Dataset update was successful, counting the number of web service method invocations.
102    * For example, if datasetService.addContact was called twice, and 2 contacts were expected to be persisted on
103    * update, we can assume the service does its job correctly and just assert that the addContact() was called twice.
104    */
105   @Test
106   public void testSaveUpdatedDatasets() throws Exception {
107     // generate SyncResult, including a single updated Dataset
108     SyncResult syncResult = synchroniser.syncInstallation(installation, Lists.newArrayList(dataset));
109 
110     // update Dataset in Registry
111     updater.saveUpdatedDatasets(syncResult);
112 
113     // update dataset 1 time
114     verify(updater.getDatasetService(), times(1)).update(any(Dataset.class));
115 
116     // delete 1 existing machine tag, add 2 new
117     verify(updater.getDatasetService(), times(1)).deleteMachineTag(any(UUID.class), anyInt());
118     verify(updater.getDatasetService(), times(2)).addMachineTag(any(UUID.class), any(MachineTag.class));
119 
120     // delete 1 existing contact, add 2 new
121     verify(updater.getDatasetService(), times(1)).deleteContact(any(UUID.class), anyInt());
122     verify(updater.getDatasetService(), times(2)).addContact(any(UUID.class), any(Contact.class));
123 
124     // delete 1 endpoint, add 1 endpoint
125     verify(updater.getDatasetService(), times(1)).deleteEndpoint(any(UUID.class), anyInt());
126     verify(updater.getDatasetService(), times(1)).addEndpoint(any(UUID.class), any(Endpoint.class));
127 
128     // delete 0 identifiers, add 0 identifiers
129     verify(updater.getDatasetService(), times(0)).deleteIdentifier(any(UUID.class), anyInt());
130     verify(updater.getDatasetService(), times(0)).addIdentifier(any(UUID.class), any(Identifier.class));
131 
132     // delete 0 tags, add 0 tags
133     verify(updater.getDatasetService(), times(0)).deleteTag(any(UUID.class), anyInt());
134     verify(updater.getDatasetService(), times(0)).addTag(any(UUID.class), any(Tag.class));
135   }

137   /**
138    * Synchronize a TAPIR installation, whose single existing Dataset has been migrated to DwC-A already. An endpoint of
139    * type DwC-A is evidence of migration. Since the Dataset has been migrated, its metadata won't be updated from
140    * TAPIR anymore.
141    * </br>
142    * The test verifies that the Dataset update was skipped, counting the number of web service method invocations.
143    */
144   @Test
145   public void testSaveUpdatedDatasetsSkipped() throws Exception {
146     Endpoint dwca = new Endpoint();
147     dwca.setKey(100);
148     dwca.setType(EndpointType.DWC_ARCHIVE);
149     dwca.setUrl(URI.create("http://ipt.gbif.org/archive.do?r=test"));
150     dataset.getEndpoints().add(dwca);
151 
152     // generate SyncResult, including a single updated Dataset
153     SyncResult syncResult = synchroniser.syncInstallation(installation, Lists.newArrayList(dataset));
154 
155     // update Dataset in Registry
156     updater.saveUpdatedDatasets(syncResult);
157 
158     // update dataset was skipped - service was called 0 times!
159     verify(updater.getDatasetService(), times(0)).update(any(Dataset.class));
160   }

162   /**
163    * Prepare TAPIR Dataset for use in tests, having 1 contact, 1 identifier, 2 machine tags, 1 tag, and 1 endpoint.
164    */
165   private Dataset prepareDataset() {
166     // create TAPIR dataset with single endpoint
167     Dataset dataset = new Dataset();
168     dataset.setKey(UUID.randomUUID());
169     dataset.setTitle("Beavers");
170     dataset.addEndpoint(endpoint);
171 
172     // add single Contact
173     Contact contact = new Contact();
174     contact.setKey(1);
175     contact.addEmail("test@gbif.org");
176     dataset.setContacts(Lists.newArrayList(contact));
177 
178     // add single Identifier
179     Identifier identifier = new Identifier();
180     identifier.setKey(1);
181     identifier.setType(IdentifierType.GBIF_PORTAL);
182     identifier.setIdentifier("450");
183     dataset.setIdentifiers(Lists.newArrayList(identifier));
184 
185     // add 2 MachineTags 1 with metasync.gbif.org namespace, and another not having it
186     List<MachineTag> machineTags = Lists.newArrayList();
187 
188     MachineTag machineTag = new MachineTag();
189     machineTag.setKey(1);
190     machineTag.setNamespace(Constants.METADATA_NAMESPACE);
191     machineTag.setName(Constants.DECLARED_COUNT);
192     machineTag.setValue("1000");
193     machineTags.add(machineTag);
194 
195     MachineTag machineTag2 = new MachineTag();
196     machineTag2.setKey(2);
197     machineTag2.setNamespace("public");
198     machineTag2.setName("IsoCountryCode");
199     machineTag2.setValue("DK");
200     machineTags.add(machineTag2);
201 
202     dataset.setMachineTags(machineTags);
203 
204     // add 1 Tag
205     Tag tag = new Tag();
206     tag.setKey(1);
207     tag.setValue("Invasive");
208     dataset.setTags(Lists.newArrayList(tag));
209 
210     return dataset;
211   }

213   private static HttpResponse prepareResponse(int responseStatus, String fileName) throws IOException {
214     HttpResponse response =
215       new BasicHttpResponse(new BasicStatusLine(new ProtocolVersion("HTTP", 1, 1), responseStatus, ""));
216     response.setStatusCode(responseStatus);
217     byte[] bytes = Resources.toByteArray(Resources.getResource(fileName));
218     response.setEntity(new ByteArrayEntity(bytes));
219     return response;
220   }
