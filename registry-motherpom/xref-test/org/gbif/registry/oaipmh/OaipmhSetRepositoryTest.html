28    /**
29     * Prepare the MockDatasetMapper with controlled mock data.
30     *
31     * @return
32     */
33    private static DatasetMapper prepareDatasetMapperMock(){
34      MockDatasetMapper datasetMapper = new MockDatasetMapper();
35      datasetMapper.mockDatasetForCountry(Country.DENMARK);
36      datasetMapper.mockDatasetForCountry(Country.SWEDEN);
37      datasetMapper.mockDatasetForCountry(Country.NORWAY);
38  
39      Installation installation = new Installation();
40      installation.setKey(installationUUIDRepitle);
41      installation.setTitle("My reptile IPT");
42      datasetMapper.mockDatasetForInstallation(installation);
43  
44      installation = new Installation();
45      installation.setKey(installationUUIDGulo);
46      installation.setTitle("My gulo observations IPT");
47      datasetMapper.mockDatasetForInstallation(installation);
48  
49      return datasetMapper;
50    }

52    @Test
53    public void testParseSetName(){
54      assertTrue(OaipmhSetRepository.parseSetName("country:DK").isPresent());
55      assertFalse(OaipmhSetRepository.parseSetName("null:Country").isPresent());
56    }
57  
58    @Test
59    public void testRetrieveSets(){
60      DatasetMapper mockDatasetMapper = prepareDatasetMapperMock();
61      OaipmhSetRepository setRepository = new OaipmhSetRepository(mockDatasetMapper);
62  
63      // the +1 to each list size covers the top level set
64      int total = (DatasetType.values().length + 1) + (mockDatasetMapper.listDistinctCountries(null).size() + 1) + (mockDatasetMapper.listDistinctInstallations(null).size() + 1);
65  
66      //test retrieving all Sets one by one
67      for(int i=0; i < total; i++){
68        ListSetsResult result = setRepository.retrieveSets(i, 1);
69        assertEquals(1, result.getResults().size());
70        assertEquals(i+1 < total, result.hasMore());
71      }
72    }
73  
74    @Test
75    public void testSetExists(){
76      DatasetMapper mockDatasetMapper = prepareDatasetMapperMock();
77      OaipmhSetRepository setRepository = new OaipmhSetRepository(mockDatasetMapper);
78      assertTrue("Should find Set provided in test data:" + installationUUIDRepitle.toString(),
79              setRepository.exists(OaipmhSetRepository.SetType.INSTALLATION.getSubsetPrefix() + installationUUIDRepitle.toString()));
80      assertFalse("Should not find a Set for random Installation key",
81              setRepository.exists(OaipmhSetRepository.SetType.INSTALLATION.getSubsetPrefix() + UUID.randomUUID().toString()));
82  
83      assertTrue("Should find Set provided in test data:" + Country.DENMARK.getIso2LetterCode(),
84              setRepository.exists(OaipmhSetRepository.SetType.COUNTRY.getSubsetPrefix() + Country.DENMARK.getIso2LetterCode()));
85      assertFalse("Should not find Set not provided in test data:" + Country.AUSTRALIA.getIso2LetterCode(),
86              setRepository.exists(OaipmhSetRepository.SetType.COUNTRY.getSubsetPrefix() + Country.AUSTRALIA.getIso2LetterCode()));
87    }
