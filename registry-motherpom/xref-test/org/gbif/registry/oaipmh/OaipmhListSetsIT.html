25  /**
26   * Test the ListSets verb of the OAI-PMH endpoint.
27   *
28   * @author cgendreau
29   */
30  public class OaipmhListSetsIT  extends AbstractOaipmhEndpointIT {
31  
32    // OaipmhSetRepository.SetType represents the root of the set hierarchy to the respective node,
33    // DatasetType.values() represents the static subsets of OaipmhSetRepository.SetType.DATASET_TYPE
34    private static final int NUMBER_OF_STATIC_SETS =  OaipmhSetRepository.SetType.values().length + (DatasetType.values().length);
35  
36    private static final Function<Set, String> SET_TO_SETSPEC =
37      new Function<Set, String>() {
38        public String apply(Set set) {
39          return set.getSpec();
40        }
41      };
42  
43    public OaipmhListSetsIT(NodeService nodeService, OrganizationService organizationService, InstallationService installationService, DatasetService datasetService) {
44      super(nodeService, organizationService, installationService, datasetService);
45    }
46  
47    @Test
48    public void testSetsWithEmptyDatabase() throws NoSetHierarchyException {
49  
50      Iterator<Set> sets = serviceProvider.listSets();
51      List<Set> setsList = Lists.newArrayList(sets);
52  
53      assertEquals("ListSets verb returns only static Sets when database is empty", NUMBER_OF_STATIC_SETS, setsList.size());
54    }
55  
56    @Test
57    public void testSetsWithDatabaseContent() throws Exception {
58      Organization orgIceland = createOrganization(Country.ICELAND);
59      Installation orgIcelandInstallation1 = createInstallation(orgIceland.getKey());
60  
61      Iterator<Set> sets = serviceProvider.listSets();
62      List<Set> setsList = Lists.newArrayList(sets);
63      assertEquals("ListSets verb returns only static Sets when there is not datasets", NUMBER_OF_STATIC_SETS, setsList.size());
64  
65      createDataset(orgIceland.getKey(), orgIcelandInstallation1.getKey(), DatasetType.OCCURRENCE, new Date());
66  
67      // refresh data
68      sets = serviceProvider.listSets();
69      setsList = Lists.newArrayList(sets);
70  
71      List<String> setsSpecList = Lists.transform(setsList, SET_TO_SETSPEC);
72      assertTrue("ListSets verb returns a Set corresponding to the country of the datatset publishing Organization",
73              setsSpecList.contains(OaipmhSetRepository.SetType.COUNTRY.getSubsetPrefix() + Country.ICELAND.getIso2LetterCode()));
74  
75      assertTrue("ListSets verb returns a Set corresponding to the Installation from which the datatset was published",
76              setsSpecList.contains(OaipmhSetRepository.SetType.INSTALLATION.getSubsetPrefix() + orgIcelandInstallation1.getKey().toString()));
77    }
