14  public class TimeSeriesExtractorTest {

16    @Test
17    public void testExtractDecades() throws Exception {
18  
19      TimeSeriesExtractor ex = new TimeSeriesExtractor(1000, 2500, 1800, 2050);
20  
21      List<TemporalCoverage> coverages = Lists.newArrayList();
22      coverages.add(range(1977, 1995));
23      coverages.add(range(1914, 1917));
24      coverages.add(range(987, 999));
25      coverages.add(range(2222, 3333));
26  
27      List<Integer> periods = ex.extractDecades(coverages);
28      List<Integer> expected = Lists.newArrayList(1000, 1910, 1970, 1980, 1990, 2200, 2300, 2400, 2500);
29      assertEquals(expected, periods);
30    }
31  
32    @Test
33    public void testSwappedDecades() throws Exception {
34  
35      TimeSeriesExtractor ex = new TimeSeriesExtractor(1000, 2500, 1800, 2050);
36  
37      List<TemporalCoverage> coverages = Lists.newArrayList();
38      coverages.add(range(1995, 1977));
39      coverages.add(range(1914, 1917));
40      coverages.add(range(987, 999));
41      coverages.add(range(3333, 2222));
42  
43      List<Integer> periods = ex.extractDecades(coverages);
44      List<Integer> expected = Lists.newArrayList(1000, 1910);
45      assertEquals(expected, periods);
46    }
47  
48    private DateRange range(int year1, int year2) {
49      DateRange dr = new DateRange();
50  
51      Calendar cal = Calendar.getInstance();
52      cal.set(Calendar.YEAR, year1);
53      dr.setStart(cal.getTime());
54  
55      cal.set(Calendar.YEAR, year2);
56      dr.setEnd(cal.getTime());
57  
58      return dr;
59    }
