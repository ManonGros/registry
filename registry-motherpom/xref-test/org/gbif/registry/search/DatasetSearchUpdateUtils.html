3   import java.util.concurrent.TimeUnit;
4   
5   import com.google.common.base.Preconditions;
6   import com.google.common.base.Stopwatch;
7   import com.google.common.base.Throwables;
8   import org.slf4j.Logger;
9   import org.slf4j.LoggerFactory;
10  
11  
12  public class DatasetSearchUpdateUtils {
13  
14    // how often to poll and wait for SOLR to update
15    private static final int SOLR_UPDATE_TIMEOUT_SECS = 10;
16    private static final int SOLR_UPDATE_POLL_MSECS = 10;
17    private static final Logger LOG = LoggerFactory.getLogger(DatasetSearchUpdateUtils.class);
18  
19    /**
20     * Waits for SOLR update threads to finish.
21     */
22    public static void awaitUpdates(DatasetIndexUpdateListener datasetIndexUpdater) {
23      Preconditions.checkNotNull(datasetIndexUpdater, "Index updater is required");
24      try {
25        Stopwatch stopWatch = Stopwatch.createStarted();
26        while (datasetIndexUpdater.queuedUpdates() > 0) {
27          Thread.sleep(SOLR_UPDATE_POLL_MSECS);
28          if (stopWatch.elapsed(TimeUnit.SECONDS) > SOLR_UPDATE_TIMEOUT_SECS) {
29            throw new IllegalStateException("Failing test due to unreasonable timeout on SOLR update");
30          }
31        }
32        LOG.info("Waited {} msecs for SOLR update backlog to clear successfully",
33          stopWatch.elapsed(TimeUnit.MILLISECONDS));
34  
35      } catch (InterruptedException e) {
36        throw Throwables.propagate(e);
37      }
38    }
39  }
