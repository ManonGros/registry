11  /**
12   * Simple utility class to build RepositoryConfiguration for OAI-PMH testing.
13   *
14   * @author cgendreau
15   */
16  public class OaipmhTestConfiguration {
17  
18    public static final int MAX_LIST_RECORDS = 2;
19  
20    public static RepositoryConfiguration buildTestRepositoryConfiguration(Properties properties) {
21  
22      return new RepositoryConfiguration()
23              .withRepositoryName("GBIF Test Registry")
24              .withAdminEmail(properties.getProperty(OaipmhModule.OAIPMH_ADMIN_EMAIL_PROPERTY))
25              .withBaseUrl(properties.getProperty(OaipmhModule.OAIPMH_BASE_URL_PROPERTY))
26              .withEarliestDate(OaipmhModule.EARLIEST_DATE)
27              .withMaxListIdentifiers(2)
28              .withMaxListSets(2)
29              .withMaxListRecords(MAX_LIST_RECORDS)
30              .withGranularity(Granularity.Second)
31              .withDeleteMethod(DeletedRecord.PERSISTENT)
32              .withDescription("<TestDescription xmlns=\"\">Test description</TestDescription>");
33    }
34  }
