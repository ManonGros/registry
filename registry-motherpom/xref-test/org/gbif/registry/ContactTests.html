44  public class ContactTests {
45  
46    public static <T extends NetworkEntity> void testAddDeleteUpdate(ContactService service, T entity) {
47  
48      // check there are none on a newly created entity
49      List<Contact> contacts = service.listContacts(entity.getKey());
50      assertNotNull("Contact list should be empty, not null when no contacts exist", contacts);
51      assertTrue("Contact should be empty when none added", contacts.isEmpty());
52  
53      // test additions, both being primary
54      service.addContact(entity.getKey(), Contacts.newInstance());
55      service.addContact(entity.getKey(), Contacts.newInstance());
56      contacts = service.listContacts(entity.getKey());
57      assertNotNull(contacts);
58      assertEquals("2 contacts have been added", 2, contacts.size());
59      assertFalse("Older contact should not be primary anymore", contacts.get(1).isPrimary());
60      assertTrue("Newer contact should be primary", contacts.get(0).isPrimary());
61  
62      // test deletion, ensuring correct one is deleted
63      service.deleteContact(entity.getKey(), contacts.get(1).getKey());
64      contacts = service.listContacts(entity.getKey());
65      assertNotNull(contacts);
66      assertEquals("1 contact should remain after the deletion", 1, contacts.size());
67      Contact expected = Contacts.newInstance();
68      Contact created = contacts.get(0);
69      assertLenientEquals("Created contact does not read as expected", expected, created);
70  
71      // try and update a contact
72      contacts = service.listContacts(entity.getKey());
73      contacts.get(0).setFirstName("Timmay");
74      service.updateContact(entity.getKey(), contacts.get(0));
75      contacts = service.listContacts(entity.getKey());
76      assertNotNull(contacts);
77      assertEquals("The update does not reflect the change", "Timmay", contacts.get(0).getFirstName());
78  
79      try {
80        service.updateContact(UUID.randomUUID(), contacts.get(0));
81        fail("Contact update supplied an illegal entity key but was not caught");
82      } catch (IllegalArgumentException e) {
83        // expected
84      }
85    }
86  
87    /**
88     * Tests that adding a contact means the entity is found in the search.
89     */
90    public static <T extends NetworkEntity> void testSimpleSearch(ContactService service,
91      NetworkEntityService<T> networkService, T entity) {
92      assertEquals("There should be no results for this search", Long.valueOf(0), networkService.search("Frankie", null)
93        .getCount());
94      Contact c = Contacts.newInstance();
95      c.setLastName("Frankie");
96      service.addContact(entity.getKey(), c);
97      assertEquals("There should a search result for Frankie", Long.valueOf(1), networkService.search("Frankie", null)
98        .getCount());
99  
100   }
