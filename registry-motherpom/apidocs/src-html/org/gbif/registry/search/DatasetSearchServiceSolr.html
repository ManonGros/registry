001package org.gbif.registry.search;
002
003import org.gbif.api.model.registry.search.DatasetSearchParameter;
004import org.gbif.api.model.registry.search.DatasetSearchRequest;
005import org.gbif.api.model.registry.search.DatasetSearchResult;
006import org.gbif.api.model.registry.search.DatasetSuggestRequest;
007import org.gbif.api.model.registry.search.DatasetSuggestResult;
008import org.gbif.api.service.registry.DatasetSearchService;
009import org.gbif.common.search.service.SolrSearchSuggestService;
010
011import java.util.Map;
012
013import com.google.common.collect.ImmutableSortedMap;
014import com.google.inject.Inject;
015import com.google.inject.name.Named;
016import org.apache.solr.client.solrj.SolrClient;
017import org.apache.solr.client.solrj.SolrQuery;
018
019/**
020 * Dataset search implementation using the provided SOLR instance.
021 */
022public class DatasetSearchServiceSolr
023  extends
024  SolrSearchSuggestService<DatasetSearchResult, DatasetSearchParameter, SolrAnnotatedDataset, DatasetSearchRequest, DatasetSuggestResult, SolrAnnotatedSuggestDataset, DatasetSuggestRequest>
025  implements DatasetSearchService {
026
027  // Order by best score, then alphabetically by title
028  private static final Map<String, SolrQuery.ORDER> PRIMARY_SORT_ORDER = ImmutableSortedMap.of(
029    "score", SolrQuery.ORDER.desc,
030    "dataset_title", SolrQuery.ORDER.asc);
031
032  @Inject
033  public DatasetSearchServiceSolr(@Named("Dataset") SolrClient solrClient) {
034    super(solrClient, DatasetSearchResult.class, SolrAnnotatedDataset.class, DatasetSearchParameter.class,
035      PRIMARY_SORT_ORDER, SolrAnnotatedSuggestDataset.class);
036  }
037
038}
