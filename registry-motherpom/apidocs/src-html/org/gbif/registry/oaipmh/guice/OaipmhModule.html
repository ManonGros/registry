001package org.gbif.registry.oaipmh.guice;
002
003import org.gbif.registry.oaipmh.OaipmhItemRepository;
004import org.gbif.registry.oaipmh.OaipmhSetRepository;
005
006import java.util.Calendar;
007import java.util.Date;
008import java.util.Properties;
009import java.util.TimeZone;
010
011import com.google.inject.AbstractModule;
012import com.google.inject.Scopes;
013import org.dspace.xoai.dataprovider.repository.ItemRepository;
014import org.dspace.xoai.dataprovider.repository.RepositoryConfiguration;
015import org.dspace.xoai.dataprovider.repository.SetRepository;
016import org.dspace.xoai.model.oaipmh.DeletedRecord;
017import org.dspace.xoai.model.oaipmh.Granularity;
018
019/**
020 * A Guice module that binds implementations of XOAI related classes.
021 *
022 * DatasetResource, DatasetMapper, OrganizationMapper, CubeService must be bind externally.
023 *
024 */
025public class OaipmhModule extends AbstractModule {
026
027  // earliest date a dataset was created
028  public static final Date EARLIEST_DATE;
029  static {
030    Calendar cal = Calendar.getInstance();
031    cal.setTimeZone(TimeZone.getTimeZone("UTC"));
032    cal.set(2007, 0, 1, 0, 0, 1);
033    EARLIEST_DATE = cal.getTime();
034  }
035
036  public static final String OAIPMH_BASE_URL_PROPERTY = "oaipmh.baseUrl";
037  public static final String OAIPMH_ADMIN_EMAIL_PROPERTY = "oaipmh.adminEmail";
038
039  private static final String REPO_NAME = "GBIF Registry";
040
041  private final RepositoryConfiguration repositoryConfiguration;
042
043  /**
044   * @param properties OAI-PMH root url
045   */
046  public OaipmhModule(Properties properties) {
047    this(new RepositoryConfiguration()
048      .withRepositoryName(REPO_NAME)
049      .withAdminEmail(properties.getProperty(OAIPMH_ADMIN_EMAIL_PROPERTY))
050      .withBaseUrl(properties.getProperty(OAIPMH_BASE_URL_PROPERTY))
051      .withEarliestDate(EARLIEST_DATE)
052      .withMaxListIdentifiers(1000)
053      .withMaxListSets(1000)
054      .withMaxListRecords(100)
055      .withGranularity(Granularity.Second)
056      .withDeleteMethod(DeletedRecord.PERSISTENT)
057      .withDescription(
058              "<oai_dc:dc xmlns:oai_dc=\"http://www.openarchives.org/OAI/2.0/oai_dc/\" xmlns:dc=\"http://purl.org/dc/elements/1.1/\" xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" xsi:schemaLocation=\"http://www.openarchives.org/OAI/2.0/oai_dc/ http://www.openarchives.org/OAI/2.0/oai_dc.xsd\">\n" +
059                      "\t<dc:title>GBIF Registry</dc:title>\n" +
060                      "\t<dc:creator>Global Biodiversity Information Facility Secretariat</dc:creator>\n" +
061                      "\t<dc:description>\n" +
062                      "\t\tThe GBIF Registry â€” the entities that make up the GBIF network.\n" +
063                      "\t\tThis OAI-PMH service exposes Datasets, organized into sets of country, installation and resource type.\n" +
064                      "\t\tFor more information, see http://www.gbif.org/developer/registry\n" +
065                      "\t</dc:description>\n" +
066                      "</oai_dc:dc>\n"
067      ));
068  }
069
070  /**
071   *
072   * @param repositoryConfiguration an already configured RepositoryConfiguration object
073   */
074  public OaipmhModule(RepositoryConfiguration repositoryConfiguration) {
075    this.repositoryConfiguration = repositoryConfiguration;
076  }
077
078  @Override
079  protected void configure() {
080    bind(RepositoryConfiguration.class).toInstance(repositoryConfiguration);
081    bind(ItemRepository.class).to(OaipmhItemRepository.class).in(Scopes.SINGLETON);
082    bind(SetRepository.class).to(OaipmhSetRepository.class).in(Scopes.SINGLETON);
083  }
084
085}
