001package org.gbif.registry.metasync.resulthandler;
002
003import org.gbif.registry.metasync.api.SyncResult;
004
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008/**
009 * A simple utility class that prints some information about each synchronisation result. Does not generate aggregate
010 * statistics at the moment.
011 */
012public final class DebugHandler {
013
014  private static final Logger LOG = LoggerFactory.getLogger(DebugHandler.class);
015
016  public static void processResults(Iterable<SyncResult> results) {
017    for (SyncResult result : results) {
018      processResult(result);
019    }
020  }
021
022  public static void processResult(SyncResult result) {
023    if (result == null) {
024      LOG.warn("Null SyncResult!");
025    } else if (result.exception == null) {
026      LOG.info("Installation [{}] synced successfully. [{}] added, [{}] deleted, [{}] updated",
027               result.installation.getKey(),
028               result.addedDatasets.size(),
029               result.deletedDatasets.size(),
030               result.existingDatasets.size());
031    } else {
032      LOG.info("Installation [{}] failed sync. Reason: [{}]",
033               result.installation.getKey(),
034               result.exception.getErrorCode());
035      if (result.exception.getMessage() != null) {
036        LOG.info("Message: [{}]", result.exception.getMessage());
037      }
038      if (result.exception.getCause() != null) {
039        LOG.info("Cause: [{}], [{}]",
040                 result.exception.getCause().getClass().toString(),
041                 result.exception.getCause().getMessage());
042      }
043    }
044  }
045
046  private DebugHandler() {
047    throw new UnsupportedOperationException("Can't initialize class");
048  }
049
050}
