001package org.gbif.registry.metasync.api;
002
003import org.gbif.api.model.registry.Dataset;
004import org.gbif.api.model.registry.Installation;
005import org.gbif.api.model.registry.metasync.MetasyncHistory;
006import org.gbif.api.model.registry.metasync.MetasyncResult;
007
008import java.util.Collection;
009import java.util.List;
010import java.util.Map;
011
012import com.google.common.base.Preconditions;
013
014/**
015 * A simple holder object used to pass around the result of metadata synchronisation.
016 */
017@SuppressWarnings({"PublicField", "AssignmentToCollectionOrArrayFieldFromParameter"})
018public class SyncResult {
019
020  /**
021   * Maps from the existing Dataset in the Registry to a new object that we just parsed from the Endpoint.
022   */
023  public Map<Dataset, Dataset> existingDatasets;
024  public List<Dataset> addedDatasets;
025  public List<Dataset> deletedDatasets;
026  public Installation installation;
027  public MetadataException exception;
028
029  public SyncResult(
030    Map<Dataset, Dataset> existingDatasets,
031    List<Dataset> addedDatasets,
032    List<Dataset> deletedDatasets,
033    Installation installation
034  ) {
035    this.existingDatasets = existingDatasets;
036    this.addedDatasets = addedDatasets;
037    this.deletedDatasets = deletedDatasets;
038    this.installation = installation;
039  }
040
041  public SyncResult(Installation installation, MetadataException exception) {
042    this.installation = installation;
043    this.exception = exception;
044  }
045
046  /**
047   * @return A metasync history summary of the result
048   */
049  public MetasyncHistory buildHistory() {
050    MetasyncHistory history = new MetasyncHistory();
051    history.setInstallationKey(installation.getKey());
052
053    if (exception != null) {
054      history.setResult(buildCode(exception.getErrorCode()));
055      String message = String.format(
056        "Synchronization failed with error [%s]. %d datasets were updated. %d datasets were added. %d datasets were deleted.",
057        exception.getMessage(),
058        sizeOf(existingDatasets),
059        sizeOf(addedDatasets),
060        sizeOf(deletedDatasets));
061      history.setDetails(message);
062    } else {
063      history.setResult(MetasyncResult.OK);
064      String message = String.format(
065        "Synchronization succeeded. %d datasets were updated. %d datasets were added. %d datasets were deleted.",
066        sizeOf(existingDatasets),
067        sizeOf(addedDatasets),
068        sizeOf(deletedDatasets));
069      history.setDetails(message);
070    }
071    return history;
072  }
073
074  // NPE safe version
075  private int sizeOf(Collection<?> collection) {
076    return collection == null ? 0 : collection.size();
077  }
078
079  // NPE safe version
080  private int sizeOf(Map<?, ?> collection) {
081    return collection == null ? 0 : collection.size();
082  }
083
084  private MetasyncResult buildCode(ErrorCode code) {
085    Preconditions.checkNotNull(code, "Cannot build a code from null input");
086    switch (code) {
087      case IO_EXCEPTION:
088        return MetasyncResult.IO_EXCEPTION;
089      case PROTOCOL_ERROR:
090        return MetasyncResult.PROTOCOL_ERROR;
091      case HTTP_ERROR:
092        return MetasyncResult.HTTP_ERROR;
093      case OTHER_ERROR:
094        return MetasyncResult.OTHER_ERROR;
095      default:
096        return MetasyncResult.OTHER_ERROR;
097    }
098  }
099}
