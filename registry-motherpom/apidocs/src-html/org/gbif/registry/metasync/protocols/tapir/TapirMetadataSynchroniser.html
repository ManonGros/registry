056/**
057 * Synchronises metadata from a TAPIR Installation.
058 * <p/>
059 * Every TAPIR Installation can have multiple Endpoints. Each of these Endpoints represents a single Dataset and each
060 * of the Datasets can have a single Endpoint. Note: TAPIR supports Archives but no one seems to be using them so they
061 * are not supported here.
062 * <p/>
063 * The process is as follows:
064 * <p/>
065 * <ol>
066 * <li>For every Endpoint do a {@code capabilities} as well as a {@code metadata} request</li>
067 * <li>Convert the data into a {@link Dataset} object</li>
068 * <li>When we have all new {@code Dataset} objects try to map them to existing ones using the
069 * {@code local id} (which is the last part of the URL)</li>
070 * </ol>
071 * Note: If there is an exception during processing one of the Endpoints the whole synchronisation process will be
072 * aborted. I'm doing this to prevent inconsistencies.
073 */
074public class TapirMetadataSynchroniser extends BaseProtocolHandler {
075  private static final Logger LOG = LoggerFactory.getLogger(TapirMetadataSynchroniser.class);
076
077  // TAPIR supports multiple output formats, so we map a namespace to a search template with appropriate output model
078  // The maps are ordered by output model, with the preferred (highest priority) output for DwC/ABCD coming first
079  private static final LinkedListMultimap<String, String> ORDERED_TEMPLATE_MAPPING = LinkedListMultimap.create();
080
081  static {
082    // DwC
083    ORDERED_TEMPLATE_MAPPING.put("http://rs.tdwg.org/dwc/dwcore/",
084      "http://rs.gbif.org/templates/tapir/dwc/1.4/sci_name_range.xml");
085    ORDERED_TEMPLATE_MAPPING.put("http://rs.tdwg.org/dwc/geospatial/",
086      "http://rs.gbif.org/templates/tapir/dwc/1.4/sci_name_range.xml");
087    ORDERED_TEMPLATE_MAPPING.put("http://rs.tdwg.org/dwc/curatorial/",
088      "http://rs.gbif.org/templates/tapir/dwc/1.4/sci_name_range.xml");
089    ORDERED_TEMPLATE_MAPPING.put("http://rs.tdwg.org/dwc/terms/",
090      "http://rs.tdwg.org/tapir/cs/dwc/terms/2009-09-23/template/dwc_sci_name_range.xml");
091    ORDERED_TEMPLATE_MAPPING.put("http://digir.net/schema/conceptual/darwin/2003/1.0",
092      "http://rs.gbif.org/templates/tapir/dwc/1.0/sci_name_range.xml");
093    ORDERED_TEMPLATE_MAPPING.put("http://www.tdwg.org/schemas/abcd/2.06",
094      "http://rs.gbif.org/templates/tapir/abcd/206/sci_name_range.xml");
095    // ABCD (Default to 2.0.6 whenever 2.0.5 is encountered)
096    ORDERED_TEMPLATE_MAPPING.put("http://www.tdwg.org/schemas/abcd/2.05",
097      "http://rs.gbif.org/templates/tapir/abcd/206/sci_name_range.xml");
098    ORDERED_TEMPLATE_MAPPING.put("http://www.tdwg.org/schemas/abcd/1.2",
099      "http://rs.gbif.org/templates/tapir/abcd/1.2/sci_name_range.xml");
100  }

102  public TapirMetadataSynchroniser(HttpClient httpClient) {
103    super(httpClient);

106  @Override
107  public boolean canHandle(Installation installation) {
108    return installation.getType() == InstallationType.TAPIR_INSTALLATION;
109  }
110
111  @Override
112  public SyncResult syncInstallation(Installation installation, List<Dataset> datasets) throws MetadataException {
113    checkArgument(installation.getType() == InstallationType.TAPIR_INSTALLATION, "Only supports TAPIR Installations");
114
115    List<Dataset> added = Lists.newArrayList();
116    List<Dataset> deleted = Lists.newArrayList();
117    Map<Dataset, Dataset> updated = Maps.newHashMap();
118
119    // This metadata will be used to update the Installation itself
120    TapirMetadata updaterMetadata = null;
121
122    for (Endpoint endpoint : installation.getEndpoints()) {
123      String localId = getLocalId(endpoint);
124
125      Capabilities capabilities = getCapabilities(endpoint);
126      if (capabilities == null) {
127        throw new MetadataException("Did not receive a valid Capabilities response for [" + endpoint.getKey() + "]",
128                                    ErrorCode.PROTOCOL_ERROR);
129      }
130
131      TapirMetadata metadata = getTapirMetadata(endpoint);
132
133      updateInstallationEndpoint(metadata, endpoint);
134
135      String outputModelTemplate = getPreferredOutputModelTemplate(capabilities.getSchemas());
136      URI searchRequestUrl = buildSearchRequestUrl(endpoint, outputModelTemplate);
137      TapirSearch search = getTapirSearch(searchRequestUrl);
138
139      Dataset newDataset = convertToDataset(capabilities, metadata, search);
140      Dataset existingDataset = findDataset(localId, datasets);
141      if (existingDataset == null) {
142        added.add(newDataset);
143      } else {
144        updated.put(existingDataset, newDataset);
145      }
146
147      updaterMetadata = metadata;
148    }
149
150    // All Datasets that weren't updated must have been deleted
151    for (Dataset dataset : datasets) {
152      if (!updated.containsKey(dataset)) {
153        deleted.add(dataset);
154      }
155    }
156
157    updateInstallation(installation, updaterMetadata);
158
159    return new SyncResult(updated, added, deleted, installation);
160  }

162  /**
163   * Gets the <em>local id</em> from the Endpoint. This is the last part of the URL and the only uniquely identifying
164   * piece for a TAPIR Dataset.
165   */
166  private String getLocalId(Endpoint endpoint) throws MetadataException {
167    String[] split = endpoint.getUrl().toASCIIString().split("/");
168
169    if (split.length < 2) {
170      throw new MetadataException("Could not find local Id for [" + endpoint.getUrl() + "]", ErrorCode.OTHER_ERROR);
171    }
172
173    return split[split.length - 1];
174  }
175
176  /**
177   * Does a Capabilities request against the TAPIR Endpoint.
178   */
179  private Capabilities getCapabilities(Endpoint endpoint) throws MetadataException {
180    URI uri;
181    try {
182      uri = new URIBuilder(endpoint.getUrl()).addParameter("op", "capabilities").build();
183    } catch (URISyntaxException e) {
184      throw new MetadataException(e, ErrorCode.OTHER_ERROR);
185    }
186
187    return doHttpRequest(uri, newDigester(Capabilities.class));
188  }
189
190  /**
191   * Does a Metadata request against the TAPIR Endpoint.
192   */
193  private TapirMetadata getTapirMetadata(Endpoint endpoint) throws MetadataException {
194    return doHttpRequest(endpoint.getUrl(), newDigester(TapirMetadata.class));
195  }

198   * Does a Search request against the TAPIR Endpoint.
199   *
200   * @param request request URI to search request used to retrieve number of records

202   * @return TapirSearch, or null if the response could not be parsed into a TapirSearch
203   *
204   * @throws MetadataException in case anything goes wrong during the request

206  private TapirSearch getTapirSearch(URI request) throws MetadataException {
207    return doHttpRequest(request, newDigester(TapirSearch.class));

210  /**
211   * Updates the Endpoint of the Installation that we're currently working on.
212   */
213  private void updateInstallationEndpoint(TapirMetadata metadata, Endpoint endpoint) {
214    endpoint.setDescription(metadata.getDescriptions().toString());
215  }

217  /**
218   * Converts the Capabilities and Metadata response from TAPIR into a GBIF Dataset.
219   */
220  private Dataset convertToDataset(Capabilities capabilities, TapirMetadata metadata, @Nullable TapirSearch search) {
221    Dataset dataset = new Dataset();
222    dataset.setTitle(metadata.getTitles().toString());
223    dataset.setDescription(metadata.getDescriptions().toString());
224    dataset.setHomepage(metadata.getAccessPoint());
225    dataset.setLanguage(metadata.getDefaultLanguage());
226    // Respect publisher issued DOIs if provided.
227    if (DOI.isParsable(metadata.getIdentifier())) {
228      dataset.setDoi(new DOI(metadata.getIdentifier()));
229    }
230
231    List<Contact> contacts = Lists.newArrayList();
232    for (TapirRelatedEntity tapirRelatedEntity : metadata.getRelatedEntities()) {
233      for (TapirContact tapirContact : tapirRelatedEntity.getContacts()) {
234        Contact contact = new Contact();
235        for (String r : tapirContact.getRoles()) {
236          contact.addPosition(r);
237        }
238        contact.setFirstName(tapirContact.getFullName());
239        contact.addPhone(tapirContact.getTelephone());
240        contact.addEmail(tapirContact.getEmail());
241        contact.setDescription(tapirContact.getTitle());
242        contacts.add(contact);
243      }
244    }
245    dataset.setContacts(contacts);
246
247    Endpoint endpoint = new Endpoint();
248    endpoint.setType(EndpointType.TAPIR);
249    endpoint.setDescription(metadata.getTitles().toString());
250    endpoint.setUrl(metadata.getAccessPoint());
251    dataset.addEndpoint(endpoint);
252
253    // This makes sure that we only save one content namespace/conceptual schema to the machine tags. This is done
254    // by picking the first match from the ORDERED_TEMPLATE_MAPPING
255    boolean found = false;
256    for (Map.Entry<String, String> entry : ORDERED_TEMPLATE_MAPPING.entries()) {
257      if (found) break;
258      for (Schema schema : capabilities.getSchemas()) {
259        if (schema.getNamespace().toASCIIString().equalsIgnoreCase(entry.getKey())) {
260          dataset.addMachineTag(MachineTag.newInstance(METADATA_NAMESPACE,
261                                                       Constants.CONCEPTUAL_SCHEMA,
262                                                       schema.getNamespace().toASCIIString()));
263          found = true;
264          break;
265        }
266      }
267    }
268    if (!found) {
269      LOG.warn("Could not find a supported conceptual schema");
270    }
271
272    // if search response brought back the number of records, add corresponding tag
273    if (search != null && search.getNumberOfRecords() != 0) {
274      dataset.addMachineTag(MachineTag.newInstance(METADATA_NAMESPACE,
275        Constants.DECLARED_COUNT,
276        String.valueOf(search.getNumberOfRecords())));
277    }
278
279    return dataset;
280  }
281
282  /**
283   * Tries to find a matching Dataset in the list of provided Datasets.
284   *
285   * @return the matching Dataset or {@code null} if it could not be found
286   */
287  @Nullable
288  private Dataset findDataset(String localId, Iterable<Dataset> datasets) throws MetadataException {
289    for (Dataset dataset : datasets) {
290      for (Endpoint endpoint : dataset.getEndpoints()) {
291        if (localId.equals(getLocalId(endpoint))) {
292          return dataset;
293        }
294      }
295    }
296    return null;
297  }

299  /**
300   * Updates the Installation with the data that is universal across all Datasets. As the Installation itself doesn't
301   * have any Metadata Endpoint there is very little information we can extract.
302   */
303  private void updateInstallation(Installation installation, TapirMetadata updaterMetadata) {
304    installation.addMachineTag(MachineTag.newInstance(METADATA_NAMESPACE,
305                                                      "version",
306                                                      updaterMetadata.getSoftwareVersion()));
307    installation.addMachineTag(MachineTag.newInstance(METADATA_NAMESPACE,
308                                                      "software_name",
309                                                      updaterMetadata.getSoftwareName()));
310  }

312  /**
313   * Construct a map of the atoms for the TAPIR search request.
314   *
315   * @param outputModelTemplate output model template URI
316   *
317   * @return The atoms for the TAPIR search request (always in the same order which is important for testing) used only
318   *         to return the count.
319   */
320  private Map<String, String> buildTapirSearchRequestParameters(String outputModelTemplate) {
321    Map<String, String> params = Maps.newLinkedHashMap();
322    params.put("op", "s");
323    params.put("t", outputModelTemplate);
324    params.put("count", "true");
325
326    params.put("start", "0");
327    params.put("limit", "1");
328
329    params.put("lower", "AAA");
330    params.put("upper", "zzz");
331
332    return params;
333  }
334
335  /**
336   * Build the TAPIR search request URL that will return the number of records count. It has scientific name range
337   * AAA-zzz, offset 0, limit 1, and count equal to true.
338   *
339   * @param endpoint            Endpoint
340   * @param outputModelTemplate output model template to use in request
341   *
342   * @return TAPIR search request URL
343   */
344  public URI buildSearchRequestUrl(Endpoint endpoint, String outputModelTemplate) {
345    try {
346      URIBuilder uriBuilder = new URIBuilder(endpoint.getUrl());
347      for (Map.Entry<String, String> paramEntry : buildTapirSearchRequestParameters(outputModelTemplate).entrySet()) {
348        uriBuilder.addParameter(paramEntry.getKey(), paramEntry.getValue());
349      }
350      return uriBuilder.build();
351    } catch (URISyntaxException e) {
352      // It's coming from a valid URL so it shouldn't ever throw this, so we swallow this
353      return null;
354    }
355  }

357  /**
358   * Gets the preferred output model template to use for requests to the TAPIR endpoint using the list of supported
359   * namespaces coming from the capabilities response's list of Schemas.
360   * </br>
361   * E.g. schema with namespace ABCD 2.06 is preferred over ABCD 1.2, so the output model corresponding to ABCD 2.06 is
362   * returned.
363   *
364   * @param schemas TAPIR capabilities list of Schemas
365   *
366   * @return the preferred output model template
367   *
368   * @throws MetadataException if no preferred output model template was found
369   */
370  private String getPreferredOutputModelTemplate(Iterable<Schema> schemas) throws MetadataException {
371    // iterate through ordered namespace -> output model mappings
372    for (Map.Entry<String, String> entry : ORDERED_TEMPLATE_MAPPING.entries()) {
373      // was this namespace listed in the capabilities?
374      for (Schema schema : schemas) {
375        if (entry.getKey().equalsIgnoreCase(schema.getNamespace().toString())) {
376          // return the output model template, corresponding to the highest priority namespace encountered
377          return entry.getValue();
378        }
379      }
380    }
381    throw new MetadataException("No namespace found matching a DwC or ABCD output model", ErrorCode.PROTOCOL_ERROR);
382  }
