001package org.gbif.registry.metasync.protocols.biocase.model;
002
003import java.util.List;
004
005import com.google.common.base.Objects;
006import com.google.common.collect.Lists;
007import org.apache.commons.digester3.annotations.rules.CallMethod;
008import org.apache.commons.digester3.annotations.rules.CallParam;
009import org.apache.commons.digester3.annotations.rules.ObjectCreate;
010
011/**
012 * This is the inventory retrieved by issuing a {@code scan} request used prior to BioCASe 3.4.
013 *
014 * @see NewDatasetInventory
015 */
016@ObjectCreate(pattern = "response/content")
017public class OldDatasetInventory {
018
019  private List<String> datasets = Lists.newArrayList();
020
021  public List<String> getDatasets() {
022    return datasets;
023  }
024
025  public void setDatasets(List<String> datasets) {
026    this.datasets = datasets;
027  }
028
029  @CallMethod(pattern = "response/content/scan/value")
030  public void addDataset(
031    @CallParam(pattern = "response/content/scan/value") String dataset
032  ) {
033    datasets.add(dataset);
034  }
035
036  @Override
037  public String toString() {
038    return Objects.toStringHelper(this).add("datasets", datasets).toString();
039  }
040}
