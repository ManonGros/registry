050/**
051 * DiGIR synchronisation happens by issuing a metadata request to the single endpoint that the installation should
052 * have. The response contains a list of resources which we parse to GBIF {@link Dataset} objects, each having a single
053 * Endpoint.
054 */
055public class DigirMetadataSynchroniser extends BaseProtocolHandler {
056
057  // keyword used to identify if an endpoint is of type DIGIR_MANIS
058  private static final String MANIS_KEYWORD = "manis";
059  // only schemaLocation of type DIGIR_MANIS not containing the word "manis"
060  private static final String MANIS_SCHEMA_LOCATION = "http://bnhm.berkeley.edu/DwC/bnhm_dc2_schema.xsd";
061
062  public DigirMetadataSynchroniser(HttpClient httpClient) {
063    super(httpClient);
064  }
065
066  @Override
067  public boolean canHandle(Installation installation) {
068    return installation.getType() == InstallationType.DIGIR_INSTALLATION;
069  }
070
071  @Override
072  public SyncResult syncInstallation(Installation installation, List<Dataset> datasets) throws MetadataException {
073    checkArgument(installation.getType() == InstallationType.DIGIR_INSTALLATION, "Only supports DiGIR Installations");
074
075    if (installation.getEndpoints().size() != 1) {
076      throw new MetadataException("A DiGIR Installation should only ever have one Endpoint, this one has ["
077                                  + installation.getEndpoints().size()
078                                  + "]", ErrorCode.OTHER_ERROR);
079    }
080    Endpoint endpoint = installation.getEndpoints().get(0);
081
082    DigirMetadata metadata = getDigirMetadata(endpoint);
083    updateInstallation(metadata, installation);
084    updateInstallationEndpoint(metadata, endpoint);
085    return mapToDatasets(metadata, datasets, endpoint.getUrl(), installation);
086  }
087
088  private DigirMetadata getDigirMetadata(Endpoint endpoint) throws MetadataException {
089    return doHttpRequest(endpoint.getUrl(), newDigester(DigirMetadata.class));
090  }
091
092  /**
093   * Updates the Installation in in place with all the data gathered from the Endpoint.
094   */
095  private void updateInstallation(DigirMetadata metadata, Installation installation) {
096    installation.setContacts(matchContacts(installation.getContacts(),
097                                           convertToRegistryContacts(metadata.getHost().getContacts())));
098
099    installation.setDescription(metadata.getHost().getDescription());
100
101    installation.addMachineTag(MachineTag.newInstance(METADATA_NAMESPACE,
102                                                      Constants.DIGIR_CODE,
103                                                      metadata.getHost().getCode()));
104    installation.addMachineTag(MachineTag.newInstance(METADATA_NAMESPACE,
105                                                      Constants.INSTALLATION_VERSION,
106                                                      metadata.getImplementation()));
107  }
108
109  /**
110   * Updates the single Endpoint that a DiGIR Installation has.
111   */
112  private void updateInstallationEndpoint(DigirMetadata metadata, Endpoint endpoint) {
113    endpoint.setDescription(metadata.getHost().getDescription());
114  }
115
116  /**
117   * Maps the resources we got from the metadata response to Datasets that are currently hosted by this Installation.
118   * We identify Datasets by the {@code code} attribute that we're getting from the metadata response. We're saving
119   * this code on the Dataset itself as a machine tag.
120   */
121  private SyncResult mapToDatasets(
122    DigirMetadata metadata, Iterable<Dataset> datasets, URI url, Installation installation
123  ) {
124    List<Dataset> added = Lists.newArrayList();
125    List<Dataset> deleted = Lists.newArrayList();
126    Map<Dataset, Dataset> updated = Maps.newHashMap();
127
128    // Maps currently existing DiGIR codes to the Datasets from our Registry that use those codes
129    Map<String, Dataset> codeMap = Maps.newHashMap();
130    for (Dataset dataset : datasets) {
131
132      // Find the "code" machine tag
133      for (MachineTag tag : dataset.getMachineTags()) {
134        if (tag.getNamespace().equals(METADATA_NAMESPACE) && tag.getName().equals(Constants.DIGIR_CODE)) {
135          codeMap.put(tag.getValue(), dataset);
136        }
137      }
138    }
139
140    // Sort in either updated or added Datasets using the just built Map
141    for (DigirResource resource : metadata.getResources()) {
142      Dataset newDataset = convertToDataset(resource, url);
143      if (codeMap.containsKey(resource.getCode())) {
144        updated.put(codeMap.get(resource.getCode()), newDataset);
145      } else {
146        added.add(newDataset);
147      }
148    }
149
150    // All Datasets that weren't updated must have been deleted
151    for (Dataset dataset : datasets) {
152      if (!updated.containsKey(dataset)) {
153        deleted.add(dataset);
154      }
155    }
156
157    return new SyncResult(updated, added, deleted, installation);
158  }
159
160  /**
161   * Converts a DiGIR resource to a GBIF Dataset.
162   */
163  private Dataset convertToDataset(DigirResource resource, URI url) {
164    Dataset dataset = new Dataset();
165    dataset.setTitle(resource.getName());
166    dataset.setDescription(resource.getDescription());
167
168    // We're only using the very first related URI even though there might be more
169    if (!resource.getRelatedInformation().isEmpty()) {
170      dataset.setHomepage(resource.getRelatedInformation().iterator().next());
171    }
172    dataset.setCitation(new Citation(resource.getCitation(), null));
173    dataset.setRights(resource.getUseRestrictions());
174    dataset.setContacts(convertToRegistryContacts(resource.getContacts()));
175    dataset.addMachineTag(MachineTag.newInstance(METADATA_NAMESPACE, Constants.DIGIR_CODE, resource.getCode()));
176
177    // Respect publisher issued DOIs if provided.
178    if (DOI.isParsable(resource.getCode())) {
179      dataset.setDoi(new DOI(resource.getCode()));
180    }
181
182    if (resource.getNumberOfRecords() != 0) {
183      dataset.addMachineTag(MachineTag.newInstance(METADATA_NAMESPACE,
184                                                   Constants.DECLARED_COUNT,
185                                                   String.valueOf(resource.getNumberOfRecords())));
186    }
187
188    if (resource.getMaxSearchResponseRecords() != 0) {
189      dataset.addMachineTag(MachineTag.newInstance(METADATA_NAMESPACE,
190                                                   Constants.DIGIR_MAX_SEARCH_RESPONSE_RECORDS,
191                                                   String.valueOf(resource.getMaxSearchResponseRecords())));
192    }
193
194    if (resource.getDateLastUpdated() != null) {
195      dataset.addMachineTag(MachineTag.newInstance(METADATA_NAMESPACE,
196                                                   Constants.DATE_LAST_UPDATED,
197                                                   resource.getDateLastUpdated().toString()));
198    }
199
200    for (Map.Entry<String, URI> entry : resource.getConceptualSchemas().entrySet()) {
201      dataset.addMachineTag(MachineTag.newInstance(METADATA_NAMESPACE,
202                                                   Constants.CONCEPTUAL_SCHEMA,
203                                                   entry.getValue().toASCIIString()));
204    }
205
206    // Each DiGIR Dataset has exactly one Endpoint, we create and populate it here
207    Endpoint endpoint = new Endpoint();
208    endpoint.setDescription(resource.getName());
209    endpoint.setUrl(url);
210    // normal DiGIR vs MaNIS DiGIR?
211    endpoint.setType(determineEndpointType(resource.getConceptualSchemas()));
212    dataset.addEndpoint(endpoint);
213
214    return dataset;
215  }
216
217  /**
218   * Iterates through the resource's map of namespace (conceptualSchemas) / schemaLocation key value pairs.
219   * If a DIGIR_MANIS endpoint is found in the list, the EndpointType is equal to DiGIR_MANIS. A DiGIR_MANIS
220   * endpoint is identified, by checking if the schemaLocation 1)contains the word "manis" or 2) is equal to
221   * "http://bnhm.berkeley.edu/DwC/bnhm_dc2_schema.xsd".
222   *
223   * @param conceptualSchemas map with namespace (conceptualSchemas), schemaLocation key value pairs
224   *
225   * @return endpoint type, defaulting to (normal) DiGIR
226   */
227  EndpointType determineEndpointType(Map<String, URI> conceptualSchemas) {
228    for (URI schemaLocation : conceptualSchemas.values()) {
229      if (schemaLocation.toString().equalsIgnoreCase(MANIS_SCHEMA_LOCATION) || schemaLocation.toString().toLowerCase()
230        .contains(MANIS_KEYWORD)) {
231        return EndpointType.DIGIR_MANIS;
232      }
233    }
234    return EndpointType.DIGIR;
235  }
236
237  /**
238   * Converts a list of DiGIR contacts to GBIF {@link Contact} objects.
239   */
240  private List<Contact> convertToRegistryContacts(Iterable<DigirContact> contacts) {
241    List<Contact> resultList = Lists.newArrayList();
242    for (DigirContact contact : contacts) {
243      resultList.add(convertToRegistryContact(contact));
244    }
245    return resultList;
246  }
247
248  /**
249   * Converts a single DiGIR contact to a GBIF {@link Contact}
250   */
251  private Contact convertToRegistryContact(DigirContact digirContact) {
252    Contact contact = new Contact();
253    contact.setFirstName(trimToNull(digirContact.getName()));
254    contact.addPosition(trimToNull(digirContact.getTitle()));
255    contact.addEmail(trimToNull(digirContact.getEmail()));
256    contact.addPhone(trimToNull(digirContact.getPhone()));
257    contact.setType(ContactType.inferType(digirContact.getType()));
258    return contact;
259  }
