054/**
055 * Synchronises metadata from a BioCASe Installation.
056 * <p/>
057 * Every BioCASe Installation can have multiple Endpoints, each of those Endpoints can have multiple Datasets and each
058 * of those Datasets can have multiple Endpoints (Web service, ABCD Archive, DwC-A).
059 * <p/>
060 * BioCASe synchronisation happens in the following steps:
061 * <ul>
062 * <li>For all endpoints of an Installation make a {@code capabilities} request, followed by a dataset inventory
063 * request, then a metadata and count (to get the number of records) request for each dataset</li>
064 * <li>The type of inventory request depends on the capabilities. Newer version of BioCASe (3.4 and greater) support a
065 * separate inventory request.</li>
066 * </ul>
067 * <p/>
068 * Unfortunately BioCASe does not have good (stable) identifiers for Datasets so we need to rely on the Dataset title.
069 */
070public class BiocaseMetadataSynchroniser extends BaseProtocolHandler {
071  private static final Logger LOG = LoggerFactory.getLogger(BiocaseMetadataSynchroniser.class);
072
073  public BiocaseMetadataSynchroniser(HttpClient httpClient) {
074    super(httpClient);
075  }
076
077  @Override
078  public boolean canHandle(Installation installation) {
079    return installation.getType() == InstallationType.BIOCASE_INSTALLATION;
080  }
081
082  @Override
083  public SyncResult syncInstallation(
084    Installation installation, List<Dataset> datasets
085  ) throws MetadataException {
086    checkArgument(installation.getType() == InstallationType.BIOCASE_INSTALLATION,
087                  "Only supports BioCASe Installations");
088
089    List<Dataset> added = Lists.newArrayList();
090    List<Dataset> deleted = Lists.newArrayList();
091    Map<Dataset, Dataset> updated = Maps.newHashMap();
092
093    for (Endpoint endpoint : installation.getEndpoints()) {
094      LOG.info("Starting synchronization of endpoint: {}", endpoint.getUrl());
095
096      Capabilities capabilities = getCapabilities(endpoint);
097      if (capabilities.getPreferredSchema() == null) {
098        throw new MetadataException("No preferred schema", ErrorCode.PROTOCOL_ERROR);
099      }
100      List<String> datasetInventory = getDatasetInventory(capabilities, endpoint);
101      for (String datasetTitle : datasetInventory) {
102        Dataset newDataset;
103
104        if (capabilities.getPreferredSchema().equals(Constants.ABCD_12_SCHEMA)) {
105          SimpleAbcd12Metadata metadata = get12Metadata(endpoint, datasetTitle, capabilities);
106          newDataset = convertToDataset(metadata, endpoint, capabilities);
107        } else {
108          SimpleAbcd206Metadata metadata = get206Metadata(endpoint, datasetTitle, capabilities);
109          newDataset = convertToDataset(metadata, endpoint, capabilities);
110        }
111
112        Dataset existingDataset = findDataset(datasetTitle, datasets);
113        if (existingDataset == null) {
114          added.add(newDataset);
115        } else {
116          updated.put(existingDataset, newDataset);
117        }
118      }
119    }
120
121    // All Datasets that weren't updated must have been deleted
122    for (Dataset dataset : datasets) {
123      if (!updated.containsKey(dataset)) {
124        deleted.add(dataset);
125      }
126    }
127
128    return new SyncResult(updated, added, deleted, installation);
129  }

131  public URI buildUri(URI url, String parameter, String value) throws MetadataException {
132    try {
133      return new URIBuilder(url).addParameter(parameter, value).build();
134    } catch (URISyntaxException e) {
135      throw new MetadataException(e, ErrorCode.OTHER_ERROR);
136    }
137  }

139  /**
140   * Does a Capabilities request against the Endpoint.
141   */
142  private Capabilities getCapabilities(Endpoint endpoint) throws MetadataException {
143    return doHttpRequest(endpoint.getUrl(), newDigester(Capabilities.class));
144  }

146  /**
147   * Tries to get an inventory (list) of Datasets for this BioCASe Endpoint. Depending on the version of the
148   * Installation there are two ways to do this.
149   */
150  private List<String> getDatasetInventory(Capabilities capabilities, Endpoint endpoint) throws MetadataException {
151    String version = capabilities.getVersions().get("pywrapper");
152    if (checkIfSupportsNewInventory(version)) {
153      return doNewStyleInventory(endpoint);
154    } else {
155      return doOldStyleInventory(endpoint, capabilities);
156    }
157  }
158
159  /**
160   * Given a version string in the normal "major.minor.patch" format evaluates whether this version of BioCASe supports
161   * the new style inventory request or not. All versions of BioCASe 3.4 and above do support this.
162   */
163  private boolean checkIfSupportsNewInventory(String version) {
164    if (version == null) {
165      return false;
166    }
167
168    String[] versionParts = version.split("\\.");
169    try {
170      // Trying to parse the "major" part first, if we succeed and it is greater than 3 we do support the nev inventory
171      // style, if it is less than three we don't support it. If it is equal to 3 we need to check the "minor" component
172      int majorVersion = Integer.valueOf(versionParts[0]);
173      if (majorVersion < 3) {
174        return false;
175      }
176      if (majorVersion > 3) {
177        return true;
178      }
179
180      // "major" version is 3 but there is no "minor" part
181      if (versionParts.length < 2) {
182        return false;
183      }
184
185      // Check whether the "minor" version is greater than or equal to 4
186      int minorVersion = Integer.valueOf(versionParts[1]);
187      return minorVersion >= 4;
188    } catch (NumberFormatException ignored) {
189      return false;
190    }
191  }

193  /**
194   * Does a request against the dedicated {@code inventory} endpoint which lists all Datasets that are available as well
195   * as all Archives.
196   */
197  // TODO: Need to return information about archives
198  private List<String> doNewStyleInventory(Endpoint endpoint) throws MetadataException {
199    URI uri = buildUri(endpoint.getUrl(), "inventory", "1");
200    NewDatasetInventory inventory = doHttpRequest(uri, newDigester(NewDatasetInventory.class));
201    List<String> datasets = Lists.newArrayList();
202    if (inventory == null) return datasets;
203    for (InventoryDataset inventoryDataset : inventory.getDatasets()) {
204      datasets.add(inventoryDataset.getTitle());
205    }
206    return datasets;
207  }
208
209  /**
210   * Does a search request against this Endpoint specially crafted to only find all Dataset titles.
211   */
212  private List<String> doOldStyleInventory(Endpoint endpoint, Capabilities capabilities) throws MetadataException {
213    String requestParameter = TemplateUtils.getBiocaseInventoryRequest(capabilities.getPreferredSchema());
214    URI uri = buildUri(endpoint.getUrl(), "request", requestParameter);
215    OldDatasetInventory inventory = doHttpRequest(uri, newDigester(OldDatasetInventory.class));
216    return inventory.getDatasets();
217  }

239  private Dataset convertToDataset(
240    SimpleAbcd206Metadata metadata, Endpoint installationEndpoint, Capabilities capabilities
241  ) {
242    Dataset dataset = new Dataset();
243    dataset.setTitle(metadata.getName());
244    dataset.setDescription(metadata.getDetails());
245    dataset.setHomepage(metadata.getHomepage());
246    dataset.setLogoUrl(metadata.getLogoUrl());
247    dataset.setRights(metadata.getRights());
248    // Respect publisher issued DOIs if provided.
249    if (DOI.isParsable(metadata.getDatasetGUID())) {
250      dataset.setDoi(new DOI(metadata.getDatasetGUID()));
251    }
252
253    Citation citation = new Citation();
254    citation.setText(metadata.getCitationText());
255    dataset.setCitation(citation);
256
257    dataset.setContacts(metadata.getContacts());
258
259    Endpoint endpoint = new Endpoint();
260    endpoint.setType(EndpointType.BIOCASE);
261    endpoint.setUrl(installationEndpoint.getUrl());
262    endpoint.addMachineTag(MachineTag.newInstance(Constants.METADATA_NAMESPACE,
263                                                  Constants.CONCEPTUAL_SCHEMA,
264                                                  capabilities.getPreferredSchema()));
265
266    dataset.addEndpoint(endpoint);
267    return dataset;
268  }

272    Endpoint installationEndpoint,
273    Capabilities capabilities

275    Dataset dataset = new Dataset();

277    dataset.setDescription(metadata.getDescription());

279    dataset.setLogoUrl(metadata.getLogoUrl());
280    dataset.setRights(metadata.getRights());
281
282    Citation citation = new Citation();
283    citation.setText(metadata.getCitationText());
284    dataset.setCitation(citation);
285
286    dataset.setContacts(metadata.getContacts());
287
288    Endpoint endpoint = new Endpoint();
289    endpoint.setType(EndpointType.BIOCASE);
290    endpoint.setUrl(installationEndpoint.getUrl());
291    endpoint.addMachineTag(MachineTag.newInstance(Constants.METADATA_NAMESPACE,
292                                                  Constants.CONCEPTUAL_SCHEMA,
293                                                  capabilities.getPreferredSchema()));
294
295    dataset.addEndpoint(endpoint);
296    return dataset;
297  }

299  /**
300   * Tries to find a matching Dataset in the list of provided Datasets by looking at the title.
301   *
302   * @return the matching Dataset or {@code null} if it could not be found
303   */
304  @Nullable
305  private Dataset findDataset(String datasetTitle, Iterable<Dataset> datasets) {
306    for (Dataset dataset : datasets) {
307      if (dataset.getTitle().equals(datasetTitle)) {
308        return dataset;
309      }
310    }
311
312    return null;
313  }
