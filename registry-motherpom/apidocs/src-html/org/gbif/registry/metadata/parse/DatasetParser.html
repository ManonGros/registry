001package org.gbif.registry.metadata.parse;
002
003import org.gbif.api.model.registry.Dataset;
004import org.gbif.api.vocabulary.MetadataType;
005
006import java.io.ByteArrayInputStream;
007import java.io.IOException;
008import java.io.InputStream;
009import java.util.LinkedList;
010
011import com.google.common.collect.Lists;
012import com.google.common.io.ByteStreams;
013import com.google.common.io.Closeables;
014import org.apache.commons.beanutils.ConversionException;
015import org.apache.commons.digester3.Digester;
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;
018import org.xml.sax.Attributes;
019import org.xml.sax.InputSource;
020import org.xml.sax.SAXException;
021import org.xml.sax.XMLReader;
022import org.xml.sax.helpers.DefaultHandler;
023import org.xml.sax.helpers.XMLReaderFactory;
024
025/**
026 * Main parser of dataset metadata that uses parser specific digester RuleSets for EML or Dublin Core.
027 * It can automatically detect the document type if it is unknown or be used only with a specific parser type.
028 *
029 * This parser and its digester rules use the DatasetDelegator class to wrap a dataset and set complex bean components.
030 */
031public class DatasetParser {
032
033  private static final Logger LOG = LoggerFactory.getLogger(DatasetParser.class);
034
035  private static class ParserDetectionHandler extends DefaultHandler {
036    private static final String DC_NAMESPACE = "http://purl.org/dc/terms/";
037    private MetadataType parserType;
038    private LinkedList<String> path = Lists.newLinkedList();
039
040    @Override
041    public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
042      // look for EML
043      if (path.size() == 1 && path.get(0).equals("eml") && localName.equals("dataset")) {
044        parserType = MetadataType.EML;
045      }
046
047      // look for DC title
048      if (parserType == null && DC_NAMESPACE.equals(uri)) {
049        parserType = MetadataType.DC;
050      }
051
052      path.add(localName);
053    }
054
055    @Override
056    public void endElement(String uri, String localName, String qName) throws SAXException {
057      String last = path.removeLast();
058      if (!last.equals(localName)) {
059        LOG.warn("XML path broken. Got {} but path stack gave {}", localName, last);
060      }
061    }
062  }
063
064  /**
065   * @return the detected parser type or null
066   * @throws java.lang.IllegalArgumentException in case no parser exists for this document
067   */
068  public static MetadataType detectParserType(InputStream xml) {
069    try {
070      XMLReader xmlReader = XMLReaderFactory.createXMLReader();
071      ParserDetectionHandler handler = new ParserDetectionHandler();
072      xmlReader.setContentHandler(handler);
073      InputSource inputSource = new InputSource(xml);
074      xmlReader.parse(inputSource);
075      if (handler.parserType != null) {
076        return handler.parserType;
077      }
078
079    } catch (SAXException e) {
080      LOG.error("Failed to SAX parse a document for parser type detection", e);
081    } catch (IOException e) {
082      LOG.warn("Failed to read metadata document for parser type detection", e);
083    }
084    throw new IllegalArgumentException("No parser found for this metadata document. Only EML or DC supported");
085  }
086
087
088  /**
089   * Build from stream on-top of a preexisting Dataset populating its fields from a source metadata that's parsed.
090   *
091   * @param xml to read
092   * @return The Dataset populated, never null
093   * @throws java.io.IOException If the Stream cannot be read from
094   * @throws IllegalArgumentException If the XML is not well formed or is not understood
095   */
096  public static Dataset build(InputStream xml) throws IOException {
097    // buffer entire stream first. We need it several times
098    final byte[] data = ByteStreams.toByteArray(xml);
099    // detect the parser type
100    return parse(detectParserType(new ByteArrayInputStream(data)), new ByteArrayInputStream(data));
101  }
102
103  public static Dataset parse(MetadataType type, InputStream xml) throws IOException {
104    Digester digester = new Digester();
105    digester.setNamespaceAware(true);
106
107    // add digester rules based on parser type
108    switch (type) {
109      case EML:
110        LOG.debug("Parsing EML document");
111        digester.addRuleSet(new EMLRuleSet());
112        break;
113      case DC:
114        LOG.debug("Parsing DC document");
115        digester.addRuleSet(new DublinCoreRuleSet());
116        break;
117    }
118
119    // push the Delegating object onto the stack
120    DatasetWrapper delegator = new DatasetWrapper();
121    digester.push(delegator);
122
123    // now parse and return the dataset
124    try {
125        digester.parse(xml);
126    } catch (ConversionException e) {
127      // swallow
128    } catch (SAXException e) {
129      if (e.getException() == null || !e.getException().getClass().equals(ConversionException.class)) {
130        // allow type conversions to happen
131        throw new IllegalArgumentException("Invalid metadata xml document", e);
132      }
133    } finally {
134      delegator.postProcess();
135      Closeables.closeQuietly(xml);
136    }
137
138    return delegator.getTarget();
139  }
