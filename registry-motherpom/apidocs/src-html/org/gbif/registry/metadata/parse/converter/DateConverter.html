001package org.gbif.registry.metadata.parse.converter;
002
003import org.gbif.common.parsers.core.ParseResult;
004import org.gbif.common.parsers.date.DateParseUtils;
005
006import java.util.Date;
007
008import org.apache.commons.beanutils.converters.AbstractConverter;
009
010public class DateConverter extends AbstractConverter {
011
012  public DateConverter() {
013
014  }
015
016  @Override
017  protected Object convertToType(Class type, Object value) {
018    ParseResult<Date> result = DateParseUtils.parse(value.toString());
019    return result.getStatus() == ParseResult.STATUS.SUCCESS ? result.getPayload() : null;
020  }
021
022  /**
023   * Return the default type this {@code Converter} handles.
024   *
025   * @return The default type this {@code Converter} handles.
026   */
027  @Override
028  protected Class getDefaultType() {
029    return Date.class;
030  }
031}
