001package org.gbif.registry.metadata;
002
003import org.gbif.api.model.registry.Contact;
004import org.gbif.api.model.registry.Dataset;
005import org.gbif.api.model.registry.eml.temporal.TemporalCoverage;
006import org.gbif.api.model.registry.eml.temporal.VerbatimTimePeriod;
007import org.gbif.api.model.registry.eml.temporal.VerbatimTimePeriodType;
008import org.gbif.api.vocabulary.ContactType;
009import org.gbif.registry.metadata.contact.ContactAdapter;
010
011import java.io.IOException;
012import java.io.Writer;
013import java.util.List;
014import java.util.Map;
015
016import com.google.common.base.Preconditions;
017import com.google.common.collect.ImmutableMap;
018import com.google.common.collect.Lists;
019import freemarker.template.Configuration;
020import freemarker.template.TemplateException;
021
022/**
023 * A simple tool to serialize a dataset object into an EML GBIF profile compliant xml document.
024 */
025public class EMLWriter {
026
027  private static final String TEMPLATE_PATH = "/gbif-eml-profile-template";
028  private static final String EML_TEMPLATE = String.format("eml-dataset-%s.ftl", EMLProfileVersion.GBIF_1_0_2.getVersion());
029  private final Configuration freemarkerConfig;
030  private final boolean useDoiAsIdentifier;
031
032  private static final Configuration FTL = DatasetXMLWriterConfigurationProvider.provideFreemarker(TEMPLATE_PATH);
033
034  /**
035   * Private constructor, use {@link #newInstance()}
036   * @param cfg
037   */
038  private EMLWriter(Configuration cfg, boolean useDoiAsIdentifier) {
039    this.freemarkerConfig = cfg;
040    this.useDoiAsIdentifier = useDoiAsIdentifier;
041  }
042
043  /**
044   * Get a new instance of EMLWriter with default Freemarker configuration.
045   * Same as calling {@link #newInstance(boolean)} method with useDoiAsIdentifier = false
046   *
047   * @return new instance
048   */
049  public static EMLWriter newInstance(){
050    return new EMLWriter(DatasetXMLWriterConfigurationProvider.provideFreemarker(), false);
051  }
052
053  /**
054   * Get a new instance of EMLWriter with default Freemarker configuration.
055   *
056   * @param useDoiAsIdentifier should the packageId be the dataset.doi? If true, the dataset.doi won't be included in the list of alternate identifiers
057   * @return
058   */
059  public static EMLWriter newInstance(boolean useDoiAsIdentifier){
060    return new EMLWriter(DatasetXMLWriterConfigurationProvider.provideFreemarker(), useDoiAsIdentifier);
061  }
062
063  /**
064   * Write a document from a Dataset object.
065   *
066   * @param dataset non null dataset object
067   * @param writer where the output document will go. The writer is not closed by this method.
068   * @throws IOException if an error occurs while processing the template
069   */
070  public void writeTo(Dataset dataset, Writer writer) throws IOException {
071    write(dataset, writer, useDoiAsIdentifier);
072  }
073
074
075  /**
076   * Wrapper for a dataset instance that exposes some very EML specific methods.
077   * Mostly used for generating EML, see EMLWriter.
078   * This class requires to be public to be used in the Freemarker template.
079   */
080  public static class EmlDatasetWrapper {
081
082    private final Dataset dataset;
083    private final ContactAdapter contactAdapter;
084
085    public EmlDatasetWrapper(Dataset dataset) {
086      this.dataset = dataset;
087      this.contactAdapter = new ContactAdapter(dataset.getContacts());
088    }
089
090    public List<Contact> getAssociatedParties() {
091      return contactAdapter.getAssociatedParties();
092    }
093
094    public Contact getResourceCreator() {
095      return contactAdapter.getResourceCreator();
096    }
097
098    public Contact getAdministrativeContact() {
099      return contactAdapter.getAdministrativeContact();
100    }
101
102    public Contact getMetadataProvider() {
103      return contactAdapter.getFirstPreferredType(ContactType.METADATA_AUTHOR);
104    }
105
106    public List getFormationPeriods() {
107      return getTimePeriods(VerbatimTimePeriodType.FORMATION_PERIOD);
108    }
109
110    public List getLivingTimePeriods() {
111      return getTimePeriods(VerbatimTimePeriodType.LIVING_TIME_PERIOD);
112    }
113
114    private List<VerbatimTimePeriod> getTimePeriods(VerbatimTimePeriodType type) {
115      List<VerbatimTimePeriod> periods = Lists.newArrayList();
116      for (TemporalCoverage tc : dataset.getTemporalCoverages()) {
117        if (tc instanceof VerbatimTimePeriod) {
118          VerbatimTimePeriod tp = (VerbatimTimePeriod) tc;
119          if (type.equals(tp.getType())) {
120            periods.add(tp);
121          }
122        }
123      }
124      return periods;
125    }
126
127  }
128
129  /**
130   * This method will be deprecated soon, please use an instance {@link #newInstance()}
131   *
132   * Same as calling {@link #write(Dataset, Writer, boolean) write} method with useDoiAsIdentifier = false.
133   *
134   * @param dataset
135   * @param writer
136   * @throws IOException
137   */
138  public static void write(Dataset dataset, Writer writer) throws IOException {
139    write(dataset,writer,false);
140  }
141
142  /**
143   * This method will be deprecated soon, please use an instance {@link #newInstance()}
144   *
145   * Write an EML document from a Dataset object.
146   *
147   * @param dataset non null dataset object
148   * @param writer where the output document will go. The writer is not closed by this method.
149   * @param _useDoiAsIdentifier should the packageId be the dataset.doi? If true, the dataset.doi won't be included in the list of alternate identifiers.
150   * @throws IOException if an error occurs while processing the template
151   */
152  public static void write(Dataset dataset, Writer writer, boolean _useDoiAsIdentifier) throws IOException {
153    Preconditions.checkNotNull(dataset, "Dataset can't be null");
154
155    Map<String, Object> map = ImmutableMap.of
156            ("dataset", dataset, "eml", new EmlDatasetWrapper(dataset), "useDoiAsIdentifier", _useDoiAsIdentifier);
157
158    try {
159      FTL.getTemplate(EML_TEMPLATE).process(map, writer);
160    } catch (TemplateException e) {
161      throw new IOException("Error while processing the EML Freemarker template for dataset " + dataset.getKey(), e);
162    }
163  }
164
165}
