001package org.gbif.registry.cli;
002
003import org.gbif.api.model.common.DOI;
004import org.gbif.api.model.common.DoiData;
005import org.gbif.api.model.common.DoiStatus;
006import org.gbif.common.messaging.AbstractMessageCallback;
007import org.gbif.common.messaging.api.messages.ChangeDoiMessage;
008import org.gbif.doi.service.DoiException;
009import org.gbif.doi.service.DoiExistsException;
010import org.gbif.doi.service.DoiHttpException;
011import org.gbif.doi.service.DoiService;
012import org.gbif.doi.service.InvalidMetadataException;
013import org.gbif.registry.persistence.mapper.DoiMapper;
014import org.gbif.registry.ws.util.DataCiteConverter;
015
016import java.net.URI;
017
018import org.slf4j.Logger;
019import org.slf4j.LoggerFactory;
020import org.slf4j.Marker;
021import org.slf4j.MarkerFactory;
022
023/**
024 * Message callback implementation to take DOI updates and send them to DataCite. Updates the status of the DOI in
025 * the registry database.
026 */
027public class DoiUpdateListener extends AbstractMessageCallback<ChangeDoiMessage> {
028
029    private static final Logger LOG = LoggerFactory.getLogger(DoiUpdateListener.class);
030    private static final Marker DOI_SMTP = MarkerFactory.getMarker("DOI_SMTP");
031    private final long timeToRetryInMs;
032
033    private final DoiService doiService;
034    private final DoiMapper doiMapper;
035
036    public DoiUpdateListener(DoiService doiService, DoiMapper doiMapper, long timeToRetryInMs) {
037        this.doiService = doiService;
038        this.doiMapper = doiMapper;
039        this.timeToRetryInMs = timeToRetryInMs;
040    }
041
042    @Override
043    public void handleMessage(ChangeDoiMessage msg) {
044        LOG.debug("Handling change DOI to {} message for {}", msg.getStatus(), msg.getDoi());
045        final DoiData currState = doiMapper.get(msg.getDoi());
046        if (currState == null) {
047            // this is bad, we should have an entry for the DOI in our registry table!
048            LOG.warn("Skipping unknown GBIF DOI {}", msg.getDoi());
049            return;
050        }
051
052        boolean descriptionTruncated = false;
053        for (int retry = 1; retry < 4; retry++) {
054            try {
055                switch (msg.getStatus()) {
056                    case REGISTERED:
057                        registerOrUpdate(msg.getDoi(), msg.getTarget(), msg.getMetadata(), currState);
058                        break;
059                    case RESERVED:
060                        reserve(msg.getDoi(), msg.getMetadata(), currState);
061                        break;
062                    case DELETED:
063                        delete(msg.getDoi(), currState);
064                        break;
065                    default:
066                        LOG.warn("Cannot update {} to illegal state {}.", msg.getDoi(), msg.getStatus());
067                        break;
068                }
069                break;
070
071            } catch (DoiExistsException e) {
072                writeFailedStatus(msg.getDoi(), msg.getTarget(), msg.getMetadata());
073                LOG.warn(DOI_SMTP, "DOI {} existed already when trying to change status to {}. Ignore", msg.getDoi(), msg.getStatus(), e);
074                break;
075
076            } catch (DoiHttpException e) {
077                writeFailedStatus(msg.getDoi(), msg.getTarget(), msg.getMetadata());
078                if (e.getStatus() == 413) {
079                    LOG.warn(DOI_SMTP, "Metadata of length {} is exceeding max datacite limit in attempt #{} "
080                                    + "while updating {} to {} with target {}. "
081                                    + "Trying again {}",
082                            msg.getMetadata().length(), retry, msg.getDoi(), msg.getStatus(), msg.getTarget(),
083                            descriptionTruncated ? "without constituent information" : "with truncated description", e);
084                    try {
085                        String truncatedXml;
086                        if (descriptionTruncated) {
087                            LOG.warn("Truncating all constituent relations as last resort from metadata for DOI {}", msg.getDoi());
088                            truncatedXml = DataCiteConverter.truncateConstituents(msg.getDoi(), msg.getMetadata(), msg.getTarget());
089                        } else {
090                            LOG.debug("Original metadata for DOI {}:\n\n{}", msg.getDoi(), msg.getMetadata());
091                            truncatedXml = DataCiteConverter.truncateDescription(msg.getDoi(), msg.getMetadata(), msg.getTarget());
092                            descriptionTruncated = true;
093                        }
094                        msg = new ChangeDoiMessage(msg.getStatus(), msg.getDoi(), truncatedXml, msg.getTarget());
095                    } catch (InvalidMetadataException e1) {
096                        LOG.warn("Failed to deserialize xml metadata for DOI {}", msg.getDoi(), e1);
097                    }
098                } else {
099                    LOG.warn(DOI_SMTP, "DOI http {} exception updating {} to {} with target {}. Attempt #{}",
100                            e.getStatus(), msg.getDoi(), msg.getStatus(), msg.getTarget(), retry, e);
101                    sleep();
102                }
103
104            } catch (DoiException e) {
105                writeFailedStatus(msg.getDoi(), msg.getTarget(), msg.getMetadata());
106                LOG.warn(DOI_SMTP, "DOI exception updating {} to {} with target {}. Attempt #{}",
107                        msg.getDoi(), msg.getStatus(), msg.getTarget(), retry, e);
108                sleep();
109            }
110        }
111    }
112
113    private void sleep() {
114        try {
115            Thread.sleep(timeToRetryInMs);
116        } catch (InterruptedException e1) {
117            LOG.info("Interrupted retries");
118        }
119    }
120
121    private void writeFailedStatus(DOI doi, URI target, String xml) {
122        doiMapper.update(doi, new DoiData(DoiStatus.FAILED, target), xml);
123    }
124
125    private void reserve(DOI doi, String xml, DoiData currState) throws DoiException {
126        doiService.reserve(doi, xml);
127        LOG.info("Reserved doi {}", doi);
128        DoiData newState = new DoiData(DoiStatus.RESERVED, currState.getTarget());
129        doiMapper.update(doi, newState, xml);
130    }
131
132    private void delete(DOI doi, DoiData currState) throws DoiException {
133        if (currState.getStatus() == null) {
134            doiMapper.delete(doi);
135        } else {
136            try {
137                boolean fullDeleted = doiService.delete(doi);
138                if (fullDeleted) {
139                    doiMapper.delete(doi);
140                    LOG.info("Deleted doi {}", doi);
141                } else {
142                    DoiData newState = new DoiData(DoiStatus.DELETED, currState.getTarget());
143                    doiMapper.update(doi, newState, null);
144                    LOG.info("Marked registered doi {} as deleted", doi);
145                }
146            } catch (DoiHttpException e) {
147                // in case of a 404 swallow
148                if (e.getStatus() == 404) {
149                    LOG.warn(DOI_SMTP, "Trying to delete DOI {} failed because it doesn't exist in DataCite; deleting locally",
150                            doi, e);
151                    doiMapper.delete(doi);
152                } else {
153                    throw e;
154                }
155            }
156        }
157    }
158
159    private void registerOrUpdate(DOI doi, URI target, String xml, DoiData currState) throws DoiException {
160        if (DoiStatus.REGISTERED == currState.getStatus()) {
161            // the DOI was already registered, so we only need to update the target url if changed and the metadata
162            if (!target.equals(currState.getTarget())) {
163                doiService.update(doi, target);
164            }
165            doiService.update(doi, xml);
166            LOG.info("Updated doi {} with target {}", doi, target);
167        } else {
168            doiService.register(doi, target, xml);
169            LOG.info("Registered doi {} with target {}", doi, target);
170        }
171        // store the new state in our registry
172        doiMapper.update(doi, new DoiData(DoiStatus.REGISTERED, target), xml);
173    }
174}
