001package org.gbif.registry.ws.security;
002
003import java.security.Principal;
004import java.util.UUID;
005
006public interface EditorAuthorizationService {
007
008    /**
009     * Checks whether a given user is allowed to modify machine tags belonging to the given namespace.
010     * @param user the security context user principal
011     * @param ns the namespace in question
012     * @return true if the user is allowed to modify the namespace
013     */
014  boolean allowedToModifyNamespace(Principal user, String ns);
015
016    /**
017     * @return true if rights exist for this user to delete the tag
018     */
019  boolean allowedToDeleteMachineTag(Principal user, int machineTagKey);
020  /**
021   * Checks whether a given registry entity is explicitly part of the list of entities for an editor user.
022   * @param user the security context user principal
023   * @param key the entity in question
024   * @return true if the passed entity is allowed to be modified
025   */
026  boolean allowedToModifyEntity(Principal user, UUID key);
027
028  boolean allowedToModifyDataset(Principal user, UUID datasetKey);
029
030  boolean allowedToModifyOrganization(Principal user, UUID orgKey);
031
032  boolean allowedToModifyInstallation(Principal user, UUID installationKey);
033}
