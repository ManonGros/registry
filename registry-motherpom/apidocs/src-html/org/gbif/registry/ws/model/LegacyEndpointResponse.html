001package org.gbif.registry.ws.model;
002
003import org.gbif.api.model.registry.Endpoint;
004import org.gbif.registry.ws.util.LegacyResourceConstants;
005
006import java.util.UUID;
007import javax.validation.constraints.NotNull;
008import javax.xml.bind.annotation.XmlElement;
009import javax.xml.bind.annotation.XmlRootElement;
010
011/**
012 * Class used to generate responses for legacy (GBRDS/IPT) API.
013 * </br>
014 * JAXB annotations allow the class to be converted into an XML document or JSON response. @XmlElement is used to
015 * specify element names that consumers of legacy services expect to find.
016 */
017@XmlRootElement(name = "service")
018public class LegacyEndpointResponse {
019
020  private String key;
021  private String organisationKey;
022  private String resourceKey;
023  private String description;
024  private String descriptionLanguage;
025  private String type;
026  private String typeDescription;
027  private String accessPointURL;
028
029  public LegacyEndpointResponse(Endpoint endpoint, UUID datasetKey) {
030    key = endpoint.getKey() == null ? "" : endpoint.getKey().toString();
031    type = endpoint.getType() == null ? "" : endpoint.getType().name();
032    resourceKey = datasetKey == null ? "" : datasetKey.toString();
033    description = endpoint.getDescription() == null ? "" : endpoint.getDescription();
034    accessPointURL = endpoint.getUrl() == null ? "" : endpoint.getUrl().toASCIIString();
035    // always empty - not null because they have to be included in response
036    organisationKey = "";
037    descriptionLanguage = "";
038    typeDescription = "";
039  }
040
041  /**
042   * No argument, default constructor needed by JAXB.
043   */
044  public LegacyEndpointResponse() {
045  }
046
047  @XmlElement(name = LegacyResourceConstants.KEY_PARAM)
048  @NotNull
049  public String getKey() {
050    return key;
051  }
052
053  public void setKey(String key) {
054    this.key = key;
055  }
056
057  @XmlElement(name = LegacyResourceConstants.RESOURCE_KEY_PARAM)
058  @NotNull
059  public String getResourceKey() {
060    return resourceKey;
061  }
062
063  public void setResourceKey(String resourceKey) {
064    this.resourceKey = resourceKey;
065  }
066
067  @XmlElement(name = LegacyResourceConstants.ORGANIZATION_KEY_PARAM)
068  @NotNull
069  public String getOrganisationKey() {
070    return organisationKey;
071  }
072
073  public void setOrganisationKey(String organisationKey) {
074    this.organisationKey = organisationKey;
075  }
076
077  @XmlElement(name = LegacyResourceConstants.TYPE_PARAM)
078  @NotNull
079  public String getType() {
080    return type;
081  }
082
083  public void setType(String type) {
084    this.type = type;
085  }
086
087  @XmlElement(name = LegacyResourceConstants.TYPE_DESCRIPTION_PARAM)
088  @NotNull
089  public String getTypeDescription() {
090    return typeDescription;
091  }
092
093  public void setTypeDescription(String typeDescription) {
094    this.typeDescription = typeDescription;
095  }
096
097  @XmlElement(name = LegacyResourceConstants.DESCRIPTION_PARAM)
098  @NotNull
099  public String getDescription() {
100    return description;
101  }
102
103  public void setDescription(String description) {
104    this.description = description;
105  }
106
107  @XmlElement(name = LegacyResourceConstants.DESCRIPTION_LANGUAGE_PARAM)
108  @NotNull
109  public String getDescriptionLanguage() {
110    return descriptionLanguage;
111  }
112
113  public void setDescriptionLanguage(String descriptionLanguage) {
114    this.descriptionLanguage = descriptionLanguage;
115  }
116
117  @XmlElement(name = LegacyResourceConstants.ACCESS_POINT_URL_PARAM)
118  @NotNull
119  public String getAccessPointURL() {
120    return accessPointURL;
121  }
122
123  public void setAccessPointURL(String accessPointURL) {
124    this.accessPointURL = accessPointURL;
125  }
126}
