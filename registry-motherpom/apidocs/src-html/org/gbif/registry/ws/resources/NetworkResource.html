031import java.util.UUID;

033import javax.annotation.Nullable;

043import com.google.common.base.Strings;

047import static org.gbif.registry.ws.security.UserRoles.ADMIN_ROLE;
048
049/**
050 * A MyBATIS implementation of the service.
051 */
052@Path("network")
053@Singleton
054public class NetworkResource extends BaseNetworkEntityResource<Network> implements NetworkService {
055
056  private final DatasetMapper datasetMapper;
057  private final NetworkMapper networkMapper;
058
059  @Inject
060  public NetworkResource(
061    NetworkMapper networkMapper,
062    ContactMapper contactMapper,
063    EndpointMapper endpointMapper,
064    IdentifierMapper identifierMapper,
065    MachineTagMapper machineTagMapper,
066    TagMapper tagMapper,
067    CommentMapper commentMapper,
068    DatasetMapper datasetMapper,
069    EventBus eventBus,
070    EditorAuthorizationService userAuthService) {
071    super(networkMapper,
072      commentMapper,
073      contactMapper,
074      endpointMapper,
075      identifierMapper,
076      machineTagMapper,
077      tagMapper,
078      Network.class,
079      eventBus,
080      userAuthService);
081    this.datasetMapper = datasetMapper;
082    this.networkMapper = networkMapper;
083  }
084
085
086  /**
087   * All network entities support simple (!) search with "&q=".
088   * This is to support the console user interface, and is in addition to any complex, faceted search that might
089   * additionally be supported, such as dataset search.
090   */
091  @GET
092  public PagingResponse<Network> list(@Nullable @QueryParam("q") String query,
093    @Nullable @QueryParam("identifierType") IdentifierType identifierType,
094    @Nullable @QueryParam("identifier") String identifier,
095    @Nullable @Context Pageable page) {
096    // This is getting messy: http://dev.gbif.org/issues/browse/REG-426
097    if (identifierType != null && identifier != null) {
098      return listByIdentifier(identifierType, identifier, page);
099    } else if (identifier != null) {
100      return listByIdentifier(identifier, page);
101    } else if (Strings.isNullOrEmpty(query)) {
102      return list(page);
103    } else {
104      return search(query, page);
105    }
106  }
107
108  @Path("{key}/constituents")
109  @GET
110  @Override
111  public PagingResponse<Dataset> listConstituents(@PathParam("key") UUID networkKey, @Context Pageable page) {
112    return pagingResponse(page, (long) networkMapper.countDatasetsInNetwork(networkKey),
113      datasetMapper.listDatasetsInNetwork(networkKey, page));
114  }
115
116  @Path("{key}/constituents/{datasetKey}")
117  @POST
118  @RolesAllowed(ADMIN_ROLE)
119  @Override
120  public void addConstituent(@PathParam("key") UUID networkKey, @PathParam("datasetKey") UUID datasetKey) {
121    networkMapper.addDatasetConstituent(networkKey, datasetKey);
122  }
123
124
125  @Path("{key}/constituents/{datasetKey}")
126  @DELETE
127  @RolesAllowed(ADMIN_ROLE)
128  @Override
129  public void removeConstituent(@PathParam("key") UUID networkKey, @PathParam("datasetKey") UUID datasetKey) {
130    networkMapper.deleteDatasetConstituent(networkKey, datasetKey);
131  }
132
133}
