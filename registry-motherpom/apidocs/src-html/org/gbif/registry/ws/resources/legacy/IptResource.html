061  /**
062   * Register IPT installation, handling incoming request with path /ipt/register. The primary contact and hosting
063   * organization key are mandatory. Only after both the installation and primary contact have been persisted is a
064   * Response with Status.CREATED returned.
065   *
066   * @param installation IptInstallation with HTTP form parameters having been injected from Jersey
067   * @param security SecurityContext (security related information)
068   * @return Response with Status.CREATED if successful
069   */
070  @POST
071  @Path("register")
072  @Produces(MediaType.APPLICATION_XML)
073  @Consumes(MediaType.APPLICATION_FORM_URLENCODED)
074  public Response registerIpt(@InjectParam LegacyInstallation installation, @Context SecurityContext security) {
075    if (installation != null && security.getUserPrincipal() != null) {
076      // set required fields
077      String user = security.getUserPrincipal().getName();
078      installation.setCreatedBy(user);
079      installation.setModifiedBy(user);
080      // add contact and endpoint to installation
081      installation.prepare();
082      // primary contact and hosting organization key are mandatory
083      Contact primary = installation.getPrimaryContact();
084      if (primary != null && LegacyResourceUtils.isValid(installation, organizationService)) {
085        // persist installation
086        UUID key = installationService.create(installation.toApiInstallation());
087        // persist contact
088        if (key != null) {
089          // set primary contact's required fields
090          primary.setCreatedBy(user);
091          primary.setModifiedBy(user);
092          // persist primary contact
093          installationService.addContact(key, primary);
094          // try to persist FEED endpoint (non-mandatory)
095          Endpoint endpoint = installation.getFeedEndpoint();
096          if (endpoint != null) {
097            // set endpoint's required fields
098            endpoint.setCreatedBy(user);
099            endpoint.setModifiedBy(user);
100            installationService.addEndpoint(key, endpoint);
101          }
102          LOG.info("IPT installation registered successfully, key=%s", key.toString());
103          // construct GenericEntity response object expected by IPT
104          GenericEntity<IptEntityResponse> entity =
105            new GenericEntity<IptEntityResponse>(new IptEntityResponse(key.toString())) {
106            };
107          // return Response
108          return Response.status(Response.Status.CREATED).cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED)
109            .entity(entity).build();
110        } else {
111          LOG.error("IPT installation could not be persisted!");
112        }
113      } else {
114        LOG.error("Mandatory primary contact and/or hosting organization key missing or incomplete!");
115      }
116    }
117    LOG.error("IPT installation registration failed");
118    return Response.status(Response.Status.BAD_REQUEST).cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED)
119      .build();
120  }

123  /**
124   * Update IPT installation, handling incoming request with path /ipt/update/{key}. The primary contact and hosting
125   * organization key are mandatory. Only after both the installation and primary contact have been updated is a
126   * Response with Status.CREATED returned.
127   *
128   * @param installationKey installation key (UUID) coming in as path param
129   * @param installation IptInstallation with HTTP form parameters having been injected from Jersey
130   * @param security SecurityContext (security related information)
131   * @return Response with Status.CREATED if successful
132   */
133  @POST
134  @Path("update/{key}")
135  @Consumes(MediaType.APPLICATION_FORM_URLENCODED)
136  public Response updateIpt(@PathParam("key") UUID installationKey, @InjectParam LegacyInstallation installation,
137    @Context SecurityContext security) {
138    if (installation != null && installationKey != null) {
139      // set required fields
140      String user = security.getUserPrincipal().getName();
141      installation.setCreatedBy(user);
142      installation.setModifiedBy(user);
143      // set key from path parameter
144      installation.setKey(installationKey);
145      // retrieve existing installation
146      Installation existing = installationService.get(installationKey);
147      // populate installation with existing primary contact so it gets updated, not duplicated
148      installation.setContacts(existing.getContacts());
149      // add contact and endpoint to installation
150      installation.prepare();
151      // ensure the hosting organization exists, and primary contact exists
152      Contact contact = installation.getPrimaryContact();
153      if (contact != null && LegacyResourceUtils
154        .isValidOnUpdate(installation, installationService, organizationService)) {
155        // update only fields that could have changed
156        existing.setModifiedBy(user);
157        existing.setTitle(installation.getTitle());
158        existing.setDescription(installation.getDescription());
159        existing.setType(installation.getType());
160        existing.setOrganizationKey(installation.getOrganizationKey());
161
162        // persist changes
163        installationService.update(existing);
164
165        // set primary contact's required field(s)
166        contact.setModifiedBy(user);
167        // add/update primary contact: Contacts are mutable, so try to update if the Contact already exists
168        if (contact.getKey() == null) {
169          contact.setCreatedBy(user);
170          installationService.addContact(installationKey, contact);
171        } else {
172          installationService.updateContact(installationKey, contact);
173        }
174
175        // try to persist FEED endpoint (non-mandatory): Endpoints not mutable, so delete all then re-add
176        List<Endpoint> endpoints = installationService.listEndpoints(installationKey);
177        for (Endpoint endpoint : endpoints) {
178          installationService.deleteEndpoint(installationKey, endpoint.getKey());
179        }
180        Endpoint endpoint = installation.getFeedEndpoint();
181        if (endpoint != null) {
182          // set endpoint's required fields
183          endpoint.setCreatedBy(user);
184          endpoint.setModifiedBy(user);
185          installationService.addEndpoint(installationKey, endpoint);
186        }
187
188        LOG.info("IPT installation updated successfully, key={}", installationKey.toString());
189        return Response.noContent().cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED).build();
190      } else {
191        LOG.error("Mandatory primary contact and/or hosting organization key missing or incomplete!");
192      }
193    }
194    LOG.error("IPT installation update failed");
195    return Response.status(Response.Status.BAD_REQUEST).cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED)
196      .build();
197  }

199  /**
200   * Register IPT dataset, handling incoming request with path /ipt/resource. The primary contact and publishing
201   * organization key are mandatory. Only after both the dataset and primary contact have been persisted is a
202   * Response with Status.CREATED returned.
203   *
204   * @param dataset LegacyDataset with HTTP form parameters having been injected from Jersey
205   * @param security SecurityContext (security related information)
206   * @return Response with Status.CREATED if successful
207   */
208  @POST
209  @Path("resource")
210  @Produces(MediaType.APPLICATION_XML)
211  @Consumes(MediaType.APPLICATION_FORM_URLENCODED)
212  public Response registerDataset(@InjectParam LegacyDataset dataset, @Context SecurityContext security) {
213    if (dataset != null) {
214      // set required fields
215      String user = security.getUserPrincipal().getName();
216      dataset.setCreatedBy(user);
217      dataset.setModifiedBy(user);
218      // if the installation key was missing, try to infer it from publishing organization's installations
219      if (dataset.getInstallationKey() == null) {
220        dataset.setInstallationKey(inferInstallationKey(dataset));
221      }
222      // add contact and endpoint(s) to dataset
223      dataset.prepare();
224      // primary contact, publishing organization key, and installationKey are mandatory
225      Contact contact = dataset.getPrimaryContact();
226      if (contact != null && LegacyResourceUtils.isValid(dataset, organizationService, installationService)) {
227        // persist dataset
228        UUID key = datasetService.create(dataset.toApiDataset());
229        // persist contact
230        if (key != null) {
231          // set primary contact's required fields
232          contact.setCreatedBy(user);
233          contact.setModifiedBy(user);
234          // add primary contact
235          datasetService.addContact(key, contact);
236          // try to persist endpoint(s) (non-mandatory)
237          Endpoint emlEndpoint = dataset.getEmlEndpoint();
238          if (emlEndpoint != null) {
239            // set endpoint's required fields
240            emlEndpoint.setCreatedBy(user);
241            emlEndpoint.setModifiedBy(user);
242            datasetService.addEndpoint(key, emlEndpoint);
243          }
244          Endpoint archiveEndpoint = dataset.getArchiveEndpoint();
245          if (archiveEndpoint != null) {
246            // set endpoint's required fields
247            archiveEndpoint.setCreatedBy(user);
248            archiveEndpoint.setModifiedBy(user);
249            datasetService.addEndpoint(key, archiveEndpoint);
250          }
251          LOG.info("Dataset registered successfully, key=%s", key.toString());
252          // construct response object expected by IPT
253          IptEntityResponse entity = new IptEntityResponse(key.toString());
254          // return Response
255          return Response.status(Response.Status.CREATED).cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED)
256            .entity(entity).build();
257        } else {
258          LOG.error("Dataset could not be persisted!");
259        }
260      } else {
261        LOG.error("Mandatory primary contact and/or publishing organization key missing or incomplete!");
262      }
263    }
264    LOG.error("Dataset registration failed");
265    return Response.status(Response.Status.BAD_REQUEST).cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED)
266      .build();
267  }

270  /**
271   * Update IPT Dataset, handling incoming request with path /ipt/resource/{key}. The publishing organization key is
272   * mandatory (supplied in the credentials not the parameters). The contacts are preserved from the existing dataset,
273   * careful not to duplicate contacts. Only after both the dataset and primary contact have been updated is a Response
274   * with Status.OK returned.
275   *
276   * @param datasetKey dataset key (UUID) coming in as path param
277   * @param dataset LegacyDataset with HTTP form parameters having been injected from Jersey
278   * @param security SecurityContext (security related information)
279   * @return with Status.CREATED (201) if successful
280   */
281  @POST
282  @Path("resource/{key}")
283  @Consumes(MediaType.APPLICATION_FORM_URLENCODED)
284  public Response updateDataset(@PathParam("key") UUID datasetKey, @InjectParam LegacyDataset dataset,
285    @Context SecurityContext security) {
286    if (dataset != null) {
287      // set required fields
288      String user = security.getUserPrincipal().getName();
289      dataset.setCreatedBy(user);
290      dataset.setModifiedBy(user);
291      dataset.setKey(datasetKey);
292      // retrieve existing dataset
293      Dataset existing = datasetService.get(datasetKey);
294      // populate dataset with existing primary contact so it gets updated, not duplicated
295      dataset.setContacts(existing.getContacts());
296      // update primary contact, endpoint(s), and type
297      dataset.prepare();
298      // retrieve existing dataset's installation key if it wasn't provided
299      if (dataset.getInstallationKey() == null) {
300        dataset.setInstallationKey(existing.getInstallationKey());
301      }
302      // Dataset can only have 1 installation key, log if the hosting installation is being changed
303      // Reason: IPT versions before 2.0.5 didn't supply the parameter iptKey on dataset updates
304      else if (dataset.getInstallationKey() != existing.getInstallationKey()) {
305        LOG.debug("The dataset's technical installation is being changed from {} to {}", dataset.getInstallationKey(), existing.getInstallationKey());
306      }
307      // populate publishing organization from credentials
308      dataset.setPublishingOrganizationKey( LegacyAuthorizationFilter.extractOrgKeyFromSecurity(security) );
309      // ensure the publishing organization exists, the installation exists, primary contact exists, etc
310      Contact contact = dataset.getPrimaryContact();
311      if (contact != null && LegacyResourceUtils
312        .isValidOnUpdate(dataset, datasetService, organizationService, installationService)) {
313        // update only fields that could have changed
314        existing.setModifiedBy(user);
315        existing.setTitle(dataset.getTitle());
316        existing.setDescription(dataset.getDescription());
317        existing.setHomepage(dataset.getHomepage());
318        existing.setLogoUrl(dataset.getLogoUrl());
319        existing.setType(dataset.getType());
320        existing.setInstallationKey(dataset.getInstallationKey());
321
322        existing.setPublishingOrganizationKey(dataset.getPublishingOrganizationKey());
323
324        // persist changes
325        datasetService.update(existing);
326
327        // set primary contact's required field(s)
328        contact.setModifiedBy(user);
329        // add/update primary contact: Contacts are mutable, so try to update if the Contact already exists
330        if (contact.getKey() == null) {
331          contact.setCreatedBy(user);
332          datasetService.addContact(datasetKey, contact);
333        } else {
334          datasetService.updateContact(datasetKey, contact);
335        }
336
337        // try to persist eml and archive endpoint(s) (non-mandatory): Endpoints not mutable, so delete all then re-add
338        List<Endpoint> endpoints = datasetService.listEndpoints(datasetKey);
339        for (Endpoint endpoint : endpoints) {
340          datasetService.deleteEndpoint(datasetKey, endpoint.getKey());
341        }
342        Endpoint emlEndpoint = dataset.getEmlEndpoint();
343        if (emlEndpoint != null) {
344          // set endpoint's required fields
345          emlEndpoint.setCreatedBy(user);
346          emlEndpoint.setModifiedBy(user);
347          datasetService.addEndpoint(datasetKey, emlEndpoint);
348        }
349        Endpoint archiveEndpoint = dataset.getArchiveEndpoint();
350        if (archiveEndpoint != null) {
351          // set endpoint's required fields
352          archiveEndpoint.setCreatedBy(user);
353          archiveEndpoint.setModifiedBy(user);
354          datasetService.addEndpoint(datasetKey, archiveEndpoint);
355        }
356
357        LOG.info("Dataset updated successfully, key=%s", datasetKey.toString());
358        return Response.noContent().cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED).build();
359      } else {
360        LOG.error("Request invalid. Dataset missing required fields or using stale keys!");
361      }
362    }
363    LOG.error("Dataset update failed");
364    return Response.status(Response.Status.BAD_REQUEST).cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED)
365      .build();
366  }

368  /**
369   * Delete IPT Dataset, handling incoming request with path /ipt/resource/{key}. Only credentials are mandatory.
370   * If deletion is successful, returns Response with Status.OK.
371   *
372   * @param datasetKey dataset key (UUID) coming in as path param
373   * @return Response with Status.OK if successful
374   */
375  @DELETE
376  @Path("resource/{key}")
377  @Consumes(MediaType.APPLICATION_FORM_URLENCODED + ";charset=UTF-8")
378  public Response deleteDataset(@PathParam("key") UUID datasetKey) {
379
380    if (datasetKey != null) {
381      // retrieve existing dataset
382      Dataset existing = datasetService.get(datasetKey);
383      if (existing != null) {
384
385        // logically delete dataset. Contacts and endpoints remain, referring to logically deleted dataset
386        datasetService.delete(datasetKey);
387
388        LOG.info("Dataset deleted successfully, key=%s", datasetKey.toString());
389        return Response.status(Response.Status.OK).cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED).build();
390
391      } else {
392        LOG.error("Request invalid. Dataset to be deleted no longer exists!");
393      }
394    }
395    LOG.error("Dataset delete failed");
396    return Response.status(Response.Status.BAD_REQUEST).cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED)
397      .build();
398  }

400  /**
401   * This method tries to infer the Dataset's installation key (when it is missing). Inference is done, using the rule
402   * that if the dataset's publishing organization only has 1 installation, this must be the installation that serves
403   * the dataset. Conversely, if the organization has more or less than 1 installation, no inference can be made, and
404   * null is returned instead.
405   *
406   * @param dataset LegacyDataset with HTTP form parameters having been injected from Jersey
407   * @return inferred installation key, or null if none inferred
408   */
409  private UUID inferInstallationKey(LegacyDataset dataset) {
410    if (dataset.getInstallationKey() == null) {
411      UUID organizationKey = dataset.getPublishingOrganizationKey();
412      if (organizationKey != null) {
413        PagingRequest page = new PagingRequest(0, 2);
414        PagingResponse<Installation> response = organizationService.installations(organizationKey, page);
415        // there is 1, and only 1 installation?
416        if (ONE.equals(response.getCount())) {
417          Installation installation = response.getResults().get(0);
418          if (installation != null) {
419            LOG.info("The installation key was inferred successfully from publishing organization's single installation");
420            return installation.getKey();
421          }
422        }
423      }
424    }
425    LOG.error("The installation key could not be inferred from publishing organization!");
426    return null;
427  }
