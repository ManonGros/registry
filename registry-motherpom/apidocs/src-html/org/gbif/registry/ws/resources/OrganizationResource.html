065  protected static final int MINIMUM_PASSWORD_SIZE = 6;
066  protected static final int MAXIMUM_PASSWORD_SIZE = 15;
067  private final static String PASSWORD_ALLOWED_CHARACTERS = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ1234567890";

100  /**
101   * This method overrides the create method for an organization, populating the password field with a randomly
102   * generated string before passing on to the superclass create method.
103   *
104   * @param organization organization
105   * @param security     SecurityContext (security related information)
106   *
107   * @return key of entity created
108   */
109  @POST
110  @RolesAllowed({ADMIN_ROLE, EDITOR_ROLE})
111  @Override
112  public UUID create(@NotNull Organization organization, @Context SecurityContext security) {
113    organization.setPassword(generatePassword());
114    return super.create(organization, security);
115  }

117  /**
118   * All network entities support simple (!) search with "&q=".
119   * This is to support the console user interface, and is in addition to any complex, faceted search that might
120   * additionally be supported, such as dataset search.
121   * Organizations can also be filtered by their country, but a search and country filter cannot be combined.
122   */
123  @GET
124  public PagingResponse<Organization> list(@Nullable @Context Country country,
125    @Nullable @QueryParam("identifierType") IdentifierType identifierType,
126    @Nullable @QueryParam("identifier") String identifier,
127    @Nullable @QueryParam("q") String query,
128    @Nullable @Context Pageable page) {
129    // This is getting messy: http://dev.gbif.org/issues/browse/REG-426
130    if (country != null) {
131      return listByCountry(country, page);
132    } else if (identifierType != null && identifier != null) {
133      return listByIdentifier(identifierType, identifier, page);
134    } else if (identifier != null) {
135      return listByIdentifier(identifier, page);
136    } else if (!Strings.isNullOrEmpty(query)) {
137      return search(query, page);
138    } else {
139      return list(page);
140    }
141  }

205  /**
206   * This is an HTTP only method to retrieve the password for an organization.
207   *
208   * @param organizationKey organization key
209   *
210   * @return password if set, warning message if not set, or null if organization doesn't exist
211   */
212  @Path("{key}/password")
213  @GET
214  @RolesAllowed(ADMIN_ROLE)
215  @Produces(MediaType.TEXT_PLAIN)
216  public String retrievePassword(@PathParam("key") UUID organizationKey) {
217    Organization o = get(organizationKey);
218    if (o == null) {
219      return null;
220    }
221    // Organization.password is never null according to database schema. API doesn't mirror this though.
222    return o.getPassword();
223  }
224
225  /**
226   * Randomly generates a password for an organization.
227   *
228   * @return generated password
229   */
230  @VisibleForTesting
231  protected static String generatePassword() {
232    Random random = new Random();
233    // randomly calculate the size of the password, between 0 and MAXIMUM_PASSWORD_SIZE
234    int size = random.nextInt(MAXIMUM_PASSWORD_SIZE);
235    // ensure the size is at least greater than or equal to MINIMUM_PASSWORD_SIZE
236    size = (size < MINIMUM_PASSWORD_SIZE) ? MINIMUM_PASSWORD_SIZE : size;
237
238    // generate the password
239    StringBuilder password = new StringBuilder();
240    int randomIndex;
241    while (size-- > 0) {
242      random = new Random();
243      randomIndex = random.nextInt(PASSWORD_ALLOWED_CHARACTERS.length());
244      password.append(PASSWORD_ALLOWED_CHARACTERS.charAt(randomIndex));
245    }
246
247    return password.toString();
248  }
