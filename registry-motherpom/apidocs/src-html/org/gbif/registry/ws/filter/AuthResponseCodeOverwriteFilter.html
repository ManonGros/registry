001package org.gbif.registry.ws.filter;
002
003import javax.ws.rs.core.Response;
004
005import com.sun.jersey.spi.container.ContainerRequest;
006import com.sun.jersey.spi.container.ContainerResponse;
007import com.sun.jersey.spi.container.ContainerResponseFilter;
008
009/**
010 * A filter that exists to overwrite 401 response codes with a 403 if the client instructs it to by using the header
011 * gbif-prefer-403-over-401= true.
012 * This exists because ajax based clients (e.g. Angular, JQuery etc) do not get the response before the browser pops up
013 * the Authentication window, which happens on a 401 response. This is a known limitation and this is relatively common
014 * practice.
015 */
016public class AuthResponseCodeOverwriteFilter implements ContainerResponseFilter {
017
018  private static String REQUEST_HEADER_OVERWRITE = "gbif-prefer-403-over-401";
019
020  @Override
021  public ContainerResponse filter(ContainerRequest request, ContainerResponse response) {
022    if (Boolean.valueOf(request.getHeaderValue(REQUEST_HEADER_OVERWRITE))
023      && Response.Status.UNAUTHORIZED.getStatusCode() == response.getStatus()) {
024      response.setStatus(Response.Status.FORBIDDEN.getStatusCode());
025    }
026    return response;
027  }
028}
