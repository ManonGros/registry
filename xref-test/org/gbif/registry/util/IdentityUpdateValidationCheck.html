24  /**
25   * This class does the identity function update for entities in the registry. That is to say that it reads and updates
26   * all entities.
27   * Because the database was migrated, and we have Java validation rules, this will ensure that all migrated data can be
28   * used in updates.
29   */
30  public class IdentityUpdateValidationCheck {
31  
32    private static final Logger LOG = LoggerFactory.getLogger(IdentityUpdateValidationCheck.class);
33  
34    /**
35     * Runs the validation. To run, add the necessary properties as arguments. The key and password are
36     * authenticated by the web services, so check they match a key/password combination the web services' know of.
37     *
38     * @param args Base url
39     */
40    public static void main(String[] args) {
41      Properties p = new Properties();
42      p.put("registry.ws.url", args[0]);
43      p.setProperty("application.key", args[1]);
44      p.setProperty("application.secret", args[2]);
45      // Create authentication module, and set principal name, equal to a GBIF User unique account name
46      GbifApplicationAuthModule auth = new GbifApplicationAuthModule(p);
47      auth.setPrincipal("admin");
48      Injector injector = Guice.createInjector(new RegistryWsClientModule(p), auth);
49  
50      LOG.info("Starting Node tests");
51      int nodeErrorCount = verifyEntity(injector.getInstance(NodeService.class));
52      LOG.info("Node tests produced {} errors", nodeErrorCount);
53  
54      LOG.info("Starting Organization tests");
55      int organisationErrorCount = verifyEntity(injector.getInstance(OrganizationService.class));
56      LOG.info("Organization tests produced {} errors", organisationErrorCount);
57  
58      LOG.info("Starting Installation tests");
59      int installationErrorCount = verifyEntity(injector.getInstance(InstallationService.class));
60      LOG.info("Installation tests produced {} errors", installationErrorCount);
61  
62      LOG.info("Starting Dataset tests");
63      int datasetErrorCount = verifyEntity(injector.getInstance(DatasetService.class));
64      LOG.info("Dataset tests produced {} errors", datasetErrorCount);
65  
66      LOG.info("Starting Network tests");
67      int networkErrorCount = verifyEntity(injector.getInstance(NetworkService.class));
68      LOG.info("Network tests produced {} errors", networkErrorCount);
69  
70      // summarise again jsut for ease of reading
71      LOG.info("Test results:");
72      LOG.info("Node tests produced {} errors", nodeErrorCount);
73      LOG.info("Organization tests produced {} errors", organisationErrorCount);
74      LOG.info("Installation tests produced {} errors", installationErrorCount);
75      LOG.info("Dataset tests produced {} errors", datasetErrorCount);
76      LOG.info("Network tests produced {} errors", networkErrorCount);
77  
78    }
79  
80    private static <T extends NetworkEntity> int verifyEntity(NetworkEntityService<T> service) {
81      int errorCount = 0;
82      PagingRequest page = new PagingRequest(0, 100);
83      PagingResponse<T> response = null;
84      do {
85        response = service.list(page);
86        LOG.debug("Page with offset[{}] return[{}] records", page.getOffset(), response.getCount());
87        for (T e : response.getResults()) {
88          LOG.debug("Updating entity[{}]", e.getKey());
89          try {
90            service.update(e);
91  
92            if (e instanceof Contactable) {
93              Contactable contactable = (Contactable) e;
94              for (Contact c : contactable.getContacts()) {
95                service.updateContact(e.getKey(), c);
96              }
97            }
98  
99          } catch (Exception ex) {
100           LOG.error("Unable to update entity[{}]", e.getKey(), ex);
101           errorCount++;
102         }
103       }
104       page.nextPage();
105     } while (!response.isEndOfRecords());
106     return errorCount;
107   }
