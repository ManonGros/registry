13  package org.gbif.registry.grizzly;
14  
15  import org.gbif.registry.guice.TestRegistryWsServletListener;
16  import org.gbif.registry.search.DatasetIndexUpdateListener;
17  
18  import java.io.IOException;
19  
20  import com.google.common.base.Preconditions;
21  import com.google.common.base.Throwables;
22  import com.google.inject.servlet.GuiceFilter;
23  import com.sun.grizzly.http.embed.GrizzlyWebServer;
24  import com.sun.grizzly.http.servlet.ServletAdapter;
25  import org.apache.solr.client.solrj.SolrClient;
26  import org.junit.rules.TestRule;
27  import org.junit.runner.Description;
28  import org.junit.runners.model.Statement;
29  import org.slf4j.Logger;
30  import org.slf4j.LoggerFactory;
31  
32  /**
33   * An embedded web server using Grizzly that run the registry ws application.
34   * To allow WS tests to flush SOLR between runs, this allows the registration of the SOLR server.
35   */
36  public class RegistryServer implements TestRule {
37  
38    /**
39     * The system property one can set to override the port.
40     */
41    public static final String PORT_PROPERTY = "grizzly.port"; // to override as a system property
42  
43    /**
44     * The default port to use, should no system property be supplied.
45     */
46    public static final int DEFAULT_PORT = 7001;
47  
48    private static final Logger LOG = LoggerFactory.getLogger(RegistryServer.class);
49    private final GrizzlyWebServer webServer;
50  
51    public static final RegistryServer INSTANCE = new RegistryServer();
52  
53    private SolrClient solrClient;
54    private DatasetIndexUpdateListener datasetUpdater;
55  
56    /**
57     * Gets the port that grizzly will run on. This will either be {@link #DEFAULT_PORT} or the value supplied as a
58     * system
59     * property named {@link #PORT_PROPERTY}.
60     */
61    public static int getPort() {
62      String port = System.getProperty(PORT_PROPERTY);
63      try {
64        return (port == null) ? DEFAULT_PORT : Integer.parseInt(port);
65      } catch (NumberFormatException e) {
66        throw new IllegalArgumentException(PORT_PROPERTY + " does not hold a valid port: " + port);
67      }
68    }
69  
70    // we are a singleton instance
71    private RegistryServer() {
72      webServer = new GrizzlyWebServer(getPort());
73      ServletAdapter sa = new ServletAdapter();
74      sa.addServletListener(TestRegistryWsServletListener.class.getName());
75      sa.addFilter(new GuiceFilter(), "Guice", null);
76      sa.setContextPath("/");
77      sa.setServletPath("/");
78      webServer.addGrizzlyAdapter(sa, null);
79    }
80  
81    public void start() {
82      LOG.info("Starting registry WS for tests");
83      try {
84        webServer.start();
85      } catch (IOException e) {
86        Throwables.propagate(e);
87      }
88    }
89  
90    public void stop() {
91      webServer.stop();
92      LOG.info("Stopping test registry WS");
93    }
94  
95    public synchronized void setSolrClient(SolrClient solrClient) {
96      this.solrClient = solrClient;
97    }
98  
99    public synchronized void setDatasetUpdater(DatasetIndexUpdateListener datasetUpdater) {
100     this.datasetUpdater = datasetUpdater;
101   }
102 
103   /**
104    * Utility to allow this to be used as a rule that will start and stop a server around the statement base.
105    */
106   @Override
107   public Statement apply(final Statement base, Description description) {
108     return new Statement() {
109 
110       @Override
111       public void evaluate() throws Throwable {
112         RegistryServer server = new RegistryServer();
113         server.start();
114         try {
115           base.evaluate();
116         } finally {
117           server.stop();
118         }
119       }
120     };
121   }
122 
123   public SolrClient getSolrClient() {
124     Preconditions.checkNotNull(solrClient, "Misuse of class. SolrClient has not been set");
125     return solrClient;
126   }
127 
128   public DatasetIndexUpdateListener getDatasetUpdater() {
129     Preconditions.checkNotNull(datasetUpdater, "Misuse of class. DatasetIndexUpdateListener has not been set");
130     return datasetUpdater;
131   }
132 }
