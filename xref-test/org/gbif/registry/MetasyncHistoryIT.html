54  /**
55   * Runs tests for the {@link MetasyncHistoryService} implementations.
56   * This is parameterized to run the same test routines for the following:
57   * <ol>
58   * <li>The WS service layer</li>
59   * <li>The WS service client layer</li>
60   * </ol>
61   */
62  @RunWith(Parameterized.class)
63  public class MetasyncHistoryIT {
64  
65    // Flushes the database on each run
66    @ClassRule
67    public static final LiquibaseInitializer liquibaseRule = new LiquibaseInitializer(RegistryTestModules.database());
68  
69    @ClassRule
70    public static final RegistryServer registryServer = RegistryServer.INSTANCE;
71  
72    // Tests user
73    private static String TEST_USER = "admin";
74  
75    @Rule
76    public final DatabaseInitializer databaseRule = new DatabaseInitializer(RegistryTestModules.database());
77  
78    private final MetasyncHistoryService metasyncHistoryService;
79  
80    private final SimplePrincipalProvider simplePrincipalProvider;
81  
82    private final OrganizationService organizationService;
83    private final NodeService nodeService;
84    private final InstallationService installationService;
85  
86    public MetasyncHistoryIT(
87      MetasyncHistoryService metasyncHistoryService,
88      OrganizationService organizationService, NodeService nodeService,
89      InstallationService installationService,
90      SimplePrincipalProvider simplePrincipalProvider) {
91      this.metasyncHistoryService = metasyncHistoryService;
92      this.organizationService = organizationService;
93      this.nodeService = nodeService;
94      this.installationService = installationService;
95      this.simplePrincipalProvider = simplePrincipalProvider;
96    }
97  
98    @Parameters
99    public static Iterable<Object[]> data() {
100     final Injector webservice = webservice();
101     final Injector client = webserviceClient();
102     return ImmutableList.<Object[]>of(new Object[] {webservice.getInstance(InstallationResource.class),
103       webservice.getInstance(OrganizationResource.class), webservice.getInstance(NodeResource.class),
104       webservice.getInstance(InstallationResource.class), null},
105       new Object[] {client.getInstance(MetasyncHistoryService.class),
106         client.getInstance(OrganizationService.class), client.getInstance(NodeService.class),
107         client.getInstance(InstallationService.class), client.getInstance(SimplePrincipalProvider.class)});
108   }
109 
110 
111   @Before
112   public void setup() {
113     // reset SimplePrincipleProvider, configured for web service client tests only
114     if (simplePrincipalProvider != null) {
115       simplePrincipalProvider.setPrincipal(TEST_USER);
116     }
117   }
118 
119   /**
120    * Tests the operations create and list of {@link MetasyncHistoryService}.
121    */
122   @Test
123   public void testCreateAndList() {
124     MetasyncHistory metasyncHistory = getTestInstance();
125     metasyncHistoryService.createMetasync(metasyncHistory);
126     PagingResponse<MetasyncHistory> response =
127       metasyncHistoryService.listMetasync(new PagingRequest());
128     Assert.assertTrue("The list operation should return at least 1 record", response.getResults().size() > 0);
129   }
130 
131 
132   /**
133    * Tests the {@link MetasyncHistoryService#listByInstallation(UUID, org.gbif.api.model.common.paging.Pageable)}
134    * operation.
135    */
136   @Test
137   public void testListAndListByInstallation() {
138     MetasyncHistory metasyncHistory = getTestInstance();
139     metasyncHistoryService.createMetasync(metasyncHistory);
140     PagingResponse<MetasyncHistory> response =
141       metasyncHistoryService.listMetasync(metasyncHistory.getInstallationKey(), new PagingRequest());
142     Assert.assertTrue("The list operation should return at least 1 record", response.getResults().size() > 0);
143   }
144 
145   /**
146    * Creates a test installation. The installation is persisted in the data base.
147    * The organization related to the installation are created too.
148    */
149   private Installation createTestInstallation() {
150     // endorsing node for the organization
151     UUID nodeKey = nodeService.create(Nodes.newInstance());
152 
153     // publishing organization (required field)
154     Organization org = Organizations.newInstance(nodeKey);
155     UUID organizationKey = organizationService.create(org);
156 
157     Installation inst = Installations.newInstance(organizationKey);
158     UUID installationKey = installationService.create(inst);
159     inst.setKey(installationKey);
160     return inst;
161   }
162 
163   /**
164    * Creates {@link MetasyncHistory} object to be used in test cases.
165    */
166   private MetasyncHistory getTestInstance() {
167     MetasyncHistory metasyncHistory = new MetasyncHistory();
168     metasyncHistory.setDetails("testDetails");
169     metasyncHistory.setResult(MetasyncResult.OK);
170     metasyncHistory.setSyncDate(new Date());
171     Installation installation = createTestInstallation();
172     metasyncHistory.setInstallationKey(installation.getKey());
173     return metasyncHistory;
174   }
