35    /**
36     * Initializes and configures the HttpUtil instance, which is what the IPT/Legacy WS consumers use to issue HTTP
37     * requests. This method copies the same configuration as the IPT uses, to ensure the exact same structured requests
38     * are sent during unit testing.
39     */
40    public static HttpUtil getHttp() {
41      // new threadsafe, multithreaded http client with support for http and https.
42      DefaultHttpClient client = HttpUtil.newMultithreadedClient(1000000, 1, 1);
43      // the registry requires Preemptive authentication, so make this the very first interceptor in the protocol chain
44      client.addRequestInterceptor(new PreemptiveAuthenticationInterceptor(), 0);
45      return new HttpUtil(client);
46    }
47  
48    /**
49     * Construct request URI, appending path to local Grizzly server address.
50     *
51     * @param path path
52     *
53     * @return request URI
54     */
55    public static String getRequestUri(String path) {
56      StringBuilder uri = new StringBuilder();
57      uri.append("http://localhost:");
58      uri.append(RegistryServer.getPort());
59      uri.append(path);
60      return uri.toString();
61    }
