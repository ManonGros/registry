131   /**
132    * Tests the create and get of a {@link DatasetProcessStatus}.
133    */
134   @Test
135   public void testCreateAndGet() {
136     DatasetProcessStatus datasetProcessStatus = buildProcessStatus(createTestDataset(), 1);
137     datasetProcessStatusService.createDatasetProcessStatus(datasetProcessStatus);
138     DatasetProcessStatus datasetProcessStatus2 = datasetProcessStatusService.getDatasetProcessStatus(
139       datasetProcessStatus.getDatasetKey(),
140       datasetProcessStatus.getCrawlJob().getAttempt());
141     Assert.assertNotNull(datasetProcessStatus2);
142   }

144   /**
145    * Tests the {@link DatasetProcessStatusService#listDatasetProcessStatus(org.gbif.api.model.common.paging.Pageable)}
146    * operation.
147    */
148   @Test
149   public void testListAndListByDataset() {
150     Dataset dataset = createTestDataset();
151     Dataset dataset2 = createTestDataset();
152 
153     datasetProcessStatusService.createDatasetProcessStatus(buildProcessStatus(dataset, 1));
154     datasetProcessStatusService.createDatasetProcessStatus(buildProcessStatus(dataset, 2));
155     datasetProcessStatusService.createDatasetProcessStatus(buildProcessStatus(dataset2, 1));
156 
157     PagingResponse<DatasetProcessStatus> statuses =
158       datasetProcessStatusService.listDatasetProcessStatus(new PagingRequest(0, 10));
159     assertEquals("There have been 3 crawl attempts", Long.valueOf(3), statuses.getCount());
160     assertEquals("There have been 3 crawl attempts", 3, statuses.getResults().size());
161 
162     PagingResponse<DatasetProcessStatus> statusesByDataset =
163       datasetProcessStatusService.listDatasetProcessStatus(dataset.getKey(), new PagingRequest());
164     assertEquals("The dataset has had 2 crawl attempts", Long.valueOf(2), statusesByDataset.getCount());
165     assertEquals("The dataset has had 2 crawl attempts", 2, statusesByDataset.getResults().size());
166   }

168   @Test(expected = IllegalArgumentException.class)
169   public void testIllegalCreate() {
170     Dataset dataset = createTestDataset();
171 
172     DatasetProcessStatus status = buildProcessStatus(dataset, 1);
173     datasetProcessStatusService.createDatasetProcessStatus(status);
174     // valid to update
175     status.setFragmentsEmitted(1000);
176     datasetProcessStatusService.updateDatasetProcessStatus(status);
177     // illegal to create the same attempt ID
178     datasetProcessStatusService.createDatasetProcessStatus(buildProcessStatus(dataset, 1));
179   }

181   /**
182    * Tests a create, get, update
183    */
184   @Test
185   public void testUpdate() {
186     DatasetProcessStatus orig = buildProcessStatus(createTestDataset(), 1);
187     datasetProcessStatusService.createDatasetProcessStatus(orig);
188     DatasetProcessStatus written =
189       datasetProcessStatusService.getDatasetProcessStatus(orig.getDatasetKey(), orig.getCrawlJob().getAttempt());
190     assertEquals(orig.getDatasetKey(), written.getDatasetKey());
191     assertEquals(orig.getCrawlJob().getAttempt(), written.getCrawlJob().getAttempt());
192     written.setFinishReason(FinishReason.ABORT);
193     datasetProcessStatusService.updateDatasetProcessStatus(written);
194     assertEquals(orig.getCrawlJob().getAttempt(), written.getCrawlJob().getAttempt());
195   }

197   /**
198    * Creates a test dataset. The dataset is persisted in the data base.
199    * The installation and organization related to the dataset are created too.
200    */
201   private Dataset createTestDataset() {
202     // endorsing node for the organization
203     UUID nodeKey = nodeService.create(Nodes.newInstance());
204 
205     // publishing organization (required field)
206     Organization o = Organizations.newInstance(nodeKey);
207     UUID organizationKey = organizationService.create(o);
208 
209     Installation i = Installations.newInstance(organizationKey);
210     UUID installationKey = installationService.create(i);
211     Dataset dataset = Datasets.newInstance(organizationKey, installationKey);
212     dataset.setKey(datasetService.create(dataset));
213     return dataset;
214   }

216   /**
217    * Builds a new sample process status for the given dataset and attempt number.
218    */
219   private DatasetProcessStatus buildProcessStatus(Dataset dataset, int attempt) {
220     DatasetProcessStatus.Builder builder = new DatasetProcessStatus.Builder();
221     CrawlJob crawlJob =
222       new CrawlJob(dataset.getKey(), EndpointType.DIGIR, URI.create("http://gbif.org.test"), attempt, null);
223     builder.crawlJob(crawlJob);
224     builder.crawlContext("testcontext");
225     builder.startedCrawling(new Date());
226     builder.finishedCrawling(new Date());
227     builder.fragmentsEmitted(144);
228     builder.fragmentsProcessed(144);
229     builder.fragmentsReceived(144);
230     builder.finishReason(FinishReason.NORMAL);
231     builder.processStateOccurrence(ProcessState.FINISHED);
232     builder.processStateChecklist(ProcessState.EMPTY);
233     builder.interpretedOccurrencesPersistedError(0);
234     builder.interpretedOccurrencesPersistedSuccessful(144);
235     builder.pagesCrawled(2);
236     builder.pagesFragmentedError(0);
237     builder.pagesFragmentedSuccessful(2);
238     builder.rawOccurrencesPersistedError(0);
239     builder.rawOccurrencesPersistedNew(144);
240     builder.rawOccurrencesPersistedUnchanged(0);
241     builder.rawOccurrencesPersistedUpdated(0);
242     builder.verbatimOccurrencesPersistedError(0);
243     builder.verbatimOccurrencesPersistedSuccessful(144);
244     builder.datasetKey(crawlJob.getDatasetKey());
245     return builder.build();
246   }
