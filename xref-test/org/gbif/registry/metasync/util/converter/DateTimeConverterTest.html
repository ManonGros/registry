13    /**
14     * Test ISO 8691 formats.
15     *
16     * See https://en.wikipedia.org/wiki/ISO_8601 for an overview.
17     */
18    @Test
19    public void testIso8691Dates() {
20      assertEquals(new DateTime().withDate(2016, 1, 27).withTimeAtStartOfDay(), converter.convert(DateTime.class, "2016-01-27"));
21      assertEquals(new DateTime().withDate(2016, 1, 25).withTimeAtStartOfDay(), converter.convert(DateTime.class, "2016-W04"));
22      assertEquals(new DateTime().withDate(2016, 1, 27).withTimeAtStartOfDay(), converter.convert(DateTime.class, "2016-W04-3"));
23      assertEquals(new DateTime().withDate(2016, 1, 27).withTimeAtStartOfDay(), converter.convert(DateTime.class, "2016-027"));
24      assertEquals(new DateTime().withDate(2016, 1, 27).withTimeAtStartOfDay(), converter.convert(DateTime.class, "20160127"));
25  
26      assertEquals(new DateTime().withZone(DateTimeZone.UTC).withDate(2016, 1, 27).withTime(06, 11, 22, 0).toDate(), ((DateTime) converter.convert(DateTime.class, "2016-01-27T06:11:22+00:00")).toDate());
27      assertEquals(new DateTime().withZone(DateTimeZone.UTC).withDate(2016, 1, 27).withTime(06, 11, 22, 0).toDate(), ((DateTime) converter.convert(DateTime.class, "2016-01-27T06:11:22Z")).toDate());
28      assertEquals(new DateTime().withZone(DateTimeZone.UTC).withDate(2016, 1, 27).withTime(06, 11, 22, 0).toDate(), ((DateTime) converter.convert(DateTime.class, "20160127T061122Z")).toDate());
29  
30      assertEquals(new DateTime().withZone(DateTimeZone.UTC).withDate(2016, 1, 27).withTime(06, 11, 22, 0).toDate(), ((DateTime) converter.convert(DateTime.class, "2016-01-27T09:11:22+03:00")).toDate());
31      assertEquals(new DateTime().withZone(DateTimeZone.UTC).withDate(2016, 1, 27).withTime(06, 11, 22, 0).toDate(), ((DateTime) converter.convert(DateTime.class, "2016-01-27T03:11:22-03:00")).toDate());
32      assertEquals(new DateTime().withZone(DateTimeZone.UTC).withDate(2016, 1, 27).withTime(06, 11, 22, 0).toDate(), ((DateTime) converter.convert(DateTime.class, "2016-01-27T03:11:22âˆ’03:00")).toDate());
33  
34      assertEquals(new DateTime().withZone(DateTimeZone.UTC).withDate(2016, 1, 27).withTime(06, 11, 22, 0).toDate(), ((DateTime) converter.convert(DateTime.class, "20160127T061122Z")).toDate());
35    }

37    /**
38     * Test some invalid formats, but which we are still provided with.
39     */
40    @Test
41    public void testInvalidDates() {
42      assertEquals(new DateTime().withDate(2016, 1, 27).withTimeAtStartOfDay(), converter.convert(DateTime.class, "2016/01/27"));
43      assertEquals(new DateTime().withZone(DateTimeZone.UTC).withDate(2016, 1, 27).withTime(06, 11, 22, 0).toDate(), ((DateTime) converter.convert(DateTime.class, "2016-01-27 06:11:22")).toDate());
44      assertEquals(new DateTime().withZone(DateTimeZone.UTC).withDate(2016, 1, 27).withTime(06, 11, 22, 0).toDate(), ((DateTime) converter.convert(DateTime.class, "2016-01-27 09:11:22+0300")).toDate());
45      assertEquals(new DateTime().withDate(2016, 1, 27).withTimeAtStartOfDay(), converter.convert(DateTime.class, "2016-01-27TCentral Sta:ndard Time"));
46    }

48    /**
49     * Test some partial formats.
50     */
51    @Test
52    public void testPartialDates() {
53      assertEquals(new DateTime().withDate(2016, 1, 1).withTimeAtStartOfDay(), converter.convert(DateTime.class, "2016-01"));
54      assertEquals(new DateTime().withDate(2016, 1, 1).withTimeAtStartOfDay(), converter.convert(DateTime.class, "2016"));
55  
56      assertEquals(new DateTime().withDate(2016, 1, 1).withTimeAtStartOfDay(), converter.convert(DateTime.class, "2016/01"));
57  
58      assertEquals(null, converter.convert(DateTime.class, ":"));
59      assertEquals(null, converter.convert(DateTime.class, ""));
60      assertEquals(null, converter.convert(DateTime.class, "ABCDEFGHIJKM"));
61      assertEquals(null, converter.convert(DateTime.class, "ABCD"));
62    }
