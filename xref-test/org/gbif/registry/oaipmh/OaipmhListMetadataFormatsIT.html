29   * Test the ListMetadataFormats verb of the OAI-PMH endpoint.
30   */
31  @RunWith(Parameterized.class)
32  public class OaipmhListMetadataFormatsIT extends AbstractOaipmhEndpointIT {
33  
34    public OaipmhListMetadataFormatsIT(NodeService nodeService, OrganizationService organizationService, InstallationService installationService, DatasetService datasetService) {
35      super(nodeService, organizationService, installationService, datasetService);
36    }
37  
38    @Test(expected = IdDoesNotExistException.class)
39    public void listMetadataFormats_notFound() throws IdDoesNotExistException {
40      serviceProvider.listMetadataFormats(
41              ListMetadataParameters.request()
42                      .withIdentifier("non-existent-record-identifier")
43      );
44    }
45  
46    @Test
47    public void listMetadataFormats_forARecord() throws Exception {
48      Organization org1 = createOrganization(Country.ETHIOPIA);
49      Installation org1Installation1 = createInstallation(org1.getKey());
50      Dataset org1Installation1Dataset1 = createDataset(org1.getKey(), org1Installation1.getKey(), DatasetType.OCCURRENCE, new Date());
51  
52      String key = org1Installation1Dataset1.getKey().toString();
53  
54      List<MetadataFormat> metadataFormats = Lists.newArrayList(serviceProvider.listMetadataFormats(
55              ListMetadataParameters.request()
56                      .withIdentifier(key)
57      ));
58  
59      assertThat("EML and OAIDC formats supported", metadataFormats, containsInAnyOrder(samePropertyValuesAs(OAIDC_FORMAT), samePropertyValuesAs(EML_FORMAT)));
60    }
61  
62    @Test
63    public void listMetadataFormats_forRepository() throws Exception {
64  
65      List<MetadataFormat> metadataFormats = Lists.newArrayList(serviceProvider.listMetadataFormats(
66              ListMetadataParameters.request()
67      ));
68  
69      assertThat("EML and OAIDC formats supported", metadataFormats, containsInAnyOrder(samePropertyValuesAs(OAIDC_FORMAT), samePropertyValuesAs(EML_FORMAT)));
70    }
