3   import org.gbif.api.vocabulary.Language;

5   import org.junit.Test;
6   
7   import static org.junit.Assert.assertEquals;
8   import static org.junit.Assert.assertNull;
9   
10  public class LanguageTypeConverterTest {
11    private LanguageTypeConverter converter = new LanguageTypeConverter();
12  
13    @Test
14    public void parseLanguages() throws Throwable {
15      assertLang(Language.AFRIKAANS, "af");
16      assertLang(Language.AFRIKAANS, "AF");
17      assertLang(Language.ENGLISH, "EN");
18      assertLang(Language.ENGLISH, "en_US");
19      assertLang(Language.ENGLISH, "en_UK");
20      assertLang(Language.FRENCH, "french");
21      assertLang(Language.FRENCH, "fra");
22      assertLang(Language.FRENCH, "fre");
23    }
24  
25    private void assertLang(Language expect, String value) throws Throwable {
26      Language l = (Language) converter.convertToType(Language.class, value);
27      if (expect == null) {
28        assertNull(l);
29      } else {
30        assertEquals(expect, l);
31      }
32    }
33  }
