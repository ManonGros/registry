32    private static final String DATASET_TITLE = "Dataset - Ws Client Demo";

34    private static UUID datasetKey;

37    /**
38     * Create Dataset one time only.
39     * </p>
40     * Since nodes and organizations can't be created using the API, an existing organization (endorsed already) is
41     * used as both the publishing organization for the dataset, and the hosting organization for the installation.
42     */
43    @BeforeClass
44    public static void createDataset() throws URISyntaxException {
45      // Publishing and hosting organization: Test Organization #1, endorsed by DanBIF
46      UUID organizationKey = UUID.fromString("0a16da09-7719-40de-8d4f-56a15ed52fb6");
47  
48      // create new installation, indicating which organization is hosting the installation
49      Installation installation = new Installation();
50      installation.setTitle("Installation - Ws Client Demo");
51      installation.setType(InstallationType.IPT_INSTALLATION);
52      installation.setDisabled(false);
53      installation.setDescription("Test IPT instance created using GBIF Registry ws client");
54      installation.setOrganizationKey(organizationKey);
55      installationKey = installationService().create(installation);
56  
57      // create new dataset, indicating which organization owns the dataset, and which installation hosts the dataset
58      Dataset dataset = new Dataset();
59      dataset.setTitle(DATASET_TITLE);
60      dataset.setType(DatasetType.OCCURRENCE);
61      dataset.setDescription("Test Dataset created using GBIF Registry ws client");
62      dataset.setPublishingOrganizationKey(organizationKey);
63      dataset.setInstallationKey(installationKey);
64      datasetKey = datasetService().create(dataset);
65      LOG.info("Dataset created with key: {}", datasetKey);
66    }
67  
68    /**
69     * Read dataset
70     */
71    @Test
72    public void testReadDataset() {
73      Dataset dataset = datasetServiceReadOnly().get(datasetKey);
74      assertEquals(DATASET_TITLE, dataset.getTitle());
75    }
