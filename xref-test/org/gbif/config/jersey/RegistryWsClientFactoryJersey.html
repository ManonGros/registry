1   package org.gbif.config.jersey;

3   import org.gbif.api.service.registry.DatasetService;
4   import org.gbif.api.service.registry.InstallationService;

6   import org.gbif.registry.ws.client.InstallationWsClient;
7   import org.gbif.utils.HttpUtil;
8   import org.gbif.ws.json.JacksonJsonContextResolver;
9   
10  import com.sun.jersey.api.client.WebResource;
11  import com.sun.jersey.api.client.config.ClientConfig;
12  import com.sun.jersey.api.client.config.DefaultClientConfig;
13  import com.sun.jersey.api.client.filter.HTTPBasicAuthFilter;
14  import com.sun.jersey.api.json.JSONConfiguration;
15  import com.sun.jersey.client.apache4.ApacheHttpClient4;
16  import com.sun.jersey.client.apache4.ApacheHttpClient4Handler;
17  import org.apache.http.client.HttpClient;
18  import org.codehaus.jackson.jaxrs.JacksonJsonProvider;
19  
20  public class RegistryWsClientFactoryJersey {
21  
22    private static final int DEFAULT_HTTP_TIMEOUT_MSECS = 10000;
23    private static final int DEFAULT_MAX_HTTP_CONNECTIONS = 100;
24    private static final int DEFAULT_MAX_HTTP_CONNECTIONS_PER_ROUTE = 100;
25  
26    // GBIF account, with correct role and permissions set
27    private static final String USERNAME = "ws_client_demo";
28    private static final String PASSWORD = "Demo123";
29    private static final HTTPBasicAuthFilter AUTH_FILTER = new HTTPBasicAuthFilter(USERNAME, PASSWORD);
30    private static final String REGISTRY_API_BASE_URL = "http://api.gbif-uat.org/v1/";
31  
32    private static WebResource resource;
33    private static DatasetService datasetService;
34    private static DatasetService datasetServiceReadOnly;
35    private static InstallationService installationService;
36  
37  
38    /**
39     * @return Apache HTTP Client
40     */
41    private static HttpClient buildHttpClient() {
42      return HttpUtil.newMultithreadedClient(DEFAULT_HTTP_TIMEOUT_MSECS, DEFAULT_MAX_HTTP_CONNECTIONS,
43        DEFAULT_MAX_HTTP_CONNECTIONS_PER_ROUTE);
44    }
45  
46    /**
47     * @return Jersey resource that utilizes the Apache HTTP Client to send and receive HTTP request and responses to the
48     * GBIF API.
49     */
50    public static synchronized WebResource buildJerseyClient() {
51      if (resource == null) {
52          ApacheHttpClient4Handler hch = new ApacheHttpClient4Handler(buildHttpClient(), null, false);
53          ClientConfig clientConfig = new DefaultClientConfig();
54          clientConfig.getClasses().add(JacksonJsonContextResolver.class);
55          // this line is critical! Note that this is the jersey version of this class name!
56          clientConfig.getClasses().add(JacksonJsonProvider.class);
57          clientConfig.getFeatures().put(JSONConfiguration.FEATURE_POJO_MAPPING, Boolean.TRUE);
58          resource = new ApacheHttpClient4(hch, clientConfig).resource(REGISTRY_API_BASE_URL);
59      }
60      return resource;
61    }
62  
63    /**
64     * @return read-only DatasetService
65     */
66    public static synchronized DatasetService datasetServiceReadOnly() {
67      if (datasetServiceReadOnly == null) {
68        datasetServiceReadOnly = new DatasetWsClient(buildJerseyClient(), null);
69      }
70      return datasetServiceReadOnly;
71    }
72  
73    /**
74     * @return DatasetService with authentication
75     */
76    public static synchronized DatasetService datasetService() {
77      if (datasetService == null) {
78        datasetService = new DatasetWsClient(buildJerseyClient(), AUTH_FILTER);
79      }
80      return datasetService;
81    }
82  
83    /**
84     * @return InstallationService with authentication
85     */
86    public static synchronized InstallationService installationService() {
87      if (installationService == null) {
88        installationService = new InstallationWsClient(buildJerseyClient(), AUTH_FILTER);
89      }
90      return installationService;
91    }
92  
93  }
