001package org.gbif.registry.ws.client;
002
003import org.gbif.api.model.common.paging.Pageable;
004import org.gbif.api.model.common.paging.PagingResponse;
005import org.gbif.api.model.occurrence.Download;
006import org.gbif.api.model.registry.DatasetOccurrenceDownloadUsage;
007import org.gbif.api.service.registry.DatasetOccurrenceDownloadUsageService;
008import org.gbif.registry.ws.client.guice.RegistryWs;
009import org.gbif.ws.client.BaseWsGetClient;
010
011import java.util.UUID;
012
013import javax.annotation.Nullable;
014
015import com.google.inject.Inject;
016import com.sun.jersey.api.client.WebResource;
017import com.sun.jersey.api.client.filter.ClientFilter;
018
019/**
020 * OccurrenceDownloadService web service client.
021 */
022public class DatasetOccurrenceDownloadUsageWsClient extends BaseWsGetClient<Download, String> implements
023  DatasetOccurrenceDownloadUsageService {
024
025  @Inject
026  public DatasetOccurrenceDownloadUsageWsClient(@RegistryWs WebResource resource, @Nullable ClientFilter authFilter) {
027    super(Download.class, resource.path("occurrence/download"), authFilter);
028  }
029
030  @Override
031  public void create(DatasetOccurrenceDownloadUsage downloadDataset) {
032    post(downloadDataset, "dataset");
033  }
034
035  @Override
036  public PagingResponse<DatasetOccurrenceDownloadUsage> listByDataset(UUID datasetKey, Pageable page) {
037    return get(GenericTypes.PAGING_DATASET_OCCURRENCE_DOWNLOAD, page, "dataset", String.valueOf(datasetKey));
038  }
039
040}
