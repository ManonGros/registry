145  @Override
146  public int addMachineTag(UUID targetEntityKey, MachineTag machineTag) {
147    // allow post through varnish (no chunked encoding needed)
148    int tagId;
149    try {
150      tagId = getResource()
151        .path(targetEntityKey.toString())
152        .path("machineTag")
153        .type(MediaType.APPLICATION_JSON)
154        .post(Integer.class, mapper.writeValueAsBytes(machineTag));
155
156    } catch (IOException e) {
157      throw new IllegalStateException(e);
158    }
159//    int tagId = post(Integer.class, machineTag, targetEntityKey.toString(), "machineTag");
160    return tagId;
161  }
