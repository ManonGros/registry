001package org.gbif.registry.ws.client;
002
003import org.gbif.api.model.common.search.SearchResponse;
004import org.gbif.api.model.registry.search.DatasetSearchParameter;
005import org.gbif.api.model.registry.search.DatasetSearchRequest;
006import org.gbif.api.model.registry.search.DatasetSearchResult;
007import org.gbif.api.model.registry.search.DatasetSuggestRequest;
008import org.gbif.api.model.registry.search.DatasetSuggestResult;
009import org.gbif.api.service.registry.DatasetSearchService;
010import org.gbif.registry.ws.client.guice.RegistryWs;
011import org.gbif.ws.client.BaseWsSuggestClient;
012
013import java.util.List;
014
015import com.google.inject.Inject;
016import com.sun.jersey.api.client.GenericType;
017import com.sun.jersey.api.client.WebResource;
018
019public class DatasetSearchWsClient
020  extends
021  BaseWsSuggestClient<DatasetSearchResult, DatasetSearchParameter, DatasetSearchRequest, DatasetSuggestResult, DatasetSuggestRequest>
022  implements DatasetSearchService {
023
024  private static final GenericType<SearchResponse<DatasetSearchResult, DatasetSearchParameter>> SEARCH_TYPE =
025    new GenericType<SearchResponse<DatasetSearchResult, DatasetSearchParameter>>() {
026    };
027
028  private static final GenericType<List<DatasetSuggestResult>> SUGGEST_TYPE =
029    new GenericType<List<DatasetSuggestResult>>() {
030    };
031
032  @Inject
033  public DatasetSearchWsClient(@RegistryWs WebResource resource) {
034    // Note: /search is appended in the parent
035    super(resource.path("dataset"), SEARCH_TYPE, SUGGEST_TYPE);
036  }
037
038  @Override
039  public List<DatasetSuggestResult> suggest(DatasetSuggestRequest suggestRequest) {
040    return super.suggest(suggestRequest);
041  }
042
043  @Override
044  public SearchResponse<DatasetSearchResult, DatasetSearchParameter> search(DatasetSearchRequest searchRequest) {
045    return super.search(searchRequest);
046  }
047
048
049}
