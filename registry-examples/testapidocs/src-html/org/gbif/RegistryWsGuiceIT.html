016@Ignore("Requires live UAT API")
017public class RegistryWsGuiceIT {
018
019  /**
020   * Retrieve dataset Pontaurus with key "b3e760d8-8dcc-468c-a23d-8e9772affe59". It is assumed dataset Pontaurus will
021   * always exist in every Registry database.
022   */
023  @Test
024  public void testGetDataset() {
025    DatasetService ds = webserviceClientReadOnly().getInstance(DatasetService.class);
026    Dataset dataset = ds.get(UUID.fromString("8575f23e-f762-11e1-a439-00145eb45e9a"));
027    assertEquals("PonTaurus collection", dataset.getTitle());
028  }
029
030  /**
031   * Update dataset user has permission to edit.
032   */
033  @Test
034  public void testUpdateDataset() {
035    DatasetService ds = webserviceClient().getInstance(DatasetService.class);
036    Dataset dataset = ds.get(UUID.fromString("38b0f0f5-cac1-48e6-be85-b9f8cdd5ca93"));
037    dataset.setRights("CC0");
038    ds.update(dataset);
039  }
