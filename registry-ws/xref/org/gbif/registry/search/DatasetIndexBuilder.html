45  /**
46   * A builder that will clear and build a new dataset index by paging over the given service.
47   */
48  @Singleton
49  public class DatasetIndexBuilder {
50  
51    // controls how many results we request while paging over the WS
52    private static final int WS_PAGE_SIZE = 100;
53    private static final Logger LOG = LoggerFactory.getLogger(DatasetIndexBuilder.class);
54    private final SolrClient solrClient;
55    private final DatasetService datasetService;
56    private final DatasetDocConverter docConverter;
57  
58    @Inject
59    public DatasetIndexBuilder(@Named("Dataset") SolrClient solrClient, DatasetService datasetService,
60      InstallationService installationService, OrganizationService organizationService) {
61      this.solrClient = solrClient;
62      this.datasetService = datasetService;
63      // We can use a cache at startup
64      docConverter =
65        new DatasetDocConverter(new CachingNetworkEntityService<Organization>(organizationService),
66          new CachingNetworkEntityService<Installation>(installationService), datasetService);
67    }
68  
69    public void build() throws SolrServerException, IOException {
70      LOG.info("Building a new Dataset index");
71      Stopwatch stopwatch = Stopwatch.createStarted();
72      pageAndIndex();
73      solrClient.commit();
74      solrClient.optimize();
75      LOG.info("Finished building Dataset index in {} secs", stopwatch.elapsed(TimeUnit.SECONDS));
76    }
77  
78    /**
79     * Pages over all datasets and adds to SOLR.
80     */
81    private void pageAndIndex() throws IOException, SolrServerException {
82      PagingRequest page = new PagingRequest(0, WS_PAGE_SIZE);
83      PagingResponse<Dataset> response = null;
84      do {
85        LOG.debug("Requesting {} datasets starting at offset {}", page.getLimit(), page.getOffset());
86        response = datasetService.list(page);
87        // Batching updates to SOLR proves quicker with batches of 100 - 1000 showing similar performance
88        List<SolrInputDocument> batch = Lists.newArrayList();
89        for (Dataset ds : response.getResults()) {
90          batch.add(docConverter.build(ds));
91        }
92        if (!batch.isEmpty()) {
93          solrClient.add(batch);
94          solrClient.commit(); // to allow eager users (or impatient developers) to see search data on startup quickly
95        }
96        page.nextPage();
97  
98      } while (!response.isEndOfRecords());
99    }

101   /**
102    * A lightweight cache to help improve performance of the builder.
103    *
104    * @param <T> The type of entity being wrapped
105    */
106   private static class CachingNetworkEntityService<T> implements NetworkEntityService<T> {
107 
108     private final NetworkEntityService<T> service;
109     private final LoadingCache<UUID, T> cache = CacheBuilder.newBuilder()
110       .maximumSize(1000)
111       .expireAfterWrite(10, TimeUnit.MINUTES)
112       .build(
113         new CacheLoader<UUID, T>() {
114 
115           @Override
116           public T load(UUID key) throws Exception {
117             return service.get(key);
118           }
119         });
