31  /**
32   * A utility builder to convert datasets into solr input documents.
33   */
34  class DatasetDocConverter {
35  
36    private static final Logger LOG = LoggerFactory.getLogger(DatasetDocConverter.class);
37  
38    private final DatasetService datasetService;
39    private final NetworkEntityService<Organization> organizationService;
40    private final NetworkEntityService<Installation> installationService;
41    private final TimeSeriesExtractor timeSeriesExtractor = new TimeSeriesExtractor(1000, 2400, 1800, 2050);
42    private final SAXParserFactory saxFactory = SAXParserFactory.newInstance();
43  
44    /**
45     * Holds a map of {@link org.gbif.api.model.checklistbank.NameUsage} properties (Java fields) to Solr fields name.
46     */
47    private final Map<String, String> fieldPropertyMap = ImmutableMap.<String, String>builder()
48      .put("key", "key")
49      .put("dataset_title", "title")
50      .put("description", "description")
51      // enums
52      .put("dataset_type", "type")
53      .put("dataset_subtype", "subtype")
54      .build();
55    private final Set<String> enumFields = ImmutableSet.of("dataset_type", "dataset_subtype");
56  
57    public DatasetDocConverter(
58      NetworkEntityService<Organization> organizationService, NetworkEntityService<Installation> installationService,
59      DatasetService datasetService) {
60      this.organizationService = organizationService;
61      this.installationService = installationService;
62      this.datasetService = datasetService;
63    }
64  
65    /**
66     * Creates a SolrAnnotatedDataset from the given dataset, copying only the relevant fields for Solr from
67     * the given dataset.
68     *
69     * @param d The Dataset which will be copied into this object
70     */
71    public SolrInputDocument build(Dataset d) {
72      SolrInputDocument doc = new SolrInputDocument();
73      // Uses the pre-initialized field-property map to find the corresponding Solr field of a Java field.
74      try {
75        for (String field : fieldPropertyMap.keySet()) {
76          String property = fieldPropertyMap.get(field);
77          Object val = PropertyUtils.getProperty(d, property);
78          if (val != null) {
79            // enum values are stored as ordinal ints
80            if (enumFields.contains(field)) {
81              doc.addField(field, ((Enum) val).ordinal());
82            } else {
83              doc.addField(field, val);
84            }
85          }
86        }
87      } catch (Exception e) {
88        LOG.error("Error converting dataset {} to solr document", d.getKey(), e);
89        throw new RuntimeException(e);
90      }
91  
92      // keywords
93      List<String> kw = Lists.newArrayList();
94      for (Tag t : d.getTags()) {
95        kw.add(t.getValue());
96      }
97      doc.addField("keyword", kw);
98  
99      // decade series
100     List<Integer> decades = timeSeriesExtractor.extractDecades(d.getTemporalCoverages());
101     doc.addField("decade", decades);
102 
103     // hosting org and publishing org title have to be retrieved via service calls
104     indexOrgs(doc, d);
105 
106     // index entire metadata documents
107     indexMetadata(doc, d);
108 
109     // TODO: populate country coverage and continents:
110     // http://dev.gbif.org/issues/browse/POR-522
111 
112     return doc;
113   }
114 
115   private void indexMetadata(SolrInputDocument doc, Dataset d) {
116     InputStream stream = null;
117     try {
118       stream = datasetService.getMetadataDocument(d.getKey());
119 
120       if (stream != null) {
121         FullTextSaxHandler handler = new FullTextSaxHandler();
122         SAXParser p = saxFactory.newSAXParser();
123         // parse does close the stream
124         p.parse(stream, handler);
125         doc.addField("metadata", handler.getFullText());
126       }
127     } catch (ParserConfigurationException e) {
128       throw new IllegalStateException("XML Parser not working on this system", e);
129 
130     } catch (SAXException e) {
131       LOG.warn("Cannot parse original metadata xml for dataset {}", d.getKey());
132 
133     } catch (Exception e) {
134       LOG.error("Unable to index metadata document for dataset {}", d.getKey(), e);
135 
136     } finally {
137       Closeables.closeQuietly(stream);
138     }
139   }

141   private void indexOrgs(SolrInputDocument doc, Dataset d){
142     // see http://dev.gbif.org/issues/browse/REG-405 which explains why we defend against NotFoundExceptions below
143 
144     Organization publisher = null;
145     try {
146       publisher = d.getPublishingOrganizationKey() != null ? organizationService.get(d.getPublishingOrganizationKey()) : null;
147     } catch (NotFoundException e) {
148       // server side, interceptors may trigger on a @nulltoNotFoundException which we code defensively for, but smells
149       LOG.warn("Service reports organization[{}] cannot be found for dataset[{}]", d.getPublishingOrganizationKey(),
150                d.getKey());
151     }
152 
153     Installation installation = null;
154     try {
155       installation = d.getInstallationKey() != null ? installationService.get(d.getInstallationKey()) : null;
156     } catch (NotFoundException e) {
157       // server side, interceptors may trigger on a @nulltoNotFoundException which we code defensively for, but smells
158       LOG.warn("Service reports installation[{}] cannot be found for dataset[{}]", d.getInstallationKey(), d.getKey());
159     }
160 
161     Organization host = null;
162     try {
163       host = installation != null && installation.getOrganizationKey() != null ? organizationService.get(installation
164                                                                                                            .getOrganizationKey()) : null;
165     } catch (NotFoundException e) {
166       // server side, interceptors may trigger on a @nulltoNotFoundException which we code defensively for, but smells
167       LOG.warn("Service reports organization[{}] cannot be found for installation[{}]",
168                installation.getOrganizationKey(), installation.getKey());
169     }
170 
171     if (publisher != null) {
172       doc.addField("publishing_organization_key", publisher.getKey().toString());
173       doc.addField("publishing_organization_title", publisher.getTitle());
174       if (publisher.getCountry() != null) {
175         doc.addField("publishing_country", publisher.getCountry().ordinal());
176       }
177     } else {
178       doc.addField("publishing_country", Country.UNKNOWN.ordinal());
179     }
180     if (host != null) {
181       doc.addField("hosting_organization_key", host.getKey().toString());
182       doc.addField("hosting_organization_title", host.getTitle());
183     }
184 
185   }
