40  /**
41   * A guice module that sets up implementation of the search services and gives SOLR providers for search.
42   * <p/>
43   * The current implementation uses an embedded SOLR index, that uses a SOLR RAM directory. This is satisfactory while
44   * indexes remain small, and while we opt to rebuild the index on startup. Future implementations might consider using
45   * an external SOLR server, thus this method signatures use abstract SOLR type, and not a concrete implementation to
46   * allow for future changes.
47   * <p/>
48   * <strong>A note on the gbif-common-search SolrModule {@link org.gbif.common.search.inject.SolrModule}:</strong>
49   * <p/>
50   * The {@link org.gbif.common.search.inject.SolrModule} was considered for use here, but discarded because in its current implementation:
51   * <ul>
52   * <li>The {@link org.gbif.common.search.solr.builders.EmbeddedServerBuilder} does not support multiple cores, but that is anticipated as a future
53   * requirement for this project</li>
54   * <li>The {@link org.gbif.common.search.solr.builders.EmbeddedServerBuilder} does not allow someone to supply additional configuration files (it was
55   * developed for unit test purposes only)</li>
56   * <li>The SOLR server desired requires only 3 lines of code to instantiate. The implementor felt that readability is
57   * improved keeping this together</li>
58   * <li>{@link org.gbif.common.search.inject.SolrModule} is complex as it provides configuration based declaration (e.g. to support varying
59   * deployments. The implementor felt that was an additional learning curve that is unnecessary to put on future
60   * developers of this project. That said, should this project require flexible deployment options in the future,
61   * developers would be wise to consider adding multi-core support to the {@link org.gbif.common.search.inject.SolrModule}</li>
62   * </ul>
63   */

79      bind(DatasetIndexUpdateListener.class).asEagerSingleton();
80      bind(DatasetIndexBuilder.class).in(Scopes.SINGLETON);
81      expose(DatasetSearchService.class);
82      expose(DatasetIndexUpdateListener.class); // for testing
83      expose(OrganizationService.class); // for testing
84      expose(InstallationService.class); // for testing
85      expose(DatasetService.class); // for testing
86      expose(NodeService.class); // for testing
87  
88      // 6 email properties:
89      // use dev email?
90      mapAndExposeBool("mail.devemail.enabled","useDevEmail");
91      // smpt host?
92      mapAndExpose("mail.smtp.host","smptHost");
93      // smpt port?
94      mapAndExposeInt("mail.smtp.port","smptPort");
95      // dev email address?
96      mapAndExpose("mail.devemail","devEmail");
97      // cc (helpdesk) email address?
98      mapAndExpose("mail.cc","ccEmail");
99      // from email address?
100     mapAndExpose("mail.from","fromEmail");
101     // Expose solr.home
102     mapAndExpose("solr.home","solr.home");
103   }

106    * Map the String property "fromProperty" to and String named annotation "name".
107    */
108   private void mapAndExpose(String fromProperty , String name){
109     String from = getVerbatimProperties().getProperty(fromProperty);
110     bind(String.class).annotatedWith(Names.named(name)).toInstance(from);
111     expose(String.class).annotatedWith(Names.named(name));
112   }

132   /**
133    * A provider of the SolrServer to use with a dataset search.
134    * Named as such, to allow multiple providers (e.g. solr cores) should this be required in the future.
135    *
136    * @return An embedded solr server that uses the configuration in /solr on the classpath
137    * @throws java.net.URISyntaxException If /solr configuration is not found on the classpath
138    */
139   @Provides
140   @Exposed
141   @Named("Dataset")
142   @Singleton
143   public SolrClient datasetSolr(@Named("solr.home") String solrHome) throws URISyntaxException, IOException {
144     Path tmpSolrHome = createTempSolrDirectory(solrHome);
145     File conf = new File(tmpSolrHome.toFile().getAbsolutePath(), "solr.xml");
146     return new EmbeddedSolrServer(CoreContainer.createAndLoad(Paths.get(tmpSolrHome.toFile().getAbsolutePath()), conf.toPath()), "dataset");
147   }
148 
149   /**
150    * Creates a temporary Solr directory containing all the Solr configuration files.
151    */
152   private Path createTempSolrDirectory(String solrHome) throws IOException {
153     final Path solrHomePath = Paths.get(solrHome);
154     final Path datasetCoreHomePath = Paths.get(solrHome,"dataset");
155     File solrHomeFile = solrHomePath.toFile();
156     if (solrHomeFile.exists()){
157       FileUtils.cleanDirectory(solrHomeFile);
158     } else {
159       solrHomeFile.mkdirs();
160     }
161     File confDir = new File(datasetCoreHomePath.toFile(),"conf");
162     confDir.mkdirs();
163     final Path confPath = Paths.get(confDir.toURI());
164     exportResource("/solr/solr.xml",solrHomePath);
165     exportResource("/solr/dataset/core.properties", datasetCoreHomePath);
166     exportResource("/solr/dataset/conf/mapping-FoldToASCII.txt", confPath);
167     exportResource("/solr/dataset/conf/mapping-ISOLatin1Accent.txt", confPath);
168     exportResource("/solr/dataset/conf/schema.xml", confPath);
169     exportResource("/solr/dataset/conf/solrconfig.xml", confPath);
170     return solrHomePath;
171   }

173   /**
174    * Export a resource embedded into a Jar file to the local file path.
175    *
176    * @param resourceName ie.: "/SmartLibrary.dll"
177    * @return The path to the exported resource
178    * @throws Exception
179    */
180   private static String exportResource(String resourceName, Path tmpDir) throws IOException {//todo move to Utils
181     Closer closer = Closer.create();
182     try {
183       URL resource = RegistrySearchModule.class.getResource(resourceName);
184       File fileResource = new File(resource.getFile());
185       InputStream stream = closer.register(resource.openStream());//note that each / is a directory down in the "jar tree" been the jar the root of the tree"
186       if(stream == null) {
187         throw new IOException("Cannot get resource \"" + resourceName + "\" from Jar file.");
188       }
189       int readBytes;
190       byte[] buffer = new byte[4096];
191       OutputStream resStreamOut = closer.register(new FileOutputStream(tmpDir + "/" + fileResource.getName()));
192       while ((readBytes = stream.read(buffer)) > 0) {
193         resStreamOut.write(buffer, 0, readBytes);
194       }
195     } catch (IOException ex) {
196       throw ex;
197     } finally {
198       closer.close();
199     }
200     return tmpDir + resourceName;
201   }
