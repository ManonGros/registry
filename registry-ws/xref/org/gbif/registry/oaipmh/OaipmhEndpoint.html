55  /**
56   * An OAI-PMH endpoint, using the XOAI library.
57   */
58  @Path("oai-pmh/registry")
59  @Singleton
60  public class OaipmhEndpoint {
61  
62    private static final TransformerFactory factory = TransformerFactory.newInstance();
63  
64    private static final DateProvider dateProvider = new UTCDateProvider();
65    private static final ErrorHandler errorsHandler = new ErrorHandler();
66  
67    public Transformer xsltTransformer(String xsltFile) {
68      try {
69        /*
70         * TODO: "An object of this class [Transformer] may not be used in multiple threads running concurrently!"
71         * Instead, XOAI should accept an immutable Templates object, and call .newTransformer() to get a Transformer.
72         * See https://xalan.apache.org/xalan-j/usagepatterns.html#multithreading
73         */
74        InputStream stream = this.getClass().getClassLoader().getResourceAsStream("org/gbif/registry/oaipmh/"+xsltFile);
75        return factory.newTransformer(new StreamSource(stream));
76      } catch (TransformerConfigurationException e) {
77        throw new RuntimeException("Unable to read XSLT transform "+xsltFile, e);
78      }
79    }
80  
81    private final MetadataFormat OAIDC_METADATA_FORMAT = new MetadataFormat()
82            .withPrefix("oai_dc")
83            .withNamespace("http://www.openarchives.org/OAI/2.0/oai_dc/")
84            .withSchemaLocation("http://www.openarchives.org/OAI/2.0/oai_dc.xsd")
85            .withTransformer(xsltTransformer("dc.xslt"));
86  
87    private final MetadataFormat EML_METADATA_FORMAT = new MetadataFormat()
88            .withPrefix("eml")
89            .withNamespace("eml://ecoinformatics.org/eml-2.1.1")
90            .withSchemaLocation("http://rs.gbif.org/schema/eml-gbif-profile/1.0.2/eml.xsd")
91            .withTransformer(xsltTransformer("eml.xslt"));
92  
93    private Context context = new Context()
94            .withMetadataFormat(OAIDC_METADATA_FORMAT)
95            .withMetadataFormat(EML_METADATA_FORMAT);
96  
97    private Repository repository;
98    private DataProvider dataProvider;
99  
100   @Inject
101   public OaipmhEndpoint(RepositoryConfiguration repositoryConfiguration, ItemRepository itemRepository, SetRepository setRepository) {
102 
103     this.repository = new Repository()
104             .withItemRepository(itemRepository)
105             .withSetRepository(setRepository)
106             .withResumptionTokenFormatter(new SimpleResumptionTokenFormat())
107             .withConfiguration(repositoryConfiguration);
108 
109     this.dataProvider = new DataProvider(context, repository);
110   }
111 
112   @GET
113   @Produces("application/xml;charset=UTF-8")
114   public InputStream oaipmh(
115           @QueryParam("verb") String verb,
116           @Nullable @QueryParam("identifier") String identifier,
117           @Nullable @QueryParam("metadataPrefix") String metadataPrefix,
118           @Nullable @QueryParam("from") String from,
119           @Nullable @QueryParam("until") String until,
120           @Nullable @QueryParam("set") String set,
121           @Nullable @QueryParam("resumptionToken") String resumptionToken) {
122 
123     Date fromDate = null, untilDate = null;
124     OAIRequestParametersBuilder reqBuilder = new OAIRequestParametersBuilder()
125             .withVerb(verb)
126             .withMetadataPrefix(metadataPrefix);
127 
128     if(from != null) {
129       try {
130         fromDate = dateProvider.parse(from);
131       } catch (ParseException pEx) {
132         return handleOAIRequestBadArgument(reqBuilder.build(), "from=" + from);
133       }
134     }
135 
136     if(until != null){
137       try {
138         untilDate = dateProvider.parse(from);
139       } catch (ParseException pEx) {
140         return handleOAIRequestBadArgument(reqBuilder.build(), "until=" + until);
141       }
142     }
143 
144     reqBuilder.withIdentifier(identifier)
145             .withFrom(fromDate)
146             .withUntil(untilDate)
147             .withSet(set)
148             .withResumptionToken(resumptionToken);
149 
150     return handleOAIRequest(reqBuilder.build());
151   }
152 
153   private InputStream handleOAIRequest(OAIRequest request) {
154     try {
155       OAIPMH oaipmh = dataProvider.handle(request);
156       return new ByteArrayInputStream(write(oaipmh).getBytes("UTF-8"));
157 
158     } catch (Exception e) {
159       throw new ServiceUnavailableException("OAI Failed to serialize dataset", e);
160     }
161   }
162 
163   /**
164    * Build and generate the response for a BadArgument error code.
165    *
166    * @param requestParameters origin of the BadArgument error
167    * @param errorMessage textual message to report
168    * @return
169    */
170   private InputStream handleOAIRequestBadArgument(OAIRequest requestParameters, String errorMessage) {
171 
172     Request request = new Request(repository.getConfiguration().getBaseUrl())
173             .withVerbType(requestParameters.get(Verb))
174             .withResumptionToken(requestParameters.get(ResumptionToken))
175             .withIdentifier(requestParameters.get(Identifier))
176             .withMetadataPrefix(requestParameters.get(MetadataPrefix));
177 
178     try {
179       OAIPMH errorResponse = new OAIPMH()
180               .withRequest(request)
181               .withResponseDate(dateProvider.now())
182               .withError(errorsHandler.handle(new BadArgumentException(errorMessage)));
183       return new ByteArrayInputStream(write(errorResponse).getBytes("UTF-8"));
184 
185     } catch (Exception e) {
186       throw new ServiceUnavailableException("OAI Failed to serialize dataset", e);
187     }
188   }
189 
190   protected String write(final XmlWritable handle) throws XMLStreamException, XmlWriteException {
191     return new StringBuilder()
192             .append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n")
193             .append(XmlWriter.toString(new XmlWritable() {
194               @Override
195               public void write(XmlWriter writer) throws XmlWriteException {
196                 writer.write(handle);
197               }
198             }))
199             .toString();
200   }
