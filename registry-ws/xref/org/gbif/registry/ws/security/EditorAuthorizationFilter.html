3   import java.security.Principal;
4   import java.util.UUID;
5   import java.util.regex.Matcher;
6   import java.util.regex.Pattern;
7   import javax.ws.rs.WebApplicationException;
8   import javax.ws.rs.core.Context;
9   import javax.ws.rs.core.Response;
10  import javax.ws.rs.core.SecurityContext;
11  
12  import com.google.common.annotations.VisibleForTesting;
13  import com.google.inject.Inject;
14  import com.sun.jersey.spi.container.ContainerRequest;
15  import com.sun.jersey.spi.container.ContainerRequestFilter;
16  import org.slf4j.Logger;
17  import org.slf4j.LoggerFactory;
18  
19  /**
20   * For requests authenticated with a REGISTRY_EDITOR role two levels of authorization need to be passed.
21   * First of all any resource method is required to have the role included in the RolesAllowed annotation.
22   * Secondly this request filter needs to be passed for POST/PUT/DELETE requests that act on existing and UUID identified
23   * main registry entities such as dataset, organization, node, installation and network.
24   *
25   * In order to do authorization the key of these entities is extracted from the requested path.
26   * An exception to this is the create method for those main entities themselves.
27   * This is covered by the BaseNetworkEntityResource.create() method directly.
28   */
29  public class EditorAuthorizationFilter implements ContainerRequestFilter {
30  
31    private static final Logger LOG = LoggerFactory.getLogger(EditorAuthorizationFilter.class);
32  
33    private static final String ENTITY_KEY = "^%s/([a-fA-F0-9-]+)";
34    private static final Pattern NODE_NETWORK_PATTERN = Pattern.compile(String.format(ENTITY_KEY, "(?:network|node)"));
35    private static final Pattern ORGANIZATION_PATTERN = Pattern.compile(String.format(ENTITY_KEY, "organization"));
36    private static final Pattern DATASET_PATTERN = Pattern.compile(String.format(ENTITY_KEY, "dataset"));
37    private static final Pattern INSTALLATION_PATTERN = Pattern.compile(String.format(ENTITY_KEY, "installation"));
38  
39    private final EditorAuthorizationService userAuthService;
40  
41    @Context
42    private SecurityContext secContext;
43  
44    @Inject
45    public EditorAuthorizationFilter(EditorAuthorizationService userAuthService) {
46      this.userAuthService = userAuthService;
47    }
48  
49  
50    @Override
51    public ContainerRequest filter(ContainerRequest request) {
52      // only verify non GET methods with an authenticated REGISTRY_EDITOR
53      // all other roles are taken care by simple JSR250 annotations on the resource methods
54      Principal user = secContext.getUserPrincipal();
55  
56      if (user != null
57          && (!secContext.isUserInRole(UserRoles.ADMIN_ROLE) && secContext.isUserInRole(UserRoles.EDITOR_ROLE))
58          && !request.getMethod().equals("GET")) {
59  
60        String path = request.getPath().toLowerCase();
61  
62        try {
63  
64          Matcher m = ORGANIZATION_PATTERN.matcher(path);
65          if (m.find()) {
66            if (!userAuthService.allowedToModifyOrganization(user, UUID.fromString(m.group(1)))) {
67              throw new WebApplicationException(Response.Status.FORBIDDEN);
68            }
69          }
70  
71          m = DATASET_PATTERN.matcher(path);
72          if (m.find()) {
73            if (!userAuthService.allowedToModifyDataset(user, UUID.fromString(m.group(1)))) {
74              throw new WebApplicationException(Response.Status.FORBIDDEN);
75            }
76          }
77  
78          m = INSTALLATION_PATTERN.matcher(path);
79          if (m.find()) {
80            if (!userAuthService.allowedToModifyInstallation(user, UUID.fromString(m.group(1)))) {
81              throw new WebApplicationException(Response.Status.FORBIDDEN);
82            }
83          }
84  
85          m = NODE_NETWORK_PATTERN.matcher(path);
86          if (m.find()) {
87            if (!userAuthService.allowedToModifyEntity(user, UUID.fromString(m.group(1)))) {
88              throw new WebApplicationException(Response.Status.FORBIDDEN);
89            }
90          }
91        } catch (IllegalArgumentException e) {
92          // no valid UUID, do nothing as it should not be a valid request anyway
93        }
94      }
95  
96      return request;
97    }
98  
99    @VisibleForTesting
100   protected void setSecContext(SecurityContext secContext) {
101     this.secContext = secContext;
102   }
103 }
