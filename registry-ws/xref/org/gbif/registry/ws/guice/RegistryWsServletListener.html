44  /**
45   * The Registry WS module.
46   */
47  public class RegistryWsServletListener extends GbifServletListener {
48  
49    public static final String APP_CONF_FILE = "registry.properties";
50  
51    private static final String API_URL_PROPERTY = "api.url";
52    // fail fast by designed, see CubeService usage
53    private static final String METRICS_WS_HTTP_TIMEOUT = "100";
54  
55    public static final List<Class<? extends ContainerRequestFilter>> requestFilters = Lists.newArrayList();
56    public static final List<Class<? extends ContainerResponseFilter>> responseFilters = Lists.newArrayList();
57  
58    static {
59      requestFilters.add(LegacyAuthorizationFilter.class);
60      requestFilters.add(EditorAuthorizationFilter.class);
61      responseFilters.add(AuthResponseCodeOverwriteFilter.class);
62    }
63  
64    private static final String PACKAGES = "org.gbif.registry.ws.resources, org.gbif.registry.ws.provider, org.gbif.registry.oaipmh";
65  
66    /**
67     * Get a subset of properties related to metrics.
68     * Uses the api.url to fill metrics.ws.url and set a small timeout (100 ms) for http
69     * This methods only exists because {@link org.gbif.ws.client.guice.GbifWsClientModule} uses Names.bindProperties(binder(), properties);
70     * which would lead to multiple bindings for the same property.
71     *
72     * @param properties
73     * @return
74     */
75    private static Properties getMetricsProperties(Properties properties){
76      Properties metricsProperties = new Properties();
77      metricsProperties.setProperty("metrics.ws.url", properties.getProperty(API_URL_PROPERTY));
78      metricsProperties.setProperty(MetricsWsClientModule.HttpClientConnParams.HTTP_TIMEOUT, METRICS_WS_HTTP_TIMEOUT);
79      return metricsProperties;
80    }
81  
82    public RegistryWsServletListener() throws IOException {
83      super(PropertiesUtil.readFromFile(ConfUtils.getAppConfFile(APP_CONF_FILE)), PACKAGES, true, responseFilters, requestFilters);
84    }
85  
86    @VisibleForTesting
87    public RegistryWsServletListener(Properties properties) {
88      super(properties, PACKAGES, true, null, requestFilters);
89    }
90  
91    @Override
92    protected List<Module> getModules(Properties properties) {
93      return Lists.newArrayList(new DoiModule(properties),
94                                new RegistryMyBatisModule(properties),
95                                new DrupalMyBatisModule(properties),
96                                new DirectoryModule(properties),
97                                StringTrimInterceptor.newMethodInterceptingModule(),
98                                new ValidationModule(),
99                                new EventModule(properties),
100                               new RegistrySearchModule(properties),
101                               new SecurityModule(properties),
102                               new VarnishPurgeModule(properties),
103                               new TitleLookupModule(true, properties.getProperty(API_URL_PROPERTY)),
104                               new MetricsWsClientModule(getMetricsProperties(properties)),
105                               new OaipmhModule(properties));
106   }
107 
108   @VisibleForTesting
109   @Override
110   protected Injector getInjector() {
111     return super.getInjector();
112   }
113 }
