37  /**
38   * Handle all legacy web service Endpoint requests (excluding IPT requests), previously handled by the GBRDS.
39   */
40  @Singleton
41  @Path("registry/service")
42  public class LegacyEndpointResource {
43  
44    private static final Logger LOG = LoggerFactory.getLogger(LegacyEndpointResource.class);
45  
46    private final DatasetService datasetService;
47  
48    @Inject
49    public LegacyEndpointResource(DatasetService datasetService) {
50      this.datasetService = datasetService;
51    }
52  
53    /**
54     * Register Dataset Endpoint, handling incoming request with path /resource/service. The access point URL, type, and
55     * dataset key are mandatory. Only after both the endpoint has been persisted is a Response with Status.CREATED
56     * returned.
57     *
58     * @param endpoint LegacyEndpoint with HTTP form parameters having been injected from Jersey
59     * @param security SecurityContext (security related information)
60     *
61     * @return Response with Status.CREATED if successful
62     */
63    @POST
64    @Produces(MediaType.APPLICATION_XML)
65    @Consumes(MediaType.APPLICATION_FORM_URLENCODED)
66    public Response registerEndpoint(@InjectParam LegacyEndpoint endpoint, @Context SecurityContext security) {
67      if (endpoint != null) {
68        // set required fields
69        String user = security.getUserPrincipal().getName();
70        endpoint.setCreatedBy(user);
71        endpoint.setModifiedBy(user);
72  
73        // required fields present, and corresponding dataset exists?
74        if (LegacyResourceUtils.isValid(endpoint, datasetService)) {
75  
76          // persist endpoint
77          int key = datasetService.addEndpoint(endpoint.getDatasetKey(), endpoint);
78          LOG.info("Endpoint created successfully, key=%s", String.valueOf(key));
79  
80          // generate response
81          return Response.status(Response.Status.CREATED).cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED)
82            .entity(endpoint).build();
83        } else {
84          LOG.error("Mandatory parameter(s) missing or invalid!");
85        }
86      }
87      LOG.error("Endpoint creation failed");
88      return Response.status(Response.Status.BAD_REQUEST).cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED)
89        .build();
90    }
91  
92    /**
93     * Delete all Endpoints for a Dataset, handling incoming request with path /resource/service and query parameter
94     * resourceKey. Only credentials are mandatory. If deletion is successful, returns Response with Status.OK.
95     *
96     * @param datasetKey dataset key (UUID) coming in as query param
97     *
98     * @return Response with Status.OK if successful
99     */
100   @DELETE
101   @Consumes(MediaType.APPLICATION_FORM_URLENCODED)
102   public Response deleteAllDatasetEndpoints(@QueryParam("resourceKey") UUID datasetKey) {
103     if (datasetKey != null) {
104       // retrieve existing dataset
105       Dataset existing = datasetService.get(datasetKey);
106       if (existing != null) {
107 
108         // delete dataset's endpoints
109         List<Endpoint> endpointList = existing.getEndpoints();
110         for (Endpoint endpoint : endpointList) {
111           datasetService.deleteEndpoint(datasetKey, endpoint.getKey());
112         }
113 
114         LOG.info("Dataset's endpoints deleted successfully, key=%s", datasetKey.toString());
115         return Response.status(Response.Status.OK).cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED).build();
116 
117       } else {
118         LOG.error("Request invalid. Dataset (whose endpoints are to be deleted) no longer exists!");
119       }
120     }
121     LOG.error("Endpoint deletion failed");
122     return Response.status(Response.Status.BAD_REQUEST).cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED)
123       .build();
124   }

126   /**
127    * Retrieve all Endpoints associated to a Dataset, handling incoming request with path /service and query parameter
128    * resourceKey. The dataset key query parameter is mandatory. Only after both the datasetKey is verified to
129    * correspond to an existing Dataset, is a Response including the list of Endpoints returned.
130    * </br>
131    * Alternatively, get a list of all service types, handling incoming request with path /service.json and query
132    * parameter op=types
133    *
134    * @param datasetKey dataset key (UUID) coming in as query param
135    *
136    * @return Response with list of Endpoints or empty list with error message if none found
137    */
138   @GET
139   @Produces({MediaType.APPLICATION_XML, MediaType.APPLICATION_JSON})
140   @Consumes(MediaType.TEXT_PLAIN)
141   public Response endpointsForDataset(@QueryParam("resourceKey") UUID datasetKey, @QueryParam("op") String op) {
142 
143     // get all service types?
144     if (op != null && op.equalsIgnoreCase("types")) {
145       // TODO: replace static list http://dev.gbif.org/issues/browse/REG-394
146       try {
147         String content = Resources.toString(Resources.getResource("legacy/service_types.json"), Charsets.UTF_8);
148         LOG.debug("Get service types finished");
149         return Response.status(Response.Status.OK).entity(content)
150           .cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED).build();
151       } catch (IOException e) {
152         LOG.error("An error occurred retrieving service types");
153         return Response.status(Response.Status.INTERNAL_SERVER_ERROR)
154           .cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED).build();
155       }
156     } else if (datasetKey != null) {
157       try {
158         // verify Dataset with key exists, otherwise NotFoundException gets thrown
159         datasetService.get(datasetKey);
160 
161         LOG.debug("Get all Endpoints for Dataset, key={}", datasetKey);
162         List<LegacyEndpointResponse> endpoints = Lists.newArrayList();
163 
164         LOG.debug("Requesting all endpoints for dataset, key={}", datasetKey);
165         List<Endpoint> response = datasetService.listEndpoints(datasetKey);
166         for (Endpoint e : response) {
167           endpoints.add(new LegacyEndpointResponse(e, datasetKey));
168         }
169 
170         LOG.debug("Get all Endpoints for Dataset finished");
171         // return array, required for serialization otherwise get com.sun.jersey.api.MessageException: A message body
172         // writer for Java class java.util.ArrayList
173         LegacyEndpointResponse[] array = endpoints.toArray(new LegacyEndpointResponse[endpoints.size()]);
174         return Response.status(Response.Status.OK).entity(array)
175           .cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED).build();
176       } catch (NotFoundException e) {
177         LOG.error("The dataset with key {} specified by query parameter does not exist", datasetKey);
178         // the dataset didn't exist, and expected response is "{Error: "No services associated to the organisation}"
179         return Response.status(Response.Status.OK).entity(new ErrorResponse("No dataset matches the key provided"))
180           .cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED).build();
181       }
182     }
183     return Response.status(Response.Status.BAD_REQUEST).cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED)
184       .build();
185   }
