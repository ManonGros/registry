16  package org.gbif.registry.persistence.mapper;
17  
18  import org.gbif.api.model.registry.Metadata;
19  import org.gbif.api.vocabulary.MetadataType;
20  import org.gbif.registry.persistence.mapper.handler.ByteArrayWrapper;
21  
22  import java.util.List;
23  import java.util.UUID;
24  import javax.annotation.Nullable;
25  
26  import org.apache.ibatis.annotations.Param;
27  
28  public interface MetadataMapper {
29    /**
30     * This gets the instance in question.
31     *
32     * @param key of the metadata record to fetch
33     * @return either the requested metadata or {@code null} if it couldn't be found
34     */
35    Metadata get(@Param("key") int key);
36  
37    /**
38     * Return the content of a metadata entry.
39     *
40     * @param key of the metadata record to fetch
41     * @return either the requested metadata or {@code null} if it couldn't be found
42     */
43    ByteArrayWrapper getDocument(@Param("key") int key);
44  
45    /**
46     * Stores a new metadata document with its source document as a byte array exactly as it was.
47     */
48    int create(@Param("meta") Metadata metadata, @Param("data") byte[] content);
49  
50    void delete(@Param("key") int key);
51  
52    /**
53     * Return all metadata entries for a given dataset key ordered by priority and creation date, i.e. first come the
54     * EML documents ordered by creation, then the Dublin Core ones.
55     * @param datasetKey the dataset key the returned metadata belongs to
56     * @param type optional metadata type to filter
57     */
58    List<Metadata> list(@Param("key") UUID datasetKey, @Param("type") @Nullable MetadataType type);
59  
60  }
