39    @Transactional
40    public static <T extends NetworkEntity> UUID create(NetworkEntityMapper<T> mapper, T entity) {
41      checkArgument(entity.getKey() == null, "Unable to create an entity which already has a key");
42      // REVIEW: If this call fails the entity will have been modified anyway! We could make a copy and return that
43  // instead
44      entity.setKey(UUID.randomUUID());
45      mapper.create(entity);
46      return entity.getKey();
47    }

53    @Transactional
54    public static <T extends NetworkEntity> void update(NetworkEntityMapper<T> mapper, T entity) {
55      checkNotNull(entity, "Unable to update an entity when it is not provided");
56      T existing = mapper.get(entity.getKey());
57      checkNotNull(existing, "Unable to update a non existing entity");
58  
59      if (existing.getDeleted() != null) {
60        // allow updates ONLY if they are undeleting too
61        checkArgument(entity.getDeleted() == null,
62          "Unable to update a previously deleted entity unless you clear the deletion timestamp");
63      } else {
64        // do not allow deletion here (for safety) - we have an explicity deletion service
65        checkArgument(entity.getDeleted() == null, "Cannot delete using the update service.  Use the deletion service");
66      }
67  
68      mapper.update(entity);
69    }

81    /**
82     * The simple search option of the list.
83     *
84     * @param mapper To use for the search
85     * @param query A simple query string such as "Pontaurus"
86     * @param page To support paging
87     * @return A paging response
88     */
89    public static <T extends NetworkEntity> PagingResponse<T> search(NetworkEntityMapper<T> mapper, String query,
90      Pageable page) {
91      Preconditions.checkNotNull(page, "To search you must supply a page");
92      long total = mapper.count(query);
93      return new PagingResponse<T>(page.getOffset(), page.getLimit(), total, mapper.search(query, page));
94    }

106   @Transactional
107   public static int addContact(
108     ContactMapper contactMapper, ContactableMapper contactableMapper, UUID targetEntityKey, Contact contact
109     ) {
110     checkArgument(contact.getKey() == null, "Unable to create an entity which already has a key");
111     contactMapper.createContact(contact);
112     // is this a primary contact? We need to make sure it only exists once per type
113     if (contact.isPrimary()) {
114       contactableMapper.updatePrimaryContacts(targetEntityKey, contact.getType());
115     }
116     contactableMapper.addContact(targetEntityKey, contact.getKey(), contact.getType(), contact.isPrimary());
117     return contact.getKey();
118   }
119 
120   @Transactional
121   public static int updateContact(ContactMapper contactMapper, ContactableMapper contactableMapper,
122     UUID targetEntityKey, Contact contact) {
123     checkArgument(contact.getKey() != null, "Unable to update an entity with no key");
124     // null safe checking follows
125     checkArgument(Boolean.TRUE.equals(contactableMapper.areRelated(targetEntityKey, contact.getKey())),
126       "The provided contact is not connected to the given entity");
127     // is this a primary contact? We need to make sure it only exists once per type
128     if (contact.isPrimary()) {
129       contactableMapper.updatePrimaryContacts(targetEntityKey, contact.getType());
130     }
131     contactMapper.updateContact(contact);
132     // update the type and is_primary (is_primary will have been set to false by updatePrimaryContacts above)
133     contactableMapper.updateContact(targetEntityKey, contact.getKey(), contact.getType(), contact.isPrimary());
134     return contact.getKey();
135   }

145   @Transactional
146   public static int addEndpoint(
147     EndpointMapper endpointMapper, EndpointableMapper endpointableMapper, UUID targetEntityKey, Endpoint endpoint,
148     MachineTagMapper machineTagMapper
149     ) {
150     checkArgument(endpoint.getKey() == null, "Unable to create an entity which already has a key");
151     endpointMapper.createEndpoint(endpoint);
152     endpointableMapper.addEndpoint(targetEntityKey, endpoint.getKey());
153 
154     for (MachineTag machineTag : endpoint.getMachineTags()) {
155       machineTag.setCreatedBy(endpoint.getCreatedBy());
156       WithMyBatis.addMachineTag(machineTagMapper, endpointMapper, endpoint.getKey(), machineTag);
157     }
158     return endpoint.getKey();
159   }
