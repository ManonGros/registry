59    @Test
60    public void testListRecordsBySets() throws Exception {
61      Calendar calendar = Calendar.getInstance();
62      calendar.set(2015,9,16);
63      Organization orgIceland = createOrganization(Country.ICELAND);
64      Installation orgIcelandInstallation1 = createInstallation(orgIceland.getKey());
65      Dataset orgIcelandInstallation1Dataset1 = createDataset(orgIceland.getKey(), orgIcelandInstallation1.getKey(), DatasetType.CHECKLIST, calendar.getTime());
66  
67      Installation orgIcelandInstallation2 = createInstallation(orgIceland.getKey());
68      Dataset orgIcelandInstallation2Dataset1 = createDataset(orgIceland.getKey(), orgIcelandInstallation2.getKey(), DatasetType.OCCURRENCE, new Date());
69  
70      Organization org2 = createOrganization(Country.NEW_ZEALAND);
71      Installation org2Installation1 = createInstallation(org2.getKey());
72      createDataset(org2.getKey(), org2Installation1.getKey(), DatasetType.CHECKLIST, new Date());
73  
74      // SetType.COUNTRY
75      Iterator<Record> records = serviceProvider.listRecords(
76              ListRecordsParameters.request()
77                      .withMetadataPrefix(EML_FORMAT.getMetadataPrefix())
78                      .withSetSpec(OaipmhSetRepository.SetType.COUNTRY.getSubsetPrefix() + Country.ICELAND.getIso2LetterCode()));
79      assertTrue("ListRecords verb with set Country return at least 1 record", records.hasNext());
80  
81      // SetType.INSTALLATION
82      records = serviceProvider.listRecords(
83              ListRecordsParameters.request()
84                      .withMetadataPrefix(EML_FORMAT.getMetadataPrefix())
85                      .withSetSpec(OaipmhSetRepository.SetType.INSTALLATION.getSubsetPrefix() + orgIcelandInstallation1.getKey().toString()));
86      assertTrue("ListRecords verb with set Installation return at least 1 record", records.hasNext());
87      assertEquals(orgIcelandInstallation1Dataset1.getKey().toString(), records.next().getHeader().getIdentifier());
88  
89      // SetType.DATASET_TYPE
90      records = serviceProvider.listRecords(
91              ListRecordsParameters.request()
92                      .withMetadataPrefix(EML_FORMAT.getMetadataPrefix())
93                      .withSetSpec(OaipmhSetRepository.SetType.DATASET_TYPE.getSubsetPrefix() + DatasetType.OCCURRENCE.toString()));
94      assertTrue("ListRecords verb with set DatasetType return at least 1 record", records.hasNext());
95      assertEquals(orgIcelandInstallation2Dataset1.getKey().toString(), records.next().getHeader().getIdentifier());
96  
97      // Non-existing Set
98      records = serviceProvider.listRecords(
99              ListRecordsParameters.request()
100                     .withMetadataPrefix(EML_FORMAT.getMetadataPrefix())
101                     .withSetSpec("non-existing-set"));
102     assertFalse("ListRecords verb with non-existing set should return no record", records.hasNext());
103   }

105   /**
106    *
107    * Test that ListRecords verb return all records when the number of records is higher than 'MaxListRecords'.
108    * When the number of records is higher than 'MaxListRecords', at least 2 requests will be sent and a 'resumptionToken'
109    * will be used.
110    *
111    * @throws Exception
112    */
113   @Test
114   public void testListRecordsPaging() throws Exception {
115     int numberOfDataset = 3;
116     Organization orgIceland = createOrganization(Country.ICELAND);
117     Installation orgIcelandInstallation1 = createInstallation(orgIceland.getKey());
118     createDataset(orgIceland.getKey(), orgIcelandInstallation1.getKey(), DatasetType.CHECKLIST, new Date());
119 
120     Installation orgIcelandInstallation2 = createInstallation(orgIceland.getKey());
121     createDataset(orgIceland.getKey(), orgIcelandInstallation2.getKey(), DatasetType.OCCURRENCE, new Date());
122 
123     Organization org2 = createOrganization(Country.NEW_ZEALAND);
124     Installation org2Installation1 = createInstallation(org2.getKey());
125     createDataset(org2.getKey(), org2Installation1.getKey(), DatasetType.CHECKLIST, new Date());
126 
127     // ensure the test will run under the expected configuration
128     assertTrue("OaipmhMockModule 'MaxListRecords' should be set to a value less than " + numberOfDataset, numberOfDataset > OaipmhTestConfiguration.MAX_LIST_RECORDS);
129 
130     Iterator<Record> records = serviceProvider.listRecords(
131             ListRecordsParameters.request()
132                     .withMetadataPrefix(EML_FORMAT.getMetadataPrefix()));
133     List<Record> recordList = Lists.newArrayList(records);
134     assertEquals("ListRecords verb return all records when the number of records is higher than 'MaxListRecords'", numberOfDataset, recordList.size());
135   }
