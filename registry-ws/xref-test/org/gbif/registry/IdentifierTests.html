13  package org.gbif.registry;
14  
15  import org.gbif.api.model.common.paging.PagingResponse;
16  import org.gbif.api.model.registry.Identifier;
17  import org.gbif.api.model.registry.NetworkEntity;
18  import org.gbif.api.service.registry.IdentifierService;
19  import org.gbif.api.service.registry.NetworkEntityService;
20  import org.gbif.registry.utils.Identifiers;
21  
22  import java.util.List;
23  
24  import static org.gbif.registry.LenientAssert.assertLenientEquals;
25  
26  import static org.junit.Assert.assertEquals;
27  import static org.junit.Assert.assertNotNull;
28  import static org.junit.Assert.assertTrue;
29  
30  public class IdentifierTests {
31  
32    public static <T extends NetworkEntity> void testAddDelete(IdentifierService service,
33      NetworkEntityService<T> networkEntityService, T entity) {
34  
35      // check there are none on a newly created entity
36      List<Identifier> identifiers = service.listIdentifiers(entity.getKey());
37      assertNotNull("Identifier list should be empty, not null when no identifiers exist", identifiers);
38      assertTrue("Identifiers should be empty when none added", identifiers.isEmpty());
39  
40      // test additions
41      service.addIdentifier(entity.getKey(), Identifiers.newInstance());
42      service.addIdentifier(entity.getKey(), Identifiers.newInstance());
43      identifiers = service.listIdentifiers(entity.getKey());
44      assertNotNull(identifiers);
45      assertEquals("2 identifiers have been added", 2, identifiers.size());
46  
47      // ensure the search works for this test. One entity with 2 duplicate identifiers is still one entity
48      Identifier identifier = Identifiers.newInstance();
49      PagingResponse<T> entities =
50        networkEntityService.listByIdentifier(identifier.getType(), identifier.getIdentifier(), null);
51      assertEquals("Only one entity should have the identifier", (Long) 1L, entities.getCount());
52      entities =
53        networkEntityService.listByIdentifier(identifier.getIdentifier(), null);
54      assertEquals("Only one entity should have the identifier", (Long) 1L, entities.getCount());
55  
56  
57      // test deletion, ensuring correct one is deleted
58      service.deleteIdentifier(entity.getKey(), identifiers.get(0).getKey());
59      identifiers = service.listIdentifiers(entity.getKey());
60      assertNotNull(identifiers);
61      assertEquals("1 identifier should remain after the deletion", 1, identifiers.size());
62      Identifier expected = Identifiers.newInstance();
63      Identifier created = identifiers.get(0);
64      assertLenientEquals("Created identifier does not read as expected", expected, created);
65    }
66  }
