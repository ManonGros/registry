23  public class DoiMapperTest {
24  
25    private DoiMapper mapper;
26  
27    @ClassRule
28    public static LiquibaseInitializer liquibase = new LiquibaseInitializer(RegistryTestModules.database());
29  
30    @Rule
31    public final DatabaseInitializer databaseRule = new DatabaseInitializer(RegistryTestModules.database());
32  
33    @Before
34    public void setup() {
35      Injector inj = RegistryTestModules.mybatis();
36      mapper = inj.getInstance(DoiMapper.class);
37    }
38  
39    @Test
40    public void testCreate() throws Exception {
41      DOI doi = new DOI("10.998/dead.moon");
42      assertNull(mapper.get(doi));
43      mapper.create(doi, DoiType.DOWNLOAD);
44      DoiData data = mapper.get(doi);
45      assertNotNull(data);
46      assertEquals(DoiStatus.NEW, data.getStatus());
47      assertNull(data.getTarget());
48    }
49  
50    @Test
51    public void testUpdate() throws Exception {
52      DOI doi = new DOI("10.998/dead.kennedys");
53      mapper.create(doi, DoiType.DOWNLOAD);
54  
55      mapper.update(doi, new DoiData(DoiStatus.NEW, null), null);
56      DoiData data = mapper.get(doi);
57      assertEquals(DoiStatus.NEW, data.getStatus());
58      assertNull(data.getTarget());
59  
60      mapper.update(doi, new DoiData(DoiStatus.RESERVED, null), null);
61      data = mapper.get(doi);
62      assertEquals(DoiStatus.RESERVED, data.getStatus());
63      assertNull(data.getTarget());
64  
65      URI uri = URI.create("ftp://ftp.bands.com");
66      mapper.update(doi, new DoiData(DoiStatus.REGISTERED, uri), null);
67      data = mapper.get(doi);
68      assertEquals(DoiStatus.REGISTERED, data.getStatus());
69      assertEquals(uri, data.getTarget());
70    }
71  
72    @Test
73    public void testDelete() throws Exception {
74      DOI doi = new DOI("10.998/dead.kennedys");
75      mapper.create(doi, DoiType.DOWNLOAD);
76      assertNotNull(mapper.get(doi));
77      mapper.delete(doi);
78      assertNull(mapper.get(doi));
79    }
80  }
