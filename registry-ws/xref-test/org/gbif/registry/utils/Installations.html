16  package org.gbif.registry.utils;
17  
18  import org.gbif.api.model.registry.Installation;
19  import org.gbif.api.service.registry.InstallationService;
20  import org.gbif.registry.guice.RegistryTestModules;
21  import org.gbif.registry.ws.resources.InstallationResource;
22  
23  import java.util.UUID;
24  
25  import com.google.inject.Injector;
26  import org.apache.http.auth.UsernamePasswordCredentials;
27  import org.codehaus.jackson.type.TypeReference;
28  
29  public class Installations extends JsonBackedData<Installation> {
30  
31    private static final Installations INSTANCE = new Installations();
32    private static InstallationService installationService;
33  
34    private Installations() {
35      super("data/installation.json", new TypeReference<Installation>() {});
36      Injector i = RegistryTestModules.webservice();
37      installationService = i.getInstance(InstallationResource.class);
38    }
39  
40    public static Installation newInstance(UUID organizationKey) {
41      Installation i = INSTANCE.newTypedInstance();
42      i.setOrganizationKey(organizationKey);
43      return i;
44    }
45  
46    /**
47     * Persist a new Installation associated to a hosting organization for use in Unit Tests.
48     *
49     * @param organizationKey hosting organization key
50     *
51     * @return persisted Installation
52     */
53    public static Installation newPersistedInstance(UUID organizationKey) {
54      Installation i = Installations.newInstance(organizationKey);
55      // password was not included in installation.json, so set it here
56      i.setPassword("password");
57      UUID key = installationService.create(i);
58      // some properties like created, modified are only set when the installation is retrieved anew
59      return installationService.get(key);
60    }
61  
62    /**
63     * Populate credentials used in Installation update ws request.
64     *
65     * @param installation Installation
66     *
67     * @return credentials
68     */
69    public static UsernamePasswordCredentials credentials(Installation installation) {
70      return new UsernamePasswordCredentials(installation.getKey().toString(), installation.getPassword());
71    }
72  }
