047/**
048 * This is parameterized to run the same test routines for the following:
049 * <ol>
050 * <li>The persistence layer</li>
051 * <li>The WS service layer</li>
052 * <li>The WS service client layer</li>
053 * </ol>
054 */
055@RunWith(Parameterized.class)
056public class InstallationIT extends NetworkEntityTest<Installation> {
057
058  private final OrganizationService organizationService;
059  private final NodeService nodeService;
060
061  @Parameters
062  public static Iterable<Object[]> data() {
063    final Injector webservice = webservice();
064    final Injector client = webserviceClient();
065    return ImmutableList.<Object[]>of(new Object[] {webservice.getInstance(InstallationResource.class),
066      webservice.getInstance(OrganizationResource.class),
067      webservice.getInstance(NodeResource.class),
068      null},
069      new Object[] {client.getInstance(InstallationService.class),
070        client.getInstance(OrganizationService.class),
071        client.getInstance(NodeService.class),
072        client.getInstance(SimplePrincipalProvider.class)});
073  }
074
075  public InstallationIT(InstallationService service, OrganizationService organizationService, NodeService nodeService,
076    @Nullable SimplePrincipalProvider pp) {
077    super(service, pp);
078    this.organizationService = organizationService;
079    this.nodeService = nodeService;
080  }
081
082  @Override
083  protected Installation newEntity() {
084    UUID nodeKey = nodeService.create(Nodes.newInstance());
085    Organization o = Organizations.newInstance(nodeKey);
086    UUID key = organizationService.create(o);
087    Organization organization = organizationService.get(key);
088    Installation i = Installations.newInstance(organization.getKey());
089    return i;
090  }
091
092  /**
093   * Tests that we can successfully disable and undisable an installation.
094   */
095  @Test
096  public void disableInstallation() {
097    Installation e = newEntity();
098    UUID key = getService().create(e);
099    e = getService().get(key);
100    assertEquals("Should not be disabled to start", false, e.isDisabled());
101    e.setDisabled(true);
102    getService().update(e);
103    e = getService().get(e.getKey());
104    assertEquals("We just disabled it", true, e.isDisabled());
105    e.setDisabled(false);
106    getService().update(e);
107    e = getService().get(e.getKey());
108    assertEquals("We just un-disabled it", false, e.isDisabled());
109
110  }
111
112  // Easier to test this here than other places due to our utility factory
113  @Test
114  public void testHostedByInstallationList() {
115    Installation installation = create(newEntity(), 1);
116    Organization organization = organizationService.get(installation.getOrganizationKey());
117    Node node = nodeService.get(organization.getEndorsingNodeKey());
118
119    PagingResponse<Installation> resp = nodeService.installations(node.getKey(), new PagingRequest());
120    assertEquals("Paging counts are not being set", Long.valueOf(1), resp.getCount());
121
122    resp = organizationService.installations(organization.getKey(), new PagingRequest());
123    assertEquals("Paging counts are not being set", Long.valueOf(1), resp.getCount());
124  }
