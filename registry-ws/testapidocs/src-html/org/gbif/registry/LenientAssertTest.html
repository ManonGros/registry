001package org.gbif.registry;
002
003import org.gbif.api.model.registry.Contact;
004import org.gbif.registry.utils.Contacts;
005
006import org.junit.Test;
007
008import static org.gbif.registry.LenientAssert.assertLenientEquals;
009
010public class LenientAssertTest {
011
012  @Test
013  public void testLenientAssert() {
014    Contact c = Contacts.newInstance();
015    assertLenientEquals("Same object should be the same", c, c);
016    Contact c2 = Contacts.newInstance();
017    assertLenientEquals("Equivalent object should be the same", c, c2);
018    c2.setKey(1);
019    assertLenientEquals("Key should be ignored", c, c2);
020    c2.setCity("should fail");
021    try {
022      assertLenientEquals("Different objects should through assertion error", c, c2);
023    } catch (AssertionError e) {
024      // expected
025    }
026  }
027}
