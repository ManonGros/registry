001package org.gbif.registry;
002
003import org.gbif.api.model.registry.LenientEquals;
004
005import static org.junit.Assert.fail;
006
007/**
008 * Provides a lenient equals assertion, borrowing shamelessly from {@link org.junit.Assert}.
009 */
010public class LenientAssert {
011
012  /**
013   * Asserts that two objects are leniently equal. If they are not, an {@link AssertionError} is thrown with the given
014   * message. If <code>expected</code> and <code>actual</code> are <code>null</code>,
015   * they are considered equal.
016   * 
017   * @param message the identifying message for the {@link AssertionError} (<code>null</code> okay)
018   * @param expected expected value
019   * @param actual actual value
020   */
021  public static <T, L extends LenientEquals<T>> void assertLenientEquals(String message, L expected, T actual) {
022    if (lenientlyEqualRegardingNull(expected, actual)) {
023      return;
024    } else {
025      failNotEquals(message, expected, actual);
026    }
027  }
028
029  private static <T, L extends LenientEquals<T>> boolean lenientlyEqualRegardingNull(L expected, T actual) {
030    if (expected == null) {
031      return actual == null;
032    }
033
034    return expected.lenientEquals(actual);
035  }
036
037  static private void failNotEquals(String message, Object expected,
038    Object actual) {
039    fail(format(message, expected, actual));
040  }
041
042  static String format(String message, Object expected, Object actual) {
043    String formatted = "";
044    if (message != null && !message.equals("")) {
045      formatted = message + " ";
046    }
047    String expectedString = String.valueOf(expected);
048    String actualString = String.valueOf(actual);
049    if (expectedString.equals(actualString)) {
050      return formatted + "expected: "
051        + formatClassAndValue(expected, expectedString)
052        + " but was: " + formatClassAndValue(actual, actualString);
053    } else {
054      return formatted + "expected:<" + expectedString + "> but was:<"
055        + actualString + ">";
056    }
057  }
058
059  private static String formatClassAndValue(Object value, String valueString) {
060    String className = value == null ? "null" : value.getClass().getName();
061    return className + "<" + valueString + ">";
062  }
063
064}
