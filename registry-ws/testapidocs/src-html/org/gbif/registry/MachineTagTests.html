013package org.gbif.registry;
014
015import org.gbif.api.model.registry.MachineTag;
016import org.gbif.api.model.registry.NetworkEntity;
017import org.gbif.api.service.registry.MachineTagService;
018import org.gbif.registry.utils.MachineTags;
019
020import java.util.List;
021
022import static org.gbif.registry.LenientAssert.assertLenientEquals;
023
024import static org.junit.Assert.assertEquals;
025import static org.junit.Assert.assertNotNull;
026import static org.junit.Assert.assertTrue;
027
028public class MachineTagTests {
029
030  public static <T extends NetworkEntity> void testAddDelete(MachineTagService service, T entity) {
031
032    // check there are none on a newly created entity
033    List<MachineTag> machineTags = service.listMachineTags(entity.getKey());
034    assertNotNull("Machine tag list should be empty, not null when no machine tags exist", machineTags);
035    assertTrue("Machine Tag should be empty when none added", machineTags.isEmpty());
036
037    // test additions
038    service.addMachineTag(entity.getKey(), MachineTags.newInstance());
039    service.addMachineTag(entity.getKey(), MachineTags.newInstance());
040    machineTags = service.listMachineTags(entity.getKey());
041    assertNotNull(machineTags);
042    assertEquals("2 machine tags have been added", 2, machineTags.size());
043
044    // test deletion, ensuring correct one is deleted
045    service.deleteMachineTag(entity.getKey(), machineTags.get(0).getKey());
046    machineTags = service.listMachineTags(entity.getKey());
047    assertNotNull(machineTags);
048    assertEquals("1 machine tag should remain after the deletion", 1, machineTags.size());
049    MachineTag expected = MachineTags.newInstance();
050    MachineTag created = machineTags.get(0);
051    assertLenientEquals("Created machine tag does not read as expected", expected, created);
052  }
053}
