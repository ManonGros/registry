053/**
054 * Utility to provide the different Guice configurations for:
055 * <ol>
056 * <li>The WS service layer</li>
057 * <li>The WS service client layer</li>
058 * <li>A management configuration to allow utilities to manipulate the database (Liquibase etc)</li>
059 * </ol>
060 * Everything is cached, and reused on subsequent calls.
061 */
062public class RegistryTestModules {
063
064  // cache everything, for reuse in repeated calls (e.g. IDE test everything)
065  private static Injector mybatis;
066  private static Injector webservice;
067  private static Injector webserviceClient;
068  private static Injector webserviceBasicAuthClient;
069  private static Injector management;
070  private static HikariDataSource managementDatasource;
071
072  /**
073   * @return An injector that is bound for the mybatis layer and exposes mappers.
074   */
075  public static synchronized Injector mybatis() {
076    if (mybatis == null) {
077      try {
078        final Properties p = new Properties();
079        p.load(Resources.getResourceAsStream("registry-test.properties"));
080        mybatis =
081          Guice.createInjector(new RegistryMyBatisModule(p));
082      } catch (IOException e) {
083        throw Throwables.propagate(e);
084      }
085    }
086    return mybatis;
087  }
088
089  /**
090   * @return An injector that is bound for the webservice layer without SOLR capabilities.
091   */
092  public static synchronized Injector webservice() {
093    if (webservice == null) {
094      try {
095        final Properties p = new Properties();
096        p.load(Resources.getResourceAsStream("registry-test.properties"));
097        webservice =
098          Guice.createInjector(new AbstractModule() {
099
100            @Override
101            protected void configure() {
102              bind(NodeResource.class);
103              bind(OrganizationResource.class);
104              bind(InstallationResource.class);
105              bind(DatasetResource.class);
106              bind(NetworkResource.class);
107              bind(IptResource.class);
108              bind(SecurityContext.class).toInstance(mockAdmin());
109            }
110          }, TestValidateInterceptor.newMethodInterceptingModule(),
111            new DrupalMockModule(), new RegistryMyBatisModule(p), new ImsModule(), new RegistrySearchModule(p),
112            new EventModule(p), new ValidationModule(), new SecurityModule(p), new DoiModule(p), new RabbitMockModule(),
113            new TitleLookupMockModule());
114      } catch (IOException e) {
115        throw Throwables.propagate(e);
116      }
117    }
118    return webservice;
119  }

135  /**
136   * @return An injector that is bound for the webservice client layer.
137   */
138  public static synchronized Injector webserviceClient() {
139    if (webserviceClient == null) {
140      Properties props = new Properties();
141      props.setProperty("registry.ws.url", "http://localhost:" + RegistryServer.getPort());
142      props.setProperty("application.key", "gbif.registry-ws-client-it");
143      props.setProperty("application.secret", "6a55ca16c053e269a9602c02922b30ce49c49be3a68bb2d8908b24d7c1");
144      // Create authentication module, and set principal name, equal to a GBIF User unique account name
145      GbifApplicationAuthModule auth = new GbifApplicationAuthModule(props);
146      auth.setPrincipal("admin");
147      webserviceClient = Guice.createInjector(new RegistryWsClientModule(props), auth);
148    }
149    return webserviceClient;
150  }

153  /**
154   * @return An injector that is bound for the webservice client layer.
155   */
156  public static synchronized Injector webserviceBasicAuthClient(String username, String password) {
157    if (webserviceBasicAuthClient == null) {

162      webserviceClient = Guice.createInjector(new RegistryWsClientModule(props), auth);
163    }
164    return webserviceClient;
165  }

167  /**
168   * @return A datasource that is for use in management activities such as Liquibase, or cleaning between tests.
169   */
170  public static HikariDataSource database() {
171    if (managementDatasource == null) {
172      managementDatasource = RegistryTestModules.management().getInstance(HikariDataSource.class);
173    }
174    return managementDatasource;
175  }
176
177  /**
178   * @return An injector configured to issue a Datasource suitable for database management activities (Liquibase etc).
179   */
180  private static synchronized Injector management() {
181    if (management == null) {
182      try {
183        final Properties p = new Properties();
184        p.load(Resources.getResourceAsStream("registry-test.properties"));
185        management = Guice.createInjector(new AbstractModule() {
186
187          @Override
188          protected void configure() {
189            //Names.bindProperties(binder(), p);
190            //bind(DataSource.class).toProvider(new ManagementProvider());
191          }
192
193          @Provides
194          @Singleton
195          /**
196           * Provides a hikari datasource that can issue connections for management activities, such as Liquibase or
197           * clearing tables before tests run etc.
198           * We provide an implementation specific datasource here so the pool can be properly closed at the end!
199           */
200          public HikariDataSource provideDs() {
201            HikariConfig config = new HikariConfig(filterProperties(p, "registry.db."));
202            config.setConnectionTimeout(5000);
203            config.setMaximumPoolSize(1);
204            return new HikariDataSource(config);
205          }
206        });
207      } catch (Exception e) {
208        throw Throwables.propagate(e);
209      }
210    }
211    return management;
212  }
213
214  private static Properties filterProperties(Properties properties, final String prefix) {
215    Properties filtered = new Properties();
216    for(String key : properties.stringPropertyNames()) {
217      if (key.startsWith(prefix)) {
218        filtered.setProperty(key.substring(prefix.length()), properties.getProperty(key));
219      }
220    }
221    return filtered;
222  }
