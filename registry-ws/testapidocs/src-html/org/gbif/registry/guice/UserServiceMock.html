001package org.gbif.registry.guice;
002
003import org.gbif.api.model.common.User;
004import org.gbif.api.service.common.UserService;
005import org.gbif.api.vocabulary.UserRole;
006
007import java.util.Date;
008import java.util.EnumSet;
009
010import com.google.common.collect.ImmutableMap;
011
012/**
013 * Class that implements an in-memory users store.
014 * This class is intended to be use in IT and unit tests only.
015 */
016public class UserServiceMock implements UserService {
017
018  private static final ImmutableMap<String, String> USERS = new ImmutableMap.Builder<String, String>()
019    .put("admin", "admin").put("user", "user").put("editor", "editor").build();
020
021  private static final ImmutableMap<String, EnumSet<UserRole>> USER_ROLES =
022    new ImmutableMap.Builder<String, EnumSet<UserRole>>()
023      .put("admin", EnumSet.of(UserRole.REGISTRY_ADMIN))
024      .put("user", EnumSet.of(UserRole.USER))
025      .put("editor", EnumSet.of(UserRole.REGISTRY_EDITOR))
026      .build();
027
028  public static final String EMAIL_AT_GIBF = "%s@gbif.test.org";
029
030  /**
031   * Authenticates the user by a simple match of username and pwd.
032   */
033  @Override
034  public User authenticate(String username, String password) {
035    if (USERS.containsKey(username) && USERS.get(username).equals(password)) {
036      return mockUser(username, password);
037    }
038    return null;
039  }
040
041  /**
042   * Creates a full user instance.
043   */
044  private User mockUser(String username, String password) {
045    User user = new User();
046    user.setUserName(username);
047    user.setPasswordHash(password);
048    user.setLastLogin(new Date());
049    user.setEmail(String.format(EMAIL_AT_GIBF, username));
050    user.setFirstName(username);
051    user.setLastName(username);
052    user.setKey(USERS.keySet().asList().indexOf(username));
053    user.setRoles(USER_ROLES.get(username));
054    return user;
055  }
056
057
058  @Override
059  public User get(String username) {
060    if (USERS.containsKey(username)) {
061      return mockUser(username, null);
062    }
063    return null;
064  }
065
066  /**
067   * The session key must be equals to the user name.
068   */
069  @Override
070  public User getBySession(String session) {
071    return mockUser(session, null);
072  }
073
074}
