111  //@Ignore("Problems with IMS connection. See issue: http://dev.gbif.org/issues/browse/REG-407")
112  @Test
113  public void testGetByCountry() {
114    initVotingCountryNodes();
115    insertTestNode(Country.TAIWAN, ParticipationStatus.ASSOCIATE, NodeType.OTHER);
116
117    Node n = nodeService.getByCountry(Country.ANGOLA);
118    assertNull(n);
119
120    for (Country c : TEST_COUNTRIES.keySet()) {
121      n = nodeService.getByCountry(c);
122      assertEquals(c, n.getCountry());
123    }
124
125    // test taiwan hack
126    n = nodeService.getByCountry(Country.TAIWAN);
127    assertEquals(Country.TAIWAN, n.getCountry());
128  }
129
130  private void initVotingCountryNodes() {
131    count = 0;
132    for (Country c : TEST_COUNTRIES.keySet()) {
133      insertTestNode(c, ParticipationStatus.VOTING, NodeType.COUNTRY);
134    }
135  }
136
137  private void insertTestNode(Country c, ParticipationStatus status, NodeType nodeType) {
138      Node n = newEntity();
139      n.setCountry(c);
140      n.setTitle("GBIF Node " + c.getTitle());
141      n.setType(nodeType);
142      n.setParticipationStatus(status);
143      n.setGbifRegion(GbifRegion.AFRICA);
144      n = create(n, count + 1);
145      count++;
146
147      if (TEST_COUNTRIES.containsKey(c)) {
148        // create IMS identifiers
149        Identifier id = new Identifier();
150        id.setType(IdentifierType.GBIF_PARTICIPANT);
151        id.setIdentifier(TEST_COUNTRIES.get(c).toString());
152        nodeService.addIdentifier(n.getKey(), id);
153      }
154  }

169  public void testCountries() {
170    initVotingCountryNodes();
171    List<Country> countries = nodeService.listNodeCountries();
172    assertEquals(TEST_COUNTRIES.size(), countries.size());

178  @Test
179  public void testActiveCountries() {
180    initVotingCountryNodes();
181    List<Country> countries = nodeService.listActiveCountries();
182    assertEquals(TEST_COUNTRIES.size() + 1, countries.size());
183    for (Country c : countries) {
184      assertTrue("Unexpected node country" + c, Country.TAIWAN == c || TEST_COUNTRIES.containsKey(c));
185    }
186    assertTrue("Taiwan missing", countries.contains(Country.TAIWAN));
187
188    // insert extra observer nodes and make sure we get the same list
189    insertTestNode(Country.BOTSWANA, ParticipationStatus.OBSERVER, NodeType.COUNTRY);
190    insertTestNode(Country.HONG_KONG, ParticipationStatus.FORMER, NodeType.COUNTRY);
191
192    List<Country> countries2 = nodeService.listActiveCountries();
193    assertEquals(countries, countries2);
194  }

196  @Test
197  public void testDatasets() {
198    // endorsing node for the organization
199    Node node = create(newEntity(), 1);
200    // publishing organization (required field)
201    Organization o = Organizations.newInstance(node.getKey());
202    o.setEndorsementApproved(true);
203    o.setEndorsingNodeKey(node.getKey());
204    UUID organizationKey = organizationService.create(o);
205    // hosting technical installation (required field)
206    Installation i = Installations.newInstance(organizationKey);
207    UUID installationKey = installationService.create(i);
208    // 2 datasets
209    Dataset d1 = Datasets.newInstance(organizationKey, installationKey);
210    datasetService.create(d1);
211    Dataset d2 = Datasets.newInstance(organizationKey, installationKey);
212    UUID d2Key = datasetService.create(d2);
213
214    // test node service
215    PagingResponse<Dataset> resp = nodeService.endorsedDatasets(node.getKey(), null);
216    assertEquals(2, resp.getResults().size());
217    assertEquals("Paging is not returning the correct count", Long.valueOf(2), resp.getCount());
218
219    // the last created dataset should be the first in the list
220    assertEquals(d2Key, resp.getResults().get(0).getKey());
221  }

240    assertTrue(es.getContacts().size() > 5);
241
242    Node notInIms = nodeService.getByCountry(Country.AFGHANISTAN);
243    assertNotNull(notInIms);
244  }
