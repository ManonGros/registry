043/**
044 * Tests for the OaipmhEndpoint implementation.
045 */
046@RunWith(Parameterized.class)
047public abstract class AbstractOaipmhEndpointIT {
048
049  // used by OAIClient to access the OAI-PMH web service locally
050  private String BASE_URL_FORMAT = "http://localhost:%d/oai-pmh/registry";
051
052  protected MetadataFormat OAIDC_FORMAT = new MetadataFormat()
053          .withMetadataPrefix("oai_dc")
054          .withMetadataNamespace("http://www.openarchives.org/OAI/2.0/oai_dc/")
055          .withSchema("http://www.openarchives.org/OAI/2.0/oai_dc.xsd");
056
057  protected MetadataFormat EML_FORMAT = new MetadataFormat()
058          .withMetadataPrefix("eml")
059          .withMetadataNamespace("eml://ecoinformatics.org/eml-2.1.1")
060          .withSchema("http://rs.gbif.org/schema/eml-gbif-profile/1.0.2/eml.xsd");
061
062  // Flushes the database on each run
063  @ClassRule
064  public static final LiquibaseInitializer liquibaseRule = new LiquibaseInitializer(RegistryTestModules.database());
065
066  @ClassRule
067  public static final RegistryServer registryServer = RegistryServer.INSTANCE;
068
069  @Rule
070  public final DatabaseInitializer databaseRule = new DatabaseInitializer(RegistryTestModules.database());
071
072  private final NodeService nodeService;
073  private final OrganizationService organizationService;
074  private final InstallationService installationService;
075  private final DatasetService datasetService;
076
077  protected final String baseUrl;
078  final ServiceProvider serviceProvider;
079
080  public AbstractOaipmhEndpointIT(NodeService nodeService, OrganizationService organizationService, InstallationService installationService,
081                                  DatasetService datasetService){
082    this.nodeService = nodeService;
083    this.organizationService = organizationService;
084    this.installationService = installationService;
085    this.datasetService = datasetService;
086
087    baseUrl = String.format(BASE_URL_FORMAT, registryServer.getPort());
088    OAIClient oaiClient = new HttpOAIClient(baseUrl);
089    Context context = new Context().withOAIClient(oaiClient).withMetadataTransformer(EML_FORMAT.getMetadataPrefix(), org.dspace.xoai.dataprovider.model.MetadataFormat.identity());
090    serviceProvider = new ServiceProvider(context);
091  }
092
093  @Parameters
094  public static Iterable<Object[]> data() {
095    final Injector webservice = webservice();
096    //final Injector client = webserviceClient();
097    return ImmutableList.<Object[]>of(new Object[]{
098            webservice.getInstance(NodeService.class),
099            webservice.getInstance(OrganizationService.class),
100            webservice.getInstance(InstallationService.class),
101            webservice.getInstance(DatasetService.class)});
102  }
103
104  /**
105   * Creates an Organization in the test database.
106   *
107   * @param publishingCountry
108   * @return
109   */
110  Organization createOrganization(Country publishingCountry) {
111    // endorsing node for the organization
112    UUID nodeKey = nodeService.create(Nodes.newInstance());
113    // publishing organization (required field)
114    Organization o = Organizations.newInstance(nodeKey);
115    o.setCountry(publishingCountry);
116    organizationService.create(o);
117    return o;
118  }
119
120  /**
121   * Creates an Installation in the test database.
122   *
123   * @param organizationKey
124   * @return
125   */
126  Installation createInstallation(UUID organizationKey) {
127    Installation i = Installations.newInstance(organizationKey);
128    installationService.create(i);
129    return i;
130  }
131
132  /**
133   * Creates a Dataset in the test database.
134   *
135   * @param organizationKey
136   * @param installationKey
137   * @param type
138   * @param modifiedDate
139   * @return the newly created Dataset
140   * @throws Throwable
141   */
142  Dataset createDataset(UUID organizationKey, UUID installationKey, DatasetType type, Date modifiedDate) throws Exception {
143
144    Dataset d = Datasets.newInstance(organizationKey, installationKey);
145    d.setType(type);
146    datasetService.create(d);
147
148    // since modifiedDate is set automatically by the datasetMapper we update it manually
149    changeDatasetModifiedDate(d.getKey(), modifiedDate);
150    d.setModified(modifiedDate);
151
152    return d;
153  }
154
155  /**
156   * Get the specified Dataset
157   * @param datasetKey
158   * @return
159   * @throws Exception
160   */
161  Dataset getDataset(UUID datasetKey) throws Exception {
162    return datasetService.get(datasetKey);
163  }
164
165
166  /**
167   * Delete the specified Dataset.
168   *
169   * @param datasetKey
170   * @throws Exception
171   */
172  void deleteDataset(UUID datasetKey) throws Exception {
173    datasetService.delete(datasetKey);
174  }
175
176  /**
177   * Update the provided Dataset.
178   *
179   * @param dataset
180   * @throws Exception
181   */
182  void updateDataset(Dataset dataset) throws Exception {
183    datasetService.update(dataset);
184  }
185
186  /**
187   * Insert metadata associated to the provided Dataset.
188   *
189   * @param key dataset key
190   * @param document
191   * @throws Exception
192   */
193  Metadata insertMetadata(UUID key, InputStream document) throws Exception {
194    return datasetService.insertMetadata(key, document);
195  }
196
197  /**
198   * This method is used to change the modified date of a dataset in order to test date queries.
199   *
200   * @param key
201   * @param modifiedDate new modified date to set
202   *
203   */
204  void changeDatasetModifiedDate(UUID key, Date modifiedDate) throws Exception {
205    Connection connection = null;
206    try {
207      connection = RegistryTestModules.database().getConnection();
208      connection.setAutoCommit(false);
209
210      PreparedStatement p = connection.prepareStatement("UPDATE dataset SET modified = ? WHERE key = ?");
211
212      p.setDate(1, new java.sql.Date(modifiedDate.getTime()));
213      p.setObject(2, key);
214
215      p.execute();
216      connection.commit();
217    }finally {
218      if (connection != null) {
219        connection.close();
220      }
221    }
222  }
