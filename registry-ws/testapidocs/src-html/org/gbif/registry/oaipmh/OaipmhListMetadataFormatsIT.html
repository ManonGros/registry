015
016import com.google.common.collect.Lists;
017import org.dspace.xoai.model.oaipmh.MetadataFormat;
018import org.dspace.xoai.serviceprovider.exceptions.IdDoesNotExistException;

023
024import static org.hamcrest.Matchers.containsInAnyOrder;
025import static org.hamcrest.Matchers.samePropertyValuesAs;
026import static org.junit.Assert.assertThat;
027
028/**
029 * Test the ListMetadataFormats verb of the OAI-PMH endpoint.
030 */
031@RunWith(Parameterized.class)
032public class OaipmhListMetadataFormatsIT extends AbstractOaipmhEndpointIT {
033
034  public OaipmhListMetadataFormatsIT(NodeService nodeService, OrganizationService organizationService, InstallationService installationService, DatasetService datasetService) {
035    super(nodeService, organizationService, installationService, datasetService);
036  }
037
038  @Test(expected = IdDoesNotExistException.class)
039  public void listMetadataFormats_notFound() throws IdDoesNotExistException {
040    serviceProvider.listMetadataFormats(
041            ListMetadataParameters.request()
042                    .withIdentifier("non-existent-record-identifier")
043    );
044  }
045
046  @Test
047  public void listMetadataFormats_forARecord() throws Exception {
048    Organization org1 = createOrganization(Country.ETHIOPIA);
049    Installation org1Installation1 = createInstallation(org1.getKey());
050    Dataset org1Installation1Dataset1 = createDataset(org1.getKey(), org1Installation1.getKey(), DatasetType.OCCURRENCE, new Date());
051
052    String key = org1Installation1Dataset1.getKey().toString();
053
054    List<MetadataFormat> metadataFormats = Lists.newArrayList(serviceProvider.listMetadataFormats(
055            ListMetadataParameters.request()
056                    .withIdentifier(key)
057    ));
058
059    assertThat("EML and OAIDC formats supported", metadataFormats, containsInAnyOrder(samePropertyValuesAs(OAIDC_FORMAT), samePropertyValuesAs(EML_FORMAT)));
060  }
061
062  @Test
063  public void listMetadataFormats_forRepository() throws Exception {
064
065    List<MetadataFormat> metadataFormats = Lists.newArrayList(serviceProvider.listMetadataFormats(
066            ListMetadataParameters.request()
067    ));
068
069    assertThat("EML and OAIDC formats supported", metadataFormats, containsInAnyOrder(samePropertyValuesAs(OAIDC_FORMAT), samePropertyValuesAs(EML_FORMAT)));
070  }
071}
