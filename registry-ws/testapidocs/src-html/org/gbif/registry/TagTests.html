031package org.gbif.registry;
032
033import org.gbif.api.model.registry.NetworkEntity;
034import org.gbif.api.model.registry.Tag;
035import org.gbif.api.service.registry.TagService;
036
037import java.util.List;
038import java.util.UUID;
039
040import static org.junit.Assert.assertEquals;
041import static org.junit.Assert.assertNotNull;
042import static org.junit.Assert.assertTrue;
043
044public class TagTests {
045
046  public static <T extends NetworkEntity> void testTagErroneousDelete(TagService service, T entity) {
047    int tagKey = service.addTag(entity.getKey(), "tag1");
048    service.deleteTag(UUID.randomUUID(), tagKey); // wrong parent UUID
049    // nothing happens - expected?
050  }
051
052  public static <T extends NetworkEntity> void testAddDelete(TagService service, T entity) {
053    List<Tag> tags = service.listTags(entity.getKey(), null);
054    assertNotNull("Tag list should be empty, not null when no tags exist", tags);
055    assertTrue("Tags should be empty when none added", tags.isEmpty());
056    service.addTag(entity.getKey(), "tag1");
057    service.addTag(entity.getKey(), "tag2");
058    tags = service.listTags(entity.getKey(), null);
059    assertNotNull(tags);
060    assertEquals("2 tags have been added", 2, tags.size());
061    service.deleteTag(entity.getKey(), tags.get(0).getKey());
062    tags = service.listTags(entity.getKey(), null);
063    assertNotNull(tags);
064    assertEquals("1 tag should remain after the deletion", 1, tags.size());
065  }
066
067}
