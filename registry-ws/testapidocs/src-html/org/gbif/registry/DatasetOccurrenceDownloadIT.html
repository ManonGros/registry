136  /**
137   * Creates a test dataset. The dataset is persisted in the data base.
138   * The installation and organization related to the dataset are created too.
139   */
140  private Dataset createTestDataset() {
141    // endorsing node for the organization
142    UUID nodeKey = nodeService.create(Nodes.newInstance());
143
144    // publishing organization (required field)
145    Organization o = Organizations.newInstance(nodeKey);
146    UUID organizationKey = organizationService.create(o);
147
148    Installation i = Installations.newInstance(organizationKey);
149    UUID installationKey = installationService.create(i);
150    Dataset dataset = Datasets.newInstance(organizationKey, installationKey);
151    dataset.setKey(datasetService.create(dataset));
152    return dataset;
153  }
154
155  /**
156   * Tests the process of persist a {@link DatasetOccurrenceDownload} and list the downloads by dataset key.
157   */
158  @Test
159  public void testAddAndGetOccurrenceDataset() {
160    Download occurrenceDownload = OccurrenceDownloadIT.getTestInstance();
161    final Dataset testDataset = createTestDataset();
162    DatasetOccurrenceDownloadUsage downloadDataset = new DatasetOccurrenceDownloadUsage();
163    downloadDataset.setDownloadKey(occurrenceDownload.getKey());
164    downloadDataset.setDatasetKey(testDataset.getKey());
165    downloadDataset.setNumberRecords(1000L);
166    downloadDataset.setDatasetTitle(testDataset.getTitle());
167    downloadDataset.setDatasetDOI(new DOI("doi:10.1234/1ASCDU"));
168    downloadDataset.setDatasetCitation(testDataset.getCitation().getText());
169    occurrenceDownloadService.create(occurrenceDownload);
170    datasetOccurrenceDownloadUsageService.create(downloadDataset);
171    assertEquals("List operation should return 1 record", 1,
172      datasetOccurrenceDownloadUsageService.listByDataset(testDataset.getKey(), new PagingRequest(0, 3))
173        .getResults().size());
174    Download occDownload2 = occurrenceDownloadService.get(occurrenceDownload.getKey());
175    assertEquals(occDownload2.getNumberDatasets(), 1);
176  }
