001package org.gbif.registry.persistence.mapper;
002
003import org.gbif.api.model.common.DOI;
004import org.gbif.api.model.common.DoiData;
005import org.gbif.api.model.common.DoiStatus;
006import org.gbif.registry.database.DatabaseInitializer;
007import org.gbif.registry.database.LiquibaseInitializer;
008import org.gbif.registry.doi.DoiType;
009import org.gbif.registry.guice.RegistryTestModules;
010
011import java.net.URI;
012
013import com.google.inject.Injector;
014import org.junit.Before;
015import org.junit.ClassRule;
016import org.junit.Rule;
017import org.junit.Test;
018
019import static org.junit.Assert.assertEquals;
020import static org.junit.Assert.assertNotNull;
021import static org.junit.Assert.assertNull;
022
023public class DoiMapperTest {
024
025  private DoiMapper mapper;
026
027  @ClassRule
028  public static LiquibaseInitializer liquibase = new LiquibaseInitializer(RegistryTestModules.database());
029
030  @Rule
031  public final DatabaseInitializer databaseRule = new DatabaseInitializer(RegistryTestModules.database());
032
033  @Before
034  public void setup() {
035    Injector inj = RegistryTestModules.mybatis();
036    mapper = inj.getInstance(DoiMapper.class);
037  }
038
039  @Test
040  public void testCreate() throws Exception {
041    DOI doi = new DOI("10.998/dead.moon");
042    assertNull(mapper.get(doi));
043    mapper.create(doi, DoiType.DOWNLOAD);
044    DoiData data = mapper.get(doi);
045    assertNotNull(data);
046    assertEquals(DoiStatus.NEW, data.getStatus());
047    assertNull(data.getTarget());
048  }
049
050  @Test
051  public void testUpdate() throws Exception {
052    DOI doi = new DOI("10.998/dead.kennedys");
053    mapper.create(doi, DoiType.DOWNLOAD);
054
055    mapper.update(doi, new DoiData(DoiStatus.NEW, null), null);
056    DoiData data = mapper.get(doi);
057    assertEquals(DoiStatus.NEW, data.getStatus());
058    assertNull(data.getTarget());
059
060    mapper.update(doi, new DoiData(DoiStatus.RESERVED, null), null);
061    data = mapper.get(doi);
062    assertEquals(DoiStatus.RESERVED, data.getStatus());
063    assertNull(data.getTarget());
064
065    URI uri = URI.create("ftp://ftp.bands.com");
066    mapper.update(doi, new DoiData(DoiStatus.REGISTERED, uri), null);
067    data = mapper.get(doi);
068    assertEquals(DoiStatus.REGISTERED, data.getStatus());
069    assertEquals(uri, data.getTarget());
070  }
071
072  @Test
073  public void testDelete() throws Exception {
074    DOI doi = new DOI("10.998/dead.kennedys");
075    mapper.create(doi, DoiType.DOWNLOAD);
076    assertNotNull(mapper.get(doi));
077    mapper.delete(doi);
078    assertNull(mapper.get(doi));
079  }
080}
