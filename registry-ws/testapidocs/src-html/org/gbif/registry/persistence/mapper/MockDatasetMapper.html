040  /**
041   * Mock specific method
042   * Warning: this method will add the country as a distinct country and increment the dataset counts for that country
043   * @param country
044   */
045  public void mockDatasetForCountry(Country country) {
046    distinctCountries.add(country);
047    if(datasetsByCountry.containsKey(country)){
048      datasetsByCountry.put(country, datasetsByCountry.get(country) + 1);
049    }
050    else{
051      datasetsByCountry.put(country, 1l);
052    }
053  }

055  /**
056   * Mock specific method
057   * Warning: this method sets datasetsByInstallation to 1 for the specified country
058   * @param installation
059   */
060  public void mockDatasetForInstallation(Installation installation) {
061    distinctInstallation.add(installation);
062    UUID installationKey = installation.getKey();
063    if(datasetsByInstallation.containsKey(installationKey)){
064      datasetsByInstallation.put(installationKey, datasetsByInstallation.get(installationKey) + 1);
065    }
066    else{
067      datasetsByInstallation.put(installationKey, 1l);
068    }
069  }
