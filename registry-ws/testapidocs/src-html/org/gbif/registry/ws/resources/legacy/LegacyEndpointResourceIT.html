068  /**
069   * The test begins by persisting a new Organization, Installation associated to the Organization, and Dataset
070   * associated to the Organization.
071   * </br>
072   * Then, it sends a create Endpoint (POST) request to create a new Endpoint associated to this dataset.
073   * </br>
074   * Upon receiving an HTTP Response, the test parses its XML content in order to extract the created Endpoint key.
075   * The content is parsed exactly the same way as the GBRDS WS consumer would do it.
076   * </br>
077   * Last, the test validates that the endpoint was persisted correctly.
078   */
079  @Test
080  public void testRegisterLegacyEndpoint() throws Exception {
081    // persist new organization (IPT hosting organization)
082    Organization organization = Organizations.newPersistedInstance();
083    UUID organizationKey = organization.getKey();
084
085    // persist new installation
086    Installation installation = Installations.newPersistedInstance(organizationKey);
087    UUID installationKey = installation.getKey();
088
089    // persist new Dataset associated to installation
090    Dataset dataset = Datasets.newPersistedInstance(organizationKey, installationKey);
091    UUID datasetKey = dataset.getKey();
092
093    // populate params for ws
094    List<NameValuePair> data = buildLegacyEndpointParameters(datasetKey);
095    UrlEncodedFormEntity uefe = new UrlEncodedFormEntity(data, Charset.forName("UTF-8"));
096
097    // construct request uri
098    String uri = Requests.getRequestUri("/registry/service");
099
100    // before sending the update POST request, count the number of endpoints
101    assertEquals(0, datasetService.listEndpoints(datasetKey).size());
102
103    // send POST request with credentials
104    HttpUtil.Response result = Requests.http.post(uri, null, null, Organizations.credentials(organization), uefe);
105
106    // parse newly registered endpoint key (id)
107    Parsers.saxParser.parse(Parsers.getUtf8Stream(result.content), Parsers.legacyEndpointResponseHandler);
108    assertNotNull("Registered Endpoint key should be in response", Parsers.legacyEndpointResponseHandler.key);
109    assertEquals(datasetKey.toString(), Parsers.legacyEndpointResponseHandler.resourceKey);
110    assertEquals(ENDPOINT_ACCESS_POINT_URL, Parsers.legacyEndpointResponseHandler.accessPointURL);
111    assertEquals(ENDPOINT_TYPE, Parsers.legacyEndpointResponseHandler.type);
112    assertEquals(ENDPOINT_DESCRIPTION, Parsers.legacyEndpointResponseHandler.description);
113
114    // count the number of endpoints
115    assertEquals(1, datasetService.listEndpoints(datasetKey).size());
116  }

154   * is missing. The test must check that the server responds  with a 400 BAD_REQUEST Response.
155   */
156  @Test
157  public void testRegisterLegacyEndpointWithNoType() throws Exception {
158    // persist new organization (IPT hosting organization)
159    Organization organization = Organizations.newPersistedInstance();
160    UUID organizationKey = organization.getKey();
161
162    // persist new installation
163    Installation installation = Installations.newPersistedInstance(organizationKey);
164    UUID installationKey = installation.getKey();
165
166    // persist new Dataset associated to installation
167    Dataset dataset = Datasets.newPersistedInstance(organizationKey, installationKey);
168    UUID datasetKey = dataset.getKey();
169
170    // populate params for ws
171    List<NameValuePair> data = buildLegacyEndpointParameters(datasetKey);
172    assertEquals(4, data.size());
173    // remove mandatory key/value before sending
174    Iterator<NameValuePair> iter = data.iterator();
175    while (iter.hasNext()) {
176      NameValuePair pair = iter.next();
177      if (pair.getName().equals("type")) {
178        iter.remove();
179      }
180    }
181    assertEquals(3, data.size());
182    UrlEncodedFormEntity uefe = new UrlEncodedFormEntity(data, Charset.forName("UTF-8"));
183
184    // construct request uri
185    String uri = Requests.getRequestUri("/registry/service");
186
187    // send POST request with credentials
188    HttpUtil.Response result = Requests.http.post(uri, null, null, Organizations.credentials(organization), uefe);
189
190    // 400 expected
191    assertEquals(Response.Status.BAD_REQUEST.getStatusCode(), result.getStatusCode());
192  }

235   * The test sends a get all endpoints for dataset (GET) request, the JSON response having at the very least
236   * the dataset key, endpoint key, endpoint type, and endpoint url.
237   */
238  @Test
239  public void testGetLegacyEndpointsForDatasetJSON() throws Exception {
240    // persist new organization (IPT hosting organization)
241    Organization organization = Organizations.newPersistedInstance();
242    UUID organizationKey = organization.getKey();
243
244    // persist new installation of type IPT
245    Installation installation = Installations.newPersistedInstance(organizationKey);
246    UUID installationKey = installation.getKey();
247
248    // persist new Dataset associated to installation
249    Dataset dataset = Datasets.newPersistedInstance(organizationKey, installationKey);
250    UUID datasetKey = dataset.getKey();
251
252    // add endpooint for Dataset
253    Endpoint endpoint = Endpoints.newInstance();
254    int endpointKey = datasetService.addEndpoint(datasetKey, endpoint);
255
256    // count the number of endpoints
257    assertEquals(1, datasetService.listEndpoints(datasetKey).size());
258
259    // construct request uri
260    String uri = Requests.getRequestUri("/registry/service.json?resourceKey=" + datasetKey.toString());
261
262    // send GET request with no credentials
263    HttpUtil.Response result = Requests.http.get(uri);
264
265    // JSON array expected, with single endpoint
266    assertTrue(result.content.startsWith("[") && result.content.endsWith("]"));
267    JsonNode rootNode = objectMapper.readTree(result.content);
268    assertEquals(1, rootNode.size());
269    // all the expected keys in response?
270    assertEquals(String.valueOf(endpointKey), rootNode.get(0).get(LegacyResourceConstants.KEY_PARAM).getTextValue());
271    assertEquals(datasetKey.toString(), rootNode.get(0).get(LegacyResourceConstants.RESOURCE_KEY_PARAM).getTextValue());
272    assertEquals(endpoint.getType().toString(), rootNode.get(0).get(LegacyResourceConstants.TYPE_PARAM).getTextValue());
273    assertEquals(endpoint.getUrl().toASCIIString(),
274                 rootNode.get(0).get(LegacyResourceConstants.ACCESS_POINT_URL_PARAM).getTextValue());
275    // simply empty values, but expected nonetheless
276    assertNotNull(rootNode.get(0).get(LegacyResourceConstants.DESCRIPTION_LANGUAGE_PARAM).getTextValue());
277    assertNotNull(rootNode.get(0).get(LegacyResourceConstants.TYPE_DESCRIPTION_PARAM).getTextValue());
278  }

342  /**
343   * Populate a list of name value pairs used in the common ws requests for GBRDS endpoint creation.
344   *
345   * @param datasetKey dataset key
346   *
347   * @return list of name value pairs
348   */
349  private List<NameValuePair> buildLegacyEndpointParameters(UUID datasetKey) {
350    List<NameValuePair> data = new ArrayList<NameValuePair>();
351    data.add(new BasicNameValuePair(LegacyResourceConstants.RESOURCE_KEY_PARAM, datasetKey.toString()));
352    data.add(new BasicNameValuePair(LegacyResourceConstants.DESCRIPTION_PARAM, ENDPOINT_DESCRIPTION));
353    data.add(new BasicNameValuePair(LegacyResourceConstants.TYPE_PARAM, ENDPOINT_TYPE));
354    data.add(new BasicNameValuePair(LegacyResourceConstants.ACCESS_POINT_URL_PARAM, ENDPOINT_ACCESS_POINT_URL));
355    return data;
356  }
