137  /**
138   * The test begins by persisting a new Organization, and Installation associated to the Organization.
139   * </br>
140   * Then, it sends an update IPT (POST) request to update the same Installation. The request is issued against the
141   * web services running on the local Grizzly test server. The request is sent in exactly the same way as the IPT
142   * would send it, using the URL path (/ipt/update/{key}), URL encoded form parameters, and basic authentication. The
143   * web service authorizes the request, and then persists the Installation, updating its information.
144   * </br>
145   * Upon receiving an HTTP Response, the test parses its XML content in order to extract the registered IPT UUID for
146   * example. The content is parsed exactly the same way as the IPT would do it.
147   * </br>
148   * Next, the test validates that the Installation's information was updated correctly. The same request is then
149   * resent once more, and the test validates that no duplicate installation, contact, or endpoint was created.
150   */
151  @Test
152  public void testUpdateIpt() throws Exception {
153    // persist new organization (IPT hosting organization)
154    Organization organization = Organizations.newPersistedInstance();
155    UUID organizationKey = organization.getKey();
156
157    // persist new installation of type IPT
158    Installation installation = Installations.newPersistedInstance(organizationKey);
159    UUID installationKey = installation.getKey();
160
161    // validate it
162    validateExistingIptInstallation(installation, organizationKey);
163
164    // some information never going to change
165    Date created = installation.getCreated();
166    assertNotNull(created);
167    String createdBy = installation.getCreatedBy();
168    assertNotNull(createdBy);
169
170    // populate params for ws
171    List<NameValuePair> data = buildIPTParameters(organizationKey);
172
173    UrlEncodedFormEntity uefe = new UrlEncodedFormEntity(data, Charset.forName("UTF-8"));
174
175    // construct request uri
176    String uri = Requests.getRequestUri("/registry/ipt/update/" + installationKey.toString());
177
178    // send POST request with credentials
179    HttpUtil.Response result = Requests.http.post(uri, null, null, Installations.credentials(installation), uefe);
180
181    // correct response code? Jersey resource should really respond with 201, but 2XX means success
182    assertEquals(Response.Status.CREATED.getStatusCode(), result.getStatusCode());
183
184    // some information that should have been updated
185    installation = validatePersistedIptInstallation(installationKey, organizationKey);
186
187    // some additional information that should not have been updated
188    assertEquals(created, installation.getCreated());
189    assertEquals(createdBy, installation.getCreatedBy());
190
191    // before sending the same POST request..
192    // count the number of installations, contacts and endpoints
193    assertEquals(1, installationService.list(new PagingRequest(0, 10)).getResults().size());
194    assertEquals(1, installation.getContacts().size());
195    assertEquals(1, installation.getEndpoints().size());
196    // keep track of contact and endpoint key
197    int contactKey = installation.getContacts().get(0).getKey();
198    int endpointKey = installation.getEndpoints().get(0).getKey();
199
200    // send same POST request again, to check that duplicate contact and endpoints don't get persisted
201    Requests.http.post(uri, null, null, Installations.credentials(installation), uefe);
202
203    // retrieve newly updated installation, and make sure the same number of installations, contacts and endpoints exist
204    assertEquals(1, installationService.list(new PagingRequest(0, 10)).getResults().size());
205    installation = validatePersistedIptInstallation(installationKey, organizationKey);
206    assertEquals(1, installation.getContacts().size());
207    assertEquals(1, installation.getEndpoints().size());
208
209    // compare contact key and make sure it doesn't change after update (Contacts are mutable)
210    assertEquals(String.valueOf(contactKey), String.valueOf(installation.getContacts().get(0).getKey()));
211    // compare endpoint key and make sure it does change after update (Endpoints are not mutable)
212    assertNotEquals(String.valueOf(endpointKey), String.valueOf(installation.getEndpoints().get(0).getKey()));
213  }

916    data.add(new BasicNameValuePair(LegacyResourceConstants.DESCRIPTION_PARAM, IPT_DESCRIPTION));
917
918    // primary contact
919    data.add(new BasicNameValuePair(LegacyResourceConstants.PRIMARY_CONTACT_TYPE_PARAM, IPT_PRIMARY_CONTACT_TYPE));
920    data.add(new BasicNameValuePair(LegacyResourceConstants.PRIMARY_CONTACT_NAME_PARAM, IPT_PRIMARY_CONTACT_NAME));
921    data.add(new BasicNameValuePair(LegacyResourceConstants.PRIMARY_CONTACT_EMAIL_PARAM, IPT_PRIMARY_CONTACT_EMAIL.get(0)));
922
923    // service/endpoint
924    data.add(new BasicNameValuePair(LegacyResourceConstants.SERVICE_TYPES_PARAM, IPT_SERVICE_TYPE));
925    data.add(new BasicNameValuePair(LegacyResourceConstants.SERVICE_URLS_PARAM, IPT_SERVICE_URL.toASCIIString()));

928    data.add(new BasicNameValuePair(LegacyResourceConstants.WS_PASSWORD_PARAM, IPT_WS_PASSWORD));
929
930    return data;
