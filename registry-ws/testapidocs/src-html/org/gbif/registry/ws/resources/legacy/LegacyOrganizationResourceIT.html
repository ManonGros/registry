059  /**
060   * The test sends a get Organization (GET) request with no parameters and .json, signifying that the response must be
061   * JSON.
062   */
063  @Test
064  public void testGetOrganizationJSON() throws IOException, URISyntaxException, SAXException {
065    // persist new organization (IPT hosting organization)
066    Organization organization = Organizations.newPersistedInstance();
067    Contact c = Contacts.newInstance();
068    c.setType(ContactType.TECHNICAL_POINT_OF_CONTACT);
069    organizationService.addContact(organization.getKey(), c);
070
071    // construct request uri
072    String uri = Requests.getRequestUri("/registry/organisation/" + organization.getKey().toString() + ".json");
073
074    // send GET request with no credentials
075    HttpUtil.Response result = Requests.http.get(uri);
076
077    // JSON expected
078    assertTrue(result.content.startsWith("{"));
079    JsonNode rootNode = objectMapper.readTree(result.content);
080
081    // JSON object expected
082    assertEquals(15, rootNode.size());
083
084    assertEquals(organization.getKey().toString(), rootNode.get("key").getTextValue());
085    assertEquals(organization.getTitle(), rootNode.get("name").getTextValue());
086    assertEquals(organization.getLanguage().getIso2LetterCode(), rootNode.get("nameLanguage").getTextValue());
087    assertEquals(organization.getDescription(), rootNode.get("description").getTextValue());
088    assertEquals(organization.getLanguage().getIso2LetterCode(), rootNode.get("descriptionLanguage").getTextValue());
089    assertEquals(organization.getHomepage().toString(), rootNode.get("homepageURL").getTextValue());
090    assertEquals(LegacyResourceConstants.TECHNICAL_CONTACT_TYPE, rootNode.get("primaryContactType").getTextValue());
091    assertEquals("Tim Robertson", rootNode.get("primaryContactName").getTextValue());
092    assertEquals("trobertson@gbif.org", rootNode.get("primaryContactEmail").getTextValue());
093    assertEquals("Universitetsparken 15", rootNode.get("primaryContactAddress").getTextValue());
094    assertEquals("+45 28261487", rootNode.get("primaryContactPhone").getTextValue());
095    assertEquals(organization.getEndorsingNodeKey().toString(), rootNode.get("nodeKey").getTextValue());
096    assertEquals("The UK National Node", rootNode.get("nodeName").getTextValue());
097    assertEquals("", rootNode.get("nodeContactEmail").getTextValue());
098  }
