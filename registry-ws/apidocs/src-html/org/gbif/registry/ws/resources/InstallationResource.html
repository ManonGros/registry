127  /**
128   * All network entities support simple (!) search with "&q=".
129   * This is to support the console user interface, and is in addition to any complex, faceted search that might
130   * additionally be supported, such as dataset search.
131   */
132  @GET
133  public PagingResponse<Installation> list(@Nullable @QueryParam("q") String query,
134    @Nullable @QueryParam("identifierType") IdentifierType identifierType,
135    @Nullable @QueryParam("identifier") String identifier,
136    @Nullable @Context Pageable page) {
137    // This is getting messy: http://dev.gbif.org/issues/browse/REG-426
138    if (identifierType != null && identifier != null) {
139      return listByIdentifier(identifierType, identifier, page);
140    } else if (identifier != null) {
141      return listByIdentifier(identifier, page);
142    } else if (Strings.isNullOrEmpty(query)) {
143      return list(page);
144    } else {
145      return search(query, page);
146    }
147  }

171  /**
172   * This is a REST only (e.g. not part of the Java API) method that allows the registry console to trigger the
173   * synchronization of the installation. This simply emits a message to rabbitmq requesting the sync, and applies
174   * necessary security.
175   */
176  @POST
177  @Path("{key}/synchronize")
178  @RolesAllowed(ADMIN_ROLE)
179  public void synchronize(@PathParam("key") UUID installationKey) {
180    if (messagePublisher != null) {
181      LOG.info("Requesting synchronizing installation[{}]", installationKey);
182      try {
183        messagePublisher.send(new StartMetasyncMessage(installationKey));
184      } catch (IOException e) {
185        LOG.error("Unable to send message requesting synchronization", e);
186      }
187
188    } else {
189      LOG.warn("Registry is configured to run without messaging capabilities.  Unable to synchronize installation[{}]",
190        installationKey);
191    }
192  }

194  /**
195   * This is a REST only (e.g. not part of the Java API) method that allows you to get the locations of installations as
196   * GeoJSON. This method exists primarily to produce the content for the "locations of organizations hosting an IPT".
197   * The response holds the distinct organizations running the installations of the specified type.
198   */
199  @GET
200  @Path("location/{type}")
201  public String organizationsAsGeoJSON(@PathParam("type") InstallationType type) {
202    List<Organization> orgs = organizationMapper.hostingInstallationsOf(type, true);
203
204    // to increment the count on duplicates
205    Map<Organization, AtomicInteger> counts = Maps.newHashMap();
206    for (Organization o : orgs) {
207      if (counts.containsKey(o)) {
208        counts.get(o).incrementAndGet();
209      } else {
210        counts.put(o, new AtomicInteger(1));
211      }
212    }
213
214    JSONObject featureCollection = new JSONObject();
215    try {
216      featureCollection.put("type", "FeatureCollection");
217
218      List<JSONObject> features = Lists.newArrayList();
219      for (Organization o : counts.keySet()) {
220        JSONObject feature = new JSONObject();
221        feature.put("type", "Feature");
222        feature.put("properties", ImmutableMap.<String, Object>of(
223          "key", o.getKey(),
224          "title", o.getTitle(),
225          "count", counts.get(o).get()));
226        JSONObject geom = new JSONObject();
227        geom.put("type", "Point");
228        geom.put("coordinates", ImmutableList.<BigDecimal>of(
229          o.getLongitude(), o.getLatitude()));
230        feature.put("geometry", geom);
231        features.add(feature);
232      }
233      featureCollection.put("features", features);
234    } catch (JSONException e) {
235      LOG.error("Unable to build GeoJSON", e);
236    }
237
238    return featureCollection.toString();
239  }
