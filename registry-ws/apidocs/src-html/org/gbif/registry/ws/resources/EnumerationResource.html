013package org.gbif.registry.ws.resources;
014
015import org.gbif.api.util.VocabularyUtils;
016import org.gbif.api.vocabulary.Country;
017import org.gbif.ws.server.interceptor.NullToNotFound;
018import org.gbif.ws.util.ExtraMediaTypes;
019
020import java.util.List;
021import java.util.Map;
022import java.util.Set;
023
024import javax.validation.constraints.NotNull;
025import javax.ws.rs.GET;
026import javax.ws.rs.Path;
027import javax.ws.rs.PathParam;
028import javax.ws.rs.Produces;
029import javax.ws.rs.core.MediaType;
030
031import com.google.common.collect.ImmutableList;
032import com.google.common.collect.ImmutableMap;
033import com.google.common.collect.Lists;
034import com.google.common.collect.Maps;
035import com.google.common.reflect.ClassPath;
036import com.google.common.reflect.ClassPath.ClassInfo;
037import com.google.inject.Singleton;
038import org.slf4j.Logger;
039import org.slf4j.LoggerFactory;
040
041/**
042 * A resource that provides a JSON serialization of all Enumerations in the GBIF API suitable for building Javascript
043 * based clients. This has no Java client, since Java clients have access to the Enums directly.
044 * Reflection is used to generate the inventory of enumerations.
045 */
046@Path("enumeration")
047@Produces({MediaType.APPLICATION_JSON, ExtraMediaTypes.APPLICATION_JAVASCRIPT})
048@Singleton
049public class EnumerationResource {
050
051  private static Logger LOG = LoggerFactory.getLogger(EnumerationResource.class);
052
053  // Uses reflection to find the enumerations in the API
054  private static Map<String, Enum<?>[]> PATH_MAPPING = enumerations();
055
056
057  private static List<Map<String, String>> COUNTRIES;
058  static {
059    List<Map<String, String>> countries = Lists.newArrayList();
060    for (Country c : Country.values()) {
061      if (c.isOfficial()) {
062        Map<String, String> info = Maps.newHashMap();
063        info.put("iso2", c.getIso2LetterCode());
064        info.put("iso3", c.getIso3LetterCode());
065        info.put("isoNumerical", String.valueOf(c.getIsoNumericalCode()));
066        info.put("title", c.getTitle());
067        info.put("enumName", c.name());
068        info.put("official", String.valueOf(c.isOfficial()));
069        countries.add(info);
070      }
071    }
072    COUNTRIES = ImmutableList.copyOf(countries);
073  }
074
075  /**
076   * An inventory of the enumerations supported.
077   * 
078   * @return The enumerations in the GBIF API.
079   */
080  @GET
081  @Path("basic")
082  public Set<String> inventory() {
083    return PATH_MAPPING.keySet();
084  }
085
086  // reflect over the package to find suitable enumerations
087  private static Map<String, Enum<?>[]> enumerations() {
088    try {
089      ClassPath cp = ClassPath.from(EnumerationResource.class.getClassLoader());
090      ImmutableMap.Builder<String, Enum<?>[]> builder = ImmutableMap.builder();
091
092      List<ClassInfo> infos = cp.getTopLevelClasses(Country.class.getPackage().getName()).asList();
093      for (ClassInfo info : infos) {
094        Class<? extends Enum<?>> vocab = VocabularyUtils.lookupVocabulary(info.getName());
095        // verify that it is an Enumeration
096        if (vocab != null && vocab.getEnumConstants() != null) {
097          builder.put(info.getSimpleName(), vocab.getEnumConstants());
098        }
099      }
100      return builder.build();
101
102    } catch (Exception e) {
103      LOG.error("Unable to read the classpath for enumerations", e);
104      return ImmutableMap.<String, Enum<?>[]>of(); // empty
105    }
106  }
107
108  /**
109   * @return list of country informations based on our enum.
110   */
111  @Path("country")
112  @GET
113  public List<Map<String, String>> listCountries() {
114    return COUNTRIES;
115  }
116
117  /**
118   * Gets the values of the named enumeration should the enumeration exist.
119   * Note this is used by the AngularJS console.
120   * 
121   * @param name Which should be the enumeration name in the GBIF vocabulary package (e.g. Country)
122   * @return The enumeration values or null if the enumeration does not exist.
123   */
124  @Path("basic/{name}")
125  @GET()
126  @NullToNotFound
127  public Enum<?>[] getEnumeration(@PathParam("name") @NotNull String name) {
128    if (PATH_MAPPING.containsKey(name)) {
129      return PATH_MAPPING.get(name);
130    } else {
131      return null;
132    }
133  }
134}
