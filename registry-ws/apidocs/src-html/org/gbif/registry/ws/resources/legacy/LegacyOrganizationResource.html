039/**
040 * Handle all legacy web service Organization requests, previously handled by the GBRDS.
041 */
042@Singleton
043@Path("registry/organisation")
044public class LegacyOrganizationResource {
045
046  private static final Logger LOG = LoggerFactory.getLogger(LegacyOrganizationResource.class);
047
048  private final OrganizationService organizationService;
049  private final NodeService nodeService;
050  private final OrganizationMapper organizationMapper;
051  // if true, send mails to disposable email address service
052  private final boolean useDevEmail;
053  private final String smtpHost;
054  private final int smptPort;
055  private final String devEmail;
056  private final String ccEmail;
057  private final String fromEmail;
058
059  @Inject
060  public LegacyOrganizationResource(OrganizationService organizationService, NodeService nodeService,
061    OrganizationMapper organizationMapper, @Named("useDevEmail") boolean useDevEmail,
062    @Named("smptHost") String smptHost, @Named("smptPort") int smptPort, @Named("devEmail") String devEmail,
063    @Named("ccEmail") String ccEmail, @Named("fromEmail") String fromEmail) {
064    this.organizationService = organizationService;
065    this.nodeService = nodeService;
066    this.organizationMapper = organizationMapper;
067    this.useDevEmail = useDevEmail;
068    this.smtpHost = smptHost;
069    this.smptPort = smptPort;
070    this.devEmail = devEmail;
071    this.ccEmail = ccEmail;
072    this.fromEmail = fromEmail;
073  }

075  /**
076   * This sub-resource can be called for various reasons:
077   * </br>
078   * 1. Get an Organization, handling incoming request with path /registry/organization/{key}.json?callback=?,
079   * signifying that the response must be JSONP. This request is made in order to verify that an organization exists.
080   * No authorization is required for this request.
081   * </br>
082   * 2. Validate the organization credentials sent with incoming GET request. Handling request with path
083   * /registry/organization/{key}.json?op=login. Only after the credentials have been verified, is the
084   * Response with Status.OK returned.
085   * 3. Trigger an email reminder for the organization, sent to the primary contact email. Handling request with path
086   * /registry/organization/{key}.json?op=password. An HTML response indicating successful delivery is included in 200
087   * response.
088   *
089   * @param organisationKey organization key (UUID) coming in as path param
090   * @param callback        parameter
091   *
092   * @return 1. Organization, wrapped with callback parameter in JSONP, or null if organization with key does not
093   *         exist.
094   *         2. (case: op=login) Response with Status.OK if credentials were verified, or Response with
095   *         Status.UNAUTHORIZED if they weren't
096   *         3. (case: op=password) Response with Status.OK if email reminder was delivered successfully
097   */
098  @GET
099  @Path("{key}")
100  @Produces({MediaType.APPLICATION_XML, MediaType.APPLICATION_JSON,
101    "application/x-javascript", "application/javascriptx-javascript"})
102  @Consumes(
103    {MediaType.TEXT_PLAIN, MediaType.APPLICATION_FORM_URLENCODED, "application/x-javascript", "application/javascript"})
104  public Object getOrganization(@PathParam("key") UUID organisationKey, @QueryParam("callback") String callback,
105    @QueryParam("op") String op, @Context SecurityContext security) {
106
107    // incoming path parameter for organization key required
108    if (organisationKey == null) {
109      return Response.status(Response.Status.BAD_REQUEST).cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED)
110        .build();
111    }
112    LOG.info("Get Organization with key={}", organisationKey.toString());
113
114    Organization organization = organizationService.get(organisationKey);
115    if (organization == null) {
116      // the organization didn't exist, and expected response is "{Error: "No organisation matches the key provided}"
117      return Response.status(Response.Status.OK).entity(new ErrorResponse("No organisation matches the key provided"))
118        .cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED).build();
119    }
120
121    if (op != null) {
122      // ?op=login
123      if (op.equalsIgnoreCase("login")) {
124        // are the organization credentials matching with the path?
125        UUID authKey = LegacyAuthorizationFilter.extractOrgKeyFromSecurity(security);
126
127        if (!organisationKey.equals(authKey)) {
128          LOG.error("Authorization failed for organization with key={}", organisationKey.toString());
129          return Response.status(Response.Status.UNAUTHORIZED)
130            .cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED).build();
131        }
132        return Response.status(Response.Status.OK).cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED).build();
133      }
134      // ?op=password
135      // Email a password reminder to organization's primary contact, and notify requester of success in response
136      else if (op.equalsIgnoreCase("password")) {
137        // contact email address is nullable, but mandatory for sending a mail
138        Contact contact = LegacyResourceUtils.getPrimaryContact(organization);
139        String emailAddress = (contact == null || contact.getEmail().isEmpty()) ? null: contact.getEmail().get(0);
140        if (emailAddress == null) {
141          LOG.error("Password reminder failed: organization primary contact has no email address");
142          return Response.status(Response.Status.INTERNAL_SERVER_ERROR)
143            .entity(new ErrorResponse("Password reminder failed: organization primary contact has no email address"))
144            .cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED).build();
145        } else {
146          try {
147            Email email = new SimpleEmail();
148            email.setHostName(smtpHost);
149            email.setSmtpPort(smptPort);
150            email.setFrom(fromEmail);
151            email.setSubject("GBIF: Password reminder for: " + organization.getTitle());
152            email.setMsg(getEmailBody(contact, organization));
153
154            // add recipients, depending on whether development mode is on for sending email?
155            if (useDevEmail) {
156              email.addTo(devEmail);
157            } else {
158              email.addTo(emailAddress);
159              email.addCc(ccEmail);
160            }
161
162            email.send();
163          } catch (EmailException e) {
164            LOG.error("Password reminder failed", e);
165            return Response.status(Response.Status.INTERNAL_SERVER_ERROR)
166              .entity(new ErrorResponse("Password reminder failed: " + e.getMessage()))
167              .type(MediaType.APPLICATION_JSON).cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED).build();
168          }
169          LOG.debug("Password reminder sent to: {}", emailAddress);
170          return Response.status(Response.Status.OK).entity(
171            "<html><body><b>The password reminder was sent successfully to the email: </b>" + emailAddress
172            + "</body></html>").cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED).build();
173        }
174      }
175    }
176
177    // retrieve primary contact for organization
178    Contact contact = LegacyResourceUtils.getPrimaryContact(organization);
179    // construct organization response object
180    Node node = nodeService.get(organization.getEndorsingNodeKey());
181    LegacyOrganizationResponse o = new LegacyOrganizationResponse(organization, contact, node);
182
183    // callback?
184    if (callback != null) {
185      return new JSONPObject(callback, o);
186    }
187    // simple read?
188    else {
189      return Response.status(Response.Status.OK).entity(o).cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED)
190        .build();
191    }
192  }

194  /**
195   * Get a list of all Organizations, handling incoming request with path /registry/organisation.json. For each
196   * Organization, only the key and title(name) fields are required. No authorization is required for this request.
197   *
198   * @return list of all Organizations
199   */
200  @GET
201  @Produces({MediaType.APPLICATION_XML, MediaType.APPLICATION_JSON})
202  public Response getOrganizations() {
203    LOG.debug("List all Organizations for IPT");
204    List<LegacyOrganizationBriefResponse> organizations = organizationMapper.listLegacyOrganizationsBrief();
205
206    // return array, required for serialization otherwise get com.sun.jersey.api.MessageException: A message body
207    // writer for Java class java.util.ArrayList
208    LegacyOrganizationBriefResponse[] array =
209      organizations.toArray(new LegacyOrganizationBriefResponse[organizations.size()]);
210    return Response.status(Response.Status.OK).entity(array)
211      .cacheControl(LegacyResourceConstants.CACHE_CONTROL_DISABLED).build();
212  }

214  /**
215   * Build the email body, sent to the primary contact of the organization reminding them of the password.
216   *
217   * @param contact primary contact of the organization
218   * @param organization organization
219   * @return email body
220   */
221  private String getEmailBody(Contact contact, Organization organization) {
222    StringBuilder body = new StringBuilder();
223    body.append("Dear ");
224    body.append(contact.getFirstName());
225    body.append(": \n\n");
226
227    body.append("You, or someone else, has requested the password for the organisation '");
228    body.append(organization.getTitle());
229    body.append("' to be sent to your e-mail address (");
230    body.append(contact.getEmail());
231    body.append(")\n\n");
232
233    body.append("The information requested is: \n\n");
234    body.append("Username: ");
235    body.append(organization.getKey());
236    body.append("\n");
237    body.append("Password: ");
238    body.append(organization.getPassword());
239    body.append("\n\n");
240
241    body.append("If you did not request this information, please disregard this message\n\n");
242
243    body.append("GBIF (Global Biodiversity Information Facility)\n");
244    body.append("http://www.gbif.org\n");
245    body.append(ccEmail);
246
247    return body.toString();
248  }
