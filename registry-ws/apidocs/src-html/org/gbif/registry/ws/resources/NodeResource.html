060@Singleton
061@Path("node")
062public class NodeResource extends BaseNetworkEntityResource<Node> implements NodeService {
063
064  private final NodeMapper nodeMapper;
065  private final OrganizationMapper organizationMapper;
066  private final InstallationMapper installationMapper;
067  private final DatasetMapper datasetMapper;
068  private final Augmenter nodeAugmenter;
069
070  @Inject
071  public NodeResource(
072    NodeMapper nodeMapper,
073    IdentifierMapper identifierMapper,
074    CommentMapper commentMapper,
075    ContactMapper contactMapper,
076    EndpointMapper endpointMapper,
077    MachineTagMapper machineTagMapper,
078    TagMapper tagMapper,
079    OrganizationMapper organizationMapper,
080    DatasetMapper datasetMapper,
081    InstallationMapper installationMapper,
082    EventBus eventBus,
083    Augmenter nodeAugmenter,
084    EditorAuthorizationService userAuthService) {
085    super(nodeMapper, commentMapper, contactMapper, endpointMapper, identifierMapper, machineTagMapper, tagMapper,
086      Node.class, eventBus, userAuthService);
087    this.nodeMapper = nodeMapper;
088    this.organizationMapper = organizationMapper;
089    this.nodeAugmenter = nodeAugmenter;
090    this.datasetMapper = datasetMapper;
091    this.installationMapper = installationMapper;
092  }
093
094  @GET
095  @Path("{key}")
096  @Nullable
097  @NullToNotFound
098  @Override
099  public Node get(@PathParam("key") UUID key) {
100    return nodeAugmenter.augment(super.get(key));
101  }
102
103
104  /**
105   * All network entities support simple (!) search with "&q=".
106   * This is to support the console user interface, and is in addition to any complex, faceted search that might
107   * additionally be supported, such as dataset search.
108   */
109  @GET
110  public PagingResponse<Node> list(@Nullable @QueryParam("q") String query,
111    @Nullable @QueryParam("identifierType") IdentifierType identifierType,
112    @Nullable @QueryParam("identifier") String identifier,
113    @Nullable @Context Pageable page) {
114    // This is getting messy: http://dev.gbif.org/issues/browse/REG-426
115    if (identifierType != null && identifier != null) {
116      return listByIdentifier(identifierType, identifier, page);
117    } else if (identifier != null) {
118      return listByIdentifier(identifier, page);
119    } else if (Strings.isNullOrEmpty(query)) {
120      return list(page);
121    } else {
122      return search(query, page);
123    }
124  }
125
126
127  @Override
128  public PagingResponse<Node> list(@Nullable Pageable page) {
129    PagingResponse<Node> resp = super.list(page);
130    for (Node n : resp.getResults()) {
131      nodeAugmenter.augment(n);
132    }
133    return resp;
134  }
135
136  @GET
137  @Path("{key}/organization")
138  @Override
139  public PagingResponse<Organization> endorsedOrganizations(@PathParam("key") UUID nodeKey, @Context Pageable page) {
140    return new PagingResponse<Organization>(page, organizationMapper.countOrganizationsEndorsedBy(nodeKey),
141      organizationMapper.organizationsEndorsedBy(nodeKey, page));
142  }
143
144  @GET
145  @Path("pendingEndorsement")
146  @Override
147  public PagingResponse<Organization> pendingEndorsements(@Context Pageable page) {
148    return new PagingResponse<Organization>(page, organizationMapper.countPendingEndorsements(null),
149      organizationMapper.pendingEndorsements(null, page));
150  }

153  @Path("{key}/pendingEndorsement")
