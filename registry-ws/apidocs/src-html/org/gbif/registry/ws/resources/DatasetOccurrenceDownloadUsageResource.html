001package org.gbif.registry.ws.resources;
002
003import org.gbif.api.model.common.paging.Pageable;
004import org.gbif.api.model.common.paging.PagingResponse;
005import org.gbif.api.model.registry.DatasetOccurrenceDownloadUsage;
006import org.gbif.api.model.registry.PrePersist;
007import org.gbif.api.service.registry.DatasetOccurrenceDownloadUsageService;
008import org.gbif.registry.persistence.mapper.DatasetOccurrenceDownloadMapper;
009import org.gbif.ws.server.interceptor.NullToNotFound;
010import org.gbif.ws.util.ExtraMediaTypes;
011
012import java.util.List;
013import java.util.UUID;
014
015import javax.annotation.security.RolesAllowed;
016import javax.validation.Valid;
017import javax.validation.constraints.NotNull;
018import javax.validation.groups.Default;
019import javax.ws.rs.Consumes;
020import javax.ws.rs.GET;
021import javax.ws.rs.POST;
022import javax.ws.rs.Path;
023import javax.ws.rs.PathParam;
024import javax.ws.rs.Produces;
025import javax.ws.rs.core.Context;
026import javax.ws.rs.core.MediaType;
027import javax.ws.rs.core.SecurityContext;
028
029import com.google.inject.Inject;
030import com.google.inject.Singleton;
031import org.apache.bval.guice.Validate;
032import org.mybatis.guice.transactional.Transactional;
033
034import static org.gbif.registry.ws.security.UserRoles.ADMIN_ROLE;
035import static org.gbif.registry.ws.util.DownloadSecurityUtils.clearSensitiveData;
036
037/**
038 * Occurrence download resource/web service.
039 */
040@Singleton
041@Path("occurrence/download/dataset")
042@Produces({MediaType.APPLICATION_JSON, ExtraMediaTypes.APPLICATION_JAVASCRIPT})
043@Consumes(MediaType.APPLICATION_JSON)
044public class DatasetOccurrenceDownloadUsageResource implements DatasetOccurrenceDownloadUsageService {
045
046  private final DatasetOccurrenceDownloadMapper datasetOccurrenceDownloadMapper;
047
048  @Context
049  private SecurityContext securityContext;
050
051  @Inject
052  public DatasetOccurrenceDownloadUsageResource(DatasetOccurrenceDownloadMapper datasetOccurrenceDownloadMapper) {
053    this.datasetOccurrenceDownloadMapper = datasetOccurrenceDownloadMapper;
054  }
055
056
057  @POST
058  @Transactional
059  @Validate(groups = {PrePersist.class, Default.class})
060  @RolesAllowed(ADMIN_ROLE)
061  @Override
062  public void create(@Valid @NotNull DatasetOccurrenceDownloadUsage downloadDataset) {
063    datasetOccurrenceDownloadMapper.create(downloadDataset);
064  }
065
066  @GET
067  @Path("/{datasetKey}")
068  @NullToNotFound
069  @Override
070  public PagingResponse<DatasetOccurrenceDownloadUsage> listByDataset(
071    @PathParam("datasetKey") UUID datasetKey, @Context Pageable page) {
072    List<DatasetOccurrenceDownloadUsage> usages = datasetOccurrenceDownloadMapper.listByDataset(datasetKey, page);
073    clearSensitiveData(securityContext, usages);
074    return new PagingResponse<DatasetOccurrenceDownloadUsage>(page,
075      (long) datasetOccurrenceDownloadMapper.countByDataset(datasetKey), usages);
076  }
077
078}
