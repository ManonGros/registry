134  /**
135   * This method ensures that the caller is authorized to perform the action and then adds the server
136   * controlled fields for createdBy and modifiedBy. It then creates the entity.
137   *
138   * @param entity entity that extends NetworkEntity
139   * @param security SecurityContext (security related information)
140   * @return key of entity created
141   */
142  @POST
143  @Trim
144  @Transactional
145  @RolesAllowed({ADMIN_ROLE, EDITOR_ROLE})
146  public UUID create(@NotNull @Trim T entity, @Context SecurityContext security) {
147    // if not admin, verify rights
148    if (!security.isUserInRole(ADMIN_ROLE)) {
149      UUID entityKeyToBeAssesed = owningEntityKey(entity);
150      if (entityKeyToBeAssesed == null || !userAuthService.allowedToModifyEntity(security.getUserPrincipal(), entityKeyToBeAssesed)) {
151        throw new WebApplicationException(Response.Status.FORBIDDEN);
152      }
153    }
154    entity.setCreatedBy(security.getUserPrincipal().getName());
155    entity.setModifiedBy(security.getUserPrincipal().getName());
156    return create(entity);
157  }

164    return null;
165  }

335
336    machineTag.setCreatedBy(security.getUserPrincipal().getName());
337    return addMachineTag(targetEntityKey, machineTag);
