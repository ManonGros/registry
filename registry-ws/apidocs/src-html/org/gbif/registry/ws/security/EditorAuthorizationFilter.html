001package org.gbif.registry.ws.security;
002
003import java.security.Principal;
004import java.util.UUID;
005import java.util.regex.Matcher;
006import java.util.regex.Pattern;
007import javax.ws.rs.WebApplicationException;
008import javax.ws.rs.core.Context;
009import javax.ws.rs.core.Response;
010import javax.ws.rs.core.SecurityContext;
011
012import com.google.common.annotations.VisibleForTesting;
013import com.google.inject.Inject;
014import com.sun.jersey.spi.container.ContainerRequest;
015import com.sun.jersey.spi.container.ContainerRequestFilter;
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;
018
019/**
020 * For requests authenticated with a REGISTRY_EDITOR role two levels of authorization need to be passed.
021 * First of all any resource method is required to have the role included in the RolesAllowed annotation.
022 * Secondly this request filter needs to be passed for POST/PUT/DELETE requests that act on existing and UUID identified
023 * main registry entities such as dataset, organization, node, installation and network.
024 *
025 * In order to do authorization the key of these entities is extracted from the requested path.
026 * An exception to this is the create method for those main entities themselves.
027 * This is covered by the BaseNetworkEntityResource.create() method directly.
028 */
029public class EditorAuthorizationFilter implements ContainerRequestFilter {
030
031  private static final Logger LOG = LoggerFactory.getLogger(EditorAuthorizationFilter.class);
032
033  private static final String ENTITY_KEY = "^%s/([a-fA-F0-9-]+)";
034  private static final Pattern NODE_NETWORK_PATTERN = Pattern.compile(String.format(ENTITY_KEY, "(?:network|node)"));
035  private static final Pattern ORGANIZATION_PATTERN = Pattern.compile(String.format(ENTITY_KEY, "organization"));
036  private static final Pattern DATASET_PATTERN = Pattern.compile(String.format(ENTITY_KEY, "dataset"));
037  private static final Pattern INSTALLATION_PATTERN = Pattern.compile(String.format(ENTITY_KEY, "installation"));
038
039  private final EditorAuthorizationService userAuthService;
040
041  @Context
042  private SecurityContext secContext;
043
044  @Inject
045  public EditorAuthorizationFilter(EditorAuthorizationService userAuthService) {
046    this.userAuthService = userAuthService;
047  }
048
049
050  @Override
051  public ContainerRequest filter(ContainerRequest request) {
052    // only verify non GET methods with an authenticated REGISTRY_EDITOR
053    // all other roles are taken care by simple JSR250 annotations on the resource methods
054    Principal user = secContext.getUserPrincipal();
055
056    if (user != null
057        && (!secContext.isUserInRole(UserRoles.ADMIN_ROLE) && secContext.isUserInRole(UserRoles.EDITOR_ROLE))
058        && !request.getMethod().equals("GET")) {
059
060      String path = request.getPath().toLowerCase();
061
062      try {
063
064        Matcher m = ORGANIZATION_PATTERN.matcher(path);
065        if (m.find()) {
066          if (!userAuthService.allowedToModifyOrganization(user, UUID.fromString(m.group(1)))) {
067            throw new WebApplicationException(Response.Status.FORBIDDEN);
068          }
069        }
070
071        m = DATASET_PATTERN.matcher(path);
072        if (m.find()) {
073          if (!userAuthService.allowedToModifyDataset(user, UUID.fromString(m.group(1)))) {
074            throw new WebApplicationException(Response.Status.FORBIDDEN);
075          }
076        }
077
078        m = INSTALLATION_PATTERN.matcher(path);
079        if (m.find()) {
080          if (!userAuthService.allowedToModifyInstallation(user, UUID.fromString(m.group(1)))) {
081            throw new WebApplicationException(Response.Status.FORBIDDEN);
082          }
083        }
084
085        m = NODE_NETWORK_PATTERN.matcher(path);
086        if (m.find()) {
087          if (!userAuthService.allowedToModifyEntity(user, UUID.fromString(m.group(1)))) {
088            throw new WebApplicationException(Response.Status.FORBIDDEN);
089          }
090        }
091      } catch (IllegalArgumentException e) {
092        // no valid UUID, do nothing as it should not be a valid request anyway
093      }
094    }
095
096    return request;
097  }
098
099  @VisibleForTesting
100  protected void setSecContext(SecurityContext secContext) {
101    this.secContext = secContext;
102  }
103}
