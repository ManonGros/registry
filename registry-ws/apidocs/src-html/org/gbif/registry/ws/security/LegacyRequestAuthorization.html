013import java.util.NoSuchElementException;

018import javax.ws.rs.core.SecurityContext;

020import com.google.common.base.Splitter;
021import com.google.common.base.Strings;
022import com.google.inject.Inject;

031/**
032 * Class providing temporary authorization for legacy web service requests (GBRDS/IPT).
033 */
034public class LegacyRequestAuthorization implements SecurityContext {
035
036  private static final Logger LOG = LoggerFactory.getLogger(LegacyRequestAuthorization.class);
037  private static final Splitter COLON_SPLITTER = Splitter.on(":").limit(2);
038
039  private final OrganizationService organizationService;
040  private final InstallationService installationService;
041  private final DatasetService datasetService;
042
043  private final UUID userKey;
044  private final HttpContext httpContext;
045
046  @Inject
047  public LegacyRequestAuthorization(HttpContext httpContext, OrganizationService organizationService, DatasetService datasetService, InstallationService installationService) {
048    this.httpContext = httpContext;
049    this.organizationService = organizationService;
050    this.datasetService = datasetService;
051    this.installationService = installationService;
052    userKey = authenticateUser(httpContext);
053  }
054
055  /**
056   * Evaluates the Basic authentication header and verifies provided password matches with the registered one.
057   * Checks both for organizations and installation keys.
058   * @return the registry entity this request has authenticated with
059   */
060  private UUID authenticateUser(HttpContext httpContext) {
061    String authentication = httpContext.getRequest().getHeaderValue(ContainerRequest.AUTHORIZATION);
062    if (Strings.isNullOrEmpty(authentication) || !authentication.startsWith("Basic ")) {
063      LOG.info("No basic authorization header found in legacy ws request");
064      return null;
065    }
066
067    String decrypted = Base64.base64Decode(authentication.substring("Basic ".length()));
068    Iterator<String> iter = COLON_SPLITTER.split(decrypted).iterator();
069    try {
070      UUID user = UUID.fromString(iter.next());
071      String password = iter.next();
072
073      // try to validate organization key first
074      try {
075        Organization org = organizationService.get(user);
076        if (password.equals(org.getPassword())) {
077          return user;
078        } else {
079          throw new WebApplicationException(Response.Status.UNAUTHORIZED);
080        }
081
082      } catch (NotFoundException e) {
083        // maybe an installation?
084        try {
085          Installation installation = installationService.get(user);
086          if (password.equals(installation.getPassword())) {
087            return user;
088          }
089        } catch (NotFoundException e1) {
090          // throw
091        }
092        throw new WebApplicationException(Response.Status.UNAUTHORIZED);
093      }
094
095
096    } catch (NoSuchElementException e) {
097      LOG.warn("Invalid Basic Authentication syntax: {}", authentication);
098      throw new WebApplicationException(Response.Status.BAD_REQUEST);
099
100    } catch (IllegalArgumentException e) {
101      // no valid username UUID
102      LOG.warn("Invalid username, UUID required: {}", decrypted);
103      throw new WebApplicationException(Response.Status.UNAUTHORIZED);
104    }
105  }
106
107  /**
108   * Determine if HTTP request is authorized to modify Organization. The difference between this method and
109   * isAuthorizedToModifyOrganization(organizationKey) is that the organizationKey must first be parsed from the
110   * form parameters.
111   *
112   * @return true if the HTTP request is authorized to modify Organization
113   *
114   * @see LegacyRequestAuthorization#isAuthorizedToModifyOrganization(UUID)
115   */
116  public boolean isAuthorizedToModifyOrganization() {
117    // retrieve HTTP param for hosting organization key and convert incoming key into UUID
118    UUID organizationKey = getOrganizationKeyFromParams();
119
120    return isAuthorizedToModifyOrganization(organizationKey);
121  }
122
123  public boolean isAuthorizedToModifyOrganization(UUID organizationKey) {
124    if (organizationKey == null) {
125      return false;
126    }
127    // validate installation key belongs to an existing installation
128    Organization org = organizationService.get(organizationKey);
129    return org != null && org.getKey().equals(userKey);
130  }
131
132
133  /**
134   * Retrieves the organizationKey from the requests' form parameters.
135   *
136   * @return key or null if not found
137   */
138  public UUID getOrganizationKeyFromParams() {
139    MultivaluedMap<String, String> params = httpContext.getRequest().getFormParameters();
140    if (params == null) {
141      return null;
142    }
143
144    // retrieve HTTP param for hosting organization key
145    String organizationKeyFormParam = params.getFirst(LegacyResourceConstants.ORGANIZATION_KEY_PARAM);
146    if (Strings.isNullOrEmpty(organizationKeyFormParam)) {
147      return null;
148    }
149    // convert incoming key into UUID
150    try {
151      return UUID.fromString(organizationKeyFormParam);
152    } catch (IllegalArgumentException e) {
153      return null;
154    }
155  }
156
157  /**
158   * Determine if HTTP request is authorized to modify a Dataset belonging to an Organization.
159   * This method checks that the same organizationKey is specified in the credentials and HTTP form parameters,
160   * that the Organization corresponding to that organizationKey exists, that the Dataset corresponding to the
161   * datasetKey exists, that the Dataset is owned by that Organization, and that the correct organization password has
162   * been supplied.
163   *
164   * @param datasetKey Dataset key
165   *
166   * @return true if the HTTP request is authorized to modify a Dataset belonging to Organization
167   */
168  public boolean isAuthorizedToModifyOrganizationsDataset(UUID datasetKey) {
169    if (datasetKey == null) {
170      LOG.error("Dataset key was null");
171      return false;
172    }
173    // retrieve dataset to ensure it exists
174    Dataset dataset = datasetService.get(datasetKey);
175    if (dataset == null) {
176      LOG.error("Dataset with key={} does not exist", datasetKey.toString());
177      return false;
178    }
179
180    // check the dataset belongs to organization
181    if (dataset.getPublishingOrganizationKey().compareTo(userKey) != 0) {
182      LOG.error("The Dataset is not owned by the organization specified in the credentials");
183      return false;
184    }
185
186    // check if an organisationKey was included in form params, that the organization keys match
187    UUID organizationKeyFromFormParams = getOrganizationKeyFromParams();
188    if (organizationKeyFromFormParams != null) {
189      if (organizationKeyFromFormParams.compareTo(userKey) != 0) {
190        LOG.error("Different organization keys were specified in the form parameters and credentials");
191        return false;
192      }
193    }
194    LOG.info("Authorization succeeded, can modify dataset with key={} belonging to organization with key={}",
195      datasetKey.toString(), userKey.toString());
196    return true;
197  }
198
199  /**
200   * Determine if HTTP request is authorized to modify Installation.
201   *
202   * @param installationKey Installation key
203   *
204   * @return true if the HTTP request is authorized to modify Installation
205   */
206  public boolean isAuthorizedToModifyInstallation(UUID installationKey) {
207    // retrieve path param for installation key
208    if (installationKey == null) {
209      return false;
210    }
211    // validate installation key belongs to an existing installation
212    Installation installation = installationService.get(installationKey);
213    return installation != null && installation.getKey().equals(userKey);
214  }
215
216  @Override
217  public Principal getUserPrincipal() {
218    return new BasicUserPrincipal(userKey.toString());
219  }
220
221  @Override
222  public boolean isUserInRole(String role) {
223    return false;
224  }
225
226  @Override
227  public boolean isSecure() {
228    return false;
229  }
230
231  @Override
232  public String getAuthenticationScheme() {
233    return SecurityContext.BASIC_AUTH;
234 }
235
236  public UUID getUserKey() {
237    return userKey;
238  }
