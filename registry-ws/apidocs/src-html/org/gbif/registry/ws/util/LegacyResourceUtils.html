001package org.gbif.registry.ws.util;
002
003import org.gbif.api.model.registry.Contact;
004import org.gbif.api.model.registry.Contactable;
005import org.gbif.api.model.registry.Dataset;
006import org.gbif.api.model.registry.Installation;
007import org.gbif.api.service.registry.DatasetService;
008import org.gbif.api.service.registry.InstallationService;
009import org.gbif.api.service.registry.OrganizationService;
010import org.gbif.registry.ws.model.LegacyDataset;
011import org.gbif.registry.ws.model.LegacyEndpoint;
012import org.gbif.registry.ws.model.LegacyInstallation;
013
014import javax.annotation.Nullable;
015
016import com.google.common.base.Strings;
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;
019
020/**
021 * Class containing utility methods used across legacy resources, in answering legacy web service requests (GBRDS/IPT).
022 */
023public class LegacyResourceUtils {
024
025  private static final Logger LOG = LoggerFactory.getLogger(LegacyResourceUtils.class);
026
027  /**
028   * Checks if a field consists of the minimum number of characters. If it contains fewer characters than required,
029   * a warning text is appended to the field text, padding it until it exceeds the minimum required size.
030   *
031   * @param text field text
032   * @param size minimum field size (in number of characters)
033   *
034   * @return validated field meeting the minimum required length, or the original null or empty string
035   */
036  public static String validateField(@Nullable String text, int size) {
037    String validated = Strings.emptyToNull(text);
038    if (validated != null) {
039      while (validated.length() < size) {
040        validated += " [Field must be at least " + String.valueOf(size) + " characters long]";
041      }
042      return validated;
043    }
044    return text;
045  }
046
047  /**
048   * Iterate through list of contacts, return first primary contact encountered, or null if none found.
049   *
050   * @param contactable contactable
051   *
052   * @return primary contact or null if none found
053   */
054  public static Contact getPrimaryContact(Contactable contactable) {
055    for (Contact contact : contactable.getContacts()) {
056      if (contact.isPrimary()) {
057        return contact;
058      }
059    }
060    return null;
061  }
062
063  /**
064   * Check installation is valid before updating. Check the installation still exists, the hosting organization
065   * still exists, and the same hosting organization is specified before updating.
066   *
067   * @param installation installation with fields injected from HTTP request
068   *
069   * @return true if the installation is valid for updating, or false otherwise
070   */
071  public static boolean isValidOnUpdate(LegacyInstallation installation, InstallationService installationService,
072    OrganizationService organizationService) {
073    Installation existing = installationService.get(installation.getKey());
074    if (existing == null) {
075      LOG.error("Installation update uses an installation that does not exist, key={}", installation.getKey());
076      return false;
077    }
078    return isValid(installation, organizationService);
079  }
080
081  /**
082   * Check installation is valid before persisting. Check required fields, and the hosting organization
083   * still exists.
084   *
085   * @param installation installation with fields injected from HTTP request
086   *
087   * @return true if the installation is valid, or false otherwise
088   */
089  public static boolean isValid(LegacyInstallation installation, OrganizationService organizationService) {
090    if (installation.getType() == null) {
091      LOG.error("Installation is missing mandatory field type, key={}", installation.getKey());
092      return false;
093    }
094    if (installation.getOrganizationKey() == null) {
095      LOG.error("Hosting org key not included in HTTP parameters for installation, key={}", installation.getKey());
096      return false;
097    }
098    if (organizationService.get(installation.getOrganizationKey()) == null) {
099      LOG.error("Installation uses a hosting org that does not exist, key={}", installation.getOrganizationKey());
100      return false;
101    }
102    return true;
103  }
104
105  /**
106   * Check dataset is valid before persisting. Check required fields, the publishing organization
107   * still exists, and the installation still exists before updating.
108   *
109   * @param dataset dataset with fields injected from HTTP request
110   *
111   * @return true if the dataset is valid, or false otherwise
112   */
113  public static boolean isValid(LegacyDataset dataset, OrganizationService organizationService,
114    InstallationService installationService) {
115    // title is never null in IPT requests, but is mandatory in GBRDS Resource requests
116    if (dataset.getTitle() == null) {
117      LOG.error("Dataset is missing mandatory field title, key={}", dataset.getKey());
118      return false;
119    }
120    if (dataset.getLanguage() == null) {
121      LOG.error("Dataset is missing mandatory field language, key={}", dataset.getKey());
122      return false;
123    }
124    if (dataset.getType() == null) {
125      LOG.error("Dataset is missing mandatory field type, key={}", dataset.getKey());
126      return false;
127    }
128    if (dataset.getPublishingOrganizationKey() == null) {
129      LOG.error("Publishing org key not included in HTTP parameters for dataset, key={}", dataset.getKey());
130      return false;
131    }
132    if (organizationService.get(dataset.getPublishingOrganizationKey()) == null) {
133      LOG.error("Dataset uses an publishing org that does not exist, key={}", dataset.getPublishingOrganizationKey());
134      return false;
135    }
136    if (dataset.getInstallationKey() == null) {
137      LOG.error("Installation key not included in HTTP parameters for dataset, or could not be inferred, key={}",
138        dataset.getKey());
139      return false;
140    }
141    if (installationService.get(dataset.getInstallationKey()) == null) {
142      LOG.error("Dataset uses an IPT that does not exist, key={}", dataset.getInstallationKey());
143      return false;
144    }
145    return true;
146  }
147
148  /**
149   * Check dataset is valid before updating. Check the dataset still exists, the publishing organization
150   * still exists, the same publishing organization is specified, and the installation still exists before updating.
151   *
152   * @param dataset dataset with fields injected from HTTP request
153   *
154   * @return true if the dataset is valid for updating, or false otherwise
155   */
156  public static boolean isValidOnUpdate(LegacyDataset dataset, DatasetService datasetService,
157    OrganizationService organizationService, InstallationService installationService) {
158    Dataset existing = datasetService.get(dataset.getKey());
159    if (existing == null) {
160      LOG.error("Dataset update uses a dataset that does not exist, key={}", dataset.getKey());
161      return false;
162    }
163    return isValid(dataset, organizationService, installationService);
164  }
165
166  /**
167   * Check endpoint is valid before persisting. Check required fields, the access point URL, type, and dataset key, and
168   * that the dataset still exists before updating.
169   *
170   * @param endpoint endpoint with fields injected from HTTP request
171   *
172   * @return true if the endpoint is valid, or false otherwise
173   */
174  public static boolean isValid(LegacyEndpoint endpoint, DatasetService datasetService) {
175    if (endpoint.getType() == null) {
176      LOG.error("type not included in HTTP parameters for LegacyEndpiont (Service)");
177      return false;
178    }
179    if (endpoint.getUrl() == null) {
180      LOG.error("accessPointURL not included in HTTP parameters for LegacyEndpiont (Service)");
181      return false;
182    }
183    if (endpoint.getDatasetKey() == null) {
184      LOG.error("Dataset key (resourceKey) not included in HTTP parameters for LegacyEndpiont (Service)");
185      return false;
186    }
187    if (datasetService.get(endpoint.getDatasetKey()) == null) {
188      LOG.error("LegacyEndpiont (Service) refers to a dataset that does not exist, key={}", endpoint.getDatasetKey());
189      return false;
190    }
191    return true;
192  }
193}
