004import org.gbif.api.model.registry.Installation;

006import org.gbif.api.model.registry.Organization;

009import org.gbif.api.service.registry.OrganizationService;

012import java.net.URI;
013import java.util.HashSet;
014import java.util.Set;
015import java.util.UUID;
016
017import com.google.common.base.Joiner;
018import com.google.common.eventbus.EventBus;
019import com.google.common.eventbus.Subscribe;
020import com.google.inject.Inject;
021import com.google.inject.Singleton;
022import org.apache.http.impl.client.CloseableHttpClient;
023import org.slf4j.Logger;
024import org.slf4j.LoggerFactory;

026/**
027 * A event bus listener that will flush varnish if registry entities like datasets and organizations have been
028 * created, modified or deleted.
029 * <p/>
030 * Varnish provides two main ways of invalidating its cache: PURGE and BAN.
031 * PURGE truly frees the cache, but only works on individual resource URLs while BANs work with regular expressions
032 * and can banRegex entire subresources from being served. BANs do not remove the object from the varnish memory though.
033 *
034 * @see <h ref="https://www.varnish-software.com/static/book/Cache_invalidation.html">Varnish Book</h>
035 *
036 * <h3>Purging cascade logic</h3>
037 * A quick overview of the logic how entity changes trigger purges on other resources.
038 * In case of updates all keys used from the changed objects need be taken from both the old and new version.
039 *
040 * <h4>Node</h4>
041 * <p>Nodes are only modified in the IMS and no events are ever triggered.</p>
042 * <ul>
043 *   <li>Node detail PURGE</li>
044 *   <li>List all nodes BAN</li>
045 * </ul>
046 *
047 * <h4>Organization</h4>
048 * <ul>
049 *   <li>Organization detail PURGE</li>
050 *   <li>List all organizations BAN</li>
051 *   <li>Publisher search & suggest BAN</li>
052 *   <li>/node/{org.endorsingNodeKey}/organization</li>
053 * </ul>
054 *
055 * <h4>Network</h4>
056 * <ul>
057 *   <li>Network detail PURGE</li>
058 *   <li>List all networks BAN</li>
059 * </ul>
060 *
061 * <h4>Installation</h4>
062 * <ul>
063 *   <li>Installation detail PURGE</li>
064 *   <li>List all installations BAN</li>
065 *   <li>/node/{i.organization.endorsingNodeKey}/installation BAN</li>
066 *   <li>/organization/{UUID}/installation BAN</li>
067 * </ul>
068 *
069 * <h4>Dataset</h4>
070 * <ul>
071 *   <li>dataset/{key} PURGE</li>
072 *   <li>dataset/{d.parentKey} PURGE</li>
073 *   <li>dataset/{d.parentKey}/constituents BAN</li>
074 *   <li>dataset BAN</li>
075 *   <li>dataset/search|suggest BAN</li>
076 *   <li>/installation/{d.installationKey}/dataset BAN</li>
077 *   <li>/organization/{d.installation.organizationKey}/hostedDataset BAN</li>
078 *   <li>/organization/{d.publishingOrganizationKey}/publishedDataset BAN</li>
079 *   <li>/node/{d.publishingOrganization.endorsingNodeKey}/dataset BAN</li>
080 *   <li>/network/{any UUID}/constituents BAN</li>
081 * </ul>
082 *
083 */
084@Singleton
085public class VarnishPurgeListener {
086  private static final Logger LOG = LoggerFactory.getLogger(VarnishPurgeListener.class);
087  private final OrganizationService organizationService;
088  private final InstallationService installationService;
089  private final DatasetService datasetService;
090  private final VarnishPurger purger;
091  private static final Joiner PATH_JOINER = Joiner.on("/").skipNulls();
092
093  @Inject
094  public VarnishPurgeListener(CloseableHttpClient client, EventBus eventBus, URI apiBaseUrl,
095                              OrganizationService organizationService,InstallationService installationService,
096                              DatasetService datasetService) {
097    this.organizationService = organizationService;
098    this.installationService = installationService;
099    this.datasetService = datasetService;
100    eventBus.register(this);
101    purger = new VarnishPurger(client, apiBaseUrl);
102  }
103
104  @Subscribe
105  public final <T extends NetworkEntity> void created(CreateEvent<T> event) {
106    purgeEntityAndBanLists(event.getObjectClass(), event.getNewObject().getKey());
107
108    if (event.getObjectClass().equals(Organization.class)) {
109      cascadeOrganizationChange((Organization) event.getNewObject());
110    } else if (event.getObjectClass().equals(Dataset.class)) {
111      cascadeDatasetChange((Dataset) event.getNewObject());
112    } else if (event.getObjectClass().equals(Installation.class)) {
113      cascadeInstallationChange((Installation) event.getNewObject());
114    }
115  }
116
117  @Subscribe
118  public final <T extends NetworkEntity> void updated(UpdateEvent<T> event) {
119    purgeEntityAndBanLists(event.getObjectClass(), event.getOldObject().getKey());
120
121    if (event.getObjectClass().equals(Organization.class)) {
122      cascadeOrganizationChange((Organization)event.getOldObject(), (Organization)event.getNewObject());
123    } else if (event.getObjectClass().equals(Dataset.class)) {
124      cascadeDatasetChange((Dataset)event.getOldObject(), (Dataset)event.getNewObject());
125    } else if (event.getObjectClass().equals(Installation.class)) {
126      cascadeInstallationChange((Installation)event.getOldObject(), (Installation)event.getNewObject());
127    }
128  }
129
130  @Subscribe
131  public final <T extends NetworkEntity> void deleted(DeleteEvent<T> event) {
132    purgeEntityAndBanLists(event.getObjectClass(), event.getOldObject().getKey());
133
134    if (event.getObjectClass().equals(Organization.class)) {
135      cascadeOrganizationChange((Organization) event.getOldObject());
136    } else if (event.getObjectClass().equals(Dataset.class)) {
137      cascadeDatasetChange((Dataset) event.getOldObject());
138    } else if (event.getObjectClass().equals(Installation.class)) {
139      cascadeInstallationChange((Installation) event.getOldObject());
140    }
141  }
142
143  @Subscribe
144  public final void componentChange(ChangedComponentEvent event) {
145    purgeEntityAndBanLists(event.getTargetClass(), event.getTargetEntityKey());
146    // keys have not changed, only some component of the entity itself
147    if (event.getTargetClass().equals(Organization.class)) {
148      cascadeOrganizationChange(organizationService.get(event.getTargetEntityKey()));
149    } else if (event.getTargetClass().equals(Dataset.class)) {
150      cascadeDatasetChange(datasetService.get(event.getTargetEntityKey()));
151    } else if (event.getTargetClass().equals(Installation.class)) {
152      cascadeInstallationChange(installationService.get(event.getTargetEntityKey()));
153    }
154  }
155
156  // group bans by entity class to avoid too many ban rules and thus bad varnish performance
157  private void cascadeDatasetChange(Dataset ... datasets) {
158    // UUIDHashSet ignores null values
159    Set<UUID> instKeys = new UUIDHashSet();
160    Set<UUID> orgKeys = new UUIDHashSet();
161    Set<UUID> nodeKeys = new UUIDHashSet();
162    Set<UUID> parentKeys = new UUIDHashSet();
163    for (Dataset d : datasets) {
164      if (!orgKeys.contains(d.getPublishingOrganizationKey())) {
165        Organization o = organizationService.get(d.getPublishingOrganizationKey());
166        nodeKeys.add(o.getEndorsingNodeKey());
167      }
168      if (!instKeys.contains(d.getInstallationKey())) {
169        instKeys.add(d.getInstallationKey());
170        Installation i = installationService.get(d.getInstallationKey());
171        orgKeys.add(i.getOrganizationKey());
172      }
173      orgKeys.add(d.getPublishingOrganizationKey());
174      if (d.getParentDatasetKey() != null) {
175        parentKeys.add(d.getParentDatasetKey());
176        purger.purge(path("dataset", d.getParentDatasetKey()));
177      }
178    }
179    purger.ban(String.format("dataset/%s/constituents", purger.anyKey(parentKeys)));
180    // /installation/{d.installationKey}/dataset BAN
181    purger.ban(String.format("installation/%s/dataset", purger.anyKey(instKeys)));
182    // /organization/{d.publishingOrganizationKey}/publishedDataset BAN
183    // /organization/{d.installation.organizationKey}/hostedDataset BAN
184    purger.ban(String.format("organization/%s/(published|hosted)Dataset", purger.anyKey(orgKeys)));
185    // /node/{d.publishingOrganization.endorsingNodeKey}/dataset BAN
186    purger.ban(String.format("node/%s/dataset", purger.anyKey(nodeKeys)));
187    // /network/{any UUID}/constituents BAN
188    purger.ban(String.format("network/.+/constituents"));
189  }
190
191  private void cascadeOrganizationChange(Organization ... orgs) {
192    // /node/{org.endorsingNodeKey}/
193    Set<UUID> nodeKeys = new UUIDHashSet();
194    for (Organization o : orgs) {
195      nodeKeys.add(o.getEndorsingNodeKey());
196    }
197    purger.ban(String.format("node/%s/organization", purger.anyKey(nodeKeys)));
198  }
199
200  private void cascadeInstallationChange(Installation ... installations) {
201    Set<UUID> keys = new UUIDHashSet();
202    // /organization/{i.organizationKey}/installation BAN
203    for (Installation i : installations) {
204      keys.add(i.getOrganizationKey());
205    }
206    purger.ban(String.format("organization/%s/installation", purger.anyKey(keys)));
207
208    // /node/{i.organization.endorsingNodeKey}/installation BAN
209    Set<UUID> nodekeys = new UUIDHashSet();
210    for (UUID orgKey : keys) {
211      Organization o = organizationService.get(orgKey);
212      nodekeys.add(o.getEndorsingNodeKey());
213    }
214    purger.ban(String.format("%node/%s/organization", purger.anyKey(nodekeys)));
215  }
216
217  /**
218   * Removes the specific entity from varnish and bans search & list pages.
219   * This method does not check which entity class was supplied, but as it is some type of NetworkEntity
220   * we deal with the right urls.
221   */
222  private void purgeEntityAndBanLists(Class cl, UUID key) {
223    // purge entity detail
224    purger.purge( path(cl.getSimpleName().toLowerCase(), key) );
225
226    // banRegex lists and searches
227    purger.ban(String.format("%s(/search|/suggest)?[^/]*$", cl.getSimpleName().toLowerCase()));
228  }
229
230  private static String path(Object ... parts) {
231    return PATH_JOINER.join(parts);
232  }
233
234  /**
235   * HashSet with an overriden add method that silently ignores null values being added.
236   */
237  public class UUIDHashSet extends HashSet<UUID> {
238    @Override
239    public boolean add(UUID t) {
240      return t != null && super.add(t);
241    }
242  }
