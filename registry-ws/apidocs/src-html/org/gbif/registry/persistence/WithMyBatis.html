040  @Transactional
041  public static <T extends NetworkEntity> UUID create(NetworkEntityMapper<T> mapper, T entity) {
042    checkArgument(entity.getKey() == null, "Unable to create an entity which already has a key");
043    // REVIEW: If this call fails the entity will have been modified anyway! We could make a copy and return that
044// instead
045    entity.setKey(UUID.randomUUID());
046    mapper.create(entity);
047    return entity.getKey();
048  }

051    return mapper.get(key);
052  }
053
054  @Transactional
055  public static <T extends NetworkEntity> void update(NetworkEntityMapper<T> mapper, T entity) {
056    checkNotNull(entity, "Unable to update an entity when it is not provided");
057    T existing = mapper.get(entity.getKey());
058    checkNotNull(existing, "Unable to update a non existing entity");
059
060    if (existing.getDeleted() != null) {
061      // allow updates ONLY if they are undeleting too
062      checkArgument(entity.getDeleted() == null,
063        "Unable to update a previously deleted entity unless you clear the deletion timestamp");
064    } else {
065      // do not allow deletion here (for safety) - we have an explicity deletion service
066      checkArgument(entity.getDeleted() == null, "Cannot delete using the update service.  Use the deletion service");
067    }
068
069    mapper.update(entity);
070  }

077  public static <T extends NetworkEntity> PagingResponse<T> list(NetworkEntityMapper<T> mapper, Pageable page) {
078    long total = mapper.count();
079    return new PagingResponse<T>(page.getOffset(), page.getLimit(), total, mapper.list(page));
080  }

082  /**
083   * The simple search option of the list.
084   * 
085   * @param mapper To use for the search
086   * @param query A simple query string such as "Pontaurus"
087   * @param page To support paging
088   * @return A paging response
089   */
090  public static <T extends NetworkEntity> PagingResponse<T> search(NetworkEntityMapper<T> mapper, String query,
091    Pageable page) {
092    Preconditions.checkNotNull(page, "To search you must supply a page");
093    long total = mapper.count(query);
094    return new PagingResponse<T>(page.getOffset(), page.getLimit(), total, mapper.search(query, page));
095  }
096
097  public static <T extends NetworkEntity> PagingResponse<T> listByIdentifier(NetworkEntityMapper<T> mapper,
098    @Nullable IdentifierType type, String identifier, @Nullable Pageable page) {
099    Preconditions.checkNotNull(page, "To list by identifier you must supply a page");
100    Preconditions.checkNotNull(identifier, "To list by identifier you must supply an identifier");
101    long total = mapper.countByIdentifier(type, identifier);
102    return new PagingResponse<T>(page.getOffset(), page.getLimit(), total, mapper.listByIdentifier(type, identifier,
103      page));
104  }

107  @Transactional
108  public static int addContact(
109    ContactMapper contactMapper, ContactableMapper contactableMapper, UUID targetEntityKey, Contact contact
110    ) {
111    checkArgument(contact.getKey() == null, "Unable to create an entity which already has a key");
112    contactMapper.createContact(contact);
113    // is this a primary contact? We need to make sure it only exists once per type
114    if (contact.isPrimary()) {
115      contactableMapper.updatePrimaryContacts(targetEntityKey, contact.getType());
116    }
117    contactableMapper.addContact(targetEntityKey, contact.getKey(), contact.getType(), contact.isPrimary());
118    return contact.getKey();
119  }
120
121  @Transactional
122  public static int updateContact(ContactMapper contactMapper, ContactableMapper contactableMapper,
123    UUID targetEntityKey, Contact contact) {
124    checkArgument(contact.getKey() != null, "Unable to update an entity with no key");
125    // null safe checking follows
126    checkArgument(Boolean.TRUE.equals(contactableMapper.areRelated(targetEntityKey, contact.getKey())),
127      "The provided contact is not connected to the given entity");
128    // is this a primary contact? We need to make sure it only exists once per type
129    if (contact.isPrimary()) {
130      contactableMapper.updatePrimaryContacts(targetEntityKey, contact.getType());
131    }
132    contactMapper.updateContact(contact);
133    // update the type and is_primary (is_primary will have been set to false by updatePrimaryContacts above)
134    contactableMapper.updateContact(targetEntityKey, contact.getKey(), contact.getType(), contact.isPrimary());
135    return contact.getKey();
136  }

146  @Transactional
147  public static int addEndpoint(
148    EndpointMapper endpointMapper, EndpointableMapper endpointableMapper, UUID targetEntityKey, Endpoint endpoint,
149    MachineTagMapper machineTagMapper
150    ) {
151    checkArgument(endpoint.getKey() == null, "Unable to create an entity which already has a key");
152    endpointMapper.createEndpoint(endpoint);
153    endpointableMapper.addEndpoint(targetEntityKey, endpoint.getKey());
154
155    for (MachineTag machineTag : endpoint.getMachineTags()) {
156      machineTag.setCreatedBy(endpoint.getCreatedBy());
157      WithMyBatis.addMachineTag(machineTagMapper, endpointMapper, endpoint.getKey(), machineTag);
158    }
159    return endpoint.getKey();
160  }
