046/**
047 * A builder that will clear and build a new dataset index by paging over the given service.
048 */
049@Singleton
050public class DatasetIndexBuilder {
051
052  // controls how many results we request while paging over the WS
053  private static final int WS_PAGE_SIZE = 100;
054  private static final Logger LOG = LoggerFactory.getLogger(DatasetIndexBuilder.class);
055  private final SolrClient solrClient;
056  private final DatasetService datasetService;
057  private final DatasetDocConverter docConverter;
058
059  @Inject
060  public DatasetIndexBuilder(@Named("Dataset") SolrClient solrClient, DatasetService datasetService,
061    InstallationService installationService, OrganizationService organizationService) {
062    this.solrClient = solrClient;
063    this.datasetService = datasetService;
064    // We can use a cache at startup
065    docConverter =
066      new DatasetDocConverter(new CachingNetworkEntityService<Organization>(organizationService),
067        new CachingNetworkEntityService<Installation>(installationService), datasetService);
068  }
069
070  public void build() throws SolrServerException, IOException {
071    LOG.info("Building a new Dataset index");
072    Stopwatch stopwatch = Stopwatch.createStarted();
073    pageAndIndex();
074    solrClient.commit();
075    solrClient.optimize();
076    LOG.info("Finished building Dataset index in {} secs", stopwatch.elapsed(TimeUnit.SECONDS));
077  }
078
079  /**
080   * Pages over all datasets and adds to SOLR.
081   */
082  private void pageAndIndex() throws IOException, SolrServerException {
083    PagingRequest page = new PagingRequest(0, WS_PAGE_SIZE);
084    PagingResponse<Dataset> response = null;
085    do {
086      LOG.debug("Requesting {} datasets starting at offset {}", page.getLimit(), page.getOffset());
087      response = datasetService.list(page);
088      // Batching updates to SOLR proves quicker with batches of 100 - 1000 showing similar performance
089      List<SolrInputDocument> batch = Lists.newArrayList();
090      for (Dataset ds : response.getResults()) {
091        batch.add(docConverter.build(ds));
092      }
093      if (!batch.isEmpty()) {
094        solrClient.add(batch);
095        solrClient.commit(); // to allow eager users (or impatient developers) to see search data on startup quickly
096      }
097      page.nextPage();
098
099    } while (!response.isEndOfRecords());
100  }

102  /**
103   * A lightweight cache to help improve performance of the builder.
104   *
105   * @param <T> The type of entity being wrapped
106   */
107  private static class CachingNetworkEntityService<T> implements NetworkEntityService<T> {
108
109    private final NetworkEntityService<T> service;
110    private final LoadingCache<UUID, T> cache = CacheBuilder.newBuilder()
111      .maximumSize(1000)
112      .expireAfterWrite(10, TimeUnit.MINUTES)
113      .build(
114        new CacheLoader<UUID, T>() {
115
116          @Override
117          public T load(UUID key) throws Exception {
118            return service.get(key);
119          }
120        });
121
122    public CachingNetworkEntityService(NetworkEntityService<T> service) {
123      this.service = service;
124    }
