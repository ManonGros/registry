001package org.gbif.registry.search;
002
003import java.util.regex.Pattern;
004
005import org.xml.sax.Attributes;
006import org.xml.sax.SAXException;
007import org.xml.sax.helpers.DefaultHandler;
008
009/**
010 * A simple metadata sax base handler that collects all character data inside elements into a string buffer, resetting
011 * the buffer with every element start and storing the string version of the buffer in this.content when the end of the
012 * element is reached.
013 */
014public class FullTextSaxHandler extends DefaultHandler {
015
016  private StringBuffer content;
017  private Pattern normWhitespace = Pattern.compile("[\\s\n\r\t]+");
018
019  @Override
020  public void startDocument() throws SAXException {
021    super.startDocument();
022    content = new StringBuffer(" ");
023  }
024
025  @Override
026  public void characters(char[] ch, int start, int length) {
027    content.append(ch, start, length);
028  }
029
030  @Override
031  public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
032    // as we remove tags we add whitespace to delimit words
033    content.append(" ");
034    // add attribute content to the full text
035    for (int idx = attributes.getLength(); idx > 0; idx--) {
036      content.append(attributes.getValue(idx - 1));
037      content.append(" ");
038    }
039  }
040
041  public String getFullText() {
042    return normWhitespace.matcher(content.toString()).replaceAll(" ").trim();
043  }
044}
