040/**
041 * A guice module that sets up implementation of the search services and gives SOLR providers for search.
042 * <p/>
043 * The current implementation uses an embedded SOLR index, that uses a SOLR RAM directory. This is satisfactory while
044 * indexes remain small, and while we opt to rebuild the index on startup. Future implementations might consider using
045 * an external SOLR server, thus this method signatures use abstract SOLR type, and not a concrete implementation to
046 * allow for future changes.
047 * <p/>
048 * <strong>A note on the gbif-common-search SolrModule {@link org.gbif.common.search.inject.SolrModule}:</strong>
049 * <p/>
050 * The {@link org.gbif.common.search.inject.SolrModule} was considered for use here, but discarded because in its current implementation:
051 * <ul>
052 * <li>The {@link org.gbif.common.search.solr.builders.EmbeddedServerBuilder} does not support multiple cores, but that is anticipated as a future
053 * requirement for this project</li>
054 * <li>The {@link org.gbif.common.search.solr.builders.EmbeddedServerBuilder} does not allow someone to supply additional configuration files (it was
055 * developed for unit test purposes only)</li>
056 * <li>The SOLR server desired requires only 3 lines of code to instantiate. The implementor felt that readability is
057 * improved keeping this together</li>
058 * <li>{@link org.gbif.common.search.inject.SolrModule} is complex as it provides configuration based declaration (e.g. to support varying
059 * deployments. The implementor felt that was an additional learning curve that is unnecessary to put on future
060 * developers of this project. That said, should this project require flexible deployment options in the future,
061 * developers would be wise to consider adding multi-core support to the {@link org.gbif.common.search.inject.SolrModule}</li>
062 * </ul>
063 */
064public class RegistrySearchModule extends PrivateServiceModule {
065
066  private static final String REGISTRY_PROPERTY_PREFIX = "registry.search.";
067
068  public RegistrySearchModule(Properties properties) {
069    super(REGISTRY_PROPERTY_PREFIX, properties);
070  }

072  @Override
073  public void configureService() {
074    bind(DatasetSearchService.class).to(DatasetSearchServiceSolr.class).in(Scopes.SINGLETON);
075    bind(DatasetService.class).to(DatasetResource.class).in(Scopes.SINGLETON);
076    bind(OrganizationService.class).to(OrganizationResource.class).in(Scopes.SINGLETON);
077    bind(NodeService.class).to(NodeResource.class).in(Scopes.SINGLETON);
078    bind(InstallationService.class).to(InstallationResource.class).in(Scopes.SINGLETON);
079    bind(DatasetIndexUpdateListener.class).asEagerSingleton();
080    bind(DatasetIndexBuilder.class).in(Scopes.SINGLETON);
081    expose(DatasetSearchService.class);
082    expose(DatasetIndexUpdateListener.class); // for testing
083    expose(OrganizationService.class); // for testing
084    expose(InstallationService.class); // for testing
085    expose(DatasetService.class); // for testing
086    expose(NodeService.class); // for testing
087
088    // 6 email properties:
089    // use dev email?
090    mapAndExposeBool("mail.devemail.enabled","useDevEmail");
091    // smpt host?
092    mapAndExpose("mail.smtp.host","smptHost");
093    // smpt port?
094    mapAndExposeInt("mail.smtp.port","smptPort");
095    // dev email address?
096    mapAndExpose("mail.devemail","devEmail");
097    // cc (helpdesk) email address?
098    mapAndExpose("mail.cc","ccEmail");
099    // from email address?
100    mapAndExpose("mail.from","fromEmail");
101    // Expose solr.home
102    mapAndExpose("solr.home","solr.home");
103  }
104
105  /**
106   * Map the String property "fromProperty" to and String named annotation "name".
107   */
108  private void mapAndExpose(String fromProperty , String name){
109    String from = getVerbatimProperties().getProperty(fromProperty);
110    bind(String.class).annotatedWith(Names.named(name)).toInstance(from);
111    expose(String.class).annotatedWith(Names.named(name));
112  }
113
114  /**
115   * Map the String property "fromProperty" to and Boolean named annotation "name".
116   */
117  private void mapAndExposeBool(String fromProperty , String name){
118    boolean from = Boolean.valueOf(getVerbatimProperties().getProperty(fromProperty));
119    bind(Boolean.class).annotatedWith(Names.named(name)).toInstance(from);
120    expose(Boolean.class).annotatedWith(Names.named(name));
121  }
122
123  /**
124   * Map the String property "fromProperty" to and Integer named annotation "name".
125   */
126  private void mapAndExposeInt(String fromProperty , String name){
127    int from = Integer.valueOf(getVerbatimProperties().getProperty(fromProperty));
128    bind(Integer.class).annotatedWith(Names.named(name)).toInstance(from);
129    expose(Integer.class).annotatedWith(Names.named(name));
130  }

132  /**
133   * A provider of the SolrServer to use with a dataset search.
134   * Named as such, to allow multiple providers (e.g. solr cores) should this be required in the future.
135   *
136   * @return An embedded solr server that uses the configuration in /solr on the classpath
137   * @throws java.net.URISyntaxException If /solr configuration is not found on the classpath
138   */
139  @Provides
140  @Exposed
141  @Named("Dataset")
142  @Singleton
143  public SolrClient datasetSolr(@Named("solr.home") String solrHome) throws URISyntaxException, IOException {
144    Path tmpSolrHome = createTempSolrDirectory(solrHome);
145    File conf = new File(tmpSolrHome.toFile().getAbsolutePath(), "solr.xml");
146    return new EmbeddedSolrServer(CoreContainer.createAndLoad(tmpSolrHome.toFile().getAbsolutePath(), conf), "dataset");
147  }
148
149  /**
150   * Creates a temporary Solr directory containing all the Solr configuration files.
151   */
152  private Path createTempSolrDirectory(String solrHome) throws IOException {
153    final Path solrHomePath = Paths.get(solrHome);
154    final Path datasetCoreHomePath = Paths.get(solrHome,"dataset");
155    File solrHomeFile = solrHomePath.toFile();
156    if (solrHomeFile.exists()){
157      FileUtils.cleanDirectory(solrHomeFile);
158    } else {
159      solrHomeFile.mkdirs();
160    }
161    File confDir = new File(datasetCoreHomePath.toFile(),"conf");
162    confDir.mkdirs();
163    final Path confPath = Paths.get(confDir.toURI());
164    exportResource("/solr/solr.xml",solrHomePath);
165    exportResource("/solr/dataset/core.properties", datasetCoreHomePath);
166    exportResource("/solr/dataset/conf/mapping-FoldToASCII.txt", confPath);
167    exportResource("/solr/dataset/conf/mapping-ISOLatin1Accent.txt", confPath);
168    exportResource("/solr/dataset/conf/schema.xml", confPath);
169    exportResource("/solr/dataset/conf/solrconfig.xml", confPath);
170    return solrHomePath;
171  }

173  /**
174   * Export a resource embedded into a Jar file to the local file path.
175   *
176   * @param resourceName ie.: "/SmartLibrary.dll"
177   * @return The path to the exported resource
178   * @throws Exception
179   */
180  private static String exportResource(String resourceName, Path tmpDir) throws IOException {//todo move to Utils
181    Closer closer = Closer.create();
182    try {
183      URL resource = RegistrySearchModule.class.getResource(resourceName);
184      File fileResource = new File(resource.getFile());
185      InputStream stream = closer.register(resource.openStream());//note that each / is a directory down in the "jar tree" been the jar the root of the tree"
186      if(stream == null) {
187        throw new IOException("Cannot get resource \"" + resourceName + "\" from Jar file.");
188      }
189      int readBytes;
190      byte[] buffer = new byte[4096];
191      OutputStream resStreamOut = closer.register(new FileOutputStream(tmpDir + "/" + fileResource.getName()));
192      while ((readBytes = stream.read(buffer)) > 0) {
193        resStreamOut.write(buffer, 0, readBytes);
194      }
195    } catch (IOException ex) {
196      throw ex;
197    } finally {
198      closer.close();
199    }
200    return tmpDir + resourceName;
201  }
