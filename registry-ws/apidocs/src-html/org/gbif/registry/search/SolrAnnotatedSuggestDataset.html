001package org.gbif.registry.search;
002
003import org.gbif.api.model.registry.search.DatasetSuggestResult;
004import org.gbif.api.vocabulary.DatasetSubtype;
005import org.gbif.api.vocabulary.DatasetType;
006import org.gbif.common.search.model.Key;
007import org.gbif.common.search.model.SuggestMapping;
008
009import java.util.UUID;
010
011import org.apache.solr.client.solrj.beans.Field;
012
013/**
014 * This class contains the annotations required by {@link org.gbif.api.service.common.SuggestService} and the Solr
015 * result/object mapping.
016 * This class holds the Solr response mapping and the annotation required by the suggest service.
017 */
018@SuggestMapping(field = "dataset_title_ngram", phraseQueryField = "dataset_title_nedge")
019public class SolrAnnotatedSuggestDataset extends DatasetSuggestResult {
020
021  @Field("description")
022  @Override
023  public void setDescription(String description) {
024    super.setDescription(description);
025  }
026
027  @Field("key")
028  @Key
029  public void setKey(String key) {
030    setKey(UUID.fromString(key));
031  }
032
033  @Field("dataset_subtype")
034  public void setSubtype(Integer datasetSubtypeOrdinal) {
035    setSubtype(datasetSubtypeOrdinal == null ? null : DatasetSubtype.values()[datasetSubtypeOrdinal]);
036  }
037
038  @Field("dataset_title")
039  @Override
040  public void setTitle(String title) {
041    super.setTitle(title);
042  }
043
044  @Field("dataset_type")
045  public void setType(Integer datasetTypeOrdinal) {
046    setType(datasetTypeOrdinal == null ? null : DatasetType.values()[datasetTypeOrdinal]);
047  }
048}
