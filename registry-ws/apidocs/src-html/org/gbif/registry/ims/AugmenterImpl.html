001package org.gbif.registry.ims;
002
003import org.gbif.api.model.registry.Identifier;
004import org.gbif.api.model.registry.Node;
005import org.gbif.api.vocabulary.IdentifierType;
006import org.gbif.drupal.mybatis.ImsNodeMapper;
007
008import com.google.inject.Inject;
009import com.google.inject.Singleton;
010import org.slf4j.Logger;
011import org.slf4j.LoggerFactory;
012
013@Singleton
014public class AugmenterImpl implements Augmenter {
015  private static Logger LOG = LoggerFactory.getLogger(AugmenterImpl.class);
016
017  private ImsNodeMapper mapper;
018
019  @Inject
020  public AugmenterImpl(ImsNodeMapper mapper) {
021    this.mapper = mapper;
022  }
023
024  private Integer findImsParticipantID(Node node) {
025    for (Identifier id : node.getIdentifiers()) {
026      if (IdentifierType.GBIF_PARTICIPANT == id.getType()) {
027        try {
028          return Integer.parseInt(id.getIdentifier());
029        } catch (NumberFormatException e) {
030          LOG.error("IMS Participant ID is no integer: %s", id.getIdentifier());
031        }
032      }
033    }
034    return null;
035  }
036
037  @Override
038  public Node augment(Node node) {
039    if (node != null) {
040      try {
041        Integer imsId = findImsParticipantID(node);
042        if (imsId != null) {
043          Node imsNode = mapper.get(imsId);
044          if (imsNode != null) {
045            // update node with IMS info if it exists
046            node.setParticipantTitle(imsNode.getParticipantTitle());
047            node.setContacts(imsNode.getContacts());
048            node.setAbbreviation(imsNode.getAbbreviation());
049            node.setDescription(imsNode.getDescription());
050            node.setParticipantSince(imsNode.getParticipantSince());
051            node.setOrganization(imsNode.getOrganization());
052            node.setAddress(imsNode.getAddress());
053            node.setPostalCode(imsNode.getPostalCode());
054            node.setCity(imsNode.getCity());
055            node.setProvince(imsNode.getProvince());
056            node.setEmail(imsNode.getEmail());
057            node.setPhone(imsNode.getPhone());
058            node.setHomepage(imsNode.getHomepage());
059          }
060        }
061      } catch (Exception e) {
062        LOG.error("Failed to augment node %s with IMS information", node.getKey(), e);
063      }
064    }
065
066    return node;
067  }
068}
