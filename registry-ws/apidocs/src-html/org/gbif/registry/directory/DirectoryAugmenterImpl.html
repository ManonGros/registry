029@Singleton
030public class DirectoryAugmenterImpl implements Augmenter {
031
032  /**
033   * Maps contact types to participant roles.
034   */
035  private static final ImmutableMap<ParticipantPersonRole,ContactType> PARTICIPANT_ROLE_TO_CONTACT_TYPE =
036          ImmutableMap.of(ParticipantPersonRole.ADDITIONAL_DELEGATE, ContactType.ADDITIONAL_DELEGATE,
037                  ParticipantPersonRole.HEAD_OF_DELEGATION, ContactType.HEAD_OF_DELEGATION,
038                  ParticipantPersonRole.TEMPORARY_DELEGATE, ContactType.TEMPORARY_DELEGATE,
039                  ParticipantPersonRole.TEMPORARY_HEAD_OF_DELEGATION, ContactType.TEMPORARY_HEAD_OF_DELEGATION);
040
041  private static final ImmutableMap<NodePersonRole,ContactType> NODE_ROLE_TO_CONTACT_TYPE =
042          ImmutableMap.of(NodePersonRole.NODE_MANAGER, ContactType.NODE_MANAGER,
043                  NodePersonRole.NODE_STAFF, ContactType.NODE_STAFF);
044
045  private static Logger LOG = LoggerFactory.getLogger(DirectoryAugmenterImpl.class);
046
047  private ParticipantService participantService;
048  private NodeService nodeService;
049  private PersonService personService;
050
051  @Inject
052  public DirectoryAugmenterImpl(ParticipantService participantService, NodeService nodeService, PersonService personService) {
053    this.participantService = participantService;
054    this.nodeService = nodeService;
055    this.personService = personService;
056  }
057
058  /**
059   * Gets the participantID from the Node.
060   */
061  private static Integer findParticipantID(Node node) {
062    for (Identifier id : node.getIdentifiers()) {
063      if (IdentifierType.GBIF_PARTICIPANT == id.getType()) {
064        try {
065          return Integer.parseInt(id.getIdentifier());
066        } catch (NumberFormatException e) {
067          LOG.error("Directory participantId is no integer: %s", id.getIdentifier());
068        }
069      }
070    }
071    return null;
072  }
073
074  @Override
075  public Node augment(Node registryNode) {
076    if (registryNode != null) {
077      try {
078        Integer participantID = findParticipantID(registryNode);
079        if (participantID != null) {
080          Participant participant = participantService.get(participantID);
081          if (participant != null) {
082            List<Contact> contacts = Lists.newArrayList();
083            // update node with Directory info if it exists
084            List<org.gbif.api.model.directory.Node> participantNodes = getParticipantNodes(participant);
085            registryNode.setParticipantTitle(participant.getName());
086            contacts.addAll(getContactsForParticipant(participant));
087
088            registryNode.setAbbreviation(participant.getAbbreviatedName());
089            registryNode.setDescription(participant.getComments());
090            if(participant.getMembershipStart() != null) {
091              registryNode.setParticipantSince(getParticipantSinceYear(participant.getMembershipStart()));
092            }
093            if(!participantNodes.isEmpty()){
094              contacts.addAll(getContactsForNode(participantNodes));
095              registryNode.setAddress(getNodesAddresses(participantNodes));
096              registryNode.setHomepage(getWebUrls(participant,participantNodes));
097              registryNode.setEmail(getEmails(participantNodes));
098              registryNode.setPhone(getPhones(participantNodes));
099            }
100            else{
101              LOG.info("Empty node for participantId %d", participantID);
102            }
103            registryNode.setContacts(contacts);
104          }
105        }
106      } catch (Exception e) {
107        LOG.error("Failed to augment node %s with Directory information", registryNode.getKey(), e);
108      }
109    }
110    return registryNode;
111  }
112
113  /**
114   * Gets the year part of the membershipStart field.
115   */
116  private static Integer getParticipantSinceYear(String membershipStart) {
117    String[] membershipStartComponents = membershipStart.split(" ");
118    if(membershipStartComponents.length > 1) {
119      return Ints.tryParse(membershipStartComponents[1].trim());
120    }
121    return null;
122  }
123
124  /**
125   * Gets all the nodes associated to a participant.
126   */
127  private List<org.gbif.api.model.directory.Node> getParticipantNodes(Participant participant){
128    List<org.gbif.api.model.directory.Node> nodes = Lists.newArrayList();
129    if(participant.getNodes() != null){
130      for(org.gbif.api.model.directory.Node node : participant.getNodes()) {
131        nodes.add(nodeService.get(node.getId()));
132      }
133    }
134    return nodes;
135  }
136
137  /**
138   * Gets all the addresses associated to a participant.
139   */
140  private static List<String> getNodesAddresses(List<org.gbif.api.model.directory.Node> participantNodes){
141    List<String> addresses = Lists.newArrayList();
142    for(org.gbif.api.model.directory.Node node : participantNodes) {
143      addresses.add(node.getAddress());
144    }
145    return  addresses;
146  }
147
148  /**
149   * Gets all the web pages/urls of participant and its nodes.
150   */
151  private static List<URI> getWebUrls(Participant participant, List<org.gbif.api.model.directory.Node> participantNodes){
152    List<URI> addresses = Lists.newArrayList();
153    if(participant.getParticipantUrl() != null) {
154      addresses.add(URI.create(participant.getParticipantUrl()));
155    }
156    for(org.gbif.api.model.directory.Node node : participantNodes) {
157      if(node.getNodeUrl() != null) {
158        addresses.add(URI.create(node.getNodeUrl()));
159      }
160    }
161    return  addresses;
162  }
163
164  /**
165   * Gets all the emails from the nodes list.
166   */
167  private static List<String> getEmails(List<org.gbif.api.model.directory.Node> participantNodes){
168    List<String> emails = Lists.newArrayList();
169    for(org.gbif.api.model.directory.Node node : participantNodes) {
170      if(node.getEmail() != null) {
171        emails.add(node.getEmail());
172      }
173    }
174    return  emails;
175  }
176
177  /**
178   * Gets all the phones numbers from the nodes list.
179   */
180  private static List<String> getPhones(List<org.gbif.api.model.directory.Node> participantNodes){
181    List<String> phones = Lists.newArrayList();
182    for(org.gbif.api.model.directory.Node node : participantNodes) {
183      if(node.getPhone() != null) {
184        phones.add(node.getPhone());
185      }
186    }
187    return  phones;
188  }
189
190  /**
191   * Transforms the persons associated to a participant into a list of contacts.
192   */
193  private List<Contact> getContactsForParticipant(Participant participant){
194    List<Contact> contacts = Lists.newArrayList();
195    if(participant.getPeople() != null){
196      Person person;
197      Contact contact;
198      ContactType contactType;
199      for(ParticipantPerson participantPerson : participant.getPeople()) {
200        person = personService.get(participantPerson.getPersonId());
201        contactType = null;
202        if( participantPerson.getRole() != null) {
203          contactType = PARTICIPANT_ROLE_TO_CONTACT_TYPE.get(participantPerson.getRole());
204        }
205        contact = personToContact(person, participant.getName(), contactType);
206        contacts.add(contact);
207      }
208    }
209    return contacts;
210  }
211
212  /**
213   * Transforms the persons associated to a node(s) into a list of contacts.
214   * @param directoryNodes it theory it should never be more than one
215   */
216  private List<Contact> getContactsForNode(List<org.gbif.api.model.directory.Node> directoryNodes){
217    List<Contact> contacts = Lists.newArrayList();
218    if(directoryNodes != null){
219      Person person;
220      Contact contact;
221      ContactType contactType;
222      for(org.gbif.api.model.directory.Node currentNode : directoryNodes) {
223        if(currentNode.getPeople() != null && !currentNode.getPeople().isEmpty()) {
224          for (NodePerson nodePerson : currentNode.getPeople()) {
225            person = personService.get(nodePerson.getPersonId());
226            contactType = null;
227            if (nodePerson.getRole() != null) {
228              contactType = NODE_ROLE_TO_CONTACT_TYPE.get(nodePerson.getRole());
229            }
230            contact = personToContact(person, currentNode.getName(), contactType);
231            contacts.add(contact);
232          }
233        }
234        else{
235          LOG.info("No people found for nodeId %d", currentNode.getId());
236        }
237      }
238    }
239    return contacts;
240  }
241
242  /**
243   * Transform a Directory Person into a Registry Contact.
244   * @param person
245   * @param organization
246   * @param contactType
247   * @return
248   */
249  private Contact personToContact(Person person, String organization, ContactType contactType){
250    Contact contact = new Contact();
251    contact.setKey(person.getId());
252    contact.addAddress(person.getAddress());
253    contact.setCountry(person.getCountryCode());
254    if(person.getEmail() != null) {
255      contact.addEmail(person.getEmail());
256    }
257    contact.setFirstName(person.getFirstName());
258    contact.setLastName(person.getSurname());
259    if(person.getPhone() != null) {
260      contact.addPhone(person.getPhone());
261    }
262    contact.setModified(person.getModified());
263    contact.setOrganization(organization);
264    if( contactType != null) {
265      contact.setType(contactType);
266    }
267    return contact;
268  }
