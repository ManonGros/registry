059/**
060 * An OAI-PMH endpoint, using the XOAI library.
061 */
062@Path("oai-pmh/registry")
063@Singleton
064public class OaipmhEndpoint {
065
066  private static final TransformerFactory factory = TransformerFactory.newInstance();
067
068  private static final DateProvider dateProvider = new UTCDateProvider();
069  private static final ErrorHandler errorsHandler = new ErrorHandler();
070
071  public Transformer xsltTransformer(String xsltFile) {
072    try {
073      /*
074       * TODO: "An object of this class [Transformer] may not be used in multiple threads running concurrently!"
075       * Instead, XOAI should accept an immutable Templates object, and call .newTransformer() to get a Transformer.
076       * See https://xalan.apache.org/xalan-j/usagepatterns.html#multithreading
077       */
078      InputStream stream = this.getClass().getClassLoader().getResourceAsStream("org/gbif/registry/oaipmh/"+xsltFile);
079      return factory.newTransformer(new StreamSource(stream));
080    } catch (TransformerConfigurationException e) {
081      throw new RuntimeException("Unable to read XSLT transform "+xsltFile, e);
082    }
083  }
084
085  private final MetadataFormat OAIDC_METADATA_FORMAT = new MetadataFormat()
086          .withPrefix("oai_dc")
087          .withNamespace("http://www.openarchives.org/OAI/2.0/oai_dc/")
088          .withSchemaLocation("http://www.openarchives.org/OAI/2.0/oai_dc.xsd")
089          .withTransformer(xsltTransformer("dc.xslt"));
090
091  private final MetadataFormat EML_METADATA_FORMAT = new MetadataFormat()
092          .withPrefix("eml")
093          .withNamespace("eml://ecoinformatics.org/eml-2.1.1")
094          .withSchemaLocation("http://rs.gbif.org/schema/eml-gbif-profile/1.0.2/eml.xsd")
095          .withTransformer(xsltTransformer("eml.xslt"));
096
097  private Context context = new Context()
098          .withMetadataFormat(OAIDC_METADATA_FORMAT)
099          .withMetadataFormat(EML_METADATA_FORMAT);
100
101  private Repository repository;
102  private DataProvider dataProvider;
103
104  @Inject
105  public OaipmhEndpoint(RepositoryConfiguration repositoryConfiguration, ItemRepository itemRepository, SetRepository setRepository) {
106
107    this.repository = new Repository()
108            .withItemRepository(itemRepository)
109            .withSetRepository(setRepository)
110            .withResumptionTokenFormatter(new SimpleResumptionTokenFormat())
111            .withConfiguration(repositoryConfiguration);
112
113    this.dataProvider = new DataProvider(context, repository);
114  }
115
116  @GET
117  @Produces("application/xml;charset=UTF-8")
118  public InputStream oaipmh(
119          @QueryParam("verb") String verb,
120          @Nullable @QueryParam("identifier") String identifier,
121          @Nullable @QueryParam("metadataPrefix") String metadataPrefix,
122          @Nullable @QueryParam("from") String from,
123          @Nullable @QueryParam("until") String until,
124          @Nullable @QueryParam("set") String set,
125          @Nullable @QueryParam("resumptionToken") String resumptionToken) {
126
127    Date fromDate = null, untilDate = null;
128    OAIRequestParametersBuilder reqBuilder = new OAIRequestParametersBuilder()
129            .withVerb(verb)
130            .withMetadataPrefix(metadataPrefix);
131
132    if(from != null) {
133      try {
134        fromDate = dateProvider.parse(from);
135      } catch (ParseException pEx) {
136        return handleOAIRequestBadArgument(reqBuilder.build(), "from=" + from);
137      }
138    }
139
140    if(until != null){
141      try {
142        untilDate = dateProvider.parse(from);
143      } catch (ParseException pEx) {
144        return handleOAIRequestBadArgument(reqBuilder.build(), "until=" + until);
145      }
146    }
147
148    reqBuilder.withIdentifier(identifier)
149            .withFrom(fromDate)
150            .withUntil(untilDate)
151            .withSet(set)
152            .withResumptionToken(resumptionToken);
153
154    return handleOAIRequest(reqBuilder.build());
155  }
156
157  private InputStream handleOAIRequest(OAIRequest request) {
158    try {
159      OAIPMH oaipmh = dataProvider.handle(request);
160      return new ByteArrayInputStream(write(oaipmh).getBytes("UTF-8"));
161
162    } catch (Exception e) {
163      throw new ServiceUnavailableException("OAI Failed to serialize dataset", e);
164    }
165  }
166
167  /**
168   * Build and generate the response for a BadArgument error code.
169   *
170   * @param requestParameters origin of the BadArgument error
171   * @param errorMessage textual message to report
172   * @return
173   */
174  private InputStream handleOAIRequestBadArgument(OAIRequest requestParameters, String errorMessage) {
175
176    Request request = new Request(repository.getConfiguration().getBaseUrl())
177            .withVerbType(requestParameters.get(Verb))
178            .withResumptionToken(requestParameters.get(ResumptionToken))
179            .withIdentifier(requestParameters.get(Identifier))
180            .withMetadataPrefix(requestParameters.get(MetadataPrefix));
181
182    try {
183      OAIPMH errorResponse = new OAIPMH()
184              .withRequest(request)
185              .withResponseDate(dateProvider.now())
186              .withError(errorsHandler.handle(new BadArgumentException(errorMessage)));
187      return new ByteArrayInputStream(write(errorResponse).getBytes("UTF-8"));
188
189    } catch (Exception e) {
190      throw new ServiceUnavailableException("OAI Failed to serialize dataset", e);
191    }
192  }
193
194  protected String write(final XmlWritable handle) throws XMLStreamException, XmlWriteException {
195    return new StringBuilder()
196            .append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n")
197            .append(XmlWriter.toString(new XmlWritable() {
198              @Override
199              public void write(XmlWriter writer) throws XmlWriteException {
200                writer.write(handle);
201              }
202            }))
203            .toString();
204  }
