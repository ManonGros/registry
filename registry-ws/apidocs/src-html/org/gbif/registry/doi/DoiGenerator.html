001package org.gbif.registry.doi;
002
003import org.gbif.api.model.common.DOI;
004import org.gbif.doi.metadata.datacite.DataCiteMetadata;
005import org.gbif.doi.service.InvalidMetadataException;
006
007import java.util.UUID;
008
009/**
010 * Registry internal service that guarantees to issue unique new DOIs and deals with scheduling
011 * DOI metadata updates and registration via RabbitMQ.
012 */
013public interface DoiGenerator {
014
015  /**
016   * Generates a new unique GBIF dataset DOI.
017   * The new DOI is unknown to DataCite still and only lives in the GBIF registry which guarantees it to be unique.
018   * @return the new DOI
019   */
020  DOI newDatasetDOI();
021
022  /**
023   * Generates a new unique GBIF download DOI.
024   * The new DOI is unknown to DataCite still and only lives in the GBIF registry which guarantees it to be unique.
025   * @return the new DOI
026   */
027  DOI newDownloadDOI();
028
029  /**
030   * Tests a DOI to see if it was issued by GBIF.
031   * @return true if DOI was issued by GBIF
032   */
033  boolean isGbif(DOI doi);
034
035  /**
036   * Updates the doi table to status FAILED and uses the error message & stacktrace as the xml
037   * for manual debugging / cleanup.
038   */
039  void failed(DOI doi, InvalidMetadataException e);
040
041  /**
042   * Schedules a DOI metadata update with DataCite and registers the DOI if needed.
043   * For subsequent calls with the same DOI only the metadata in DataCite will be updated.
044   * If it is called for the very first time the DOI will also be properly registered with DataCite.
045   *
046   * @param doi the GBIF DOI to registerDataset
047   * @param metadata the metadata to post to datacite. Mandatory fields are validated immediately
048   * @param datasetKey the dataset key to derive the target URL from
049   *
050   * @throws InvalidMetadataException in case the metadata is missing mandatory fields or the DOI is not a GBIF one
051   */
052  void registerDataset(DOI doi, DataCiteMetadata metadata, UUID datasetKey) throws InvalidMetadataException;
053
054  /**
055   * Schedules a DOI metadata update with DataCite and registers the DOI if needed.
056   * For subsequent calls with the same DOI only the metadata in DataCite will be updated.
057   * If it is called for the very first time the DOI will also be properly registered with DataCite.
058   *
059   * @param doi the GBIF DOI to registerDataset
060   * @param metadata the metadata to post to datacite. Mandatory fields are validated immediately
061   * @param downloadKey the download key to derive the target URL from
062   *
063   * @throws InvalidMetadataException in case the metadata is missing mandatory fields or the DOI is not a GBIF one
064   */
065  void registerDownload(DOI doi, DataCiteMetadata metadata, String downloadKey) throws InvalidMetadataException;
066
067  /**
068   * Deletes a GBIF DOI.
069   * If the DOI is registered in DataCite it's deleted; otherwise, it's only removed in the GBIF Registry.
070   *
071   * @param doi the GBIF DOI to delete/unregister
072   */
073  void delete(DOI doi);
074}
