001package org.gbif;
002
003import org.gbif.api.model.registry.Contact;
004import org.gbif.api.model.registry.Dataset;
005import org.gbif.api.model.registry.Endpoint;
006import org.gbif.api.model.registry.Installation;
007import org.gbif.api.vocabulary.ContactType;
008import org.gbif.api.vocabulary.DatasetType;
009import org.gbif.api.vocabulary.EndpointType;
010import org.gbif.api.vocabulary.InstallationType;
011
012import java.net.URI;
013import java.net.URISyntaxException;
014import java.util.UUID;
015
016import com.google.common.collect.Lists;
017import org.junit.BeforeClass;
018import org.junit.Ignore;
019import org.junit.Test;
020import org.slf4j.LoggerFactory;
021
022import static org.gbif.config.jersey.RegistryWsClientFactoryJersey.datasetService;
023import static org.gbif.config.jersey.RegistryWsClientFactoryJersey.datasetServiceReadOnly;
024import static org.gbif.config.jersey.RegistryWsClientFactoryJersey.installationService;
025
026import static org.junit.Assert.assertEquals;
027import static org.junit.Assert.assertNotNull;
028
029@Ignore("Requires live UAT API")
030public class RegistryWsJerseyIT {
031  private static final org.slf4j.Logger LOG = LoggerFactory.getLogger(RegistryWsJerseyIT.class);
032  private static final String DATASET_TITLE = "Dataset - Ws Client Demo";
033
034  private static UUID datasetKey;
035  private static UUID installationKey;
036
037  /**
038   * Create Dataset one time only.
039   * </p>
040   * Since nodes and organizations can't be created using the API, an existing organization (endorsed already) is
041   * used as both the publishing organization for the dataset, and the hosting organization for the installation.
042   */
043  @BeforeClass
044  public static void createDataset() throws URISyntaxException {
045    // Publishing and hosting organization: Test Organization #1, endorsed by DanBIF
046    UUID organizationKey = UUID.fromString("0a16da09-7719-40de-8d4f-56a15ed52fb6");
047
048    // create new installation, indicating which organization is hosting the installation
049    Installation installation = new Installation();
050    installation.setTitle("Installation - Ws Client Demo");
051    installation.setType(InstallationType.IPT_INSTALLATION);
052    installation.setDisabled(false);
053    installation.setDescription("Test IPT instance created using GBIF Registry ws client");
054    installation.setOrganizationKey(organizationKey);
055    installationKey = installationService().create(installation);
056
057    // create new dataset, indicating which organization owns the dataset, and which installation hosts the dataset
058    Dataset dataset = new Dataset();
059    dataset.setTitle(DATASET_TITLE);
060    dataset.setType(DatasetType.OCCURRENCE);
061    dataset.setDescription("Test Dataset created using GBIF Registry ws client");
062    dataset.setPublishingOrganizationKey(organizationKey);
063    dataset.setInstallationKey(installationKey);
064    datasetKey = datasetService().create(dataset);
065    LOG.info("Dataset created with key: {}", datasetKey);
066  }
067
068  /**
069   * Read dataset
070   */
071  @Test
072  public void testReadDataset() {
073    Dataset dataset = datasetServiceReadOnly().get(datasetKey);
074    assertEquals(DATASET_TITLE, dataset.getTitle());
075  }
076
077  /**
078   * Update dataset, add an enpoint.
079   */
080  @Test
081  public void testAddEndpoint() throws URISyntaxException {
082    Endpoint endpoint = new Endpoint();
083    endpoint.setUrl(new URI("http://ipt.gbif.org/archive.do?r=diptera"));
084    endpoint.setType(EndpointType.DWC_ARCHIVE);
085    datasetService().addEndpoint(datasetKey, endpoint);
086  }
087
088  /**
089   * Update dataset: add a contact.
090   */
091  @Test
092  public void testAddContact() throws URISyntaxException {
093    Contact contact = new Contact();
094    contact.setFirstName("Jose");
095    contact.setLastName("Cuadra");
096    contact.setEmail(Lists.newArrayList("jcuadra@abc.org"));
097    contact.setType(ContactType.ADMINISTRATIVE_POINT_OF_CONTACT);
098    datasetService().addContact(datasetKey, contact);
099  }
100
101  /**
102   * Delete dataset, still exists but deleted flag has been set.
103   */
104  @Test
105  public void testDeleteDataset() {
106    datasetService().delete(datasetKey);
107    Dataset dataset = datasetServiceReadOnly().get(datasetKey);
108    assertNotNull(dataset.getDeleted());
109  }
110
111  /**
112   * Delete installation, still exists but deleted flag has been set.
113   */
114  @Test
115  public void testDeleteInstallation() {
116    installationService().delete(installationKey);
117    Installation installation = installationService().get(installationKey);
118    assertNotNull(installation.getDeleted());
119  }
120}
