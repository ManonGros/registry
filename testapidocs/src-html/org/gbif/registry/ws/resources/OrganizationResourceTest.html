001package org.gbif.registry.ws.resources;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertNotEquals;
006import static org.junit.Assert.assertTrue;
007
008public class OrganizationResourceTest {
009
010  /**
011   * Ensure randomly generated password has size greater than or equal to MINIMUM_PASSWORD_SIZE and less than
012   * MAXIMUM_PASSWORD_SIZE, and that subsequent passwords being generated are in fact unique.
013   */
014  @Test
015  public void testGeneratePassword() {
016    String password1 = OrganizationResource.generatePassword();
017    assertTrue(password1.length() >= OrganizationResource.MINIMUM_PASSWORD_SIZE);
018    assertTrue(password1.length() < OrganizationResource.MAXIMUM_PASSWORD_SIZE);
019
020    String password2 = OrganizationResource.generatePassword();
021    assertTrue(password2.length() >= OrganizationResource.MINIMUM_PASSWORD_SIZE);
022    assertTrue(password2.length() < OrganizationResource.MAXIMUM_PASSWORD_SIZE);
023
024    String password3 = OrganizationResource.generatePassword();
025    assertTrue(password3.length() >= OrganizationResource.MINIMUM_PASSWORD_SIZE);
026    assertTrue(password3.length() < OrganizationResource.MAXIMUM_PASSWORD_SIZE);
027
028    String password4 = OrganizationResource.generatePassword();
029    assertTrue(password4.length() >= OrganizationResource.MINIMUM_PASSWORD_SIZE);
030    assertTrue(password4.length() < OrganizationResource.MAXIMUM_PASSWORD_SIZE);
031
032    assertNotEquals(password1, password2);
033    assertNotEquals(password1, password3);
034    assertNotEquals(password1, password4);
035
036    assertNotEquals(password2, password3);
037    assertNotEquals(password2, password4);
038
039    assertNotEquals(password3, password4);
040  }
041}
