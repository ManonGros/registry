011import com.google.inject.AbstractModule;
012import com.google.inject.Module;
013import com.google.inject.matcher.Matchers;
014import org.aopalliance.intercept.MethodInterceptor;
015import org.aopalliance.intercept.MethodInvocation;
016import org.apache.bval.guice.Validate;
017import org.apache.commons.beanutils.DynaClass;
018import org.apache.commons.beanutils.WrapDynaBean;
019import org.slf4j.Logger;
020import org.slf4j.LoggerFactory;

022/**
023 * An interceptor that will set the createdBy and or modifiedBy fields on object being modified in methods having
024 * @Validate. The object must either extend NetworkEntity, or be a Contact, Endpoint, MachineTag, Tag, Identifier, or
025 * Comment.
026 * </br>
027 * This is only needed server side running the web service tests, since the tests call the interface service method,
028 * not the HTTP resource method where the createdBy and modifiedBy fields are actually set.
029 */
030public class TestValidateInterceptor implements MethodInterceptor {
031
032  private static final Logger LOG = LoggerFactory.getLogger(TestValidateInterceptor.class);
033
034  /**
035   * Sets up method level interception for those methods annotated with {@link Validate}.
036   */
037  public static Module newMethodInterceptingModule() {
038    return new AbstractModule() {
039
040      @Override
041      protected void configure() {
042        MethodInterceptor validateKyleMethodInterceptor = new TestValidateInterceptor();
043        this.binder().requestInjection(validateKyleMethodInterceptor);
044        this.bindInterceptor(Matchers.any(), Matchers.annotatedWith(Validate.class), validateKyleMethodInterceptor);
045      }
046    };
047  }
048
049  @Override
050  public Object invoke(MethodInvocation invocation) throws Throwable {
051    Validate validate = invocation.getMethod().getAnnotation(Validate.class); // ensure it is annotated
052    if (validate != null) {
053      for (int i = 0; i < invocation.getArguments().length; i++) {
054        Object arg = invocation.getArguments()[i];
055        if (arg instanceof NetworkEntity ||
056          arg instanceof Comment ||
057          arg instanceof Contact ||
058          arg instanceof Endpoint ||
059          arg instanceof Identifier ||
060          arg instanceof MachineTag ||
061          arg instanceof Tag) {
062          addRequiredFields(invocation.getArguments()[i]);
063        }
064      }
065    }
066    return invocation.proceed();
067  }
068
069  /**
070   * Add required fields createdBy and modifiedBy (where possible) to target object which either extends NetworkEntity
071   * or is a Contact, Endpoint, MachineTag, Tag, Identifier, or Comment.
072   *
073   * @param target object
074   */
075  private void addRequiredFields(Object target) {
076    WrapDynaBean wrapped = new WrapDynaBean(target);
077    DynaClass dynaClass = wrapped.getDynaClass();
078    // update createdBy field
079    if (dynaClass.getDynaProperty("createdBy") != null) {
080      wrapped.set("createdBy", "WS TEST");
081    }
082    // update modifiedBy field
083    if (dynaClass.getDynaProperty("modifiedBy") != null) {
084      wrapped.set("modifiedBy", "WS TEST");
085    }
086  }
087}
