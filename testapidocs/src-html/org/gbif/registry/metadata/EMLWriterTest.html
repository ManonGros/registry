001package org.gbif.registry.metadata;
002
003import org.gbif.api.model.common.DOI;
004import org.gbif.api.model.registry.Contact;
005import org.gbif.api.model.registry.Dataset;
006import org.gbif.registry.metadata.parse.DatasetParser;
007import org.gbif.utils.file.FileUtils;
008
009import java.io.StringWriter;
010import java.util.UUID;
011
012import org.junit.Test;
013
014import static org.junit.Assert.assertTrue;
015
016public class EMLWriterTest {
017
018  EMLWriter emlWriter = EMLWriter.newInstance();
019  EMLWriter emlWriterDOI = EMLWriter.newInstance(true);
020
021  @Test
022  public void testWrite() throws Exception {
023    Dataset d = DatasetParser.build(FileUtils.classpathStream("eml/sample.xml"));
024    d.setKey(UUID.randomUUID());
025    StringWriter writer = new StringWriter();
026    emlWriter.writeTo(d, writer);
027  }
028
029  @Test
030  public void testWriteNullContact() throws Exception {
031    Dataset d = DatasetParser.build(FileUtils.classpathStream("eml/sample.xml"));
032    d.setKey(UUID.randomUUID());
033    d.getContacts().clear();
034    StringWriter writer = new StringWriter();
035    emlWriter.writeTo(d, writer);
036  }
037
038  @Test
039  public void testNullAddress() throws Exception {
040    Dataset d = DatasetParser.build(FileUtils.classpathStream("eml/sample.xml"));
041    d.setKey(UUID.randomUUID());
042    Contact c = d.getContacts().get(0);
043    c.getAddress().add(null);
044    c.getAddress().add(null);
045    StringWriter writer = new StringWriter();
046    emlWriter.writeTo(d, writer);
047  }
048
049  @Test
050  public void testWriteDC() throws Exception {
051    Dataset d = DatasetParser.build(FileUtils.classpathStream("dc/worms_dc.xml"));
052    d.setKey(UUID.randomUUID());
053    StringWriter writer = new StringWriter();
054    emlWriter.writeTo(d, writer);
055  }
056
057  @Test
058  public void testWriteDoiAsPrimaryId() throws Exception {
059    Dataset d = DatasetParser.build(FileUtils.classpathStream("dc/worms_dc.xml"));
060    d.setKey(UUID.randomUUID());
061    d.setDoi(new DOI("10.1234/5679"));
062    StringWriter writer = new StringWriter();
063    emlWriterDOI.writeTo(d, writer);
064    assertTrue(writer.toString().contains("packageId=\"10.1234/5679\""));
065  }
066}
