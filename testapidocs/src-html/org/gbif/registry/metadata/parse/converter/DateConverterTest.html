001package org.gbif.registry.metadata.parse.converter;
002
003import java.util.Calendar;
004import java.util.Date;
005
006import org.junit.Test;
007
008import static org.junit.Assert.assertEquals;
009import static org.junit.Assert.assertNull;
010
011public class DateConverterTest {
012  private DateConverter converter = new DateConverter();
013
014  @Test
015  public void parseDates() throws Throwable {
016    assertLang(2000, 10, 20, "2000-10-20");
017    assertLang(2000, 10, 20, "20.10.2000");
018  }
019
020  private void assertLang(Integer year, Integer month, Integer day, String value) throws Throwable {
021    Date d = (Date) converter.convertToType(Date.class, value);
022
023    Calendar cal = Calendar.getInstance();
024    cal.setTime(d);
025
026    if (year != null) {
027      assertEquals(year, (Integer) cal.get(Calendar.YEAR));
028    }
029    if (month != null) {
030      assertEquals(month, (Integer) (1 + cal.get(Calendar.MONTH)));
031    }
032    if (day != null) {
033      assertEquals(day, (Integer) cal.get(Calendar.DAY_OF_MONTH));
034    }
035    if (year == null && month == null && day==null) {
036      assertNull(d);
037    }
038  }
039}
