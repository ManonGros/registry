025package org.gbif.registry;
026
027import org.gbif.api.model.registry.Network;
028import org.gbif.api.service.registry.NetworkService;
029import org.gbif.registry.guice.RegistryTestModules;
030import org.gbif.registry.utils.Networks;
031import org.gbif.registry.ws.resources.NetworkResource;
032import org.gbif.ws.client.filter.SimplePrincipalProvider;
033
034import javax.annotation.Nullable;
035
036import com.google.common.collect.ImmutableList;
037import org.junit.runner.RunWith;
038import org.junit.runners.Parameterized;
039import org.junit.runners.Parameterized.Parameters;
040
041/**
042 * This is parameterized to run the same test routines for the following:
043 * <ol>
044 * <li>The persistence layer</li>
045 * <li>The WS service layer</li>
046 * <li>The WS service client layer</li>
047 * </ol>
048 */
049@RunWith(Parameterized.class)
050public class NetworkIT extends NetworkEntityTest<Network> {
051
052  @Parameters
053  public static Iterable<Object[]> data() {
054    return ImmutableList.<Object[]>of(new Object[] {
055      RegistryTestModules.webservice().getInstance(NetworkResource.class),
056      null},
057      new Object[] {
058        RegistryTestModules.webserviceClient().getInstance(NetworkService.class),
059        RegistryTestModules.webserviceClient().getInstance(SimplePrincipalProvider.class)
060    });
061  }
062
063  public NetworkIT(NetworkService service, @Nullable SimplePrincipalProvider pp) {
064    super(service, pp);
065  }
066
067  @Override
068  protected Network newEntity() {
069    return Networks.newInstance();
070  }
071
072}
