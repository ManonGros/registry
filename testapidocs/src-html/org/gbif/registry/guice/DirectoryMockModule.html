001package org.gbif.registry.guice;
002
003import org.gbif.api.model.registry.Contact;
004import org.gbif.api.model.registry.Node;
005import org.gbif.api.vocabulary.ContactType;
006import org.gbif.registry.directory.Augmenter;
007
008import com.google.inject.AbstractModule;
009
010public class DirectoryMockModule extends AbstractModule {
011
012  @Override
013  protected void configure() {
014    bind(Augmenter.class).to(AugmenterMock.class);
015  }
016
017  private static class AugmenterMock implements Augmenter {
018
019    @Override
020    public Node augment(Node node) {
021      if(node != null) {
022        node.setParticipantSince(2001);
023        node.setAbbreviation("GBIF.ES");
024        node.setCity("Madrid");
025        node.setPostalCode("E-28014");
026        node.setOrganization("Real Jardín Botánico - CSIC");
027        for (int i = 1; i < 8; i++) {
028          Contact c = new Contact();
029          c.setKey(i);
030          c.setLastName("Name " + i);
031          c.setType(ContactType.NODE_STAFF);
032          node.getContacts().add(c);
033        }
034      }
035      return node;
036    }
037  }
038}
