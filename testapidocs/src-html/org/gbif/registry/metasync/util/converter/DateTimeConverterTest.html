001package org.gbif.registry.metasync.util.converter;
002
003import org.joda.time.DateTime;
004import org.joda.time.DateTimeZone;
005import org.junit.Test;
006
007import static org.junit.Assert.assertEquals;
008
009public class DateTimeConverterTest {
010
011  DateTimeConverter converter = new DateTimeConverter();
012
013  /**
014   * Test ISO 8691 formats.
015   *
016   * See https://en.wikipedia.org/wiki/ISO_8601 for an overview.
017   */
018  @Test
019  public void testIso8691Dates() {
020    assertEquals(new DateTime().withDate(2016, 1, 27).withTimeAtStartOfDay(), converter.convert(DateTime.class, "2016-01-27"));
021    assertEquals(new DateTime().withDate(2016, 1, 25).withTimeAtStartOfDay(), converter.convert(DateTime.class, "2016-W04"));
022    assertEquals(new DateTime().withDate(2016, 1, 27).withTimeAtStartOfDay(), converter.convert(DateTime.class, "2016-W04-3"));
023    assertEquals(new DateTime().withDate(2016, 1, 27).withTimeAtStartOfDay(), converter.convert(DateTime.class, "2016-027"));
024    assertEquals(new DateTime().withDate(2016, 1, 27).withTimeAtStartOfDay(), converter.convert(DateTime.class, "20160127"));
025
026    assertEquals(new DateTime().withZone(DateTimeZone.UTC).withDate(2016, 1, 27).withTime(06, 11, 22, 0).toDate(), ((DateTime) converter.convert(DateTime.class, "2016-01-27T06:11:22+00:00")).toDate());
027    assertEquals(new DateTime().withZone(DateTimeZone.UTC).withDate(2016, 1, 27).withTime(06, 11, 22, 0).toDate(), ((DateTime) converter.convert(DateTime.class, "2016-01-27T06:11:22Z")).toDate());
028    assertEquals(new DateTime().withZone(DateTimeZone.UTC).withDate(2016, 1, 27).withTime(06, 11, 22, 0).toDate(), ((DateTime) converter.convert(DateTime.class, "20160127T061122Z")).toDate());
029
030    assertEquals(new DateTime().withZone(DateTimeZone.UTC).withDate(2016, 1, 27).withTime(06, 11, 22, 0).toDate(), ((DateTime) converter.convert(DateTime.class, "2016-01-27T09:11:22+03:00")).toDate());
031    assertEquals(new DateTime().withZone(DateTimeZone.UTC).withDate(2016, 1, 27).withTime(06, 11, 22, 0).toDate(), ((DateTime) converter.convert(DateTime.class, "2016-01-27T03:11:22-03:00")).toDate());
032    assertEquals(new DateTime().withZone(DateTimeZone.UTC).withDate(2016, 1, 27).withTime(06, 11, 22, 0).toDate(), ((DateTime) converter.convert(DateTime.class, "2016-01-27T03:11:22âˆ’03:00")).toDate());
033
034    assertEquals(new DateTime().withZone(DateTimeZone.UTC).withDate(2016, 1, 27).withTime(06, 11, 22, 0).toDate(), ((DateTime) converter.convert(DateTime.class, "20160127T061122Z")).toDate());
035  }
036
037  /**
038   * Test some invalid formats, but which we are still provided with.
039   */
040  @Test
041  public void testInvalidDates() {
042    assertEquals(new DateTime().withDate(2016, 1, 27).withTimeAtStartOfDay(), converter.convert(DateTime.class, "2016/01/27"));
043    assertEquals(new DateTime().withZone(DateTimeZone.UTC).withDate(2016, 1, 27).withTime(06, 11, 22, 0).toDate(), ((DateTime) converter.convert(DateTime.class, "2016-01-27 06:11:22")).toDate());
044    assertEquals(new DateTime().withZone(DateTimeZone.UTC).withDate(2016, 1, 27).withTime(06, 11, 22, 0).toDate(), ((DateTime) converter.convert(DateTime.class, "2016-01-27 09:11:22+0300")).toDate());
045    assertEquals(new DateTime().withDate(2016, 1, 27).withTimeAtStartOfDay(), converter.convert(DateTime.class, "2016-01-27TCentral Sta:ndard Time"));
046  }
047
048  /**
049   * Test some partial formats.
050   */
051  @Test
052  public void testPartialDates() {
053    assertEquals(new DateTime().withDate(2016, 1, 1).withTimeAtStartOfDay(), converter.convert(DateTime.class, "2016-01"));
054    assertEquals(new DateTime().withDate(2016, 1, 1).withTimeAtStartOfDay(), converter.convert(DateTime.class, "2016"));
055
056    assertEquals(new DateTime().withDate(2016, 1, 1).withTimeAtStartOfDay(), converter.convert(DateTime.class, "2016/01"));
057
058    assertEquals(null, converter.convert(DateTime.class, ":"));
059    assertEquals(null, converter.convert(DateTime.class, ""));
060    assertEquals(null, converter.convert(DateTime.class, "ABCDEFGHIJKM"));
061    assertEquals(null, converter.convert(DateTime.class, "ABCD"));
062  }
063}
