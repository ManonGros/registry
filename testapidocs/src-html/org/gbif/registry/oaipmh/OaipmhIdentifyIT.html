001package org.gbif.registry.oaipmh;
002
003import org.gbif.api.service.registry.DatasetService;
004import org.gbif.api.service.registry.InstallationService;
005import org.gbif.api.service.registry.NodeService;
006import org.gbif.api.service.registry.OrganizationService;
007
008import org.dspace.xoai.model.oaipmh.DeletedRecord;
009import org.dspace.xoai.model.oaipmh.Identify;
010import org.hamcrest.Matchers;
011import org.junit.Test;
012import org.junit.runner.RunWith;
013import org.junit.runners.Parameterized;
014
015import static org.junit.Assert.assertEquals;
016import static org.junit.Assert.assertThat;
017
018/**
019 * Test the Identify verb of the OAI-PMH endpoint.
020 */
021@RunWith(Parameterized.class)
022public class OaipmhIdentifyIT extends AbstractOaipmhEndpointIT {
023
024  public OaipmhIdentifyIT(NodeService nodeService, OrganizationService organizationService, InstallationService installationService, DatasetService datasetService) {
025    super(nodeService, organizationService, installationService, datasetService);
026  }
027
028  /**
029   * Since the Identity verb response is all set by the OaipmhTestConfiguration (with test parameters)
030   * we only test that we actually get a response.
031   */
032  @Test
033  public void identify() {
034
035    Identify response = serviceProvider.identify();
036
037    assertThat("Identity verb send a RepositoryName", response.getRepositoryName(), Matchers.not(Matchers.isEmptyOrNullString()));
038    assertThat("Identity verb send a baseUrl", response.getBaseURL(), Matchers.not(Matchers.isEmptyOrNullString()));
039    assertEquals(DeletedRecord.PERSISTENT, response.getDeletedRecord());
040  }
041}
