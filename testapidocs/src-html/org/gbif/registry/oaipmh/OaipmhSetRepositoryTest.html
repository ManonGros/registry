001package org.gbif.registry.oaipmh;
002
003import org.gbif.api.model.registry.Installation;
004import org.gbif.api.vocabulary.Country;
005import org.gbif.api.vocabulary.DatasetType;
006import org.gbif.registry.persistence.mapper.DatasetMapper;
007import org.gbif.registry.persistence.mapper.MockDatasetMapper;
008
009import java.util.UUID;
010
011import org.dspace.xoai.dataprovider.handlers.results.ListSetsResult;
012import org.junit.Test;
013
014import static org.junit.Assert.assertEquals;
015import static org.junit.Assert.assertFalse;
016import static org.junit.Assert.assertTrue;
017
018/**
019 * Test class for OaipmhSetRepository
020 *
021 * @author cgendreau
022 */
023public class OaipmhSetRepositoryTest {
024
025  private static UUID installationUUIDRepitle = UUID.randomUUID();
026  private static UUID installationUUIDGulo = UUID.randomUUID();
027
028  /**
029   * Prepare the MockDatasetMapper with controlled mock data.
030   *
031   * @return
032   */
033  private static DatasetMapper prepareDatasetMapperMock(){
034    MockDatasetMapper datasetMapper = new MockDatasetMapper();
035    datasetMapper.mockDatasetForCountry(Country.DENMARK);
036    datasetMapper.mockDatasetForCountry(Country.SWEDEN);
037    datasetMapper.mockDatasetForCountry(Country.NORWAY);
038
039    Installation installation = new Installation();
040    installation.setKey(installationUUIDRepitle);
041    installation.setTitle("My reptile IPT");
042    datasetMapper.mockDatasetForInstallation(installation);
043
044    installation = new Installation();
045    installation.setKey(installationUUIDGulo);
046    installation.setTitle("My gulo observations IPT");
047    datasetMapper.mockDatasetForInstallation(installation);
048
049    return datasetMapper;
050  }
051
052  @Test
053  public void testParseSetName(){
054    assertTrue(OaipmhSetRepository.parseSetName("country:DK").isPresent());
055    assertFalse(OaipmhSetRepository.parseSetName("null:Country").isPresent());
056  }
057
058  @Test
059  public void testRetrieveSets(){
060    DatasetMapper mockDatasetMapper = prepareDatasetMapperMock();
061    OaipmhSetRepository setRepository = new OaipmhSetRepository(mockDatasetMapper);
062
063    // the +1 to each list size covers the top level set
064    int total = (DatasetType.values().length + 1) + (mockDatasetMapper.listDistinctCountries(null).size() + 1) + (mockDatasetMapper.listDistinctInstallations(null).size() + 1);
065
066    //test retrieving all Sets one by one
067    for(int i=0; i < total; i++){
068      ListSetsResult result = setRepository.retrieveSets(i, 1);
069      assertEquals(1, result.getResults().size());
070      assertEquals(i+1 < total, result.hasMore());
071    }
072  }
073
074  @Test
075  public void testSetExists(){
076    DatasetMapper mockDatasetMapper = prepareDatasetMapperMock();
077    OaipmhSetRepository setRepository = new OaipmhSetRepository(mockDatasetMapper);
078    assertTrue("Should find Set provided in test data:" + installationUUIDRepitle.toString(),
079            setRepository.exists(OaipmhSetRepository.SetType.INSTALLATION.getSubsetPrefix() + installationUUIDRepitle.toString()));
080    assertFalse("Should not find a Set for random Installation key",
081            setRepository.exists(OaipmhSetRepository.SetType.INSTALLATION.getSubsetPrefix() + UUID.randomUUID().toString()));
082
083    assertTrue("Should find Set provided in test data:" + Country.DENMARK.getIso2LetterCode(),
084            setRepository.exists(OaipmhSetRepository.SetType.COUNTRY.getSubsetPrefix() + Country.DENMARK.getIso2LetterCode()));
085    assertFalse("Should not find Set not provided in test data:" + Country.AUSTRALIA.getIso2LetterCode(),
086            setRepository.exists(OaipmhSetRepository.SetType.COUNTRY.getSubsetPrefix() + Country.AUSTRALIA.getIso2LetterCode()));
087  }
088
089}
