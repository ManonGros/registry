024/**
025 * This class does the identity function update for entities in the registry. That is to say that it reads and updates
026 * all entities.
027 * Because the database was migrated, and we have Java validation rules, this will ensure that all migrated data can be
028 * used in updates.
029 */
030public class IdentityUpdateValidationCheck {
031
032  private static final Logger LOG = LoggerFactory.getLogger(IdentityUpdateValidationCheck.class);
033
034  /**
035   * Runs the validation. To run, add the necessary properties as arguments. The key and password are
036   * authenticated by the web services, so check they match a key/password combination the web services' know of.
037   *
038   * @param args Base url
039   */
040  public static void main(String[] args) {
041    Properties p = new Properties();
042    p.put("registry.ws.url", args[0]);
043    p.setProperty("application.key", args[1]);
044    p.setProperty("application.secret", args[2]);
045    // Create authentication module, and set principal name, equal to a GBIF User unique account name
046    GbifApplicationAuthModule auth = new GbifApplicationAuthModule(p);
047    auth.setPrincipal("admin");
048    Injector injector = Guice.createInjector(new RegistryWsClientModule(p), auth);
049
050    LOG.info("Starting Node tests");
051    int nodeErrorCount = verifyEntity(injector.getInstance(NodeService.class));
052    LOG.info("Node tests produced {} errors", nodeErrorCount);
053
054    LOG.info("Starting Organization tests");
055    int organisationErrorCount = verifyEntity(injector.getInstance(OrganizationService.class));
056    LOG.info("Organization tests produced {} errors", organisationErrorCount);
057
058    LOG.info("Starting Installation tests");
059    int installationErrorCount = verifyEntity(injector.getInstance(InstallationService.class));
060    LOG.info("Installation tests produced {} errors", installationErrorCount);
061
062    LOG.info("Starting Dataset tests");
063    int datasetErrorCount = verifyEntity(injector.getInstance(DatasetService.class));
064    LOG.info("Dataset tests produced {} errors", datasetErrorCount);
065
066    LOG.info("Starting Network tests");
067    int networkErrorCount = verifyEntity(injector.getInstance(NetworkService.class));
068    LOG.info("Network tests produced {} errors", networkErrorCount);
069
070    // summarise again jsut for ease of reading
071    LOG.info("Test results:");
072    LOG.info("Node tests produced {} errors", nodeErrorCount);
073    LOG.info("Organization tests produced {} errors", organisationErrorCount);
074    LOG.info("Installation tests produced {} errors", installationErrorCount);
075    LOG.info("Dataset tests produced {} errors", datasetErrorCount);
076    LOG.info("Network tests produced {} errors", networkErrorCount);
077
078  }
079
080  private static <T extends NetworkEntity> int verifyEntity(NetworkEntityService<T> service) {
081    int errorCount = 0;
082    PagingRequest page = new PagingRequest(0, 100);
083    PagingResponse<T> response = null;
084    do {
085      response = service.list(page);
086      LOG.debug("Page with offset[{}] return[{}] records", page.getOffset(), response.getCount());
087      for (T e : response.getResults()) {
088        LOG.debug("Updating entity[{}]", e.getKey());
089        try {
090          service.update(e);
091
092          if (e instanceof Contactable) {
093            Contactable contactable = (Contactable) e;
094            for (Contact c : contactable.getContacts()) {
095              service.updateContact(e.getKey(), c);
096            }
097          }
098
099        } catch (Exception ex) {
100          LOG.error("Unable to update entity[{}]", e.getKey(), ex);
101          errorCount++;
102        }
103      }
104      page.nextPage();
105    } while (!response.isEndOfRecords());
106    return errorCount;
107  }
108}
