001package org.gbif.registry.doi;
002
003import org.gbif.api.model.common.DOI;
004import org.gbif.registry.database.DatabaseInitializer;
005import org.gbif.registry.guice.RegistryTestModules;
006
007import java.util.Set;
008
009import com.beust.jcommander.internal.Sets;
010import com.google.inject.Injector;
011import org.junit.Before;
012import org.junit.Rule;
013import org.junit.Test;
014
015import static org.junit.Assert.assertEquals;
016import static org.junit.Assert.assertTrue;
017
018public class DoiGeneratorMQIT {
019  private DoiGenerator generator;
020
021  @Rule
022  public final DatabaseInitializer databaseRule = new DatabaseInitializer(RegistryTestModules.database());
023
024  @Before
025  public void setup() {
026    Injector inj = RegistryTestModules.webservice();
027    generator = inj.getInstance(DoiGenerator.class);
028  }
029
030  @Test
031  public void testNewDOI() {
032    Set<DOI> dois = Sets.newHashSet();
033    for (int x = 1; x < 20; x++) {
034      DOI doi = generator.newDatasetDOI();
035      assertTrue(generator.isGbif(doi));
036      dois.add(doi);
037      assertEquals(DOI.TEST_PREFIX, doi.getPrefix());
038      assertEquals(x, dois.size());
039    }
040  }
041
042}
