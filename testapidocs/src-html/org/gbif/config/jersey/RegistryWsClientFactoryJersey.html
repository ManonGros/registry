001package org.gbif.config.jersey;
002
003import org.gbif.api.service.registry.DatasetService;
004import org.gbif.api.service.registry.InstallationService;
005import org.gbif.registry.ws.client.DatasetWsClient;
006import org.gbif.registry.ws.client.InstallationWsClient;
007import org.gbif.utils.HttpUtil;
008import org.gbif.ws.json.JacksonJsonContextResolver;
009
010import com.sun.jersey.api.client.WebResource;
011import com.sun.jersey.api.client.config.ClientConfig;
012import com.sun.jersey.api.client.config.DefaultClientConfig;
013import com.sun.jersey.api.client.filter.HTTPBasicAuthFilter;
014import com.sun.jersey.api.json.JSONConfiguration;
015import com.sun.jersey.client.apache4.ApacheHttpClient4;
016import com.sun.jersey.client.apache4.ApacheHttpClient4Handler;
017import org.apache.http.client.HttpClient;
018import org.codehaus.jackson.jaxrs.JacksonJsonProvider;
019
020public class RegistryWsClientFactoryJersey {
021
022  private static final int DEFAULT_HTTP_TIMEOUT_MSECS = 10000;
023  private static final int DEFAULT_MAX_HTTP_CONNECTIONS = 100;
024  private static final int DEFAULT_MAX_HTTP_CONNECTIONS_PER_ROUTE = 100;
025
026  // GBIF account, with correct role and permissions set
027  private static final String USERNAME = "ws_client_demo";
028  private static final String PASSWORD = "Demo123";
029  private static final HTTPBasicAuthFilter AUTH_FILTER = new HTTPBasicAuthFilter(USERNAME, PASSWORD);
030  private static final String REGISTRY_API_BASE_URL = "http://api.gbif-uat.org/v1/";
031
032  private static WebResource resource;
033  private static DatasetService datasetService;
034  private static DatasetService datasetServiceReadOnly;
035  private static InstallationService installationService;
036
037
038  /**
039   * @return Apache HTTP Client
040   */
041  private static HttpClient buildHttpClient() {
042    return HttpUtil.newMultithreadedClient(DEFAULT_HTTP_TIMEOUT_MSECS, DEFAULT_MAX_HTTP_CONNECTIONS,
043      DEFAULT_MAX_HTTP_CONNECTIONS_PER_ROUTE);
044  }
045
046  /**
047   * @return Jersey resource that utilizes the Apache HTTP Client to send and receive HTTP request and responses to the
048   * GBIF API.
049   */
050  public static synchronized WebResource buildJerseyClient() {
051    if (resource == null) {
052        ApacheHttpClient4Handler hch = new ApacheHttpClient4Handler(buildHttpClient(), null, false);
053        ClientConfig clientConfig = new DefaultClientConfig();
054        clientConfig.getClasses().add(JacksonJsonContextResolver.class);
055        // this line is critical! Note that this is the jersey version of this class name!
056        clientConfig.getClasses().add(JacksonJsonProvider.class);
057        clientConfig.getFeatures().put(JSONConfiguration.FEATURE_POJO_MAPPING, Boolean.TRUE);
058        resource = new ApacheHttpClient4(hch, clientConfig).resource(REGISTRY_API_BASE_URL);
059    }
060    return resource;
061  }
062
063  /**
064   * @return read-only DatasetService
065   */
066  public static synchronized DatasetService datasetServiceReadOnly() {
067    if (datasetServiceReadOnly == null) {
068      datasetServiceReadOnly = new DatasetWsClient(buildJerseyClient(), null);
069    }
070    return datasetServiceReadOnly;
071  }
072
073  /**
074   * @return DatasetService with authentication
075   */
076  public static synchronized DatasetService datasetService() {
077    if (datasetService == null) {
078      datasetService = new DatasetWsClient(buildJerseyClient(), AUTH_FILTER);
079    }
080    return datasetService;
081  }
082
083  /**
084   * @return InstallationService with authentication
085   */
086  public static synchronized InstallationService installationService() {
087    if (installationService == null) {
088      installationService = new InstallationWsClient(buildJerseyClient(), AUTH_FILTER);
089    }
090    return installationService;
091  }
092
093}
