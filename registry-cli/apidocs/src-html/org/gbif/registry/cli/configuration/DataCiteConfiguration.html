001package org.gbif.registry.cli.configuration;
002
003import org.gbif.doi.service.ServiceConfig;
004import org.gbif.doi.service.datacite.DataCiteService;
005import org.gbif.utils.HttpUtil;
006
007import java.net.URI;
008import javax.validation.constraints.NotNull;
009
010import com.beust.jcommander.Parameter;
011import com.beust.jcommander.internal.Nullable;
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014
015/**
016 * A configuration for the DataCite service.
017 */
018@SuppressWarnings("PublicField")
019public class DataCiteConfiguration {
020
021  private static final Logger LOG = LoggerFactory.getLogger(DataCiteConfiguration.class);
022
023  @Parameter(names = "--datacite-username")
024  @NotNull
025  public String username;
026
027  @Parameter(names = "--datacite-password")
028  @NotNull
029  public String password;
030
031  @Parameter(names = "--datacite-api")
032  @Nullable
033  public URI api;
034
035  @Parameter(names = "--datacite-threads")
036  public int threads = 10;
037
038  @Parameter(names = "--datacite-timeout")
039  public int timeout = 20000;
040
041  public DataCiteService createService() {
042    LOG.debug("Creating DataCite doi service");
043    ServiceConfig cfg = new ServiceConfig(username, password);
044    cfg.setApi(api);
045    return new DataCiteService(HttpUtil.newMultithreadedClient(timeout, threads, threads), cfg);
046  }
047
048}
