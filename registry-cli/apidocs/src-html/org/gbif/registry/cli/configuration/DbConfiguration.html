001package org.gbif.registry.cli.configuration;
002
003import org.gbif.registry.persistence.guice.RegistryMyBatisModule;
004
005import java.lang.reflect.Field;
006import java.lang.reflect.Modifier;
007import java.util.Properties;
008import java.util.Set;
009import javax.validation.constraints.NotNull;
010
011import com.beust.jcommander.Parameter;
012import com.google.common.collect.Sets;
013import org.slf4j.Logger;
014import org.slf4j.LoggerFactory;
015
016/**
017 * A configuration for the registry database connection pool
018 * as used by the mybatis layer. Knows how to insert a service guice module.
019 */
020@SuppressWarnings("PublicField")
021public class DbConfiguration {
022
023  private static final Logger LOG = LoggerFactory.getLogger(DbConfiguration.class);
024  private static final String PROPERTY_PREFIX = "registry.db.";
025  private static final Set<String> DATASOURCE_SET = Sets.newHashSet("serverName", "databaseName", "user", "password");
026
027  @NotNull
028  @Parameter(names = "--db-host")
029  public String serverName;
030
031  @NotNull
032  @Parameter(names = "--db-db")
033  public String databaseName;
034
035  @NotNull
036  @Parameter(names = "--db-user")
037  public String user;
038
039  @NotNull
040  @Parameter(names = "--db-password", password = true)
041  public String password;
042
043  @Parameter(names = "--db-maximumPoolSize")
044  public int maximumPoolSize = 3;
045
046  @Parameter(names = "--db-connectionTimeout")
047  public int connectionTimeout = 3000;
048
049  public RegistryMyBatisModule createMyBatisModule() {
050    Properties props = new Properties();
051    props.put(PROPERTY_PREFIX + "dataSourceClassName", "org.postgresql.ds.PGSimpleDataSource");
052    for (Field field : DbConfiguration.class.getDeclaredFields()) {
053      if (!field.isSynthetic() && Modifier.isPublic(field.getModifiers())) {
054        try {
055          if (DATASOURCE_SET.contains(field.getName())) {
056            props.put(PROPERTY_PREFIX + "dataSource." + field.getName(), String.valueOf(field.get(this)));
057          } else {
058            props.put(PROPERTY_PREFIX + field.getName(), String.valueOf(field.get(this)));
059          }
060        } catch (IllegalAccessException e) {
061          // cant happen, we check for public access
062          throw new RuntimeException(e);
063        }
064      }
065    }
066    LOG.info("Connecting to registry db {} on {} with user {}", databaseName, serverName, user);
067    return new RegistryMyBatisModule(props);
068  }
069
070}
