001package org.gbif.registry.cli;
002
003import org.gbif.common.messaging.MessageListener;
004import org.gbif.registry.cli.configuration.DoiUpdaterConfiguration;
005import org.gbif.registry.persistence.mapper.DoiMapper;
006
007import com.google.common.util.concurrent.AbstractIdleService;
008import com.google.inject.Guice;
009import com.google.inject.Injector;
010
011/**
012 * A CLI service that starts and stops a listener of DoiUpdate messages. Must always be only one thread - multiple
013 * will introduce a possible race (e.g. delete before create).
014 */
015public class DoiUpdaterService extends AbstractIdleService {
016
017  private final DoiUpdaterConfiguration config;
018
019  private MessageListener listener;
020
021  public DoiUpdaterService(DoiUpdaterConfiguration config) {
022    this.config = config;
023  }
024
025  @Override
026  protected void startUp() throws Exception {
027    config.ganglia.start();
028
029    Injector inj = Guice.createInjector(config.db.createMyBatisModule());
030    listener = new MessageListener(config.messaging.getConnectionParameters());
031    listener.listen(config.queueName, 1,
032      new DoiUpdateListener(config.datacite.createService(), inj.getInstance(DoiMapper.class), config.timeToRetryInMs));
033  }
034
035  @Override
036  protected void shutDown() throws Exception {
037    if (listener != null) {
038      listener.close();
039    }
040  }
041}
