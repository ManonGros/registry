58    /**
59     * Gets the participantID from the Node.
60     */
61    private static Integer findParticipantID(Node node) {
62      for (Identifier id : node.getIdentifiers()) {
63        if (IdentifierType.GBIF_PARTICIPANT == id.getType()) {
64          try {
65            return Integer.parseInt(id.getIdentifier());
66          } catch (NumberFormatException e) {
67            LOG.error("Directory participantId is no integer: %s", id.getIdentifier());
68          }
69        }
70      }
71      return null;
72    }
73  
74    @Override
75    public Node augment(Node registryNode) {
76      if (registryNode != null) {
77        try {
78          Integer participantID = findParticipantID(registryNode);
79          if (participantID != null) {
80            Participant participant = participantService.get(participantID);
81            if (participant != null) {
82              List<Contact> contacts = Lists.newArrayList();
83              // update node with Directory info if it exists
84              List<org.gbif.api.model.directory.Node> participantNodes = getParticipantNodes(participant);
85              registryNode.setParticipantTitle(participant.getName());
86              contacts.addAll(getContactsForParticipant(participant));
87  
88              registryNode.setAbbreviation(participant.getAbbreviatedName());
89              registryNode.setDescription(participant.getComments());
90              if(participant.getMembershipStart() != null) {
91                registryNode.setParticipantSince(getParticipantSinceYear(participant.getMembershipStart()));
92              }
93              if(!participantNodes.isEmpty()){
94                contacts.addAll(getContactsForNode(participantNodes));
95                registryNode.setAddress(getNodesAddresses(participantNodes));
96                registryNode.setHomepage(getWebUrls(participant,participantNodes));
97                registryNode.setEmail(getEmails(participantNodes));
98                registryNode.setPhone(getPhones(participantNodes));
99              }
100             else{
101               LOG.info("Empty node for participantId %d", participantID);
102             }
103             registryNode.setContacts(contacts);
104           }
105         }
106       } catch (Exception e) {
107         LOG.error("Failed to augment node %s with Directory information", registryNode.getKey(), e);
108       }
109     }
110     return registryNode;
111   }

113   /**
114    * Gets the year part of the membershipStart field.
115    */
116   private static Integer getParticipantSinceYear(String membershipStart) {
117     String[] membershipStartComponents = membershipStart.split(" ");
118     if(membershipStartComponents.length > 1) {
119       return Ints.tryParse(membershipStartComponents[1].trim());
120     }
121     return null;
122   }

124   /**
125    * Gets all the nodes associated to a participant.
126    */
127   private List<org.gbif.api.model.directory.Node> getParticipantNodes(Participant participant){
128     List<org.gbif.api.model.directory.Node> nodes = Lists.newArrayList();
129     if(participant.getNodes() != null){
130       for(org.gbif.api.model.directory.Node node : participant.getNodes()) {
131         nodes.add(nodeService.get(node.getId()));
132       }
133     }
134     return nodes;
135   }

149    * Gets all the web pages/urls of participant and its nodes.

152     List<URI> addresses = Lists.newArrayList();
153     if(participant.getParticipantUrl() != null) {
154       addresses.add(URI.create(participant.getParticipantUrl()));
155     }
156     for(org.gbif.api.model.directory.Node node : participantNodes) {
157       if(node.getNodeUrl() != null) {
158         addresses.add(URI.create(node.getNodeUrl()));
159       }
160     }
161     return  addresses;
162   }

190   /**
191    * Transforms the persons associated to a participant into a list of contacts.
192    */
193   private List<Contact> getContactsForParticipant(Participant participant){
194     List<Contact> contacts = Lists.newArrayList();
195     if(participant.getPeople() != null){
196       Person person;
197       Contact contact;
198       ContactType contactType;
199       for(ParticipantPerson participantPerson : participant.getPeople()) {
200         person = personService.get(participantPerson.getPersonId());
201         contactType = null;
202         if( participantPerson.getRole() != null) {
203           contactType = PARTICIPANT_ROLE_TO_CONTACT_TYPE.get(participantPerson.getRole());
204         }
205         contact = personToContact(person, participant.getName(), contactType);
206         contacts.add(contact);
207       }
208     }
209     return contacts;
210   }

212   /**
213    * Transforms the persons associated to a node(s) into a list of contacts.
214    * @param directoryNodes it theory it should never be more than one
215    */
216   private List<Contact> getContactsForNode(List<org.gbif.api.model.directory.Node> directoryNodes){
217     List<Contact> contacts = Lists.newArrayList();
218     if(directoryNodes != null){
219       Person person;
220       Contact contact;
221       ContactType contactType;
222       for(org.gbif.api.model.directory.Node currentNode : directoryNodes) {
223         if(currentNode.getPeople() != null && !currentNode.getPeople().isEmpty()) {
224           for (NodePerson nodePerson : currentNode.getPeople()) {
225             person = personService.get(nodePerson.getPersonId());
226             contactType = null;
227             if (nodePerson.getRole() != null) {
228               contactType = NODE_ROLE_TO_CONTACT_TYPE.get(nodePerson.getRole());
229             }
230             contact = personToContact(person, currentNode.getName(), contactType);
231             contacts.add(contact);
232           }
233         }
234         else{
235           LOG.info("No people found for nodeId %d", currentNode.getId());
236         }
237       }
238     }
239     return contacts;
240   }

242   /**
243    * Transform a Directory Person into a Registry Contact.
244    * @param person
245    * @param organization
246    * @param contactType
247    * @return
248    */
249   private Contact personToContact(Person person, String organization, ContactType contactType){
250     Contact contact = new Contact();
251     contact.setKey(person.getId());
252     contact.addAddress(person.getAddress());
253     contact.setCountry(person.getCountryCode());
254     if(person.getEmail() != null) {
255       contact.addEmail(person.getEmail());
256     }
257     contact.setFirstName(person.getFirstName());
258     contact.setLastName(person.getSurname());
259     if(person.getPhone() != null) {
260       contact.addPhone(person.getPhone());
261     }
262     contact.setModified(person.getModified());
263     contact.setOrganization(organization);
264     if( contactType != null) {
265       contact.setType(contactType);
266     }
267     return contact;
268   }
