11  /**
12   * Class used to generate responses for legacy (GBRDS/IPT) API.
13   * </br>
14   * JAXB annotations allow the class to be converted into an XML document or JSON response. @XmlElement is used to
15   * specify element names that consumers of legacy services expect to find.
16   */
17  @XmlRootElement(name = "service")
18  public class LegacyEndpointResponse {
19  
20    private String key;
21    private String organisationKey;
22    private String resourceKey;
23    private String description;
24    private String descriptionLanguage;
25    private String type;
26    private String typeDescription;
27    private String accessPointURL;
28  
29    public LegacyEndpointResponse(Endpoint endpoint, UUID datasetKey) {
30      key = endpoint.getKey() == null ? "" : endpoint.getKey().toString();
31      type = endpoint.getType() == null ? "" : endpoint.getType().name();
32      resourceKey = datasetKey == null ? "" : datasetKey.toString();
33      description = endpoint.getDescription() == null ? "" : endpoint.getDescription();
34      accessPointURL = endpoint.getUrl() == null ? "" : endpoint.getUrl().toASCIIString();
35      // always empty - not null because they have to be included in response
36      organisationKey = "";
37      descriptionLanguage = "";
38      typeDescription = "";
39    }
40  
41    /**
42     * No argument, default constructor needed by JAXB.
43     */
44    public LegacyEndpointResponse() {
45    }
46
