139   public int addMachineTag(UUID targetEntityKey, MachineTag machineTag) {
140     // allow post through varnish (no chunked encoding needed)
141     int tagId;
142     try {
143       tagId = getResource()
144         .path(targetEntityKey.toString())
145         .path("machineTag")
146         .type(MediaType.APPLICATION_JSON)
147         .post(Integer.class, mapper.writeValueAsBytes(machineTag));
148 
149     } catch (IOException e) {
150       throw new IllegalStateException(e);
151     }
152 //    int tagId = post(Integer.class, machineTag, targetEntityKey.toString(), "machineTag");
153     return tagId;
154   }
