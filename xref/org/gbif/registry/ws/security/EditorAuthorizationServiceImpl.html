1   package org.gbif.registry.ws.security;

3   import org.gbif.api.model.registry.Dataset;
4   import org.gbif.api.model.registry.Installation;
5   import org.gbif.api.model.registry.Organization;
6   import org.gbif.api.service.registry.DatasetService;
7   import org.gbif.api.service.registry.InstallationService;
8   import org.gbif.api.service.registry.OrganizationService;
9   import org.gbif.registry.persistence.mapper.UserRightsMapper;
10  
11  import java.security.Principal;
12  import java.util.UUID;
13  
14  import com.google.inject.Inject;
15  
16  public class EditorAuthorizationServiceImpl implements EditorAuthorizationService {
17  
18    private UserRightsMapper userRightsMapper;
19    private DatasetService datasetService;
20    private InstallationService installationService;
21    private OrganizationService organizationService;
22  
23    @Inject
24    public EditorAuthorizationServiceImpl(DatasetService datasetService, InstallationService installationService,
25      OrganizationService organizationService, UserRightsMapper userRightsMapper) {
26      this.datasetService = datasetService;
27      this.installationService = installationService;
28      this.organizationService = organizationService;
29      this.userRightsMapper = userRightsMapper;
30    }
31  
32      @Override
33      public boolean allowedToModifyNamespace(Principal user, String ns) {
34          if (user == null) {
35              return false;
36          }
37          return userRightsMapper.namespaceExistsForUser(user.getName(), ns);
38      }
39  
40      @Override
41      public boolean allowedToDeleteMachineTag(Principal user, int machineTagKey) {
42          if (user == null) {
43              return false;
44          }
45          return userRightsMapper.allowedToDeleteMachineTag(user.getName(), machineTagKey);
46      }
47  
48      @Override
49    public boolean allowedToModifyEntity(Principal user, UUID key) {
50      if (user == null) {
51        return false;
52      }
53      return userRightsMapper.keyExistsForUser(user.getName(), key);
54    }
55  
56    @Override
57    public boolean allowedToModifyDataset(Principal user, UUID datasetKey) {
58      if (user == null) {
59        return false;
60      }
61      if (allowedToModifyEntity(user, datasetKey)) {
62        return true;
63      }
64      // try higher organization or node rights
65      Dataset d = datasetService.get(datasetKey);
66      return d == null ? false : allowedToModifyOrganization(user, d.getPublishingOrganizationKey());
67    }
68  
69    @Override
70    public boolean allowedToModifyOrganization(Principal user, UUID orgKey) {
71      if (user == null) {
72        return false;
73      }
74      if (allowedToModifyEntity(user, orgKey)) {
75        return true;
76      }
77      // try endorsing node
78      Organization o = organizationService.get(orgKey);
79      return o == null ? false : allowedToModifyEntity(user, o.getEndorsingNodeKey());
80    }
81  
82    @Override
83    public boolean allowedToModifyInstallation(Principal user, UUID installationKey) {
84      if (user == null) {
85        return false;
86      }
87      if (allowedToModifyEntity(user, installationKey)) {
88        return true;
89      }
90      // try higher organization or node rights
91      Installation inst = installationService.get(installationKey);
92      return inst == null ? false : allowedToModifyOrganization(user, inst.getOrganizationKey());
93    }
94  }
