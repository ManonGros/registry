60  @Singleton
61  @Path("node")
62  public class NodeResource extends BaseNetworkEntityResource<Node> implements NodeService {
63  
64    private final NodeMapper nodeMapper;
65    private final OrganizationMapper organizationMapper;
66    private final InstallationMapper installationMapper;
67    private final DatasetMapper datasetMapper;
68    private final Augmenter nodeAugmenter;
69  
70    @Inject
71    public NodeResource(
72      NodeMapper nodeMapper,
73      IdentifierMapper identifierMapper,
74      CommentMapper commentMapper,
75      ContactMapper contactMapper,
76      EndpointMapper endpointMapper,
77      MachineTagMapper machineTagMapper,
78      TagMapper tagMapper,
79      OrganizationMapper organizationMapper,
80      DatasetMapper datasetMapper,
81      InstallationMapper installationMapper,
82      EventBus eventBus,
83      Augmenter nodeAugmenter,
84      EditorAuthorizationService userAuthService) {
85      super(nodeMapper, commentMapper, contactMapper, endpointMapper, identifierMapper, machineTagMapper, tagMapper,
86        Node.class, eventBus, userAuthService);
87      this.nodeMapper = nodeMapper;
88      this.organizationMapper = organizationMapper;
89      this.nodeAugmenter = nodeAugmenter;
90      this.datasetMapper = datasetMapper;
91      this.installationMapper = installationMapper;
92    }

94    @GET
95    @Path("{key}")
96    @Nullable
97    @NullToNotFound
98    @Override
99    public Node get(@PathParam("key") UUID key) {
100     return nodeAugmenter.augment(super.get(key));
101   }

104   /**
105    * All network entities support simple (!) search with "&q=".
106    * This is to support the console user interface, and is in addition to any complex, faceted search that might
107    * additionally be supported, such as dataset search.
108    */
109   @GET
110   public PagingResponse<Node> list(@Nullable @QueryParam("q") String query,
111     @Nullable @QueryParam("identifierType") IdentifierType identifierType,
112     @Nullable @QueryParam("identifier") String identifier,
113     @Nullable @Context Pageable page) {
114     // This is getting messy: http://dev.gbif.org/issues/browse/REG-426
115     if (identifierType != null && identifier != null) {
116       return listByIdentifier(identifierType, identifier, page);
117     } else if (identifier != null) {
118       return listByIdentifier(identifier, page);
119     } else if (Strings.isNullOrEmpty(query)) {
120       return list(page);
121     } else {
122       return search(query, page);
123     }
124   }
125 
126 
127   @Override
128   public PagingResponse<Node> list(@Nullable Pageable page) {
129     PagingResponse<Node> resp = super.list(page);
130     for (Node n : resp.getResults()) {
131       nodeAugmenter.augment(n);
132     }
133     return resp;
134   }
