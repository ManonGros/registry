15  import static org.gbif.registry.ws.security.UserRoles.ADMIN_ROLE;
16  
17  /**
18   * Class that exposes common functions used by services that expose data about occurrence downloads.
19   */
20  public class DownloadSecurityUtils {
21  
22    private static final Logger LOG = LoggerFactory.getLogger(DownloadSecurityUtils.class);
23  
24    private DownloadSecurityUtils() {
25      // private constructor
26    }
27  
28    /**
29     * Checks if the user has the ADMIN_ROLE or is the same user in the current context.
30     */
31    public static void checkUserIsInSecurityContext(String user, SecurityContext securityContext) {
32      // A null securityContext means that the class is executed locally
33      if (!isUserAuthorizedInContext(securityContext, user)) {
34        LOG.warn(String.format("Unauthorized access detected, authenticated use, requested user %s", user));
35        throw new WebApplicationException(Response.Status.UNAUTHORIZED);
36      }
37    }
38  
39    /**
40     * Checks if the user has the ADMIN_ROLE or is the same user in the current context.
41     */
42    public static boolean isUserAuthorizedInContext(SecurityContext securityContext, String user) {
43      return (securityContext == null || securityContext.isUserInRole(ADMIN_ROLE) || (securityContext.getUserPrincipal() != null
44      && securityContext.getUserPrincipal().getName().equals(user)));
45    }
46  
47    /**
48     * Remove data that shouldn't be publicly exposed.
49     */
50    public static void clearSensitiveData(SecurityContext securityContext, Download download) {
51      if (download.getRequest() != null
52        && !isUserAuthorizedInContext(securityContext, download.getRequest().getCreator())) {
53        download.getRequest().setNotificationAddresses(null);
54      }
55    }
56  
57    /**
58     * Remove data that shouldn't be publicly exposed.
59     */
60    public static void clearSensitiveData(SecurityContext securityContext,
61      Collection<DatasetOccurrenceDownloadUsage> downloadUsages) {
62      for (DatasetOccurrenceDownloadUsage downloadUsage : downloadUsages) {
63        clearSensitiveData(securityContext, downloadUsage.getDownload());
64      }
65    }
66  }
