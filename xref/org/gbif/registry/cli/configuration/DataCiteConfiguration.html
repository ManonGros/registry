3   import org.gbif.doi.service.ServiceConfig;
4   import org.gbif.doi.service.datacite.DataCiteService;
5   import org.gbif.utils.HttpUtil;
6   
7   import java.net.URI;
8   import javax.validation.constraints.NotNull;
9   
10  import com.beust.jcommander.Parameter;
11  import com.beust.jcommander.internal.Nullable;
12  import org.slf4j.Logger;
13  import org.slf4j.LoggerFactory;
14  
15  /**
16   * A configuration for the DataCite service.
17   */
18  @SuppressWarnings("PublicField")
19  public class DataCiteConfiguration {
20  
21    private static final Logger LOG = LoggerFactory.getLogger(DataCiteConfiguration.class);
22  
23    @Parameter(names = "--datacite-username")
24    @NotNull
25    public String username;
26  
27    @Parameter(names = "--datacite-password")
28    @NotNull
29    public String password;
30  
31    @Parameter(names = "--datacite-api")
32    @Nullable
33    public URI api;
34  
35    @Parameter(names = "--datacite-threads")
36    public int threads = 10;
37  
38    @Parameter(names = "--datacite-timeout")
39    public int timeout = 20000;
40  
41    public DataCiteService createService() {
42      LOG.debug("Creating DataCite doi service");
43      ServiceConfig cfg = new ServiceConfig(username, password);
44      cfg.setApi(api);
45      return new DataCiteService(HttpUtil.newMultithreadedClient(timeout, threads, threads), cfg);
46    }
47  
48  }
