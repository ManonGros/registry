23  /**
24   * Message callback implementation to take DOI updates and send them to DataCite. Updates the status of the DOI in
25   * the registry database.
26   */
27  public class DoiUpdateListener extends AbstractMessageCallback<ChangeDoiMessage> {
28  
29      private static final Logger LOG = LoggerFactory.getLogger(DoiUpdateListener.class);
30      private static final Marker DOI_SMTP = MarkerFactory.getMarker("DOI_SMTP");
31      private final long timeToRetryInMs;
32  
33      private final DoiService doiService;
34      private final DoiMapper doiMapper;
35  
36      public DoiUpdateListener(DoiService doiService, DoiMapper doiMapper, long timeToRetryInMs) {
37          this.doiService = doiService;
38          this.doiMapper = doiMapper;
39          this.timeToRetryInMs = timeToRetryInMs;
40      }
41  
42      @Override
43      public void handleMessage(ChangeDoiMessage msg) {
44          LOG.debug("Handling change DOI to {} message for {}", msg.getStatus(), msg.getDoi());
45          final DoiData currState = doiMapper.get(msg.getDoi());
46          if (currState == null) {
47              // this is bad, we should have an entry for the DOI in our registry table!
48              LOG.warn("Skipping unknown GBIF DOI {}", msg.getDoi());
49              return;
50          }
51  
52          boolean descriptionTruncated = false;
53          for (int retry = 1; retry < 4; retry++) {
54              try {
55                  switch (msg.getStatus()) {
56                      case REGISTERED:
57                          registerOrUpdate(msg.getDoi(), msg.getTarget(), msg.getMetadata(), currState);
58                          break;
59                      case RESERVED:
60                          reserve(msg.getDoi(), msg.getMetadata(), currState);
61                          break;
62                      case DELETED:
63                          delete(msg.getDoi(), currState);
64                          break;
65                      default:
66                          LOG.warn("Cannot update {} to illegal state {}.", msg.getDoi(), msg.getStatus());
67                          break;
68                  }
69                  break;
70  
71              } catch (DoiExistsException e) {
72                  writeFailedStatus(msg.getDoi(), msg.getTarget(), msg.getMetadata());
73                  LOG.warn(DOI_SMTP, "DOI {} existed already when trying to change status to {}. Ignore", msg.getDoi(), msg.getStatus(), e);
74                  break;
75  
76              } catch (DoiHttpException e) {
77                  writeFailedStatus(msg.getDoi(), msg.getTarget(), msg.getMetadata());
78                  if (e.getStatus() == 413) {
79                      LOG.warn(DOI_SMTP, "Metadata of length {} is exceeding max datacite limit in attempt #{} "
80                                      + "while updating {} to {} with target {}. "
81                                      + "Trying again {}",
82                              msg.getMetadata().length(), retry, msg.getDoi(), msg.getStatus(), msg.getTarget(),
83                              descriptionTruncated ? "without constituent information" : "with truncated description", e);
84                      try {
85                          String truncatedXml;
86                          if (descriptionTruncated) {
87                              LOG.warn("Truncating all constituent relations as last resort from metadata for DOI {}", msg.getDoi());
88                              truncatedXml = DataCiteConverter.truncateConstituents(msg.getDoi(), msg.getMetadata(), msg.getTarget());
89                          } else {
90                              LOG.debug("Original metadata for DOI {}:\n\n{}", msg.getDoi(), msg.getMetadata());
91                              truncatedXml = DataCiteConverter.truncateDescription(msg.getDoi(), msg.getMetadata(), msg.getTarget());
92                              descriptionTruncated = true;
93                          }
94                          msg = new ChangeDoiMessage(msg.getStatus(), msg.getDoi(), truncatedXml, msg.getTarget());
95                      } catch (InvalidMetadataException e1) {
96                          LOG.warn("Failed to deserialize xml metadata for DOI {}", msg.getDoi(), e1);
97                      }
98                  } else {
99                      LOG.warn(DOI_SMTP, "DOI http {} exception updating {} to {} with target {}. Attempt #{}",
100                             e.getStatus(), msg.getDoi(), msg.getStatus(), msg.getTarget(), retry, e);
101                     sleep();
102                 }
103 
104             } catch (DoiException e) {
105                 writeFailedStatus(msg.getDoi(), msg.getTarget(), msg.getMetadata());
106                 LOG.warn(DOI_SMTP, "DOI exception updating {} to {} with target {}. Attempt #{}",
107                         msg.getDoi(), msg.getStatus(), msg.getTarget(), retry, e);
108                 sleep();
109             }
110         }
111     }
112 
113     private void sleep() {
114         try {
115             Thread.sleep(timeToRetryInMs);
116         } catch (InterruptedException e1) {
117             LOG.info("Interrupted retries");
118         }
119     }
120 
121     private void writeFailedStatus(DOI doi, URI target, String xml) {
122         doiMapper.update(doi, new DoiData(DoiStatus.FAILED, target), xml);
123     }
124 
125     private void reserve(DOI doi, String xml, DoiData currState) throws DoiException {
126         doiService.reserve(doi, xml);
127         LOG.info("Reserved doi {}", doi);
128         DoiData newState = new DoiData(DoiStatus.RESERVED, currState.getTarget());
129         doiMapper.update(doi, newState, xml);
130     }
131 
132     private void delete(DOI doi, DoiData currState) throws DoiException {
133         if (currState.getStatus() == null) {
134             doiMapper.delete(doi);
135         } else {
136             try {
137                 boolean fullDeleted = doiService.delete(doi);
138                 if (fullDeleted) {
139                     doiMapper.delete(doi);
140                     LOG.info("Deleted doi {}", doi);
141                 } else {
142                     DoiData newState = new DoiData(DoiStatus.DELETED, currState.getTarget());
143                     doiMapper.update(doi, newState, null);
144                     LOG.info("Marked registered doi {} as deleted", doi);
145                 }
146             } catch (DoiHttpException e) {
147                 // in case of a 404 swallow
148                 if (e.getStatus() == 404) {
149                     LOG.warn(DOI_SMTP, "Trying to delete DOI {} failed because it doesn't exist in DataCite; deleting locally",
150                             doi, e);
151                     doiMapper.delete(doi);
152                 } else {
153                     throw e;
154                 }
155             }
156         }
157     }
158 
159     private void registerOrUpdate(DOI doi, URI target, String xml, DoiData currState) throws DoiException {
160         if (DoiStatus.REGISTERED == currState.getStatus()) {
161             // the DOI was already registered, so we only need to update the target url if changed and the metadata
162             if (!target.equals(currState.getTarget())) {
163                 doiService.update(doi, target);
164             }
165             doiService.update(doi, xml);
166             LOG.info("Updated doi {} with target {}", doi, target);
167         } else {
168             doiService.register(doi, target, xml);
169             LOG.info("Registered doi {} with target {}", doi, target);
170         }
171         // store the new state in our registry
172         doiMapper.update(doi, new DoiData(DoiStatus.REGISTERED, target), xml);
173     }
174 }
