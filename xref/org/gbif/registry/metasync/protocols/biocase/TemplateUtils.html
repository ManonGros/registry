3   import org.gbif.registry.metasync.util.Constants;

5   import java.io.StringWriter;
6   import java.util.Properties;
7   
8   import org.apache.velocity.VelocityContext;
9   import org.apache.velocity.app.VelocityEngine;
10  import org.apache.velocity.context.Context;
11  
12  /**
13   * Utility class used by the BioCASe synchroniser to load the various templates from the classpath and fill them with
14   * the proper variables.
15   */
16  final class TemplateUtils {
17  
18    private static final VelocityEngine VELOCITY_ENGINE = new VelocityEngine();
19  
20    static {
21      Properties properties = new Properties();
22      properties.setProperty("file.resource.loader.class",
23                             "org.apache.velocity.runtime.resource.loader.ClasspathResourceLoader");
24      VELOCITY_ENGINE.init(properties);
25    }
26  
27    /**
28     * Returns a string that can be used to retrieve an "old style" inventory from a BioCASe Endpoint using a {@code scan}
29     * request.
30     */
31    static String getBiocaseInventoryRequest(String contentNamespace) {
32      Context context = new VelocityContext();
33      context.put("contentNamespace", contentNamespace);
34      context.put("concept", getTitlePath(contentNamespace));
35  
36      StringWriter writer = new StringWriter();
37      VELOCITY_ENGINE.mergeTemplate("biocase/inventory.xml.vm", "UTF-8", context, writer);
38      return writer.toString();
39    }
40  
41    /**
42     * Returns a string that can be used to retrieve metadata about a specific dataset identified by a title. You get that
43     * title by first doing an inventory request. Under tho hood this does a {@code search} request.
44     */
45    public static String getBiocaseMetadataRequest(String contentNamespace, String datasetTitle) {
46      Context context = new VelocityContext();
47      context.put("contentNamespace", contentNamespace);
48      context.put("datasetTitle", datasetTitle);
49      context.put("titleConcept", getTitlePath(contentNamespace));
50  
51      if (contentNamespace.equals(Constants.ABCD_12_SCHEMA)) {
52        context.put("nameConcept",
53                    "/DataSets/DataSet/Units/Unit/Identifications/Identification/TaxonIdentified/NameAuthorYearString");
54      } else if (contentNamespace.equals(Constants.ABCD_206_SCHEMA)) {
55        context.put("nameConcept",
56                    "/DataSets/DataSet/Units/Unit/Identifications/Identification/Result/TaxonIdentified/ScientificName/FullScientificNameString");
57      }
58  
59      StringWriter writer = new StringWriter();
60      VELOCITY_ENGINE.mergeTemplate("/biocase/metadata.xml.vm", "UTF-8", context, writer);
61      return writer.toString();
62    }
63  
64    private static String getTitlePath(String contentNamespace) {
65      if (contentNamespace.equals(Constants.ABCD_12_SCHEMA)) {
66        return "/DataSets/DataSet/OriginalSource/SourceName";
67      }
68      if (contentNamespace.equals(Constants.ABCD_206_SCHEMA)) {
69        return "/DataSets/DataSet/Metadata/Description/Representation/Title";
70      }
71      return null;
72    }
73  
74    private TemplateUtils() {
75      throw new UnsupportedOperationException("Can't initialize class");
76    }
77  
78  }
