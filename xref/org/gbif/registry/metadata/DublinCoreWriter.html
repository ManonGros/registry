33  /**
34   * Writer to serialize a Dataset as DublinCore XML document.
35   * Currently using a OAI DC profile.
36   *
37   *
38   */
39  public class DublinCoreWriter {
40  
41    public static final String ADDITIONAL_PROPERTY_OCC_COUNT = "occurrence_count";
42    public static final String ADDITIONAL_PROPERTY_DC_FORMAT = "dublincore_format";
43  
44    private static final String DC_TEMPLATE = "oai-dc-profile-template/dc-dataset.ftl";
45    private final Configuration freemarkerConfig;
46  
47    //We should probably use @Named("portal.url") but it would be more appropriate to wait
48    //until we turn this class in a non static one.
49    private static final String IDENTIFIER_PREFIX = "http://www.gbif.org/dataset/";
50    private static final FastDateFormat FDF = DateFormatUtils.ISO_DATE_FORMAT;
51    private static final TemporalCoverageFormatterVisitor TC_FORMATTER = new DcTemporalTemporalCoverageFormatter();
52  
53    /**
54     * Private constructor, use {@link #newInstance()}
55     * @param cfg
56     */
57    private DublinCoreWriter(Configuration cfg) {
58      freemarkerConfig = cfg;
59    }

70    /**
71     * Write a DublinCore document from a Dataset object.
72     *
73     * @param organization organization who published this dataset, should not be null but nulls are handled.
74     * @param dataset      non null dataset object
75     * @param additionalProperties
76     * @param writer       where the output document will go. The writer is not closed by this method.
77     *
78     * @throws IOException if an error occurs while processing the template
79     */
80    public void writeTo(@Nullable Organization organization, @NotNull Dataset dataset, Map<String,Object> additionalProperties, Writer writer) throws IOException {
81      Preconditions.checkNotNull(dataset, "Dataset can't be null");
82      Map<String, Object> map = Maps.newHashMap();
83      map.put("dataset", dataset);
84      map.put("dc", new DcDatasetWrapper(dataset, additionalProperties));
85  
86      if (organization != null) {
87        map.put("organization", organization);
88      }
89      map = ImmutableMap.copyOf(map);
90      try {
91        freemarkerConfig.getTemplate(DC_TEMPLATE).process(map, writer);
92      } catch (TemplateException e) {
93        throw new IOException("Error while processing the DublinCore Freemarker template for dataset " + dataset.getKey(), e);
94      }
95    }
96  
97    /**
98     * This class requires to be public to be used in the Freemarker template.
99     */
100   public static class DcDatasetWrapper {
101     private final Dataset dataset;
102     private final ContactAdapter contactAdapter;
103     private final Map<String,Object> additionalProperties;
104 
105     public DcDatasetWrapper(Dataset dataset){
106       this(dataset, null);
107     }
108 
109     public DcDatasetWrapper(Dataset dataset, Map<String,Object> additionalProperties) {
110       this.dataset = dataset;
111       this.contactAdapter = new ContactAdapter(dataset.getContacts());
112       this.additionalProperties = additionalProperties;
113     }
114 
115     public String getFormat(){
116       if(additionalProperties !=null && additionalProperties.containsKey(ADDITIONAL_PROPERTY_DC_FORMAT)){
117         return (String)additionalProperties.get(ADDITIONAL_PROPERTY_DC_FORMAT);
118       }
119       return "";
120     }
121 
122     /**
123      * The following ordering will be respected: ADMINISTRATIVE_POINT_OF_CONTACT, METADATA_AUTHOR and ORIGINATOR
124      * @return set of formatted contact name ordered by ContactType
125      */
126     public Set<String> getCreators() {
127       Set<String> creators = Sets.newLinkedHashSet();
128       List<Contact> filteredContacts = contactAdapter.getFilteredContacts(ContactType.ADMINISTRATIVE_POINT_OF_CONTACT,
129               ContactType.METADATA_AUTHOR, ContactType.ORIGINATOR);
130       for(Contact contact : filteredContacts){
131         creators.add(ContactAdapter.formatContactName(contact));
132       }
133       return creators;
134     }
135 
136     /**
137      * Get a list of "description" for DublinCore usage.
138      * @return never null
139      */
140     public List<String> getDescription(){
141       List<String> descriptions = Lists.newArrayList();
142       if(StringUtils.isNotBlank(dataset.getDescription())){
143         descriptions.add(dataset.getDescription());
144       }
145       return descriptions;
146     }
147 
148     public Long getOccurrenceCount(){
149       if(additionalProperties !=null && additionalProperties.containsKey(ADDITIONAL_PROPERTY_OCC_COUNT)){
150         return (Long)additionalProperties.get(ADDITIONAL_PROPERTY_OCC_COUNT);
151       }
152       return null;
153     }
154 
155     /**
156      * Get a list of "coverage" formatted for DublinCore usage.
157      *
158      * @return never null
159      */
160     public List<String> getFormattedTemporalCoverage(){
161       List<String> coverage = Lists.newArrayList();
162       if(dataset.getTemporalCoverages() != null){
163         for(TemporalCoverage tc : dataset.getTemporalCoverages()){
164           coverage.add(tc.acceptFormatter(TC_FORMATTER));
165         }
166       }
167       return coverage;
168     }
169 
170     public String getIdentifier() {
171       return IDENTIFIER_PREFIX + dataset.getKey().toString();
172     }
173 
174     public Contact getResourceCreator() {
175       return contactAdapter.getResourceCreator();
176     }
177   }
178 
179   /**
180    * Implementation of TemporalCoverageFormatterVisitor for DublinCore
181    */
182   private static class DcTemporalTemporalCoverageFormatter implements TemporalCoverageFormatterVisitor {
183 
184     @Override
185     public String format(DateRange dateRange) {
186       String startDate = dateRange.getStart() != null ? FDF.format(dateRange.getStart()) : "";
187       String endDate = dateRange.getEnd() != null ? FDF.format(dateRange.getEnd()) : "";
188       return startDate + "/" + endDate;
189     }
190 
191     @Override
192     public String format(SingleDate singleDate) {
193       if(singleDate.getDate() == null){
194         return "";
195       }
196       return FDF.format(singleDate.getDate());
197     }
198 
199     // unsupported
200     @Override
201     public String format(VerbatimTimePeriod verbatimTimePeriod) {
202       return "";
203     }
204   }
