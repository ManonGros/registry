25  /**
26   * Main parser of dataset metadata that uses parser specific digester RuleSets for EML or Dublin Core.
27   * It can automatically detect the document type if it is unknown or be used only with a specific parser type.
28   *
29   * This parser and its digester rules use the DatasetDelegator class to wrap a dataset and set complex bean components.
30   */
31  public class DatasetParser {
32  
33    private static final Logger LOG = LoggerFactory.getLogger(DatasetParser.class);
34  
35    private static class ParserDetectionHandler extends DefaultHandler {
36      private static final String DC_NAMESPACE = "http://purl.org/dc/terms/";
37      private MetadataType parserType;
38      private LinkedList<String> path = Lists.newLinkedList();
39  
40      @Override
41      public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
42        // look for EML
43        if (path.size() == 1 && path.get(0).equals("eml") && localName.equals("dataset")) {
44          parserType = MetadataType.EML;
45        }
46  
47        // look for DC title
48        if (parserType == null && DC_NAMESPACE.equals(uri)) {
49          parserType = MetadataType.DC;
50        }
51  
52        path.add(localName);
53      }
54  
55      @Override
56      public void endElement(String uri, String localName, String qName) throws SAXException {
57        String last = path.removeLast();
58        if (!last.equals(localName)) {
59          LOG.warn("XML path broken. Got {} but path stack gave {}", localName, last);
60        }
61      }
62    }
63  
64    /**
65     * @return the detected parser type or null
66     * @throws java.lang.IllegalArgumentException in case no parser exists for this document
67     */
68    public static MetadataType detectParserType(InputStream xml) {
69      try {
70        XMLReader xmlReader = XMLReaderFactory.createXMLReader();
71        ParserDetectionHandler handler = new ParserDetectionHandler();
72        xmlReader.setContentHandler(handler);
73        InputSource inputSource = new InputSource(xml);
74        xmlReader.parse(inputSource);
75        if (handler.parserType != null) {
76          return handler.parserType;
77        }
78  
79      } catch (SAXException e) {
80        LOG.error("Failed to SAX parse a document for parser type detection", e);
81      } catch (IOException e) {
82        LOG.warn("Failed to read metadata document for parser type detection", e);
83      }
84      throw new IllegalArgumentException("No parser found for this metadata document. Only EML or DC supported");
85    }
86  
87  
88    /**
89     * Build from stream on-top of a preexisting Dataset populating its fields from a source metadata that's parsed.
90     *
91     * @param xml to read
92     * @return The Dataset populated, never null
93     * @throws java.io.IOException If the Stream cannot be read from
94     * @throws IllegalArgumentException If the XML is not well formed or is not understood
95     */
96    public static Dataset build(InputStream xml) throws IOException {
97      // buffer entire stream first. We need it several times
98      final byte[] data = ByteStreams.toByteArray(xml);
99      // detect the parser type
100     return parse(detectParserType(new ByteArrayInputStream(data)), new ByteArrayInputStream(data));
101   }
102 
103   public static Dataset parse(MetadataType type, InputStream xml) throws IOException {
104     Digester digester = new Digester();
105     digester.setNamespaceAware(true);
106 
107     // add digester rules based on parser type
108     switch (type) {
109       case EML:
110         LOG.debug("Parsing EML document");
111         digester.addRuleSet(new EMLRuleSet());
112         break;
113       case DC:
114         LOG.debug("Parsing DC document");
115         digester.addRuleSet(new DublinCoreRuleSet());
116         break;
117     }
118 
119     // push the Delegating object onto the stack
120     DatasetWrapper delegator = new DatasetWrapper();
121     digester.push(delegator);
122 
123     // now parse and return the dataset
124     try {
125         digester.parse(xml);
126     } catch (ConversionException e) {
127       // swallow
128     } catch (SAXException e) {
129       if (e.getException() == null || !e.getException().getClass().equals(ConversionException.class)) {
130         // allow type conversions to happen
131         throw new IllegalArgumentException("Invalid metadata xml document", e);
132       }
133     } finally {
134       delegator.postProcess();
135       Closeables.closeQuietly(xml);
136     }
137 
138     return delegator.getTarget();
139   }
