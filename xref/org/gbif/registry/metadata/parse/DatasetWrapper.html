56    /**
57     * Utility to parse an EML calendarDate in a textual format. Can be ISO date or just the year, ignoring whitespace
58     *
59     * @param dateString To set in format YYYY-MM-DD
60     *
61     * @return the parsed date
62     *
63     * @throws java.text.ParseException Should it be an erroneous format
64     * @see <a href="http://knb.ecoinformatics.org/software/eml/eml-2.1.0/eml-coverage.html#calendarDate">EML Coverage
65     *      calendarDate keyword</a>
66     */
67    private static Date calendarDate(String dateString) throws ParseException {
68      if (Strings.isNullOrEmpty(dateString)) {
69        return null;
70      }
71      // kill whitespace
72      dateString = dateString.replaceAll("\\s", "");
73      dateString = dateString.replaceAll("[\\,._#//]", "-");
74      Date date;
75      try {
76        SimpleDateFormat iso = new SimpleDateFormat("yyyy-MM-dd");
77        date = iso.parse(dateString);
78      } catch (ParseException e) {
79        if (dateString.length() == 4) {
80          SimpleDateFormat sdf = new SimpleDateFormat("yyyy");
81          date = sdf.parse(dateString);
82          date = new Date(date.getTime() + 1);
83        } else {
84          throw e;
85        }
86      }
87      return date;
88    }

110   /**
111    * @return true if the minimal required contact information exists
112    */
113   private boolean verifyContact(Contact contact) {
114     return contact.getFirstName() !=null
115            || contact.getLastName()  !=null
116            || !contact.getPosition().isEmpty()
117            || !contact.getEmail().isEmpty()
118            || !contact.getPhone().isEmpty()
119            || contact.getOrganization() !=null
120            || (!contact.getAddress().isEmpty() && contact.getCity() !=null);
121   }

154   /**
155    * Adds a comma or semicolon concatenated keyword string as keyword collection.
156    */
157   public void addSubjects(String subjects) {
158     Splitter keywordSplitter = Splitter.on(Pattern.compile("[,;]")).trimResults().omitEmptyStrings();
159     if (!Strings.isNullOrEmpty(subjects)) {
160       KeywordCollection collection = new KeywordCollection();
161       for (String keyword : keywordSplitter.split(subjects)) {
162         collection.addKeyword(keyword);
163       }
164       target.getKeywordCollections().add(collection);
165     }
166   }

366   /**
367    * Concatenates the new paragrpah to an existing one, inserting a new html break line.
368    * @param existing
369    * @param para
370    */
371   private String appendParagraph(String existing, String para) {
372     if (!para.isEmpty()) {
373       return existing + "<br/>" + para.trim();
374     }
375     return existing;
376   }

475   private void updateTaxonomicCoverageRanks() {
476     for (TaxonomicCoverages tc : target.getTaxonomicCoverages()) {
477       for (TaxonomicCoverage t : tc.getCoverages()) {
478         if (t.getRank() != null) {
479           t.getRank().setInterpreted(toRank(t.getRank().getVerbatim()));
480         }
481       }
482     }
483   }

485   /**
486    * This extracts the first DOI from alternate identifier or gbif citation identifier if no target DOI existed, e.g.
487    * found in the packageID rule before or from the initial target instance.
488    */
489   private void updatePrimaryDOI() {
490     if (target.getDoi() == null) {
491       Iterator<Identifier> iter = target.getIdentifiers().iterator();
492       while (iter.hasNext()) {
493         Identifier i = iter.next();
494         if (i.getType() == IdentifierType.DOI) {
495           if (DOI.isParsable(i.getIdentifier())) {
496             target.setDoi( new DOI(i.getIdentifier()) );
497             iter.remove();
498             return;
499           }
500         }
501       }
502       // at last also check the citation field
503       if (!Strings.isNullOrEmpty(target.getCitation().getIdentifier())) {
504         try {
505           target.setDoi( new DOI(target.getCitation().getIdentifier()) );
506         } catch (IllegalArgumentException e) {
507           // invalid DOI, skip
508         }
509       }
510     }
511   }

513   private Rank toRank(String rank) {
514     ParseResult<Rank> result = RankParser.getInstance().parse(rank);
515     return result.getStatus() == ParseResult.STATUS.SUCCESS ? result.getPayload() : null;
516   }
